class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        res = start = 0
        dic = {}
        for i, char in enumerate(s):
            if char in dic and start <= dic[char]:
                start = dic[char] + 1
            else:
                res = max(res, i - start + 1)
            dic[char] = i
        return res
