class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        ans = 0
        i, j = 0, 0
        seen = set()
        while j < len(s):
            if s[j] not in seen:
                seen.add(s[j])
                j += 1
                ans = max(ans, j - i)
            else:
                seen.remove(s[i])
                i += 1
        return ans
