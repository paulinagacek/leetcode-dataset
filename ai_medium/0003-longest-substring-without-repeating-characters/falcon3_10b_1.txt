class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        longest = 0
        start = 0
        used_chars = {}
        
        for i, char in enumerate(s):
            if char in used_chars and start <= used_chars[char]:
                start = used_chars[char] + 1
            else:
                longest = max(longest, i - start + 1)
            
            used_chars[char] = i
        
        return longest
