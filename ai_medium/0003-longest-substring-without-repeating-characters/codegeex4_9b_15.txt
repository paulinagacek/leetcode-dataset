class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        n = len(s)
        max_len = 0
        start = 0
        char_index_map = {}
        
        for i in range(n):
            if s[i] in char_index_map:
                start = max(start, char_index_map[s[i]] + 1)
            char_index_map[s[i]] = i
            max_len = max(max_len, i - start + 1)
        
        return max_len
