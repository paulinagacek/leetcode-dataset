class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        n = len(s)
        max_len = 0
        start = 0
        seen = {}
        
        for i in range(n):
            if s[i] in seen and seen[s[i]] >= start:
                start = seen[s[i]] + 1
            seen[s[i]] = i
            max_len = max(max_len, i - start + 1)
        
        return max_len
