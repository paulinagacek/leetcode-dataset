class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        visited = set()  # Keep track of visited characters
        left = 0  # Left pointer for window traversal
        longest = 0  # Length of the longest substring without duplicates

        for right in range(len(s)):
            while s[right] in visited:
                visited.remove(s[left])
                left += 1
            visited.add(s[right])
            longest = max(longest, right - left + 1)

        return longest
