class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if not s:
            return 0

        last_seen = {}
        max_len = 0
        start = 0

        for i, c in enumerate(s):
            if c in last_seen and last_seen[c] >= start:
                start = last_seen[c] + 1
            else:
                max_len = max(max_len, i - start + 1)

            last_seen[c] = i

        return max_len
