class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        n = len(s)
        longest, start, seen = 0, 0, {}

        for i, char in enumerate(s):
            if char in seen and start <= seen[char]:
                start = seen[char] + 1
            else:
                longest = max(longest, i - start + 1)
            seen[char] = i

        return longest
