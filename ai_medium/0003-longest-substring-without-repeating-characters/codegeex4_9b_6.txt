class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        max_length = 0
        last_seen = {}
        start = 0
        
        for i, char in enumerate(s):
            if char in last_seen and last_seen[char] >= start:
                start = last_seen[char] + 1
            else:
                max_length = max(max_length, i - start + 1)
            
            last_seen[char] = i
        
        return max_length
