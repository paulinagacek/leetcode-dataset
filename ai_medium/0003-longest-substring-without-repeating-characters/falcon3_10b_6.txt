class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        left = 0
        max_len = 0
        char_map = {}
        
        for right in range(len(s)):
            if s[right] in char_map:
                left = max(char_map[s[right]], left)
            max_len = max(max_len, right - left + 1)
            char_map[s[right]] = right + 1
        
        return max_len
