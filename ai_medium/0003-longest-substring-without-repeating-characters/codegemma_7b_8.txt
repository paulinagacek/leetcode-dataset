class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        seen = set()
        max_length = 0
        start_index = 0
        
        for i, char in enumerate(s):
            while char in seen:
                seen.remove(s[start_index])
                start_index += 1
            seen.add(char)
            max_length = max(max_length, i - start_index + 1)
        
        return max_length
