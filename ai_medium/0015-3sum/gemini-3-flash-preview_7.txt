class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        res = []
        
        for i in range(n - 2):
            # Skip duplicate elements for the first element of the triplet
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            
            # Optimization: If the smallest possible sum with nums[i] is > 0, stop
            if nums[i] + nums[i + 1] + nums[i + 2] > 0:
                break
            # Optimization: If the largest possible sum with nums[i] is < 0, continue
            if nums[i] + nums[n - 2] + nums[n - 1] < 0:
                continue
                
            l, r = i + 1, n - 1
            while l < r:
                s = nums[i] + nums[l] + nums[r]
                if s < 0:
                    l += 1
                elif s > 0:
                    r -= 1
                else:
                    res.append([nums[i], nums[l], nums[r]])
                    # Skip duplicate elements for the second element
                    while l < r and nums[l] == nums[l + 1]:
                        l += 1
                    # Skip duplicate elements for the third element
                    while l < r and nums[r] == nums[r - 1]:
                        r -= 1
                    l += 1
                    r -= 1
        return res
