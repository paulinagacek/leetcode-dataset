class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = []
        n = len(nums)
        for i in range(n - 2):
            # Skip duplicate elements for the first position
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            
            # Optimization: If the smallest possible sum is greater than 0, no point continuing
            if nums[i] + nums[i + 1] + nums[i + 2] > 0:
                break
            
            # Optimization: If the largest possible sum with this nums[i] is less than 0, skip to next i
            if nums[i] + nums[n - 2] + nums[n - 1] < 0:
                continue

            l, r = i + 1, n - 1
            while l < r:
                s = nums[i] + nums[l] + nums[r]
                if s < 0:
                    l += 1
                elif s > 0:
                    r -= 1
                else:
                    res.append([nums[i], nums[l], nums[r]])
                    # Skip duplicate elements for the second and third positions
                    while l < r and nums[l] == nums[l + 1]:
                        l += 1
                    while l < r and nums[r] == nums[r - 1]:
                        r -= 1
                    l += 1
                    r -= 1
        return res
