class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1:
            return s

        numRows -= 1
        n = len(s)
        result = [''] * n

        idx = 0
        step = 1
        for letter in s:
            result[idx] = letter
            idx += step
            if idx == numRows:
                idx -= 2
                step = -1
            elif idx == -1:
                idx += 2
                step = 1

        return ''.join(result)
