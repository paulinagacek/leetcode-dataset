class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1 or numRows >= len(s):
            return s
        
        s = list(s)
        result = [" "] * len(s)
        direction = 1
        row = 0
        
        for letter in s:
            result[row] = letter
            row += direction
            if row == 0 or row == numRows - 1:
                direction *= -1
                
        return "".join(result)
