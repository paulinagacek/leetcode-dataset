class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows <= 1 or numRows >= len(s):
            return s

        rows = [''] * numRows
        r = 0
        direction = 1  # 1 down, -1 up

        for ch in s:
            rows[r] += ch
            if r == 0:
                direction = 1
            elif r == numRows - 1:
                direction = -1
            r += direction

        return ''.join(rows)
