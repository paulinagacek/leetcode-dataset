class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1:
            return s
        result = [''] * numRows
        row, direction = 0, 1
        for char in s:
            result[row] += char
            row += direction
            if row == numRows:
                direction = -1
            elif row == 0:
                direction = 1
        return ''.join(result)
