class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows <= 1:
            return s
        result = [''] * numRows
        row, step = 0, 1
        for char in s:
            result[row] += char
            row += step
            if row == 0 or row == numRows - 1:
                step *= -1
        return ''.join(result)
