class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If the number of rows is 1 or greater than/equal to the string length, 
        # the pattern won't change.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Initialize a list of strings to represent each row.
        rows = [""] * numRows
        cur_row = 0
        step = 1
        
        # Iterate through each character in the string.
        for char in s:
            rows[cur_row] += char
            
            # If we hit the first row, we must start moving down.
            if cur_row == 0:
                step = 1
            # If we hit the last row, we must start moving up.
            elif cur_row == numRows - 1:
                step = -1
            
            # Update the current row based on the step direction.
            cur_row += step
            
        # Concatenate all rows to form the final string.
        return "".join(rows)
