class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1 or len(s) <= numRows:
            return s
        
        zigzag = [''] * numRows
        downward = False
        row = 0
        
        for char in s:
            zigzag[row] += char
            if row == 0:
                downward = True
            elif row == numRows - 1:
                downward = False
            if downward:
                row += 1
            else:
                row -= 1
        
        return ''.join(zigzag)
