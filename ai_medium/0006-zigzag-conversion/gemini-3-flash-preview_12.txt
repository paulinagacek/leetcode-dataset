class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If numRows is 1 or more than or equal to the length of s, 
        # the zigzag pattern is just the string itself.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Initialize a list of strings for each row
        rows = [''] * numRows
        curr_row = 0
        step = 1
        
        for char in s:
            # Add the current character to the appropriate row
            rows[curr_row] += char
            
            # If we reach the top or bottom row, change the direction
            if curr_row == 0:
                step = 1
            elif curr_row == numRows - 1:
                step = -1
            
            # Move to the next row
            curr_row += step
            
        # Combine all rows to get the final converted string
        return "".join(rows)
