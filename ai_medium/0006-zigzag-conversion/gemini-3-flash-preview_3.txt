class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If numRows is 1 or greater than or equal to the length of the string,
        # the zigzag pattern does not change the order of characters.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Create a list of strings, one for each row.
        rows = [""] * numRows
        curr_row = 0
        step = 1
        
        # Iterate through each character in the string.
        for char in s:
            # Append the character to the current row's string.
            rows[curr_row] += char
            
            # If we reach the first or last row, change the direction of traversal.
            if curr_row == 0:
                step = 1
            elif curr_row == numRows - 1:
                step = -1
                
            # Move to the next row according to the current direction.
            curr_row += step
            
        # Join all the rows' strings together to get the final zigzag pattern.
        return "".join(rows)
