class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If the number of rows is 1 or if there aren't enough characters to zigzag,
        # the string remains the same.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Initialize an array of strings for each row.
        rows = [''] * numRows
        index, step = 0, 1
        
        for char in s:
            rows[index] += char
            
            # Change direction when we hit the top or bottom row.
            if index == 0:
                step = 1
            elif index == numRows - 1:
                step = -1
            
            index += step
            
        # Combine all rows to get the final zigzag string.
        return "".join(rows)
