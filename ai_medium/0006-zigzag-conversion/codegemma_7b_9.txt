class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1 or numRows >= len(s):
            return s
        
        result = [''] * numRows
        row, direction = 0, 1
        
        for char in s:
            result[row] += char
            row += direction
            
            if row == 0 or row == numRows - 1:
                direction *= -1
        
        return ''.join(result)
