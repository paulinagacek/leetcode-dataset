class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If there's only one row or the number of rows is greater than or equal to string length,
        # the zigzag pattern is just the string itself.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Create a list of empty strings for each row.
        rows = [''] * numRows
        current_row = 0
        step = 1  # Direction: 1 for moving down, -1 for moving up.
        
        for char in s:
            rows[current_row] += char
            
            # If we reach the first row, we move downwards.
            if current_row == 0:
                step = 1
            # If we reach the last row, we move upwards.
            elif current_row == numRows - 1:
                step = -1
            
            # Move to the next row in the specified direction.
            current_row += step
        
        # Join all row strings to produce the final zigzag-transformed string.
        return "".join(rows)
