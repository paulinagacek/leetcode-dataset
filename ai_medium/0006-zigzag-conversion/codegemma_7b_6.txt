class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1:
            return s
        
        rows = [''] * numRows
        row, step = 0, 1
        
        for char in s:
            rows[row] += char
            row += step
            if row == numRows:
                step = -1
            elif row == 0:
                step = 1
        
        return ''.join(rows)
