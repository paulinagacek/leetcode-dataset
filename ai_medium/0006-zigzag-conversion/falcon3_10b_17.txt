class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1 or numRows >= len(s):
            return s
        
        result = [''] * numRows
        current_row = 0
        is_going_down = False
        
        for char in s:
            result[current_row] += char
            if current_row == 0 or current_row == numRows - 1:
                is_going_down = not is_going_down
            if is_going_down:
                current_row += 1
            else:
                current_row -= 1
        
        return ''.join(result)
