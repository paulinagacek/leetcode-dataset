class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows <= 1:
            return s
        
        zigzag = [''] * numRows
        row, direction = 0, 1
        
        for char in s:
            zigzag[row] += char
            if row == 0 or row == numRows - 1:
                direction *= -1
            row += direction
        
        return ''.join(zigzag)
