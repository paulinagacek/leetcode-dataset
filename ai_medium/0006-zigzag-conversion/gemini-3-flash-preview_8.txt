class Solution:
    def convert(self, s: str, numRows: int) -> str:
        # If numRows is 1 or more than the length of the string, 
        # the zigzag pattern is just the string itself.
        if numRows == 1 or numRows >= len(s):
            return s
        
        # Create a list of strings for each row.
        rows = [""] * numRows
        current_row = 0
        step = 1  # 1 means moving down, -1 means moving up
        
        for char in s:
            rows[current_row] += char
            
            # If we reach the first row, we must move down.
            if current_row == 0:
                step = 1
            # If we reach the last row, we must move up.
            elif current_row == numRows - 1:
                step = -1
            
            current_row += step
            
        # Join all row strings to get the final converted string.
        return "".join(rows)
