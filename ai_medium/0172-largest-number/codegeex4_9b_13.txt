from typing import List

class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        def compare(x, y):
            if x + y > y + x:
                return 1
            elif x + y == y + x:
                return 0
            else:
                return -1
        
        nums = [str(num) for num in nums]
        nums.sort(key=lambda x: x*10, cmp=compare)
        largest_num = ''.join(nums)
        
        if largest_num[0] == '0':
            return '0'
        else:
            return largest_num
