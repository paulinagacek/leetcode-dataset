class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        def compare(x, y):
            if x + y > y + x:
                return True
            else:
                return False
        
        def quicksort(arr):
            if len(arr) <= 1:
                return arr
            pivot = arr[len(arr) // 2]
            left = [x for x in arr if compare(x, pivot)]
            middle = [x for x in arr if x == pivot]
            right = [x for x in arr if not compare(x, pivot)]
            return quicksort(left) + middle + quicksort(right)
        
        largest_num = ''.join(map(str, quicksort(nums)))
        return '0' if largest_num[0] == '0' else largest_num
