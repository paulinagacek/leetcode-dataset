class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        strs = list(map(str, nums))

        def cmp(a: str, b: str) -> int:
            ab = a + b
            ba = b + a
            if ab > ba:
                return -1
            if ab < ba:
                return 1
            return 0

        strs.sort(key=cmp_to_key(cmp))
        if strs and strs[0] == "0":
            return "0"
        return "".join(strs)
