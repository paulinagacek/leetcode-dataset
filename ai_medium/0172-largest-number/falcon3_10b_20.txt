class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        def compare(x, y):
            return int(y + x) - int(x + y)

        nums = list(map(str, nums))
        nums.sort(key=cmp_to_key(compare), reverse=True)
        result = ''.join(nums)
        return result.lstrip('0') or '0'
