class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        # Convert all integers to strings for concatenation and comparison
        nums_str = [str(num) for num in nums]
        
        # Custom comparator: if x + y > y + x, then x should come before y
        def compare(n1, n2):
            if n1 + n2 > n2 + n1:
                return -1
            elif n1 + n2 < n2 + n1:
                return 1
            else:
                return 0
        
        # Sort using the custom comparator
        nums_str.sort(key=cmp_to_key(compare))
        
        # Handle the edge case where the largest number is "0" (e.g., nums = [0, 0])
        if nums_str[0] == '0':
            return '0'
        
        # Join the sorted strings to form the final largest number
        return "".join(nums_str)
