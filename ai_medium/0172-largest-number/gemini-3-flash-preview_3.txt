class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        # Convert integers to strings
        nums_str = list(map(str, nums))
        
        # Define custom comparator for sorting
        # If x + y > y + x, then x should come before y
        def compare(x, y):
            if x + y > y + x:
                return -1
            elif x + y < y + x:
                return 1
            else:
                return 0
        
        # Sort based on the custom comparator
        nums_str.sort(key=cmp_to_key(compare))
        
        # Join the sorted list into a single string
        result = "".join(nums_str)
        
        # Handle the edge case where the result is all zeros (e.g., [0, 0])
        # "00" should be returned as "0"
        return "0" if result[0] == "0" else result
