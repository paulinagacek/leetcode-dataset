class Solution:
    def largestNumber(self, nums):
        def compare(x, y):
            if x + y > y + x:
                return 1
            elif x + y == y + x:
                return 0
            else:
                return -1
        
        str_nums = [str(num) for num in nums]
        str_nums.sort(key=lambda x: functools.cmp_to_key(compare)(x, x), reverse=True)
        
        largest_num = ''.join(str_nums)
        if largest_num[0] == '0':
            return '0'
        else:
            return largest_num
