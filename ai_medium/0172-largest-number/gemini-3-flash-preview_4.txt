class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        # Convert integers to strings
        nums_str = [str(num) for num in nums]
        
        # Custom comparator: if a + b > b + a, then a should come before b
        def compare(a, b):
            if a + b > b + a:
                return -1
            elif a + b < b + a:
                return 1
            else:
                return 0
        
        # Sort based on the custom comparator
        nums_str.sort(key=cmp_to_key(compare))
        
        # Join the sorted numbers
        result = "".join(nums_str)
        
        # Handle the edge case where the result is multiple zeros (e.g., [0, 0])
        # If the largest number is '0', the result should be '0'
        return "0" if result[0] == "0" else result
