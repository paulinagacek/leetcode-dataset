import functools

class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        def custom_sort(x, y):
            return y + x > x + y

        nums = list(map(str, nums))
        nums.sort(key=functools.cmp_to_key(custom_sort))
        
        result = ''.join(nums)
        return result if result[0] != '0' else '0'
