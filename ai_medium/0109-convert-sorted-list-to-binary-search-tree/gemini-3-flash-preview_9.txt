class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        # Convert the linked list to an array for easy access to middle elements
        nodes = []
        curr = head
        while curr:
            nodes.append(curr.val)
            curr = curr.next
        
        # Recursive function to build a height-balanced BST from a sorted array
        def buildBST(left: int, right: int) -> Optional[TreeNode]:
            if left > right:
                return None
            
            # Choose the middle element as the root to ensure the tree is height-balanced
            mid = (left + right) // 2
            root = TreeNode(nodes[mid])
            
            # Recursively construct the left and right subtrees
            root.left = buildBST(left, mid - 1)
            root.right = buildBST(mid + 1, right)
            
            return root
        
        return buildBST(0, len(nodes) - 1)
