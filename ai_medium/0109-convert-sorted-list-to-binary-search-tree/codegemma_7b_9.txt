class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        def dfs(start, end):
            if start > end:
                return None
            mid = (start + end) // 2
            root = TreeNode(val=head.val)
            head = head.next
            root.left = dfs(start, mid - 1)
            root.right = dfs(mid + 1, end)
            return root
        size = 0
        dummy = ListNode(next=head)
        while dummy.next:
            size += 1
            dummy = dummy.next
        return dfs(0, size - 1)
