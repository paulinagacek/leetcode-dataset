class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        # Convert the linked list into an array for easier access
        vals = []
        curr = head
        while curr:
            vals.append(curr.val)
            curr = curr.next
        
        # Recursive helper to build a BST from the array
        def buildBST(l: int, r: int) -> Optional[TreeNode]:
            if l > r:
                return None
            
            # Use the middle element to ensure height balance
            mid = (l + r) // 2
            root = TreeNode(vals[mid])
            
            # Recursively build left and right subtrees
            root.left = buildBST(l, mid - 1)
            root.right = buildBST(mid + 1, r)
            
            return root
        
        return buildBST(0, len(vals) - 1)
