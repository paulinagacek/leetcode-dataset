class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        def get_size(node):
            size = 0
            while node:
                size += 1
                node = node.next
            return size
        
        def helper(start, end):
            if start > end:
                return None
            mid = (start + end) // 2
            left = helper(start, mid - 1)
            root = TreeNode(head.val)
            head = head.next
            root.right = helper(mid + 1, end)
            return root
        
        size = get_size(head)
        return helper(0, size - 1)
