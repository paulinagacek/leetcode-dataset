class Solution:
    def sortedListToBST(self, head):
        if not head:
            return None

        def get_size(node):
            count = 0
            while node:
                count += 1
                node = node.next
            return count

        def helper(start, end):
            if start > end:
                return None

            mid = (start + end) // 2
            left = helper(start, mid - 1)
            root = TreeNode(head.val)
            head = head.next
            root.right = helper(mid + 1, end)
            return root

        size = get_size(head)
        return helper(0, size - 1)
