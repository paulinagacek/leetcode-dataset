# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        def find_mid(head, tail):
            slow, fast = head, head
            while fast != tail and fast.next != tail:
                fast = fast.next.next
                slow = slow.next
            return slow

        def build_bst(head, tail):
            if head == tail or head == tail.next:
                return None
            mid = find_mid(head, tail)
            root = TreeNode(mid.val)
            root.left = build_bst(head, mid)
            root.right = build_bst(mid.next, tail)
            return root

        return build_bst(head, None)
