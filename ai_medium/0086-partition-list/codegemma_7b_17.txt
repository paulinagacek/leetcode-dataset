class Solution:
    def partition(self, head: ListNode, x: int) -> ListNode:
        dummy1 = ListNode()  # Dummy node for elements less than x
        dummy2 = ListNode()  # Dummy node for elements greater than or equal to x
        p1, p2 = dummy1, dummy2  # Pointers for dummy nodes

        curr = head
        while curr:
            if curr.val < x:
                p1.next = curr
                p1 = p1.next
            else:
                p2.next = curr
                p2 = p2.next
            curr = curr.next

        p2.next = None  # Ensure no cycles
        p1.next = dummy2.next  # Connect less than to greater than or equal to
        return dummy1.next  # Return head of partitioned list
