class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        less_dummy = ListNode(0)
        ge_dummy = ListNode(0)
        less_tail = less_dummy
        ge_tail = ge_dummy

        cur = head
        while cur:
            nxt = cur.next
            cur.next = None
            if cur.val < x:
                less_tail.next = cur
                less_tail = cur
            else:
                ge_tail.next = cur
                ge_tail = cur
            cur = nxt

        less_tail.next = ge_dummy.next
        return less_dummy.next
