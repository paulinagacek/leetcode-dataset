class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        dummy_less = ListNode()
        dummy_greater = ListNode()
        less, greater = dummy_less, dummy_greater
        curr = head
        while curr:
            if curr.val < x:
                less.next = curr
                less = curr
            else:
                greater.next = curr
                greater = curr
            curr = curr.next
        less.next = dummy_greater.next
        greater.next = None
        return dummy_less.next
