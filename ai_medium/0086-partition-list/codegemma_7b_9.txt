class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        dummy1 = ListNode(next=head)
        dummy2 = ListNode()

        p1, p2 = dummy1, dummy2
        curr = head

        while curr:
            if curr.val < x:
                p1.next = curr
                p1 = p1.next
            else:
                p2.next = curr
                p2 = p2.next
            curr = curr.next

        p2.next = None  # Avoid cycle
        p1.next = dummy2.next

        return dummy1.next
