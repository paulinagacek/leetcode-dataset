class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        dummy1 = ListNode(0) # Dummy node for elements less than x
        dummy2 = ListNode(0) # Dummy node for elements greater than or equal to x
        p1, p2 = dummy1, dummy2
        cur = head
        while cur:
            if cur.val < x:
                p1.next = cur
                p1 = p1.next
            else:
                p2.next = cur
                p2 = p2.next
            cur = cur.next
        p2.next = None # Ensure the second partition is properly terminated
        p1.next = dummy2.next # Connect the two partitions
        return dummy1.next
