# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        lt_head = lt_tail = ListNode()
        ge_head = ge_tail = ListNode()
        
        while head:
            if head.val < x:
                lt_tail.next = head
                lt_tail = lt_tail.next
            else:
                ge_tail.next = head
                ge_tail = ge_tail.next
            head = head.next
        
        lt_tail.next = ge_head.next
        ge_tail.next = None
        
        return lt_head.next
