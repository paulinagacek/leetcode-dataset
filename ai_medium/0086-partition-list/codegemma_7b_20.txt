class Solution:
    def partition(self, head: ListNode, x: int) -> ListNode:
        dummy1 = ListNode(next=head)
        dummy2 = ListNode()
        p1, p2 = dummy1, dummy2
        p = head
        while p:
            if p.val < x:
                p1.next = p
                p1 = p1.next
            else:
                p2.next = p
                p2 = p2.next
            p = p.next
        p1.next = dummy2.next
        return dummy1.next
