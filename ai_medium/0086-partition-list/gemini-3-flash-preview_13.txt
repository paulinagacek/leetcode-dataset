class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        # Dummy nodes to represent the start of the two partitions
        before_dummy = ListNode(0)
        after_dummy = ListNode(0)
        
        # Pointers to manage the end of the current partitioned lists
        before = before_dummy
        after = after_dummy
        
        # Traverse the original linked list
        curr = head
        while curr:
            if curr.val < x:
                # Append to the list containing values less than x
                before.next = curr
                before = before.next
            else:
                # Append to the list containing values greater than or equal to x
                after.next = curr
                after = after.next
            curr = curr.next
        
        # Terminate the after partition to prevent potential cycles
        after.next = None
        
        # Connect the end of the "before" list to the start of the "after" list
        before.next = after_dummy.next
        
        # Return the head of the combined partitioned list
        return before_dummy.next
