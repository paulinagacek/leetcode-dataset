class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        # Create dummy nodes for two separate linked lists
        less_head = ListNode(0)
        greater_head = ListNode(0)
        
        # Pointers to track the last elements of the two lists
        less = less_head
        greater = greater_head
        
        # Traverse the original list
        current = head
        while current:
            if current.val < x:
                less.next = current
                less = less.next
            else:
                greater.next = current
                greater = greater.next
            current = current.next
            
        # Ensure the end of the greater list points to None
        greater.next = None
        
        # Connect the end of the 'less' list to the head of the 'greater' list
        less.next = greater_head.next
        
        # Return the head of the newly partitioned list
        return less_head.next
