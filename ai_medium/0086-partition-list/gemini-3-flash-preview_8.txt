class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        # Create two dummy nodes to start the 'less than' and 'greater than or equal to' lists
        less_head = ListNode(0)
        greater_head = ListNode(0)
        
        # Pointers to the current end of the two lists
        less = less_head
        greater = greater_head
        
        # Traverse the original list
        current = head
        while current:
            if current.val < x:
                less.next = current
                less = less.next
            else:
                greater.next = current
                greater = greater.next
            current = current.next
        
        # Combine the two lists
        # First, ensure the end of the greater list points to None
        greater.next = None
        # Connect the end of the less list to the start of the greater list
        less.next = greater_head.next
        
        # Return the head of the concatenated list (after the dummy node)
        return less_head.next
