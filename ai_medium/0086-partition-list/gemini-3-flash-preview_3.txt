class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        # Create dummy nodes for two lists
        before_head = ListNode(0)
        after_head = ListNode(0)
        
        # Pointers to traverse/build the two lists
        before = before_head
        after = after_head
        
        curr = head
        while curr:
            if curr.val < x:
                # Add to the "before" list
                before.next = curr
                before = before.next
            else:
                # Add to the "after" list
                after.next = curr
                after = after.next
            curr = curr.next
        
        # Connect the two lists
        # Important: set the end of the after list to None to prevent cycles
        after.next = None
        before.next = after_head.next
        
        return before_head.next
