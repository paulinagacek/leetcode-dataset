class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        # Create an adjacency list to represent the graph
        adj = [[] for _ in range(numCourses)]
        # Track the number of prerequisites (in-degree) for each course
        in_degree = [0] * numCourses
        
        for dest, src in prerequisites:
            adj[src].append(dest)
            in_degree[dest] += 1
            
        # Initialize a queue with all courses that have no prerequisites
        queue = deque([i for i in range(numCourses) if in_degree[i] == 0])
        
        # Track the number of courses that can be completed
        completed_courses = 0
        
        while queue:
            curr = queue.popleft()
            completed_courses += 1
            
            # For each course that depends on the current course
            for neighbor in adj[curr]:
                in_degree[neighbor] -= 1
                # If all prerequisites for the neighbor are satisfied, add it to the queue
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
                    
        # If the number of completed courses equals numCourses, there are no cycles
        return completed_courses == numCourses
