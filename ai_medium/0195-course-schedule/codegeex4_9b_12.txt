from typing import List
from collections import defaultdict, deque

class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        # Create a graph and in-degree array
        graph = defaultdict(list)
        in_degree = [0] * numCourses
        
        # Populate the graph and in-degree array based on prerequisites
        for dest, src in prerequisites:
            graph[src].append(dest)
            in_degree[dest] += 1
        
        # Initialize queue with all courses having in-degree 0
        queue = deque([i for i in range(numCourses) if in_degree[i] == 0])
        
        # Number of courses that can be completed
        count = 0
        
        # Process the graph
        while queue:
            node = queue.popleft()
            count += 1
            
            # Visit all adjacent nodes and reduce their in-degree
            for neigh in graph[node]:
                in_degree[neigh] -= 1
                # If in-degree becomes zero, add it to the queue
                if in_degree[neigh] == 0:
                    queue.append(neigh)
        
        # If count is equal to numCourses, all courses can be completed
        return count == numCourses
