from typing import List
from collections import defaultdict, deque

class Solution:
    def canFinish(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        # Create adjacency list and in-degree array
        adj = defaultdict(list)
        in_degree = [0] * numCourses
        
        # Fill adjacency list and in-degree array
        for dest, src in prerequisites:
            adj[src].append(dest)
            in_degree[dest] += 1
        
        # Initialize queue with courses having no prerequisites
        q = deque([i for i in range(numCourses) if in_degree[i] == 0])
        
        # Count of courses that can be completed
        count = 0
        
        # Process the queue
        while q:
            course = q.popleft()
            count += 1
            
            # Decrease in-degree of neighboring nodes
            for neighbor in adj[course]:
                in_degree[neighbor] -= 1
                # If in-degree becomes zero, add to queue
                if in_degree[neighbor] == 0:
                    q.append(neighbor)
        
        # If count of completed courses is equal to total number of courses, return True
        return count == numCourses
