class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        fast = dummy
        slow = dummy
        
        # Advance the fast pointer so that the gap between fast and slow is n nodes
        for _ in range(n):
            fast = fast.next
            
        # Move both pointers until the fast pointer reaches the last node
        while fast.next:
            fast = fast.next
            slow = slow.next
            
        # slow is now pointing to the node before the target node to be removed
        slow.next = slow.next.next
        
        return dummy.next
