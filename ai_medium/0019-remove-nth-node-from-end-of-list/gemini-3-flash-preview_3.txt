class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        fast = dummy
        slow = dummy
        
        # Advance fast pointer so that the distance between slow and fast is n nodes
        for _ in range(n):
            fast = fast.next
            
        # Move both pointers until fast reaches the last node
        while fast.next:
            fast = fast.next
            slow = slow.next
            
        # slow.next is the nth node from the end, remove it
        slow.next = slow.next.next
        
        return dummy.next
