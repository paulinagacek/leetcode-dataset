class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        left, right = 0, len(nums) - 1
        
        while left <= right:
            mid = (left + right) // 2
            
            # If we found the target, return True
            if nums[mid] == target:
                return True
            
            # If left, middle, and right are all the same, we can't tell which side is sorted.
            # However, since nums[mid] is not target, we know nums[left] is not target either.
            # We can simply increment left to shrink the search space.
            if nums[left] < nums[mid]:
                # Left half is sorted
                if nums[left] <= target < nums[mid]:
                    right = mid - 1
                else:
                    left = mid + 1
            elif nums[left] > nums[mid]:
                # Right half is sorted
                if nums[mid] < target <= nums[right]:
                    left = mid + 1
                else:
                    right = mid - 1
            else:
                # nums[left] == nums[mid], shrink the search range
                left += 1
                
        return False
