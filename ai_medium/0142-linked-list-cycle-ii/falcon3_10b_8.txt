class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next: return None
        
        slow, fast = head, head.next
        while fast and fast.next:
            if slow == fast: break
            slow = slow.next
            fast = fast.next.next
        
        if not fast or not fast.next: return None
        
        slow = head
        while slow != fast:
            slow = slow.next
            fast = fast.next
        
        return slow
