# Definition for singly-linked list.
import collections

class ListNode:
    def __init__(self, x):
        self.val = x
        self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return None
        
        nodes = collections.defaultdict(bool)
        curr = head
        
        while curr:
            if nodes[curr]:
                return curr
            nodes[curr] = True
            curr = curr.next
        
        return None
