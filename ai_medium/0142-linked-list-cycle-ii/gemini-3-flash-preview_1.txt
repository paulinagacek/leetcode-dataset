class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return None
        
        slow = head
        fast = head
        
        # Phase 1: Determine if a cycle exists using Floyd's Cycle-Finding Algorithm
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            
            # If slow and fast meet, a cycle exists
            if slow == fast:
                # Phase 2: Find the entry point of the cycle
                # Move one pointer to the head and keep the other at the meeting point.
                # Move both at the same speed; they will meet at the start of the cycle.
                ptr1 = head
                ptr2 = slow
                while ptr1 != ptr2:
                    ptr1 = ptr1.next
                    ptr2 = ptr2.next
                return ptr1
                
        return None
