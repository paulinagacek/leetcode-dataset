class Solution:
    def numDecodings(self, s: str) -> int:
        dp = [0] * (len(s) + 1)
        dp[0] = 1
        dp[1] = 0 if s[0] == '0' else 1
        
        for i in range(2, len(s) + 1):
            if s[i-1] == '0':
                dp[i] = 0
            elif s[i-2] == '0':
                dp[i] += dp[i-1]
            elif 10 <= int(s[i-2:i]) <= 26:
                dp[i] += dp[i-2]
            else:
                dp[i] += dp[i-1]
                
        return dp[-1]
