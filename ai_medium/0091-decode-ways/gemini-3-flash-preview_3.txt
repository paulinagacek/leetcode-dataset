class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it cannot be decoded.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] represents the number of ways to decode the substring s[:i].
        # We initialize the DP table with 0s.
        dp = [0] * (n + 1)
        
        # Base cases:
        # An empty string has 1 way to be decoded (as an empty string).
        dp[0] = 1
        # Since the first digit s[0] is not '0', s[:1] has exactly 1 way to be decoded.
        dp[1] = 1
        
        # Iterate through the string from the second character onwards.
        for i in range(2, n + 1):
            # 1. Single-digit decoding:
            # We can decode the current character s[i-1] as a single digit 
            # if it is not '0'.
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # 2. Two-digit decoding:
            # We can decode the current character and the previous one s[i-2:i] 
            # as a single mapping if the value is between 10 and 26.
            # Leading zeros (e.g., "06") are implicitly handled because int("06") 
            # is 6, which is less than 10.
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
            
            # If at any point dp[i] is 0, it means the substring s[:i] cannot be 
            # decoded, and therefore the entire string cannot be decoded.
            if dp[i] == 0:
                # This check isn't strictly necessary for correctness but serves
                # as an optimization for invalid strings.
                pass
        
        # The result is the number of ways to decode the entire string s[:n].
        return dp[n]
