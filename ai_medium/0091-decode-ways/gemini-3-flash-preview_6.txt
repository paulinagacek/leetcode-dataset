class Solution:
    def numDecodings(self, s: str) -> int:
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] stores the number of ways to decode the substring s[0:i]
        dp = [0] * (n + 1)
        
        # Base cases
        dp[0] = 1
        # We already checked that s[0] != '0'
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check for a single-digit decoding (s[i-1])
            # A single digit must be between '1' and '9'
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check for a two-digit decoding (s[i-2:i])
            # A two-digit code must be between "10" and "26"
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
                
        return dp[n]
