class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty, there are no ways to decode it
        if not s:
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[:i]
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded
        dp[0] = 1
        
        for i in range(1, n + 1):
            # Check the single-digit code (s[i-1])
            # A single-digit code is valid if it's not '0'
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check the two-digit code (s[i-2:i])
            # A two-digit code is valid if it's between '10' and '26'
            if i > 1:
                # Convert the substring of length 2 to an integer
                two_digit = int(s[i-2:i])
                if 10 <= two_digit <= 26:
                    dp[i] += dp[i-2]
        
        # The result is the number of ways to decode the entire string
        return dp[n]
