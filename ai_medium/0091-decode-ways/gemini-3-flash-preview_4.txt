class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it cannot be decoded.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[0:i].
        # Initialize the dp array with 0s.
        dp = [0] * (n + 1)
        
        # An empty string has 1 way to be decoded.
        dp[0] = 1
        # A string of length 1 that doesn't start with '0' has 1 way.
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check the single-digit possibility.
            # If the current character is not '0', it can be decoded as a single character.
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check the two-digit possibility.
            # The two-digit code must be between "10" and "26".
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
                
        return dp[n]
