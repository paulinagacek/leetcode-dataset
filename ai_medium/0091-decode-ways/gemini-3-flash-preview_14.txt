class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it cannot be decoded.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the prefix s[:i].
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded.
        dp[0] = 1
        # Base case: a single-character string that is not '0' has one way to be decoded.
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check if the single digit s[i-1] is valid (1-9).
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check if the two-digit number s[i-2:i] is valid (10-26).
            # leading zero in two-digit numbers (like "06") is excluded by the range 10-26.
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
        
        # The result is the number of ways to decode the entire string.
        return dp[n]
