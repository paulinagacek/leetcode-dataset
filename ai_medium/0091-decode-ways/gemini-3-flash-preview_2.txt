class Solution:
    def numDecodings(self, s: str) -> int:
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] represents the number of ways to decode the substring s[0...i-1]
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded
        dp[0] = 1
        
        for i in range(1, n + 1):
            # Check if the single digit s[i-1] is valid (1-9)
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check if the two-digit number s[i-2:i] is valid (10-26)
            if i > 1:
                # We use string slicing and conversion to check the range
                two_digit = int(s[i-2:i])
                if 10 <= two_digit <= 26:
                    dp[i] += dp[i-2]
                    
        return dp[n]
