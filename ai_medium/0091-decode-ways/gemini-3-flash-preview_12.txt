class Solution:
    def numDecodings(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        # dp[i] stores the number of ways to decode the substring s[0...i-1]
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded (doing nothing)
        dp[0] = 1
        
        for i in range(1, n + 1):
            # Check for a valid one-digit decoding (must be '1'-'9')
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check for a valid two-digit decoding (must be '10'-'26')
            if i > 1:
                # Extracts the substring of the two most recent digits
                two_digit = int(s[i-2:i])
                if 10 <= two_digit <= 26:
                    dp[i] += dp[i-2]
                    
        # The result is the total number of ways to decode the whole string
        return dp[n]
