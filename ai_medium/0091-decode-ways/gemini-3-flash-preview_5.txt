class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it cannot be decoded.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[0:i].
        # dp[0] is initialized to 1 representing an empty string decoding way.
        dp = [0] * (n + 1)
        dp[0] = 1
        # dp[1] is 1 because s[0] is not '0', so there is exactly one way to decode a single digit.
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check the single-digit code at the current position s[i-1].
            # A single-digit code is valid if it is not '0'.
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check the two-digit code ending at the current position s[i-2:i].
            # A two-digit code is valid if it is between "10" and "26".
            # Note: int() will handle cases like "06" by converting to 6.
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
                
        # The result is the number of ways to decode the entire string s.
        return dp[n]
