class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it's impossible to decode.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[0:i].
        # Initialize dp array with zeros.
        dp = [0] * (n + 1)
        
        # Base cases:
        # An empty string can be considered as having one valid decoding way.
        dp[0] = 1
        # Since s[0] is not '0', the first character has only one way to be decoded.
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check for a valid single-digit grouping ending at index i-1.
            # A single digit is valid if it's not '0'.
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check for a valid two-digit grouping ending at index i-1.
            # Convert the two-digit substring s[i-2:i] to an integer.
            two_digit = int(s[i-2:i])
            
            # A two-digit grouping is valid if it's between 10 and 26 inclusive.
            # This range ensures there are no leading zeros (e.g., "06" results in 6).
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
        
        # Return the total number of ways to decode the entire string.
        # If no valid decodings are found, dp[n] will naturally be 0.
        return dp[n]
