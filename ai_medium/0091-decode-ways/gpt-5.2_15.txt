class Solution:
    def numDecodings(self, s: str) -> int:
        n = len(s)
        if n == 0:
            return 0

        prev2 = 1
        prev1 = 0 if s[0] == '0' else 1

        for i in range(1, n):
            cur = 0

            if s[i] != '0':
                cur += prev1

            two = (ord(s[i - 1]) - 48) * 10 + (ord(s[i]) - 48)
            if 10 <= two <= 26:
                cur += prev2

            prev2, prev1 = prev1, cur

        return prev1
