class Solution:
    def numDecodings(self, s: str) -> int:
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the prefix s[0...i-1]
        dp = [0] * (n + 1)
        
        # Base cases
        dp[0] = 1  # An empty string can be decoded in 1 way
        dp[1] = 1  # We already checked that s[0] != '0'
        
        for i in range(2, n + 1):
            # Check the single-digit possibility
            # s[i-1] corresponds to the current digit in the string
            if '1' <= s[i-1] <= '9':
                dp[i] += dp[i-1]
            
            # Check the two-digit possibility
            # s[i-2:i] corresponds to the last two digits ending at index i-1
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
                
        return dp[n]
