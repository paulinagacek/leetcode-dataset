class Solution:
    def numDecodings(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        # dp[i] stores the number of ways to decode the substring s[:i]
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded (as a starting point)
        dp[0] = 1
        
        # Check the first character: if it's '0', the string cannot be decoded
        dp[1] = 1 if s[0] != '0' else 0
        
        for i in range(2, n + 1):
            # Check for a single-digit decoding (s[i-1])
            # The character s[i-1] corresponds to the single digit at position i-1
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check for a two-digit decoding (s[i-2:i])
            # The number formed by the two digits must be between 10 and 26
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
                
        return dp[n]
