class Solution:
    def numDecodings(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the prefix s[0:i]
        dp = [0] * (n + 1)
        
        # Base case: an empty string has one way to be decoded (an empty sequence).
        dp[0] = 1
        
        for i in range(1, n + 1):
            # Check if single-digit decoding s[i-1] is possible (must be '1'-'9').
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check if two-digit decoding s[i-2:i] is possible (must be "10"-"26").
            # i > 1 ensures we have at least two digits to consider.
            if i > 1 and '10' <= s[i-2:i] <= '26':
                dp[i] += dp[i-2]
        
        return dp[n]
