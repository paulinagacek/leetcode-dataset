class Solution:
    def numDecodings(self, s: str) -> int:
        if not s:
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the prefix s[0...i-1]
        dp = [0] * (n + 1)
        
        # Base case: An empty string has one way to be decoded.
        dp[0] = 1
        
        for i in range(1, n + 1):
            # Check for single-digit decoding.
            # A digit '0' cannot be decoded as a single character.
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check for two-digit decoding.
            # Only combinations from "10" to "26" are valid mappings.
            if i > 1:
                two_digit = int(s[i-2:i])
                if 10 <= two_digit <= 26:
                    dp[i] += dp[i-2]
                    
        return dp[n]
