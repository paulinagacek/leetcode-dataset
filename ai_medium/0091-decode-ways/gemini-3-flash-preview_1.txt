class Solution:
    def numDecodings(self, s: str) -> int:
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[:i]
        dp = [0] * (n + 1)
        
        # Base cases
        dp[0] = 1
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check the single-digit case: s[i-1]
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check the two-digit case: s[i-2:i]
            # Must be between 10 and 26. Note that "06" results in 6, which is not >= 10.
            two_digit = int(s[i-2:i])
            if 10 <= two_digit <= 26:
                dp[i] += dp[i-2]
        
        return dp[n]
