class Solution:
    def numDecodings(self, s: str) -> int:
        # If the string is empty or starts with '0', it cannot be decoded.
        if not s or s[0] == '0':
            return 0
        
        n = len(s)
        # dp[i] will store the number of ways to decode the substring s[0:i].
        # Initialize the dp array with 0s.
        dp = [0] * (n + 1)
        
        # Base cases:
        # An empty string can be considered to have 1 way to be decoded (as a base for combinations).
        dp[0] = 1
        # The first character is guaranteed not to be '0' from our initial check.
        dp[1] = 1
        
        for i in range(2, n + 1):
            # Check if the single digit s[i-1] is a valid code ('1'-'9').
            # If so, the number of ways to decode up to i includes dp[i-1].
            if s[i-1] != '0':
                dp[i] += dp[i-1]
            
            # Check if the two-digit code s[i-2:i] is valid ('10'-'26').
            # We use string comparison which works for two-digit numeric strings.
            # This condition naturally handles leading zeros like '06' because '06' < '10'.
            if '10' <= s[i-2:i] <= '26':
                dp[i] += dp[i-2]
        
        # The answer for the full string is the last element in the dp array.
        return dp[n]
