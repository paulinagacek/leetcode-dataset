class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        if not matrix or not matrix[0]:
            return False

        start, end = 0, len(matrix) * len(matrix[0]) - 1

        while start <= end:
            mid = (start + end) // 2
            mid_value = matrix[mid // len(matrix[0])][mid % len(matrix[0])]

            if mid_value == target:
                return True
            elif mid_value < target:
                start = mid + 1
            else:
                end = mid - 1

        return False
