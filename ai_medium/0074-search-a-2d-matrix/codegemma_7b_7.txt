class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        if not matrix or not matrix[0]:
            return False

        rows, cols = len(matrix), len(matrix[0])
        top, bottom = 0, rows - 1

        while top <= bottom:
            mid = (top + bottom) // 2
            first = matrix[mid][0]
            last = matrix[mid][-1]

            if first <= target <= last:
                return binary_search(matrix[mid], target)  # Perform binary search within the current row
            elif target < first:
                bottom = mid - 1
            else:
                top = mid + 1

        return False

def binary_search(row: List[int], target: int) -> bool:
    left, right = 0, len(row) - 1

    while left <= right:
        mid = (left + right) // 2
        if row[mid] == target:
            return True
        elif row[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return False
