class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        anagram_map = defaultdict(list)
        
        for s in strs:
            # Sort the string to create a unique key for all anagrams
            key = "".join(sorted(s))
            anagram_map[key].append(s)
            
        return list(anagram_map.values())
