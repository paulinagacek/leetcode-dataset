class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        # Initialize pointers for zero, one and two
        p0 = 0
        p1 = 0
        
        # Iterate through the array
        for i in range(len(nums)):
            # If the current element is 1, swap it with the element at p1
            if nums[i] == 1:
                nums[i], nums[p1] = nums[p1], nums[i]
                p1 += 1
            # If the current element is 0, swap it with the element at p0 and also update p1 if necessary
            elif nums[i] == 0:
                nums[i], nums[p0] = nums[p0], nums[i]
                # Only increment p1 if p0 is less than p1 to avoid swapping an already sorted '1'
                if p0 < p1:
                    nums[i], nums[p1] = nums[p1], nums[i]
                p0 += 1
                p1 += 1
        
        return nums
