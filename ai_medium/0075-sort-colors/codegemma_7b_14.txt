class Solution:
    def sortColors(self, nums: List[int]) -> None:
        red, curr, blue = 0, 0, len(nums) - 1
        
        while curr <= blue:
            if nums[curr] == 0:
                nums[curr], nums[red] = nums[red], nums[curr]
                curr += 1
                red += 1
            elif nums[curr] == 2:
                nums[curr], nums[blue] = nums[blue], nums[curr]
                blue -= 1
            else:
                curr += 1
