class Solution:
    def sortColors(self, nums: List[int]) -> None:
        i, j, curr = 0, len(nums) - 1, 0
        while curr <= j:
            if nums[curr] == 0:
                nums[curr], nums[i] = nums[i], nums[curr]
                i += 1
                curr += 1
            elif nums[curr] == 2:
                nums[curr], nums[j] = nums[j], nums[curr]
                j -= 1
            else:
                curr += 1
