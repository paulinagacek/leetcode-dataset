class Solution:
    def sortColors(self, nums: List[int]) -> None:
        ptr = 0
        mid = 0
        high = len(nums) - 1

        while mid <= high:
            if nums[mid] == 0:
                nums[mid], nums[ptr] = nums[ptr], nums[mid]
                mid += 1
                ptr += 1
            elif nums[mid] == 1:
                mid += 1
            else:
                nums[mid], nums[high] = nums[high], nums[mid]
                high -= 1
