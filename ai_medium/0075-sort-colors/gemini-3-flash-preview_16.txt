class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        low = 0
        mid = 0
        high = len(nums) - 1
        
        while mid <= high:
            if nums[mid] == 0:
                nums[low], nums[mid] = nums[low], nums[mid]
                # Swap current mid with low to move 0 to the front
                nums[low], nums[mid] = nums[mid], nums[low]
                low += 1
                mid += 1
            elif nums[mid] == 1:
                # 1 is already in its correct relative section, just move forward
                mid += 1
            else:
                # Swap current mid with high to move 2 to the end
                nums[mid], nums[high] = nums[high], nums[mid]
                high -= 1
                # Do not increment mid here because the swapped value from high
                # still needs to be evaluated.
