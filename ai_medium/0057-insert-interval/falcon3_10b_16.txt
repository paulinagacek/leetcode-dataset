class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        result = []
        i, start, end = 0, newInterval[0], newInterval[1]
        while i < len(intervals):
            current_start, current_end = intervals[i]
            if current_end < start:
                # new interval doesn't overlap with current interval
                result.append([current_start, current_end])
            else:
                # new interval overlaps with current interval or ends within it
                start = min(start, current_start)
                end = max(end, current_end)
            i += 1
        # Add the newly merged interval or the remaining part of the new interval
        result.append([start, end])
        # Append any remaining intervals that don't overlap with the new interval
        result.extend(intervals[i:])
        return result
