from typing import List

class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        result = []
        placed = False
        
        for interval in intervals:
            # If new interval has not been added and the current interval is smaller than new interval
            if not placed and newInterval[0] > interval[1]:
                result.append(newInterval)
                placed = True
            
            # If there is no overlap between current interval and new interval
            if newInterval[1] < interval[0]:
                if not placed:
                    result.append(newInterval)
                    placed = True
                result.append(interval)
            else:
                # There is an overlap, so we merge the intervals
                newInterval[0] = min(newInterval[0], interval[0])
                newInterval[1] = max(newInterval[1], interval[1])
        
        # If the new interval has not been added yet, add it to the end
        if not placed:
            result.append(newInterval)
        
        return result
