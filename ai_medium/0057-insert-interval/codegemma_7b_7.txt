class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        temp = intervals + [newInterval]
        temp.sort(key = lambda x: x[0])
        result = []
        start, end = temp[0]
        for i in range(1, len(temp)):
            if temp[i][0] <= end:
                end = max(end, temp[i][1])
            else:
                result.append([start, end])
                start, end = temp[i]
        result.append([start, end])
        return result
