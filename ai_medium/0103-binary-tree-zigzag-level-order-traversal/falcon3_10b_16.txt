# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.right = right

class Solution:
    def zigzagLevelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root:
            return []
        
        queue = [root]
        result = []
        left_to_right = True
        
        while queue:
            level = []
            size = len(queue)
            
            for _ in range(size):
                node = queue.pop(0)
                level.append(node.val)
                
                if left_to_right:
                    if node.left:
                        queue.append(node.left)
                    if node.right:
                        queue.append(node.right)
                else:
                    if node.right:
                        queue.append(node.right)
                    if node.left:
                        queue.append(node.left)
            
            result.append(level[::-1] if left_to_right else level)
            left_to_right = not left_to_right
        
        return result
