class Solution:
    def zigzagLevelOrder(self, root: Optional['TreeNode']) -> List[List[int]]:
        if not root:
            return []
        
        res = []
        q = deque([root])
        left_to_right = True
        
        while q:
            level_size = len(q)
            level = [0] * level_size
            for i in range(level_size):
                node = q.popleft()
                idx = i if left_to_right else level_size - 1 - i
                level[idx] = node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.append(level)
            left_to_right = not left_to_right
        
        return res
