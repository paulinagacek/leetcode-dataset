class Solution:
    def canJump(self, nums: List[int]) -> bool:
        farthest_reach = 0
        for i in range(len(nums)):
            if i > farthest_reach:
                return False
            farthest_reach = max(farthest_reach, i + nums[i])
            if farthest_reach >= len(nums) - 1:
                return True
        return False
