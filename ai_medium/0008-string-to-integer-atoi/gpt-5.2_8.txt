class Solution:
    def myAtoi(self, s: str) -> int:
        i, n = 0, len(s)
        while i < n and s[i] == ' ':
            i += 1

        sign = 1
        if i < n and s[i] in '+-':
            if s[i] == '-':
                sign = -1
            i += 1

        while i < n and s[i] == '0':
            i += 1

        INT_MAX = 2**31 - 1
        INT_MIN = -2**31

        num = 0
        while i < n and '0' <= s[i] <= '9':
            digit = ord(s[i]) - ord('0')

            if sign == 1:
                if num > INT_MAX // 10 or (num == INT_MAX // 10 and digit > INT_MAX % 10):
                    return INT_MAX
            else:
                limit = -INT_MIN  # 2147483648
                if num > limit // 10 or (num == limit // 10 and digit > limit % 10):
                    return INT_MIN

            num = num * 10 + digit
            i += 1

        return sign * num
