class Solution:
    def myAtoi(self, s: str) -> int:
        # Step 1: Ignore leading whitespace
        s = s.lstrip(' ')
        if not s:
            return 0
        
        # Step 2: Determine signedness
        sign = 1
        i = 0
        if s[i] == '-':
            sign = -1
            i += 1
        elif s[i] == '+':
            i += 1
            
        # Step 3: Conversion
        res = 0
        while i < len(s) and s[i].isdigit():
            res = res * 10 + int(s[i])
            i += 1
            
        # Apply the sign to the numeric result
        res *= sign
        
        # Step 4: Rounding (Clamping to 32-bit signed integer range)
        INT_MIN = -2147483648
        INT_MAX = 2147483647
        
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
            
        return res
