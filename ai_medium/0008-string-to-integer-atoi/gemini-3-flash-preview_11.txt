class Solution:
    def myAtoi(self, s: str) -> int:
        i = 0
        n = len(s)
        
        # 1. Ignore leading whitespace
        while i < n and s[i] == ' ':
            i += 1
            
        if i == n:
            return 0
            
        # 2. Determine the sign
        sign = 1
        if s[i] == '-':
            sign = -1
            i += 1
        elif s[i] == '+':
            i += 1
            
        # 3. Read the integer
        res = 0
        while i < n and s[i].isdigit():
            res = res * 10 + int(s[i])
            i += 1
            
        # Apply the sign
        res *= sign
        
        # 4. Rounding to 32-bit signed integer range
        INT_MIN, INT_MAX = -2147483648, 2147483647
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
            
        return res
