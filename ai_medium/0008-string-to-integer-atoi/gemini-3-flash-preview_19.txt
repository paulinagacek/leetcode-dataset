class Solution:
    def myAtoi(self, s: str) -> int:
        # Step 1: Ignore any leading whitespace
        s = s.lstrip()
        if not s:
            return 0
        
        # Step 2: Determine the sign
        sign = 1
        idx = 0
        if s[idx] == '-':
            sign = -1
            idx += 1
        elif s[idx] == '+':
            idx += 1
            
        # Step 3: Conversion
        num = 0
        while idx < len(s) and s[idx].isdigit():
            num = num * 10 + int(s[idx])
            idx += 1
            
        res = sign * num
        
        # Step 4: Rounding to 32-bit signed integer range
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
            
        return res
