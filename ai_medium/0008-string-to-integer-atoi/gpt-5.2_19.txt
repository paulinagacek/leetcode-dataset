class Solution:
    def myAtoi(self, s: str) -> int:
        n = len(s)
        i = 0

        while i < n and s[i] == ' ':
            i += 1

        sign = 1
        if i < n and s[i] in '+-':
            if s[i] == '-':
                sign = -1
            i += 1

        while i < n and s[i] == '0':
            i += 1

        INT_MAX = 2**31 - 1
        INT_MIN = -2**31

        num = 0
        limit = INT_MAX if sign == 1 else 2**31  # abs(INT_MIN)
        while i < n and '0' <= s[i] <= '9':
            digit = ord(s[i]) - ord('0')
            if num > (limit - digit) // 10:
                return INT_MAX if sign == 1 else INT_MIN
            num = num * 10 + digit
            i += 1

        res = sign * num
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
        return res
