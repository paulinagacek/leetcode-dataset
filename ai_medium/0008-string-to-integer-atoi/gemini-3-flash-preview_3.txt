class Solution:
    def myAtoi(self, s: str) -> int:
        s = s.lstrip()
        if not s:
            return 0
        
        sign = 1
        start_idx = 0
        
        if s[0] == '-':
            sign = -1
            start_idx = 1
        elif s[0] == '+':
            start_idx = 1
            
        res = 0
        for i in range(start_idx, len(s)):
            if not s[i].isdigit():
                break
            res = res * 10 + (ord(s[i]) - ord('0'))
            
        res *= sign
        
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
            
        return res
