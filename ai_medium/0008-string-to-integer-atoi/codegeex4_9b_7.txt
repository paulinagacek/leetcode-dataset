class Solution:
    def myAtoi(self, s: str) -> int:
        INT_MAX = 2147483647
        INT_MIN = -2147483648
        
        # Step 1: Remove leading whitespace
        s = s.strip()
        
        if not s:
            return 0
        
        # Step 2: Determine the sign of the number
        sign = 1
        index = 0
        if s[0] == '-':
            sign = -1
            index += 1
        elif s[0] == '+':
            index += 1
        
        # Step 3: Read digits and convert to integer
        result = 0
        while index < len(s) and s[index].isdigit():
            digit = int(s[index])
            
            result = result * 10 + digit
            
            if sign == 1 and result > INT_MAX:
                return INT_MAX
            elif sign == -1 and -result < INT_MIN:
                return INT_MIN
            
            index += 1
        
        # Step 4: Return the final integer, considering the sign
        return sign * result
