class Solution:
    def myAtoi(self, s: str) -> int:
        n = len(s)
        i = 0
        ans = 0
        sign = 1
        while i < n and s[i] == ' ':
            i += 1
        if i < n and s[i] == '-':
            sign = -1
            i += 1
        elif i < n and s[i] == '+':
            i += 1
        while i < n and s[i].isdigit():
            ans = ans * 10 + int(s[i])
            i += 1
        ans *= sign
        ans = max(-2**31, min(ans, 2**31 - 1))
        return ans
