class Solution:
    def myAtoi(self, s: str) -> int:
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        s = s.strip()
        sign, result = 1, 0
        
        if s and s[0] in ('+', '-'):
            sign = -1 if s[0] == '-' else 1
            s = s[1:]
        
        for char in s:
            if char.isdigit():
                result = result * 10 + int(char)
            else:
                break
        
        result *= sign
        return max(INT_MIN, min(result, INT_MAX))
