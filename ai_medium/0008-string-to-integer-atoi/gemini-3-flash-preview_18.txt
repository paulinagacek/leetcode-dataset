class Solution:
    def myAtoi(self, s: str) -> int:
        INT_MAX = 2147483647
        INT_MIN = -2147483648
        
        # Step 1: Remove leading whitespace
        s = s.lstrip()
        if not s:
            return 0
        
        # Step 2: Determine sign
        sign = 1
        index = 0
        if s[0] == '-':
            sign = -1
            index = 1
        elif s[0] == '+':
            index = 1
            
        # Step 3: Read digits until a non-digit character is encountered
        res = 0
        for i in range(index, len(s)):
            if not s[i].isdigit():
                break
            res = res * 10 + int(s[i])
            
        # Apply sign
        res *= sign
        
        # Step 4: Rounding (clamping to 32-bit signed integer range)
        if res > INT_MAX:
            return INT_MAX
        if res < INT_MIN:
            return INT_MIN
        
        return res
