class Solution:
    def myAtoi(self, s: str) -> int:
        # 1. Ignore leading whitespace
        s = s.lstrip()
        if not s:
            return 0
        
        # 2. Determine signedness
        sign = 1
        i = 0
        if s[i] == '-':
            sign = -1
            i += 1
        elif s[i] == '+':
            i += 1
            
        # 3. Read integer until non-digit character or end of string
        res = 0
        while i < len(s) and s[i].isdigit():
            res = res * 10 + int(s[i])
            i += 1
            
        res *= sign
        
        # 4. Round to 32-bit signed integer range
        INT_MIN = -2**31
        INT_MAX = 2**31 - 1
        
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
            
        return res
