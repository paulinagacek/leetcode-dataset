class Solution:
    def myAtoi(self, s: str) -> int:
        a, sign, i, num = '0123456789', 1, 0, 0
        while i < len(s) and s[i] == ' ': i += 1
        if i == len(s): return 0
        if s[i] == '-':
            sign *= -1
            i += 1
        elif s[i] == '+':
            i += 1

        while i < len(s) and s[i] in a:
            num = num*10 + int(s[i])
            i += 1
        return max(-2**31, min(sign*num, 2**31-1))
