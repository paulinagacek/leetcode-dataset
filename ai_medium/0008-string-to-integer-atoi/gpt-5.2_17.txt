class Solution:
    def myAtoi(self, s: str) -> int:
        i, n = 0, len(s)
        while i < n and s[i] == ' ':
            i += 1

        sign = 1
        if i < n and s[i] in "+-":
            if s[i] == '-':
                sign = -1
            i += 1

        INT_MAX = 2**31 - 1
        INT_MIN = -2**31

        res = 0
        while i < n and '0' <= s[i] <= '9':
            digit = ord(s[i]) - ord('0')
            if sign == 1:
                if res > INT_MAX // 10 or (res == INT_MAX // 10 and digit > INT_MAX % 10):
                    return INT_MAX
            else:
                limit = 2**31
                if res > limit // 10 or (res == limit // 10 and digit > limit % 10):
                    return INT_MIN
            res = res * 10 + digit
            i += 1

        return sign * res
