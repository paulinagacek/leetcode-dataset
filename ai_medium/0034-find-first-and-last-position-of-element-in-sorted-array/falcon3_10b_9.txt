from typing import List

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        left, right = self.findBound(nums, target, True), self.findBound(nums, target, False)
        if left == len(nums) or nums[left] != target:
            return [-1, -1]
        return [left, right]

    def findBound(self, nums: List[int], target: int, is_left: bool) -> int:
        left, right = 0, len(nums)
        while left < right:
            mid = left + (right - left) // 2
            if (is_left and nums[mid] > target) or (not is_left and nums[mid] >= target):
                right = mid
            else:
                left = mid + 1
        return left
