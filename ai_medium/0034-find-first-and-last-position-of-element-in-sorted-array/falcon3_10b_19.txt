class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def bin_search(nums, target, find_left):
            left, right = 0, len(nums) - 1
            while left <= right:
                mid = (left + right) // 2
                if nums[mid] == target:
                    if find_left:
                        right = mid - 1
                    else:
                        left = mid + 1
                elif nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            return left

        left = bin_search(nums, target, True)
        if left == len(nums) or nums[left] != target:
            return [-1, -1]

        right = bin_search(nums, target, False) - 1
        return [left, right]
