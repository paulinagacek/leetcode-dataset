class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def binary_search(left, right):
            if left > right:
                return [-1, -1]

            mid = (left + right) // 2

            if nums[mid] == target:
                left_result = binary_search(left, mid - 1)
                right_result = binary_search(mid + 1, right)

                if left_result[0] != -1:
                    return left_result
                else:
                    return [mid, right_result[1]]

            elif nums[mid] < target:
                return binary_search(mid + 1, right)

            else:
                return binary_search(left, mid - 1)

        return binary_search(0, len(nums) - 1)
