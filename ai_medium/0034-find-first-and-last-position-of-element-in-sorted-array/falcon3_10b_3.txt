class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        left = self.binary_search(nums, target, True)
        if left == -1 or nums[left] != target:
            return [-1, -1]
        right = self.binary_search(nums, target, False)
        return [left, right]

    def binary_search(self, nums: List[int], target: int, find_left: bool) -> int:
        lo, hi = 0, len(nums) - 1
        while lo <= hi:
            mid = (lo + hi) // 2
            if nums[mid] > target or (find_left and nums[mid] == target):
                hi = mid - 1
            else:
                lo = mid + 1
        if find_left:
            if hi < 0 or nums[hi] != target:
                return -1
            while nums[hi] == target and hi > 0:
                hi -= 1
            return hi + 1
        else:
            if lo >= len(nums) or nums[lo] != target:
                return -1
            while nums[lo] == target and lo < len(nums) - 1:
                lo += 1
            return lo - 1
