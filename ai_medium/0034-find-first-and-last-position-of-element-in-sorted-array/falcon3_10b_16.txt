class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def search(left, right):
            while left < right:
                mid = (left + right) // 2
                if nums[mid] >= target:
                    right = mid
                else:
                    left = mid + 1
            return left

        start = search(0, len(nums))
        if start == len(nums) or nums[start] != target:
            return [-1, -1]

        end = search(0, len(nums) - 1, start)
        return [start, end]

    def search(self, left: int, right: int, mid: int = -1):
        while left < right:
            mid = (left + right) // 2
            if nums[mid] >= target:
                right = mid
            else:
                left = mid + 1
        return left
