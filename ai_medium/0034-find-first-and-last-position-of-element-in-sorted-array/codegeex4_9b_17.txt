class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def findLeftIndex(nums, target):
            left, right = 0, len(nums) - 1
            while left <= right:
                mid = (left + right) // 2
                if nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            return left
        
        def findRightIndex(nums, target):
            left, right = 0, len(nums) - 1
            while left <= right:
                mid = (left + right) // 2
                if nums[mid] <= target:
                    left = mid + 1
                else:
                    right = mid - 1
            return right
        
        leftIndex = findLeftIndex(nums, target)
        if leftIndex == len(nums) or nums[leftIndex] != target:
            return [-1, -1]
        
        rightIndex = findRightIndex(nums, target)
        
        return [leftIndex, rightIndex]
