from typing import List

class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        if not nums:
            return [-1, -1]
        
        left = 0
        right = len(nums) - 1
        
        # Find the starting position of the target
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] < target:
                left = mid + 1
            elif nums[mid] > target:
                right = mid - 1
            else:
                if mid == 0 or nums[mid-1] != target:
                    start = mid
                    break
                else:
                    right = mid - 1
        
        # If the target is not found in the array
        if left > right:
            return [-1, -1]
        
        # Find the ending position of the target
        end = start
        while end < len(nums) and nums[end] == target:
            end += 1
        
        return [start, end-1]
