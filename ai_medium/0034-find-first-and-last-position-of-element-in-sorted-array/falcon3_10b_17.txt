class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def binary_search(left, right, condition):
            while left < right:
                mid = (left + right) // 2
                if condition(mid):
                    right = mid
                else:
                    left = mid + 1
            return left

        left = binary_search(0, len(nums), lambda i: nums[i] >= target)
        if left == len(nums) or nums[left] != target:
            return [-1, -1]

        right = binary_search(0, len(nums), lambda i: nums[i] > target)
        left, right = left, right - 1

        return [left, right]
