class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def binary_search(left, right, is_first):
            result = -1
            while left <= right:
                mid = left + (right - left) // 2
                if nums[mid] == target:
                    result = mid
                    if is_first:
                        right = mid - 1
                    else:
                        left = mid + 1
                elif nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            return result
        
        left, right = binary_search(0, len(nums) - 1, True), binary_search(0, len(nums) - 1, False)
        return [left, right] if left != -1 else [-1, -1]
