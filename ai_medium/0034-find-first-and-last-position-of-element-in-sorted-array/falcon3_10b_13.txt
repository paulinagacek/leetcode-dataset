class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        left = self.binarySearch(nums, target, True)
        if left == -1 or nums[left] != target:
            return [-1, -1]
        
        right = self.binarySearch(nums, target, False)
        
        return [left, right]
    
    def binarySearch(self, nums: List[int], target: int, is_find_left: bool) -> int:
        left, right = 0, len(nums) - 1
        
        while left <= right:
            mid = left + (right - left) // 2
            
            if nums[mid] > target or (is_find_left and nums[mid] == target):
                right = mid - 1
            else:
                left = mid + 1
        
        if is_find_left:
            return right
        
        return left - 1
