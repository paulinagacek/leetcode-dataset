class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def findBound(is_first: bool) -> int:
            low, high = 0, len(nums) - 1
            res = -1
            
            while low <= high:
                mid = (low + high) // 2
                
                if nums[mid] == target:
                    res = mid
                    if is_first:
                        high = mid - 1
                    else:
                        low = mid + 1
                elif nums[mid] < target:
                    low = mid + 1
                else:
                    high = mid - 1
            return res
        
        return [findBound(True), findBound(False)]
