class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def findBound(start, end, is_left):
            while start < end:
                mid = (start + end) // 2
                if (is_left and nums[mid] > target) or (not is_left and nums[mid] >= target):
                    end = mid
                else:
                    start = mid + 1
            return start if nums[start] == target else -1

        left = findBound(0, len(nums), True)
        right = findBound(0, len(nums), False)
        return [left, right - 1] if left != -1 else [-1, -1]
