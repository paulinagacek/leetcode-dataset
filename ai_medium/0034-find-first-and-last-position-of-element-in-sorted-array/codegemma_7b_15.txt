class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        left, right = 0, len(nums) - 1
        result = [-1, -1]
        
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                result = [mid, mid]
                left, right = mid + 1, mid - 1
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        
        return result
