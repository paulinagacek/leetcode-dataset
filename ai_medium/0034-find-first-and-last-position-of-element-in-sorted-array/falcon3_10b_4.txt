class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def bin_search(left):
            lo, hi = 0, len(nums)
            while lo < hi:
                mid = (lo + hi) // 2
                if nums[mid] > target or (left and nums[mid] == target):
                    hi = mid
                else:
                    lo = mid + 1
            return lo if left and lo < len(nums) and nums[lo] == target else -1

        left, right = bin_search(True), bin_search(False)
        return [left, right - 1] if left != -1 else [-1, -1]
