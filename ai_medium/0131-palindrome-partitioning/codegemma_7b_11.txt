class Solution:
    def partition(self, s: str) -> List[List[str]]:
        def isPalindrome(s):
            return s == s[::-1]

        def dfs(index, path):
            if index == len(s):
                result.append(path[:])
                return

            for i in range(index, len(s)):
                substring = s[index:i + 1]
                if isPalindrome(substring):
                    path.append(substring)
                    dfs(i + 1, path)
                    path.pop()

        result = []
        dfs(0, [])
        return result
