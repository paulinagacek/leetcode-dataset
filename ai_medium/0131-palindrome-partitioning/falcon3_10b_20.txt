class Solution:
    def partition(self, s: str) -> List[List[str]]:
        result = []
        
        def backtrack(start, path):
            if start == len(s):
                result.append(path[:])
                return
            for end in range(start+1, len(s) + 1):
                if self.is_palindrome(s, start, end-1):
                    path.append(s[start:end])
                    backtrack(end, path)
                    path.pop()
        
        def is_palindrome(s, start, end):
            while start < end:
                if s[start] != s[end]:
                    return False
                start += 1
                end -= 1
            return True
        
        backtrack(0, [])
        return result
