from typing import List

class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        self.backtrack(s, [], res)
        return res
    
    def backtrack(self, s: str, path: List[str], res: List[List[str]]):
        if not s:
            res.append(path)
            return
        for i in range(1, len(s) + 1):
            if s[:i] == s[:i][::-1]:
                self.backtrack(s[i:], path+[s[:i]], res)
