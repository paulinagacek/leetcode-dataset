class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        n = len(s)
        
        def backtrack(start, path):
            if start == n:
                res.append(list(path))
                return
            
            for end in range(start, n):
                substring = s[start : end + 1]
                if substring == substring[::-1]:
                    path.append(substring)
                    backtrack(end + 1, path)
                    path.pop()
                    
        backtrack(0, [])
        return res
