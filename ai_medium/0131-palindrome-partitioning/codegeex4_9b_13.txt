from typing import List

class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        self.helper(res, [], s)
        return res
    
    def helper(self, res, path, s):
        if not s:
            res.append(path)
            return 
        for i in range(len(s)):
            if self.isPal(s[:i+1]):
                self.helper(res, path + [s[:i+1]], s[i+1:])
    
    def isPal(self, s):
        return s == s[::-1]
