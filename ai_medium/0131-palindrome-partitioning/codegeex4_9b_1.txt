from typing import List

class Solution:
    def partition(self, s: str) -> List[List[str]]:
        def is_palindrome(sub):
            return sub == sub[::-1]
        
        result = []
        
        def dfs(start, path):
            if start == len(s):
                result.append(path)
                return
            for i in range(start + 1, len(s) + 1):
                if is_palindrome(s[start:i]):
                    dfs(i, path + [s[start:i]])
        
        dfs(0, [])
        return result
