from typing import List

class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        self.helper(s, [], res)
        return res
    
    def helper(self, s, path, res):
        if not s:
            res.append(path)
        for i in range(1, len(s) + 1):
            if s[:i] == s[:i][::-1]:
                self.helper(s[i:], path + [s[:i]], res)
