from typing import List

class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res = []
        def backtrack(start, path):
            if start == len(s):
                res.append(path)
                return
            for end in range(start + 1, len(s) + 1):
                if self.isPalindrome(s[start:end]):
                    backtrack(end, path + [s[start:end]])
        def isPalindrome(s):
            return s == s[::-1]
        backtrack(0, [])
        return res
