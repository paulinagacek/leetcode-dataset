class Solution:
    def partition(self, s: str) -> List[List[str]]:
        def is_palindrome(sub):
            return sub == sub[::-1]

        def backtrack(index, path):
            if index == len(s):
                result.append(path[:])
                return

            for i in range(index, len(s)):
                sub = s[index:i + 1]
                if is_palindrome(sub):
                    path.append(sub)
                    backtrack(i + 1, path)
                    path.pop()

        result = []
        backtrack(0, [])
        return result
