class Solution:
    def partition(self, s: str) -> List[List[str]]:
        def dfs(start, path):
            if start == len(s):
                result.append(path[:])
                return
            for end in range(start + 1, len(s) + 1):
                if s[start:end] == s[start:end][::-1]:
                    path.append(s[start:end])
                    dfs(end, path)
                    path.pop()

        result = []
        dfs(0, [])
        return result
