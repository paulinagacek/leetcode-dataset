class Solution:
    def partition(self, s: str) -> List[List[str]]:
        result = []
        
        def backtrack(start, path):
            if start == len(s):
                result.append(path[:])
                return
            
            for i in range(start, len(s)):
                if is_palindrome(s, start, i):
                    path.append(s[start:i+1])
                    backtrack(i+1, path)
                    path.pop()
        
        def is_palindrome(sub, left, right):
            while left < right:
                if sub[left] != sub[right]:
                    return False
                left += 1
                right -= 1
            return True
        
        backtrack(0, [])
        return result
