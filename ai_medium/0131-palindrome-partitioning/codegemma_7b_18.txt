class Solution:
    def is_palindrome(self, s: str) -> bool:
        return s == s[::-1]

    def helper(self, s: str, start: int, partitions: List[str], result: List[List[str]]) -> None:
        if start == len(s):
            result.append(partitions[:])
            return

        for i in range(start, len(s)):
            substring = s[start:i+1]
            if self.is_palindrome(substring):
                partitions.append(substring)
                self.helper(s, i+1, partitions, result)
                partitions.pop()

    def partition(self, s: str) -> List[List[str]]:
        result = []
        self.helper(s, 0, [], result)
        return result
