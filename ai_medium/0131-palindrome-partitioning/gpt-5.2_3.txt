class Solution:
    def partition(self, s: str) -> List[List[str]]:
        n = len(s)
        is_pal = [[False] * n for _ in range(n)]
        for i in range(n - 1, -1, -1):
            for j in range(i, n):
                if s[i] == s[j] and (j - i <= 2 or is_pal[i + 1][j - 1]):
                    is_pal[i][j] = True

        res: List[List[str]] = []
        path: List[str] = []

        def dfs(i: int) -> None:
            if i == n:
                res.append(path.copy())
                return
            for j in range(i, n):
                if is_pal[i][j]:
                    path.append(s[i:j + 1])
                    dfs(j + 1)
                    path.pop()

        dfs(0)
        return res
