class Solution:
    def partition(self, s: str) -> List[List[str]]:
        n = len(s)
        dp = [[False] * n for _ in range(n)]

        # Base case: all substrings of length 1 are palindromes
        for i in range(n):
            dp[i][i] = True

        # Fill the table diagonally, checking for palindromes of increasing length
        for i in range(n - 1):
            if s[i] == s[i + 1]:
                dp[i][i + 1] = True

        for k in range(3, n + 1):
            for i in range(n - k + 1):
                j = i + k - 1
                if s[i] == s[j] and dp[i + 1][j - 1]:
                    dp[i][j] = True

        result = []
        self.dfs(s, 0, [], dp, result)
        return result

    def dfs(self, s, start, current_partition, dp, result):
        if start == len(s):
            result.append(current_partition[:])
            return

        for i in range(start, len(s)):
            if dp[start][i]:
                current_partition.append(s[start:i + 1])
                self.dfs(s, i + 1, current_partition, dp, result)
                current_partition.pop()
