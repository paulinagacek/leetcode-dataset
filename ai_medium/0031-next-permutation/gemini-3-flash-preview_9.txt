class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        n = len(nums)
        if n <= 1:
            return

        # 1. Find the first index i from the right such that nums[i] < nums[i+1]
        i = n - 2
        while i >= 0 and nums[i] >= nums[i + 1]:
            i -= 1
        
        # 2. If such an index exists, find the rightmost index j such that nums[j] > nums[i]
        if i >= 0:
            j = n - 1
            while nums[j] <= nums[i]:
                j -= 1
            # 3. Swap nums[i] and nums[j]
            nums[i], nums[j] = nums[j], nums[i]
        
        # 4. Reverse the sub-array starting from index i + 1 to the end
        # This turns the descending suffix into the smallest possible ascending sequence
        left, right = i + 1, n - 1
        while left < right:
            nums[left], nums[right] = nums[right], nums[left]
            left += 1
            right -= 1
