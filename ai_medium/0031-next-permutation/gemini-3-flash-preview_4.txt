class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        n = len(nums)
        if n <= 1:
            return
        
        # Step 1: Find the first pair (i, i+1) from the right such that nums[i] < nums[i+1]
        i = n - 2
        while i >= 0 and nums[i] >= nums[i+1]:
            i -= 1
        
        # Step 2: If such an index exists, find the element to swap with
        if i >= 0:
            # Step 3: Find the largest index j > i such that nums[j] > nums[i]
            j = n - 1
            while nums[j] <= nums[i]:
                j -= 1
            # Step 4: Swap nums[i] and nums[j]
            nums[i], nums[j] = nums[j], nums[i]
            
        # Step 5: Reverse the elements from index i+1 to the end of the array
        # This turns the non-increasing sequence into the smallest non-decreasing sequence
        left, right = i + 1, n - 1
        while left < right:
            nums[left], nums[right] = nums[right], nums[left]
            left += 1
            right -= 1
