class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        n = len(nums)
        if n <= 1:
            return

        # 1. Find the first element from the right that is smaller than the one after it.
        # This identifies the pivot point where the lexicographical order can be increased.
        i = n - 2
        while i >= 0 and nums[i] >= nums[i + 1]:
            i -= 1
        
        if i >= 0:
            # 2. Find the smallest element to the right of nums[i] that is larger than nums[i].
            # Since the subarray to the right of i is sorted in descending order, 
            # we just find the first element from the right that is larger than nums[i].
            j = n - 1
            while nums[j] <= nums[i]:
                j -= 1
            # Swap the found elements.
            nums[i], nums[j] = nums[j], nums[i]
        
        # 3. Reverse the subarray to the right of index i.
        # This transforms the suffix (which was in descending order) into ascending order,
        # ensuring we have the smallest possible permutation for the modified prefix.
        left = i + 1
        right = n - 1
        while left < right:
            nums[left], nums[right] = nums[right], nums[left]
            left += 1
            right -= 1
