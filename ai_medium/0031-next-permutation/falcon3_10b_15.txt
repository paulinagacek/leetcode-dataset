class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        for i in range(len(nums) - 2, -1, -1):
            if nums[i] < nums[i + 1]:
                break
        else:
            nums.reverse()
            return
        for j in range(len(nums) - 1, i, -1):
            if nums[j] > nums[i]:
                nums[i], nums[j] = nums[j], nums[i]
                break
        nums[i + 1:] = reversed(nums[i + 1:])
