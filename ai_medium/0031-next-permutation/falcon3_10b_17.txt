class Solution:
    def nextPermutation(self, nums: List[int]) -> None:
        n = len(nums)
        idx = n - 2
        while idx >= 0 and nums[idx] >= nums[idx + 1]:
            idx -= 1
        if idx >= 0:
            j = n - 1
            while nums[j] <= nums[idx]:
                j -= 1
            nums[idx], nums[j] = nums[j], nums[idx]
        nums[idx + 1:] = nums[n - 1:idx:-1]
