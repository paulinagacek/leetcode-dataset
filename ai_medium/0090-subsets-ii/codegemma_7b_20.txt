class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = [[]]
        temp = []

        def backtrack(start):
            res.append([] + temp)
            for i in range(start, len(nums)):
                if i > start and nums[i] == nums[i - 1]:
                    continue
                temp.append(nums[i])
                backtrack(i + 1)
                temp.pop()

        backtrack(0)
        return res
