class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        result = []
        nums.sort()  # Sorting helps to handle duplicates easily
        self.backtrack(nums, 0, [], result)
        return result

    def backtrack(self, nums, start, path, result):
        result.append(list(path))  # Add current subset to result
        for i in range(start, len(nums)):
            if i > start and nums[i] == nums[i - 1]:  # Skip duplicates
                continue
            path.append(nums[i])  # Include nums[i] in the current subset
            self.backtrack(nums, i + 1, path, result)  # Recur for the next element
            path.pop()  # Backtrack: remove the last element added to path
