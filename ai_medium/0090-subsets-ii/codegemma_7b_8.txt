class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = [[]]
        temp = []

        for i in range(len(nums)):
            if i > 0 and nums[i] == nums[i-1]:
                start = len(res) - len(temp)
            else:
                start = 0

            for j in range(start, len(res)):
                temp.append(nums[i])
                res.append(res[j] + temp)

        return res
