class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = [[]]

        start = 0
        for i in range(1, len(nums)):
            if nums[i] == nums[i - 1]:
                start = max(start, res[-1].index([nums[i]]) + 1)
            for j in range(start, len(res)):
                res.append(res[j] + [nums[i]])

        return res
