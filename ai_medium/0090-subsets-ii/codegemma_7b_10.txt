class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        n = len(nums)
        res = []
        def backtrack(index, subset):
            res.append(subset[:])
            for i in range(index, n):
                if i > index and nums[i] == nums[i-1]:
                    continue
                subset.append(nums[i])
                backtrack(i+1, subset)
                subset.pop()
        backtrack(0, [])
        return res
