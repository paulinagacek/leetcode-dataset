class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res: List[List[int]] = [[]]
        start = 0
        end = 0

        for i, x in enumerate(nums):
            start = 0
            if i > 0 and nums[i] == nums[i - 1]:
                start = end + 1
            end = len(res) - 1

            for j in range(start, end + 1):
                res.append(res[j] + [x])

        return res
