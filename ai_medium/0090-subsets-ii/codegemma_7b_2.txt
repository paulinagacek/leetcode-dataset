class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        result = [[]]
        start = 0
        for i in range(1, len(nums)):
            if nums[i] == nums[i - 1]:
                start = len(result)
            temp = result[:]
            for item in temp:
                result.append(item + [nums[i]])
        return result
