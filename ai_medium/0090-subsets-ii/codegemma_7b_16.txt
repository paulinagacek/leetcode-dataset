class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = [[]]
        temp = []
        for i in range(len(nums)):
            if i > 0 and nums[i] == nums[i-1]:
                temp = res[:]
            else:
                temp = res + [[nums[i]]]
            res = temp
        return res
