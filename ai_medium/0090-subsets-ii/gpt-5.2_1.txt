class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res: List[List[int]] = []
        subset: List[int] = []

        def backtrack(start: int) -> None:
            res.append(subset[:])
            i = start
            while i < len(nums):
                subset.append(nums[i])
                backtrack(i + 1)
                subset.pop()
                i += 1
                while i < len(nums) and nums[i] == nums[i - 1]:
                    i += 1

        backtrack(0)
        return res
