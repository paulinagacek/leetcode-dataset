class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m = len(s1), len(s2)
        
        # Check if the total length matches s3
        if n + m != len(s3):
            return False
        
        # Initialize the DP array with size len(s2) + 1.
        # dp[j] will represent if s1[0:i] and s2[0:j] can form s3[0:i+j].
        dp = [False] * (m + 1)
        
        # Base case: s1 and s2 are empty, s3 must be empty.
        dp[0] = True
        
        # Fill the DP array for the case where s1 is empty (i=0).
        for j in range(1, m + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Iterate through s1 characters.
        for i in range(1, n + 1):
            # Update dp[0] for the case where s2 is empty (j=0).
            # dp[0] currently represents dp[i-1][0].
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            
            # Iterate through s2 characters.
            for j in range(1, m + 1):
                # dp[j] (before update) is dp[i-1][j].
                # dp[j-1] (after update in this row) is dp[i][j-1].
                # s3 index is i + j - 1.
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or \
                        (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
        
        return dp[m]
