class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n1, n2, n3 = len(s1), len(s2), len(s3)
        if n1 + n2 != n3:
            return False
        dp = [False] * (n2 + 1)
        dp[0] = True
        for i in range(n3):
            if s3[i] == s1[0] and dp[0]:
                dp[0] = False
                dp[1:] = [dp[j] or (s3[i] == s2[j - 1] and dp[j - 1]) for j in range(1, n2 + 1)]
        return dp[n2]
