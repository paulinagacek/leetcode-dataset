class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m = len(s1), len(s2)
        if n + m != len(s3):
            return False
        
        # We can optimize to use O(min(len(s1), len(s2))) space by swapping s1 and s2
        # if len(s1) < len(s2), but the requirement asks for O(s2.length).
        
        # dp[j] will store whether s3[0:i+j] can be formed by interleaving s1[0:i] and s2[0:j].
        dp = [False] * (m + 1)
        
        # Base case: empty s1 and empty s2 form empty s3
        dp[0] = True
        
        # Initialize dp for i = 0 (using only s2 to form s3)
        for j in range(1, m + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Fill dp table row by row
        for i in range(1, n + 1):
            # Update dp[0] for the current row i (using only s1 to form s3)
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, m + 1):
                # Current state dp[j] depends on the previous state for s1 (dp[j])
                # and the previous state for s2 (dp[j-1]).
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or \
                        (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                        
        return dp[m]
