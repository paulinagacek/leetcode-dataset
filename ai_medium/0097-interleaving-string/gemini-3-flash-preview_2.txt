class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n1, n2, n3 = len(s1), len(s2), len(s3)
        if n1 + n2 != n3:
            return False
        
        # dp[j] will store the result for s1[0...i] and s2[0...j]
        # Using O(s2.length) additional space
        dp = [False] * (n2 + 1)
        dp[0] = True
        
        # Initialize the dp array for the case where s1 is empty
        for j in range(1, n2 + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Iterate through s1
        for i in range(1, n1 + 1):
            # Update dp[0] for each row (the case where s2 is empty)
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, n2 + 1):
                # To form s3[0...i+j-1], we can either:
                # 1. Use s1[0...i-1] and s2[0...j] and check if s1[i-1] matches s3[i+j-1]
                # 2. Use s1[0...i] and s2[0...j-1] and check if s2[j-1] matches s3[i+j-1]
                # dp[j] currently holds the result from the previous row (s1[0...i-1] and s2[0...j])
                # dp[j-1] currently holds the result from the current row (s1[0...i] and s2[0...j-1])
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or \
                        (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                        
        return dp[n2]
