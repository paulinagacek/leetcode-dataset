class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m = len(s1), len(s2)
        if n + m != len(s3):
            return False
        
        # dp[j] will store whether s3[0...i+j-1] can be formed by s1[0...i-1] and s2[0...j-1]
        # This implementation uses O(m) space, which is O(s2.length).
        dp = [False] * (m + 1)
        
        # Base case: s1 and s2 are both empty
        dp[0] = True
        
        # Initialize the dp array for the case where s1 is empty (i=0)
        # s3 is formed solely by picking characters from s2
        for j in range(1, m + 1):
            dp[j] = dp[j-1] and s2[j-1] == s3[j-1]
            
        # Fill the dp table row by row (i=1 to n)
        for i in range(1, n + 1):
            # Update dp[0] for the current row i (case where s2 is empty)
            # dp[0] currently holds dp[i-1][0]. We update it to dp[i][0].
            dp[0] = dp[0] and s1[i-1] == s3[i-1]
            
            for j in range(1, m + 1):
                # Transition:
                # dp[j] (representing dp[i][j]) is True if:
                # 1. We picked the last character from s1:
                #    dp[i-1][j] was True (previous value of dp[j]) AND s1[i-1] == s3[i+j-1]
                # 2. We picked the last character from s2:
                #    dp[i][j-1] was True (already updated value of dp[j-1]) AND s2[j-1] == s3[i+j-1]
                dp[j] = (dp[j] and s1[i-1] == s3[i+j-1]) or (dp[j-1] and s2[j-1] == s3[i+j-1])
                
        return dp[m]
