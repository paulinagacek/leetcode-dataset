class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        dp = [False] * (len(s2) + 1)
        dp[0] = True
        
        for i in range(1, len(s2) + 1):
            dp[i] = dp[i - 1] and s2[i - 1] == s3[i - 1]
        
        for i in range(len(s1)):
            for j in range(len(s2)):
                dp[j + 1] = dp[j + 1] or dp[j] and s1[i] == s3[i + j + 1]
        
        return dp[-1]
