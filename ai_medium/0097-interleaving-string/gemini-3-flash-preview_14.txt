class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n1, n2, n3 = len(s1), len(s2), len(s3)
        
        # The length of s3 must be the sum of lengths of s1 and s2
        if n1 + n2 != n3:
            return False
        
        # Use a 1D DP array for space optimization to O(s2.length)
        # dp[j] will store whether s1[:i] and s2[:j] can form s3[:i+j]
        dp = [False] * (n2 + 1)
        
        # Base case: both s1 and s2 are empty
        dp[0] = True
        
        # Fill DP array for the case where s1 is empty (row 0)
        for j in range(1, n2 + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Iterate through the characters of s1
        for i in range(1, n1 + 1):
            # For each new row, update dp[0] (which represents s1[:i] and s2[:0])
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            
            # Update the DP table for current row i
            for j in range(1, n2 + 1):
                # dp[j] is reachable if:
                # 1. dp[j] (from the previous row i-1) was True and s1[i-1] matches s3[i+j-1]
                # 2. dp[j-1] (from the current row i) was True and s2[j-1] matches s3[i+j-1]
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or \
                        (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                        
        return dp[n2]
