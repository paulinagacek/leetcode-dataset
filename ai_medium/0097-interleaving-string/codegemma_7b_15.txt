class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m, l = len(s1), len(s2), len(s3)
        if n + m != l:
            return False
        dp = [False] * (m + 1)
        dp[0] = True
        for i in range(1, l + 1):
            dp[i % (m + 1)] = (dp[(i - 1) % (m + 1)] and s3[i - 1] == s2[i - 1 - n]) or (dp[(i - m - 1) % (m + 1)] and s3[i - 1] == s1[i - 1 - m])
        return dp[l % (m + 1)]
