class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        m, n = len(s1), len(s2)
        if m + n != len(s3):
            return False
        
        # Ensure s2 is the smaller string for O(min(m, n)) space optimization
        # Here we follow the requirement for O(s2.length)
        
        # dp[j] will represent whether s3[0...i+j-1] can be formed by 
        # interleaving s1[0...i-1] and s2[0...j-1].
        dp = [False] * (n + 1)
        
        # Base case: empty s1 and empty s2 form empty s3
        dp[0] = True
        
        # Initialize dp array for the case where s1 is empty
        for j in range(1, n + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        for i in range(1, m + 1):
            # Update dp[0] for the case where s2 is empty
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, n + 1):
                # The condition to update dp[j] (current i, j) depends on:
                # 1. dp[j] from the previous iteration (i-1, j) and current s1[i-1] matching s3[i+j-1]
                # 2. dp[j-1] from the current iteration (i, j-1) and current s2[j-1] matching s3[i+j-1]
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                
        return dp[n]
