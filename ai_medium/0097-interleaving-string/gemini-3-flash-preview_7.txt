class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n1, n2, n3 = len(s1), len(s2), len(s3)
        if n1 + n2 != n3:
            return False
        
        # Use O(s2.length) additional space
        dp = [False] * (n2 + 1)
        
        # Base case: empty s1 and empty s2 form empty s3
        dp[0] = True
        
        # Initialize the first row (interleaving empty s1 with s2 prefixes)
        for j in range(1, n2 + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Iterate through s1 characters
        for i in range(1, n1 + 1):
            # dp[0] currently stores dp[i-1][0], update it to represent dp[i][0]
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, n2 + 1):
                # dp[j] (before update) is dp[i-1][j]
                # dp[j-1] (after update) is dp[i][j-1]
                # Transition: s3[i+j-1] matches either s1's character or s2's character
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                
        return dp[n2]
