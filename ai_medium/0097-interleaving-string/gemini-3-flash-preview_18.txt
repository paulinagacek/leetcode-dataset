class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m = len(s1), len(s2)
        if n + m != len(s3):
            return False
        
        # Ensure that s2 is the one used for the DP array size to meet the follow-up.
        # However, for O(s2.length) as requested specifically, we can just proceed.
        # If we want O(min(len(s1), len(s2))), we would swap.
        
        dp = [False] * (m + 1)
        dp[0] = True
        
        # Initialize the DP array for the case where s1 is empty.
        # This represents dp[0][j] in a 2D DP table.
        for j in range(1, m + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        for i in range(1, n + 1):
            # Update dp[0] for the case where s2 is empty.
            # This represents dp[i][0] in a 2D DP table.
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, m + 1):
                # dp[j] (before update) is dp[i-1][j]
                # dp[j-1] (after update) is dp[i][j-1]
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
                
        return dp[m]
