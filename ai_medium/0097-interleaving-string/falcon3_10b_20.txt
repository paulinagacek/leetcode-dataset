class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n1 = len(s1)
        n2 = len(s2)
        n3 = len(s3)

        # If the combined length of s1 and s2 is not equal to s3, return False
        if n3 != n1 + n2:
            return False

        # Initialize memoization table
        memo = {}

        def dp(i: int, j: int) -> bool:
            if (i, j) in memo:
                return memo[(i, j)]

            # If reached the end of all strings
            if i == n1 and j == n2:
                return True

            # Check conditions for taking characters from s1 or s2
            res = False
            if i < n1 and s1[i] == s3[i + j]:
                res |= dp(i + 1, j)
            if j < n2 and s2[j] == s3[i + j]:
                res |= dp(i, j + 1)

            # Cache the result
            memo[(i, j)] = res
            return res

        # Start the dynamic programming process
        return dp(0, 0)
