class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m, l = len(s1), len(s2), len(s3)
        if l != n + m:
            return False
        dp = [False] * (m + 1)
        dp[0] = True
        for i in range(1, m + 1):
            dp[i] = dp[i - 1] and s2[i - 1] == s3[ i - 1]
        for i in range(1, n + 1):
            dp[i] = dp[i] or (dp[i - 1] and s1[i - 1] == s3[i - 1])
        return dp[n]
