class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        # Check if the sum of lengths of s1 and s2 matches s3's length
        n, m = len(s1), len(s2)
        if n + m != len(s3):
            return False
        
        # Use O(m) additional space by reducing the 2D DP table to a 1D row array
        # dp[j] will store whether s3[:i+j] can be formed by interleaving s1[:i] and s2[:j]
        dp = [False] * (m + 1)
        
        # Base case for the DP: both s1 and s2 are empty
        dp[0] = True
        
        # Initialize dp for the case where only s2 is used (i = 0)
        for j in range(1, m + 1):
            dp[j] = dp[j - 1] and s2[j - 1] == s3[j - 1]
            
        # Iterate through s1 and s2 to fill the DP array
        for i in range(1, n + 1):
            # Update dp[0] to reflect the case where only s1 is used (i characters from s1, 0 from s2)
            dp[0] = dp[0] and s1[i - 1] == s3[i - 1]
            for j in range(1, m + 1):
                # The character s3[i + j - 1] can be formed either:
                # 1. From s1[i - 1] if s1[:i-1] and s2[:j] could form s3[:i+j-1]
                # 2. From s2[j - 1] if s1[:i] and s2[:j-1] could form s3[:i+j-1]
                dp[j] = (dp[j] and s1[i - 1] == s3[i + j - 1]) or (dp[j - 1] and s2[j - 1] == s3[i + j - 1])
        
        # The result is stored at the end of the DP array
        return dp[m]
