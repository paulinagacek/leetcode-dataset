class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        n, m, k = len(s1), len(s2), len(s3)
        if n + m != k:
            return False

        dp = [False] * (m + 1)
        dp[0] = True

        for i in range(1, k + 1):
            if s3[i - 1] == s2[i - 1]:
                dp[i] = dp[i - 1]
            if s3[i - 1] == s1[i - 1]:
                dp[i] = dp[i] or dp[i - 1]

        return dp[k]
