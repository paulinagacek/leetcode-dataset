class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        l1, l2, l3 = len(s1), len(s2), len(s3)
        if l1 + l2 != l3:
            return False
        dp = [False] * (l2+1)
        dp[0] = True
        for i in range(l3):
            if s3[i] == s2[i] and dp[i]:
                dp[i+1] = True
            if s3[i] == s1[i - l2] and i >= l2 and dp[i-l2]:
                dp[i+1] = True
        return dp[-1]
