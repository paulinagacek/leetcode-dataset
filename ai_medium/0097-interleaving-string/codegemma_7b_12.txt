class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        l1, l2, l3 = len(s1), len(s2), len(s3)
        if l1 + l2 != l3:
            return False
        dp = [False] * (l2 + 1)
        dp[0] = True
        for i in range(1, l2 + 1):
            dp[i] = dp[i - 1] and s2[i - 1] == s3[i - 1]
        for i in range(1, l1 + l2):
            dp[i] = dp[i] or (dp[i - 1] and (i >= l1 or s3[i] == s1[i - l1]))
        return dp[-1]
