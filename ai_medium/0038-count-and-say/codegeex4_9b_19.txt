class Solution:
    def countAndSay(self, n: int) -> str:
        s = '1'
        for _ in range(n - 1):
            cnt = 1
            temp = ''
            for j in range(1, len(s)):
                if s[j] == s[j-1]:
                    cnt += 1
                else:
                    temp += (str(cnt) + s[j-1])
                    cnt = 1
            temp += str(cnt) + s[-1]
            s = temp
        return s
