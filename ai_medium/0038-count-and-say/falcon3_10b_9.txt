class Solution:
    def countAndSay(self, n: int) -> str:
        result = "1"
        for _ in range(n - 1):
            result = self.runLengthEncoding(result)
        return result

    def runLengthEncoding(self, s: str) -> str:
        encoded = ""
        count = 1
        for i in range(1, len(s)):
            if s[i] == s[i-1]:
                count += 1
            else:
                encoded += str(count) + s[i-1]
                count = 1
        encoded += str(count) + s[-1]
        return encoded
