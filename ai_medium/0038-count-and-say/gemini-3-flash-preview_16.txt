class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1:
            return "1"
        
        curr = "1"
        for _ in range(n - 1):
            next_val = []
            i = 0
            while i < len(curr):
                count = 1
                while i + 1 < len(curr) and curr[i] == curr[i + 1]:
                    i += 1
                    count += 1
                next_val.append(str(count))
                next_val.append(curr[i])
                i += 1
            curr = "".join(next_val)
            
        return curr
