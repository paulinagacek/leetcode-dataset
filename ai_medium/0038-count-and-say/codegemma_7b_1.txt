class Solution:
    def countAndSay(self, n: int) -> str:
        if n < 2:
            return "1"
        
        prev = self.countAndSay(n - 1)
        cur = ""
        count = 1
        for i in range(1, len(prev)):
            if prev[i] == prev[i - 1]:
                count += 1
            else:
                cur += str(count) + prev[i - 1]
                count = 1
        
        cur += str(count) + prev[-1]
        return cur
