class Solution:
    def countAndSay(self, n: int) -> str:
        def helper(s):
            i = 0
            result = []
            while i < len(s):
                count = 1
                while i + count < len(s) and s[i] == s[i + count]:
                    count += 1
                result.append(str(count) + s[i])
                i += count
            return ''.join(result)
        
        ans = '1'
        for i in range(n - 1):
            ans = helper(ans)
        return ans
