class Solution:
    def countAndSay(self, n: int) -> str:
        previous = "1"
        for _ in range(n - 1):
            current = ""
            count = 1
            for i in range(1, len(previous)):
                if previous[i] == previous[i - 1]:
                    count += 1
                else:
                    current += str(count) + previous[i - 1]
                    count = 1
            current += str(count) + previous[-1]
            previous = current
        return previous
