class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1:
            return "1"
        
        res = "1"
        for _ in range(n - 1):
            next_res = []
            i = 0
            while i < len(res):
                count = 1
                # Move pointer while the current character is the same as the next character
                while i + 1 < len(res) and res[i] == res[i + 1]:
                    count += 1
                    i += 1
                # Append the count and the character itself
                next_res.append(str(count))
                next_res.append(res[i])
                i += 1
            # Update the result for the next iteration
            res = "".join(next_res)
            
        return res
