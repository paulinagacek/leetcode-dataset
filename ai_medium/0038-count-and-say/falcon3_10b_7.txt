class Solution:
    def countAndSay(self, n: int) -> str:
        result = "1"
        for _ in range(1, n):
            tmp, new_result = "", ""
            for char in result:
                if tmp and char != tmp[-1]:
                    new_result += f"{len(tmp)}{tmp[0]}"
                    tmp = ""
                tmp += char
            new_result += f"{len(tmp)}{tmp[0]}"
            result = new_result
        return result
