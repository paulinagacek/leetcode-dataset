class Solution:
    def countAndSay(self, n: int) -> str:
        string = "1"
        for _ in range(n - 1):
            temp = ""
            count = 1
            for i in range(1, len(string)):
                if string[i] == string[i - 1]:
                    count += 1
                else:
                    temp += str(count) + string[i - 1]
                    count = 1
            temp += str(count) + string[-1]
            string = temp
        return string
