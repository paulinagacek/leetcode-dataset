class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1:
            return "1"
        
        curr = "1"
        for _ in range(2, n + 1):
            next_val = []
            i = 0
            while i < len(curr):
                char = curr[i]
                count = 0
                while i < len(curr) and curr[i] == char:
                    count += 1
                    i += 1
                next_val.append(str(count))
                next_val.append(char)
            curr = "".join(next_val)
            
        return curr
