class Solution:
    def countAndSay(self, n: int) -> str:
        s = '1'
        for _ in range(n - 1):
            temp, count, prev = '', 0, ''
            for char in s:
                if char == prev:
                    count += 1
                else:
                    temp += str(count) + prev
                    prev = char
                    count = 1
            temp += str(count) + prev
            s = temp
        return s
