class Solution:
    def countAndSay(self, n: int) -> str:
        s = '1'
        for _ in range(n - 1):
            cnt, tmp, prev = 0, '', s[0]
            for c in s:
                if c == prev:
                    cnt += 1
                else:
                    tmp += str(cnt) + prev
                    prev = c
                    cnt = 1
            s = tmp + str(cnt) + prev
        return s
