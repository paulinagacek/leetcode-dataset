class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1:
            return "1"
        
        result = "1"
        for _ in range(2, n + 1):
            temp = []
            j = 0
            n_len = len(result)
            while j < n_len:
                count = 1
                while j + 1 < n_len and result[j] == result[j + 1]:
                    count += 1
                    j += 1
                temp.append(str(count))
                temp.append(result[j])
                j += 1
            result = "".join(temp)
            
        return result
