class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1:
            return "1"
        sequence = "1"
        for _ in range(n - 1):
            new_sequence = ""
            count = 0
            previous_char = sequence[0]
            for char in sequence:
                if char == previous_char:
                    count += 1
                else:
                    new_sequence += str(count) + previous_char
                    previous_char = char
                    count = 1
            new_sequence += str(count) + previous_char
            sequence = new_sequence
        return sequence
