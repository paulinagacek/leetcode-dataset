class Solution:
    def countAndSay(self, n: int) -> str:
        ans = "1"
        for _ in range(n-1):
            temp, ans, count = ans, "", 1
            for i in range(1, len(temp)):
                if temp[i] == temp[i-1]:
                    count += 1
                else:
                    ans += str(count) + temp[i-1]
                    count = 1
            ans += str(count) + temp[-1]
        return ans
