class Solution:
    def countAndSay(self, n: int) -> str:
        s = "1"
        for _ in range(2, n + 1):
            parts = []
            i = 0
            while i < len(s):
                j = i + 1
                while j < len(s) and s[j] == s[i]:
                    j += 1
                parts.append(str(j - i))
                parts.append(s[i])
                i = j
            s = "".join(parts)
        return s
