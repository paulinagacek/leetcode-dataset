class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        matrix = [[0]*n for _ in range(n)]
        num, dx, dy = 1, 0, 1
        x, y = 0, 0
        matrix[x][y] = num
        for _ in range(n*n):
            if not (0 <= x+dx < n and 0 <= y+dy < n and matrix[x+dx][y+dy] == 0):
                dx, dy = dy, -dx
            x, y = x+dx, y+dy
            num += 1
            matrix[x][y] = num
        return matrix
