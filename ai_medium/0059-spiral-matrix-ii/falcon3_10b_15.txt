class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        matrix = [[0]*n for _ in range(n)]
        dx = [0, 1, 0, -1]
        dy = [1, 0, -1, 0]
        x, y, c = 0, 0, 1
        for i in range(2*n - 1):
            for _ in range(n - i):
                x, y = x + dx[i % 4], y + dy[i % 4]
                matrix[y][x] = c
                c += 1
