class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        matrix = [[0]*n for _ in range(n)]
        x, y, dx, dy = 0, 0, 0, 1
        for number in range(1, n*n + 1):
            matrix[y][x] = number
            nx, ny = x + dx, y + dy
            if not (0 <= ny < n and 0 <= nx < n and matrix[ny][nx] == 0):
                dx, dy = -dy, dx
            x, y = x + dx, y + dy
        return matrix
