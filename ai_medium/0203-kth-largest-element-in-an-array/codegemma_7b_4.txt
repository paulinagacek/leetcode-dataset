class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        p = nums[0]
        small = []
        large = []
        for i in range(1, len(nums)):
            if nums[i] <= p:
                small.append(nums[i])
            else:
                large.append(nums[i])
        if len(large) >= k:
            return self.findKthLargest(large, k)
        else:
            return p
