class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        # A min-heap of size k will keep the k largest elements seen so far.
        # The root of the min-heap (index 0) will be the kth largest element.
        min_heap = []
        for num in nums:
            if len(min_heap) < k:
                heapq.heappush(min_heap, num)
            elif num > min_heap[0]:
                heapq.heapreplace(min_heap, num)
        
        return min_heap[0]
