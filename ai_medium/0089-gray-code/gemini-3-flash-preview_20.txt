class Solution:
    def grayCode(self, n: int) -> List[int]:
        # The standard Gray code sequence can be generated using the formula:
        # G(i) = i ^ (i >> 1)
        # This formula produces a sequence where each adjacent pair (including first and last)
        # differs by exactly one bit, and it covers all integers from 0 to 2^n - 1.
        return [i ^ (i >> 1) for i in range(1 << n)]
