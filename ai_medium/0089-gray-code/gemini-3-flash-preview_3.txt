class Solution:
    def grayCode(self, n: int) -> List[int]:
        # The standard formula for the i-th number in an n-bit Gray code sequence
        # is G(i) = i ^ (i >> 1). This generates a sequence of 2^n numbers 
        # where adjacent values (including the first and last) differ by exactly one bit.
        return [i ^ (i >> 1) for i in range(1 << n)]
