class Solution:
    def grayCode(self, n: int) -> List[int]:
        # Using the formula G(i) = i ^ (i >> 1) to generate the sequence
        # This standard Gray code sequence satisfies all the properties:
        # 1. Starts with 0.
        # 2. Contains 2^n unique integers in range [0, 2^n - 1].
        # 3. Adjacent elements differ by exactly one bit.
        # 4. The first and last elements differ by exactly one bit (cyclic property).
        return [i ^ (i >> 1) for i in range(1 << n)]
