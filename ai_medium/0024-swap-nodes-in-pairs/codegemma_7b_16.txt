class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        prev, cur = dummy, head

        while cur and cur.next:
            temp = cur.next.next
            cur.next, cur.next.next, prev.next = cur.next.next, cur, temp
            prev, cur = cur, temp

        return dummy.next
