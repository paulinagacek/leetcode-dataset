# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        new_head = head.next
        
        prev = None
        current = head
        while current and current.next:
            nxt_node = current.next
            next_pair = nxt_node.next
            
            if prev:
                prev.next = nxt_node
            else:
                new_head = nxt_node
            
            nxt_node.next = current
            current.next = next_pair
            
            prev = current
            current = next_pair
        
        return new_head
