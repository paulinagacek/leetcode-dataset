# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        new_head = head.next
        
        prev = None
        cur = head
        
        while True:
            n = cur.next
            
            if not n:
                break
            
            next_pair = n.next
            n.next = cur
            cur.next = next_pair
            
            if prev:
                prev.next = n
            
            prev = cur
            cur = next_pair
        
        return new_head
