class Solution:
    def swapPairs(self, head: ListNode) -> ListNode:
        dummy = ListNode(0, head)
        prev, curr = dummy, dummy.next
        
        while curr and curr.next:
            temp = curr.next.next
            curr.next, curr.next.next = curr.next.next, curr.next
            prev.next, prev = curr.next, temp
            
            curr = temp
        
        return dummy.next
