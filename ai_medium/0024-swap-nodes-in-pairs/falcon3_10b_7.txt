# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        new_head = head.next
        prev, first, second = head, head.next, second = head.next
        prev.next = second
        while second and second.next:
            first.next = second.next
            prev.next = second
            prev, first, second = first, second, second.next
        return new_head
