# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        prev, curr = None, head
        new_head = ListNode(0)
        ans = new_head
        
        while True:
            temp1 = curr.next if curr else None
            temp2 = curr.next.next if curr and curr.next else None
            
            if temp1:
                new_head.next = temp1
                temp1.next = curr
                curr.next = temp2
                
                new_head = curr
                curr = curr.next if curr and curr.next else None
            else:
                break
        
        return ans.next
