class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(next=head)
        prev, curr = dummy, head

        while curr and curr.next:
            temp = curr.next.next
            curr.next.next = curr
            curr.next = temp
            prev.next = curr.next
            prev = curr
            curr = temp

        return dummy.next
