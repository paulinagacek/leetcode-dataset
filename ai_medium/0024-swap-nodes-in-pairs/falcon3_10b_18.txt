# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        new_head = head.next
        prev = None
        curr = head
        
        while curr and curr.next:
            next1 = curr.next
            next2 = curr.next.next
            curr.next = next2
            next1.next = curr
            if prev:
                prev.next = next1
            prev = curr
            curr = curr.next
        
        return new_head
