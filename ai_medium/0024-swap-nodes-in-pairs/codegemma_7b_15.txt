class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummyHead = ListNode(0, head)
        prev, curr = dummyHead, head

        while curr and curr.next:
            temp = curr.next.next
            curr.next.next = curr
            curr.next = temp
            prev.next = curr.next

            prev = curr
            curr = temp

        return dummyHead.next
