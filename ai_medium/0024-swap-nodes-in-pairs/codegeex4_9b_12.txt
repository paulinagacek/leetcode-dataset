# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        dummy = ListNode(0)
        dummy.next = head
        prev = dummy
        
        while head and head.next:
            nex = head.next
            temp = nex.next
            
            prev.next = nex
            nex.next = head
            head.next = temp
            
            prev = head
            head = temp
        
        return dummy.next
