# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        new_head = head.next
        prev, curr = None, head
        while curr and curr.next:
            next1, next2 = curr.next, curr.next.next
            if prev:
                prev.next = next1
            curr.next = next2
            next1.next = curr
            prev = curr
            curr = next2
        return new_head
