# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        prev, cur = None, head
        new_head = ListNode()
        new_head.next = head
        
        while cur and cur.next:
            next_pair = cur.next.next
            if prev:
                prev.next = cur.next
            else:
                new_head.next = cur.next
            
            cur.next.next = cur
            cur.next = next_pair
            
            prev = cur
            cur = next_pair
        
        return new_head.next
