class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        curr = head
        prev = None
        
        while curr and curr.next:
            temp = curr.next.next
            curr.next.next = curr
            curr.next = temp
            prev = curr
            curr = temp
            
        return head
