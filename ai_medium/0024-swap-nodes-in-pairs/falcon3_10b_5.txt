# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        new_head = head.next
        current = head
        while current and current.next:
            next_node = current.next
            swap_next = next_node.next
            next_node.next = current
            if swap_next:
                current.next = swap_next
            else:
                current.next = None
            temp = current
            current = swap_next
            temp.next = current
        return new_head
