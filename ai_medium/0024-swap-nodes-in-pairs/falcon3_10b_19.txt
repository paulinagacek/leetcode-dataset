# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0)
        dummy.next = head
        prev = dummy
        while head and head.next:
            first = head
            second = head.next
            # Swap the nodes
            prev.next = second
            second.next = first
            first.next = second.next
            # Move to the next pair
            prev = first
            head = first.next
        return dummy.next
