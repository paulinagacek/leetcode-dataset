# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0)
        dummy.next = head
        prev = dummy
        
        while head and head.next:
            nxt = head.next.next
            
            # Swapping the nodes
            prev.next = head.next
            head.next.next = head
            head.next = nxt
            
            # Moving two steps forward
            prev = head
            head = nxt
        
        return dummy.next
