# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        new_head = head.next
        prev, curr = None, head
        
        while curr and curr.next:
            next_pair = curr.next.next
            if prev:
                prev.next = curr.next
            curr.next.next = curr
            curr.next = next_pair
            
            prev = curr
            curr = next_pair
        
        return new_head
