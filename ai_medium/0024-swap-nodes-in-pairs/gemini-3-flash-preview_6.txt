class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # A dummy node helps manage the head of the list during swapping
        dummy = ListNode(0, head)
        prev = dummy
        
        # Iterate while there is a pair of nodes to swap
        while prev.next and prev.next.next:
            # Identify the two nodes to be swapped
            first = prev.next
            second = prev.next.next
            
            # Perform the swap by adjusting next pointers
            first.next = second.next
            second.next = first
            prev.next = second
            
            # Move the previous pointer two steps forward (to the end of the swapped pair)
            prev = first
            
        return dummy.next
