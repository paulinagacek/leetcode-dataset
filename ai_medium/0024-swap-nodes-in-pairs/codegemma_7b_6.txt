class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        prev, cur = dummy, head
        while cur and cur.next:
            temp = cur.next.next
            cur.next.next = cur
            cur.next = temp
            prev.next = cur.next
            prev = cur
            cur = temp
        return dummy.next
