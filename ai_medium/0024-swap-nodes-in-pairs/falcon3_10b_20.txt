# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        new_head = head.next
        curr = head
        next_pair = head.next.next if head.next else None
        while curr and next_pair:
            curr.next = next_pair.next
            next_pair.next = curr
            if not curr.next:
                break
            temp = curr.next
            curr.next = temp.next
            temp.next = curr
            curr = curr.next
            if curr and curr.next:
                next_pair = curr.next.next
            else:
                break
        return new_head
