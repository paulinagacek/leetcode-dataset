# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        new_head = head.next
        
        prev, curr = None, head
        while curr and curr.next:
            nxt_pair = curr.next.next
            if prev:
                prev.next = curr.next
            else:
                prev = ListNode(-1)
                prev.next = new_head
            
            curr.next.next = curr
            curr.next = nxt_pair
            
            prev = curr
            curr = nxt_pair
        
        return new_head
