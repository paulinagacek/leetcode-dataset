class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        first = head
        second = head.next

        first.next = second.next
        second.next = first
        head = second

        temp = first.next
        while temp and temp.next:
            first = temp
            second = temp.next

            first.next = second.next
            second.next = first

            temp = first.next

        return head
