class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(val=0, next=head)
        curr = dummy
        while curr.next and curr.next.next:
            temp = curr.next.next
            curr.next.next = temp.next
            temp.next = curr.next
            curr.next = temp
            curr = curr.next.next
        return dummy.next
