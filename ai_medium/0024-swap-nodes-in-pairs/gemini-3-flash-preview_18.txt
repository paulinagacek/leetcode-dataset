class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        prev = dummy
        
        while head and head.next:
            # Nodes to be swapped
            first = head
            second = head.next
            
            # Swapping nodes
            prev.next = second
            first.next = second.next
            second.next = first
            
            # Reinitializing pointers for the next iteration
            prev = first
            head = first.next
            
        return dummy.next
