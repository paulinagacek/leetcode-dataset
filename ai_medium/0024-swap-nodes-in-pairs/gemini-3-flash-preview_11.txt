class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0)
        dummy.next = head
        current = dummy
        
        while current.next and current.next.next:
            # Nodes to be swapped
            first = current.next
            second = current.next.next
            
            # Swapping
            first.next = second.next
            second.next = first
            current.next = second
            
            # Re-position current for the next pair
            current = first
            
        return dummy.next
