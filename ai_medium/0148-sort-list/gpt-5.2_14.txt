class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        def split(start: Optional[ListNode], size: int) -> Optional[ListNode]:
            if not start:
                return None
            for _ in range(size - 1):
                if start.next:
                    start = start.next
                else:
                    break
            nxt = start.next
            start.next = None
            return nxt

        def merge(l1: Optional[ListNode], l2: Optional[ListNode]) -> tuple[Optional[ListNode], Optional[ListNode]]:
            dummy = ListNode(0)
            tail = dummy
            while l1 and l2:
                if l1.val <= l2.val:
                    tail.next = l1
                    l1 = l1.next
                else:
                    tail.next = l2
                    l2 = l2.next
                tail = tail.next
            tail.next = l1 if l1 else l2
            while tail.next:
                tail = tail.next
            return dummy.next, tail

        n = 0
        cur = head
        while cur:
            n += 1
            cur = cur.next

        dummy = ListNode(0, head)
        size = 1
        while size < n:
            prev = dummy
            cur = dummy.next
            while cur:
                left = cur
                right = split(left, size)
                cur = split(right, size)
                merged_head, merged_tail = merge(left, right)
                prev.next = merged_head
                prev = merged_tail
            size <<= 1

        return dummy.next
