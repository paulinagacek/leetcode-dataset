# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # Step 1: Use the fast and slow pointer strategy to find the middle of the linked list.
        slow, fast = head, head.next
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        
        # Step 2: Split the linked list into two halves.
        mid = slow.next
        slow.next = None
        
        # Step 3: Recursively sort both halves.
        leftsorted = self.sortList(head)
        rightsorted = self.sortList(mid)
        
        # Step 4: Merge both sorted halves.
        return self.merge(leftsorted, rightsorted)
    
    def merge(self, l1, l2):
        dummy = ListNode()
        curr = dummy
        
        while l1 and l2:
            if l1.val < l2.val:
                curr.next = l1
                l1 = l1.next
            else:
                curr.next = l2
                l2 = l2.next
            curr = curr.next
        
        if l1:
            curr.next = l1
        elif l2:
            curr.next = l2
            
        return dummy.next
