class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # Get the length of the linked list
        length = 0
        curr = head
        while curr:
            length += 1
            curr = curr.next
        
        # Helper to split the list after n nodes and return the starting node of the remainder
        def split(node, n):
            if not node:
                return None
            for _ in range(n - 1):
                if node.next:
                    node = node.next
                else:
                    break
            nxt = node.next
            node.next = None
            return nxt
        
        # Helper to merge two sorted lists and return the head and tail of the merged list
        def merge(l1, l2):
            dummy = ListNode(0)
            tail = dummy
            while l1 and l2:
                if l1.val < l2.val:
                    tail.next = l1
                    l1 = l1.next
                else:
                    tail.next = l2
                    l2 = l2.next
                tail = tail.next
            tail.next = l1 if l1 else l2
            while tail.next:
                tail = tail.next
            return dummy.next, tail
        
        dummy = ListNode(0)
        dummy.next = head
        step = 1
        
        # Iteratively merge sublists of increasing step sizes: 1, 2, 4, 8, ...
        while step < length:
            prev = dummy
            curr = dummy.next
            while curr:
                left = curr
                right = split(left, step)
                curr = split(right, step)
                
                merged_h, merged_t = merge(left, right)
                prev.next = merged_h
                prev = merged_t
            step *= 2
            
        return dummy.next
