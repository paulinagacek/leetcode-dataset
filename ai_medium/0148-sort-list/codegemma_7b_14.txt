# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(next=head)
        length = 0
        cur = head
        while cur:
            length += 1
            cur = cur.next
        
        def mergeSort(start, end):
            if start >= end:
                return
            mid = (start + end) // 2
            mergeSort(start, mid)
            mergeSort(mid+1, end)
            merge(start, mid, end)
        def merge(start, mid, end):
            dummy.next = start
            cur = dummy
            left = start
            right = mid+1
            while left <= mid and right <= end:
                if dummy.next.val <= dummy.next.next.val:
                    cur.next = dummy.next
                    dummy.next = dummy.next.next
                else:
                    cur.next = dummy.next.next
                    dummy.next.next = dummy.next
                cur = cur.next    
            if right > end:
                cur.next = dummy.next
                dummy.next = left
            else:
                cur.next = dummy.next
                dummy.next = right
        mergeSort(dummy.next, dummy.next + length - 1)
        return dummy.next
