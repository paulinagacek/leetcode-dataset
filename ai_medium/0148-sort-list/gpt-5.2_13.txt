class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        # Count length
        n = 0
        cur = head
        while cur:
            n += 1
            cur = cur.next

        def split(start, size):
            if not start:
                return None
            cur = start
            for _ in range(size - 1):
                if not cur.next:
                    break
                cur = cur.next
            nxt = cur.next
            cur.next = None
            return nxt

        def merge(l1, l2):
            dummy = ListNode(0)
            tail = dummy
            while l1 and l2:
                if l1.val <= l2.val:
                    tail.next = l1
                    l1 = l1.next
                else:
                    tail.next = l2
                    l2 = l2.next
                tail = tail.next
            tail.next = l1 if l1 else l2
            while tail.next:
                tail = tail.next
            return dummy.next, tail

        dummy = ListNode(0, head)
        size = 1
        while size < n:
            prev = dummy
            curr = dummy.next
            while curr:
                left = curr
                right = split(left, size)
                curr = split(right, size)
                merged_head, merged_tail = merge(left, right)
                prev.next = merged_head
                prev = merged_tail
            size <<= 1

        return dummy.next
