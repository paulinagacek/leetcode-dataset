# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # Step 1: Use the fast and slow pointer approach to find the middle of the linked list.
        slow, fast = head, head.next
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        
        # Step 2: Split the linked list into two halves.
        mid_next = slow.next
        slow.next = None
        
        # Step 3: Recursively sort both halves.
        left_sorted = self.sortList(head)
        right_sorted = self.sortList(mid_next)
        
        # Step 4: Merge the sorted halves.
        return self.mergeTwoLists(left_sorted, right_sorted)
    
    def mergeTwoLists(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode()
        cur = dummy
        
        while l1 and l2:
            if l1.val < l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        
        cur.next = l1 or l2
        
        return dummy.next
