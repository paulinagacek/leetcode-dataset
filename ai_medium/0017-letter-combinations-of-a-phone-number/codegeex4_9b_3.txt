from typing import List

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits:
            return []
        phone = ['abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs', 'tuv', 'wxyz']
        ans = ['']
        for digit in digits:
            tmp = []
            for j in ans:
                for i in phone[int(digit) - 2]:
                    tmp.append(j + i)
            ans = tmp
        return ans
