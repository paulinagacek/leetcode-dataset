from typing import List

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits:
            return []
        phone = ['abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs', 'tuv', 'wxyz']
        ans = ['']
        for digit in digits:
            new_ans = []
            for word in ans:
                for letter in phone[int(digit) - 2]:
                    new_ans.append(word + letter)
            ans = new_ans
        return ans
