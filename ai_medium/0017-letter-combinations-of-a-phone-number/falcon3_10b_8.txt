class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        mapping = {'2': 'abc', '3': 'def', '4': 'ghi', '5': 'jkl', '6': 'mno', '7': 'pqrs', '8': 'tuv', '9': 'wxyz'}
        if not digits:
            return []
        res, temp = [], []
        for d in digits:
            temp = [t+c for t in temp for c in mapping[d]] if temp else list(mapping[d])
        return temp
