from typing import List

class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits:
            return []
        phone = ['abc', 'def', 'ghi', 'jkl', 'mno', 'pqrs', 'tuv', 'wxyz']
        self.ans = []
        self.helper(digits, '', 0, phone)
        return self.ans
    
    def helper(self, digits: str, cur_str: str, index: int, phone: List[str]):
        if index == len(digits):
            self.ans.append(cur_str)
            return
        for c in phone[int(digits[index]) - 2]:
            self.helper(digits, cur_str + c, index + 1, phone)
