class Solution:
    def reverseWords(self, s: str) -> str:
        words = []
        start = 0
        end = len(s) - 1
        
        # Skip leading spaces
        while start < len(s) and s[start] == ' ':
            start += 1
        
        # Skip trailing spaces
        while end >= 0 and s[end] == ' ':
            end -= 1
        
        i = start
        while i <= end:
            if s[i] != ' ':
                start = i
                while i <= end and s[i] != ' ':
                    i += 1
                words.append(s[start:i])
            else:
                i += 1
        
        # Reverse the order of the words
        words.reverse()
        
        # Concatenate the words with a single space separator
        return ' '.join(words)
