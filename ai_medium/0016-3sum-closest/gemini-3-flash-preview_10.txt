class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        # Sort the numbers to use the two-pointer technique
        nums.sort()
        n = len(nums)
        # Initialize the closest sum with the sum of the first three elements
        res = nums[0] + nums[1] + nums[2]
        
        for i in range(n - 2):
            # Skip duplicate values for the first integer to avoid redundant calculations
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            
            l, r = i + 1, n - 1
            while l < r:
                current_sum = nums[i] + nums[l] + nums[r]
                
                # If the sum is exactly the target, return it immediately
                if current_sum == target:
                    return current_sum
                
                # Update the closest sum if the current sum is closer to the target
                if abs(current_sum - target) < abs(res - target):
                    res = current_sum
                
                # Adjust the pointers based on the comparison of the current sum and target
                if current_sum < target:
                    l += 1
                else:
                    r -= 1
        
        return res
