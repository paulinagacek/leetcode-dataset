from typing import List

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        diff = float('inf')
        res = 0
        
        for i in range(n - 2):
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            j, k = i + 1, n - 1
            while j < k:
                s = nums[i] + nums[j] + nums[k]
                d = abs(target - s)
                if d < diff:
                    diff = d
                    res = s
                if s < target:
                    j += 1
                elif s > target:
                    k -= 1
                else:
                    return s
        
        return res
