from typing import List

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        closest_sum = float('inf')
        
        for i in range(n - 2):
            if i > 0 and nums[i] == nums[i - 1]:
                continue
            left = i + 1
            right = n - 1
            
            while left < right:
                curr_sum = nums[i] + nums[left] + nums[right]
                closest_sum = min(closest_sum, abs(curr_sum - target), key=lambda x: abs(x))
                
                if curr_sum == target:
                    return curr_sum
                
                elif curr_sum < target:
                    left += 1
                else:
                    right -= 1
        
        return closest_sum + target
