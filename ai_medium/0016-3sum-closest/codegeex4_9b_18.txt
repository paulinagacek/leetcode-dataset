from typing import List

class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        best_diff = float('inf')
        best_sum = 0
        
        for i in range(n - 2):
            j, k = i + 1, n - 1
            while j < k:
                curr_sum = nums[i] + nums[j] + nums[k]
                curr_diff = abs(curr_sum - target)
                
                if curr_diff < best_diff:
                    best_diff = curr_diff
                    best_sum = curr_sum
                
                if curr_sum < target:
                    j += 1
                elif curr_sum > target:
                    k -= 1
                else:
                    return curr_sum
        
        return best_sum
