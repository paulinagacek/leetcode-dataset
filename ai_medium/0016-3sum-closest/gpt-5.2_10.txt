class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        best_sum = nums[0] + nums[1] + nums[2]
        best_diff = abs(best_sum - target)

        for i in range(n - 2):
            l, r = i + 1, n - 1
            while l < r:
                s = nums[i] + nums[l] + nums[r]
                diff = s - target
                ad = abs(diff)

                if ad < best_diff:
                    best_diff = ad
                    best_sum = s
                    if best_diff == 0:
                        return best_sum

                if diff < 0:
                    l += 1
                else:
                    r -= 1

        return best_sum
