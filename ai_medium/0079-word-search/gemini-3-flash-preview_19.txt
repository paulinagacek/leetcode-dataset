class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        rows, cols = len(board), len(board[0])
        
        # Pruning: If word is longer than total cells, it's impossible.
        if len(word) > rows * cols:
            return False
        
        # Pruning: If any character in word is missing or insufficient in board.
        board_counts = Counter(chain(*board))
        word_counts = Counter(word)
        for char in word_counts:
            if word_counts[char] > board_counts[char]:
                return False
        
        # Pruning: Start searching from the end that has the least frequent character.
        if board_counts[word[0]] > board_counts[word[-1]]:
            word = word[::-1]

        def dfs(r, c, i):
            if i == len(word):
                return True
            if r < 0 or r >= rows or c < 0 or c >= cols or board[r][c] != word[i]:
                return False
            
            # Temporarily mark the cell as visited.
            temp, board[r][c] = board[r][c], '#'
            
            # Search horizontally and vertically.
            res = (dfs(r + 1, c, i + 1) or 
                   dfs(r - 1, c, i + 1) or 
                   dfs(r, c + 1, i + 1) or 
                   dfs(r, c - 1, i + 1))
            
            # Backtrack: restore the cell value.
            board[r][c] = temp
            return res

        # Try starting the DFS from every cell in the grid.
        for r in range(rows):
            for c in range(cols):
                if dfs(r, c, 0):
                    return True
        return False
