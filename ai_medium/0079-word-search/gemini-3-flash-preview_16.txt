class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        R = len(board)
        C = len(board[0])
        
        # Initial check for total number of characters
        if len(word) > R * C:
            return False
        
        # Search pruning: verify character frequency requirements
        board_counts = Counter(char for row in board for char in row)
        word_counts = Counter(word)
        for char, count in word_counts.items():
            if board_counts[char] < count:
                return False
        
        # Search pruning: start searching from the end with fewer prefix occurrences
        if board_counts[word[0]] > board_counts[word[-1]]:
            word = word[::-1]

        def dfs(r, c, i):
            if i == len(word):
                return True
            if not (0 <= r < R and 0 <= c < C) or board[r][c] != word[i]:
                return False
            
            # Temporarily mark the cell as visited
            temp = board[r][c]
            board[r][c] = '#'
            
            # Explore neighbors: down, up, right, left
            if (dfs(r + 1, c, i + 1) or 
                dfs(r - 1, c, i + 1) or 
                dfs(r, c + 1, i + 1) or 
                dfs(r, c - 1, i + 1)):
                return True
            
            # Backtrack: restore the cell's original value
            board[r][c] = temp
            return False
        
        # Iterate through every cell to start the DFS search
        for r in range(R):
            for c in range(C):
                if board[r][c] == word[0]:
                    if dfs(r, c, 0):
                        return True
        return False
