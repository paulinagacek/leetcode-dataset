class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        m, n = len(board), len(board[0])

        board_count = Counter(ch for row in board for ch in row)
        word_count = Counter(word)
        for ch, cnt in word_count.items():
            if board_count[ch] < cnt:
                return False

        if board_count[word[0]] > board_count[word[-1]]:
            word = word[::-1]

        L = len(word)
        DIRS = ((1, 0), (-1, 0), (0, 1), (0, -1))

        def dfs(r: int, c: int, i: int) -> bool:
            if board[r][c] != word[i]:
                return False
            if i == L - 1:
                return True
            ch = board[r][c]
            board[r][c] = "\0"
            ni = i + 1
            for dr, dc in DIRS:
                nr, nc = r + dr, c + dc
                if 0 <= nr < m and 0 <= nc < n and board[nr][nc] == word[ni]:
                    if dfs(nr, nc, ni):
                        board[r][c] = ch
                        return True
            board[r][c] = ch
            return False

        first = word[0]
        for r in range(m):
            for c in range(n):
                if board[r][c] == first and dfs(r, c, 0):
                    return True
        return False
