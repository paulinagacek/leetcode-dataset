class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        rows = len(board)
        cols = len(board[0])
        
        # Pruning: If the word is longer than the number of cells in the grid, it's impossible.
        if len(word) > rows * cols:
            return False
        
        # Pruning: Count the occurrences of each character in the grid and the word.
        # If any character required by the word is missing or insufficient in the grid, return false.
        board_counts = Counter(char for row in board for char in row)
        word_counts = Counter(word)
        for char, count in word_counts.items():
            if board_counts[char] < count:
                return False
        
        # Pruning: Start the search from the end of the word that has fewer occurrences in the grid.
        # This significantly reduces the branching factor early in the recursion.
        if board_counts[word[0]] > board_counts[word[-1]]:
            word = word[::-1]
            
        def backtrack(r, c, i):
            # Base Case: If the entire word has been matched, return True.
            if i == len(word):
                return True
            
            # Boundary check and character match check.
            if r < 0 or r >= rows or c < 0 or c >= cols or board[r][c] != word[i]:
                return False
            
            # Mark the current cell as visited by temporarily replacing the character.
            temp = board[r][c]
            board[r][c] = "#"
            
            # Explore all four possible directions (down, up, right, left).
            if (backtrack(r + 1, c, i + 1) or
                backtrack(r - 1, c, i + 1) or
                backtrack(r, c + 1, i + 1) or
                backtrack(r, c - 1, i + 1)):
                return True
            
            # Backtrack: Restore the original character to the cell for other path explorations.
            board[r][c] = temp
            return False
        
        # Start the DFS from every cell that matches the first letter of the target word.
        for r in range(rows):
            for c in range(cols):
                if board[r][c] == word[0]:
                    if backtrack(r, c, 0):
                        return True
                        
        return False
