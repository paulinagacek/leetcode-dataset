class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        results = []
        
        def dfs(node, currentPath, currentSum):
            if not node:
                return
            
            # Add the current node's value to the path
            currentPath.append(node.val)
            currentSum += node.val
            
            # Check if it's a leaf node and the sum matches the target
            if not node.left and not node.right:
                if currentSum == targetSum:
                    results.append(list(currentPath))
            else:
                # Continue DFS on child nodes
                dfs(node.left, currentPath, currentSum)
                dfs(node.right, currentPath, currentSum)
            
            # Backtrack: remove the current node's value from the path
            currentPath.pop()
            
        dfs(root, [], 0)
        return results
