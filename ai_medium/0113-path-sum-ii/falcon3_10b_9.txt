class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        result = []

        def helper(node, path, curr_sum):
            if not node:
                return
            curr_sum += node.val
            path.append(node.val)

            if not node.left and not node.right and curr_sum == targetSum:
                result.append(list(path))
            else:
                helper(node.left, path, curr_sum)
                helper(node.right, path, curr_sum)

            path.pop()

        helper(root, [], 0)
        return result
