# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        def dfs(node, remaining_sum, path):
            if not node:
                return []
            if not node.left and not node.right and node.val == remaining_sum:
                return [path + [node.val]]
            paths = []
            if node.left:
                paths += dfs(node.left, remaining_sum - node.val, path + [node.val])
            if node.right:
                paths += dfs(node.right, remaining_sum - node.val, path + [node.val])
            return paths
        
        return dfs(root, targetSum, [])
