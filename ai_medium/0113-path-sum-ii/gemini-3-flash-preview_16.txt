class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        result = []
        
        def backtrack(node, current_path, current_sum):
            if not node:
                return
            
            # Add current node to the path
            current_path.append(node.val)
            current_sum += node.val
            
            # Check if current node is a leaf
            if not node.left and not node.right:
                # Check if the sum of values in the path equals targetSum
                if current_sum == targetSum:
                    result.append(list(current_path))
            else:
                # Recurse on children if not a leaf
                if node.left:
                    backtrack(node.left, current_path, current_sum)
                if node.right:
                    backtrack(node.right, current_path, current_sum)
            
            # Backtrack: remove current node from the path before returning to parent
            current_path.pop()
            
        backtrack(root, [], 0)
        return result
