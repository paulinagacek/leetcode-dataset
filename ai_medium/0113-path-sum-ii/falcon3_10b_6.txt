# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        result = []

        def helper(node, curr_sum, path):
            if not node:
                return
            # Include current node value in the path and update current sum
            curr_sum += node.val
            path.append(node.val)

            # If it's a leaf node and path sum equals targetSum, add path to results
            if not node.left and not node.right and curr_sum == targetSum:
                result.append(list(path))

            # Recurse on left and right subtrees, backtrack path after recursion
            helper(node.left, curr_sum, path)
            helper(node.right, curr_sum, path)
            path.pop()

        helper(root, 0, [])
        return result
