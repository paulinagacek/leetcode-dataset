class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        result = []
        
        def dfs(node, current_target, path):
            if not node:
                return
            
            # Add the current node's value to the current path
            path.append(node.val)
            
            # Check if the current node is a leaf and the sum matches the target
            if not node.left and not node.right:
                if node.val == current_target:
                    result.append(list(path))
            else:
                # Recurse into children, reducing the target sum by the current node's value
                dfs(node.left, current_target - node.val, path)
                dfs(node.right, current_target - node.val, path)
            
            # Backtrack: remove the current node's value before returning to the parent
            path.pop()
            
        dfs(root, targetSum, [])
        return result
