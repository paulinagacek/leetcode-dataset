class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        if not root:
            return []
        self.paths = []
        self.dfs(root, targetSum, [])
        return self.paths
    
    def dfs(self, node: TreeNode, remainingSum: int, currentPath: List[int]):
        if not node.left and not node.right and remainingSum == node.val:
            currentPath.append(node.val)
            self.paths.append(currentPath[:])
            currentPath.pop()
            return
        
        currentPath.append(node.val)
        if node.left:
            self.dfs(node.left, remainingSum - node.val, currentPath)
        if node.right:
            self.dfs(node.right, remainingSum - node.val, currentPath)
        currentPath.pop()
