# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        def helper(node, remaining_sum, current_path):
            if not node:
                return []
            if not node.left and not node.right:
                if node.val == remaining_sum:
                    return [current_path + [node.val]]
                else:
                    return []
            left_paths = helper(node.left, remaining_sum - node.val, current_path + [node.val])
            right_paths = helper(node.right, remaining_sum - node.val, current_path + [node.val])
            return left_paths + right_paths
        
        return helper(root, targetSum, [])
