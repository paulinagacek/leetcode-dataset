# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        def helper(node, remaining_sum, path):
            if not node:
                return []
            if not node.left and not node.right and node.val == remaining_sum:
                return [path + [node.val]]
            left_paths = helper(node.left, remaining_sum - node.val, path + [node.val])
            right_paths = helper(node.right, remaining_sum - node.val, path + [node.val])
            return left_paths + right_paths
        return helper(root, targetSum, [])
