class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        results = []
        
        def dfs(node, current_sum, current_path):
            if not node:
                return
            
            current_path.append(node.val)
            
            # Check if current node is a leaf
            if not node.left and not node.right:
                if current_sum == node.val:
                    results.append(list(current_path))
            else:
                if node.left:
                    dfs(node.left, current_sum - node.val, current_path)
                if node.right:
                    dfs(node.right, current_sum - node.val, current_path)
            
            # Backtrack: remove the current node value before returning to the parent
            current_path.pop()
            
        dfs(root, targetSum, [])
        return results
