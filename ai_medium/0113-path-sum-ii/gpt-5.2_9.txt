class Solution:
    def pathSum(self, root: Optional["TreeNode"], targetSum: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(node: Optional["TreeNode"], rem: int) -> None:
            if not node:
                return
            path.append(node.val)
            rem -= node.val
            if not node.left and not node.right:
                if rem == 0:
                    res.append(path.copy())
            else:
                dfs(node.left, rem)
                dfs(node.right, rem)
            path.pop()

        dfs(root, targetSum)
        return res
