class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        results = []
        
        def dfs(node, current_target, path):
            if not node:
                return
            
            # Add current node's value to the current path
            path.append(node.val)
            
            # Check if the node is a leaf (no children)
            if not node.left and not node.right:
                # If the current target equals the node's value, we found a path
                if current_target == node.val:
                    results.append(list(path))
            else:
                # Recurse on left and right children with updated targetSum
                if node.left:
                    dfs(node.left, current_target - node.val, path)
                if node.right:
                    dfs(node.right, current_target - node.val, path)
            
            # Backtrack: remove the current node's value before returning up the tree
            path.pop()
            
        dfs(root, targetSum, [])
        return results
