class Solution:
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> List[List[int]]:
        results = []
        
        def dfs(node, remainingSum, currentPath):
            if not node:
                return
            
            currentPath.append(node.val)
            
            # Check if it's a leaf node and the sum matches
            if not node.left and not node.right:
                if node.val == remainingSum:
                    results.append(list(currentPath))
            else:
                # Continue exploration
                dfs(node.left, remainingSum - node.val, currentPath)
                dfs(node.right, remainingSum - node.val, currentPath)
            
            # Backtrack to explore other paths
            currentPath.pop()
            
        dfs(root, targetSum, [])
        return results
