from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(start, path):
            if len(path) == k:
                if sum(path) == n:
                    result.append(path[:])
                return
            for i in range(start, 10):
                path.append(i + 1)
                backtrack(i + 1, path)
                path.pop()

        result = []
        backtrack(0, [])
        return result
