class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(start: int, k_left: int, target: int) -> None:
            if k_left == 0:
                if target == 0:
                    res.append(path.copy())
                return

            if start > 9:
                return

            if target < 0:
                return

            if start + k_left - 1 > 9:
                return

            min_sum = k_left * (2 * start + (k_left - 1)) // 2
            if min_sum > target:
                return

            max_sum = k_left * (2 * 9 - (k_left - 1)) // 2
            if max_sum < target:
                return

            for num in range(start, 10):
                if num > target:
                    break
                path.append(num)
                dfs(num + 1, k_left - 1, target - num)
                path.pop()

        dfs(1, k, n)
        return res
