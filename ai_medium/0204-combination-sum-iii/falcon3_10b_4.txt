class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        
        def backtrack(start=1, path=[]):
            if len(path) == k:
                if sum(path) == n:
                    res.append(path[:])
                return
            for i in range(start, 10):
                if i > n - sum(path):  # pruning the search space
                    break
                path.append(i)
                backtrack(i + 1, path)
                path.pop()
        
        backtrack()
        return res
