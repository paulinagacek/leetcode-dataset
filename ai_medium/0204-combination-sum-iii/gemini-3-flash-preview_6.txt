class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        
        def backtrack(start: int, target: int, path: List[int]):
            # If the combination has reached the required length k
            if len(path) == k:
                # If the sum of path matches n, add to result
                if target == 0:
                    res.append(list(path))
                return
            
            # Try numbers from 'start' to 9
            for i in range(start, 10):
                # Optimization: if current number i exceeds the remaining target sum,
                # since we iterate in increasing order, no subsequent numbers will work.
                if i > target:
                    break
                
                # Choose the number
                path.append(i)
                # Recurse with updated start index (to ensure uniqueness) and target
                backtrack(i + 1, target - i, path)
                # Backtrack: remove the number before trying the next
                path.pop()
        
        backtrack(1, n, [])
        return res
