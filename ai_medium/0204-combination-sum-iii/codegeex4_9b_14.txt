from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(start, target, k, path):
            if k == 0 and target == 0:
                result.append(path)
                return
            for i in range(start, 10):
                if i > target:
                    break
                backtrack(i + 1, target - i, k - 1, path + [i])
        
        result = []
        backtrack(1, n, k, [])
        return result
