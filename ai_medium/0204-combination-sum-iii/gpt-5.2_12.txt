class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(start: int, k_left: int, target: int) -> None:
            if k_left == 0:
                if target == 0:
                    res.append(path.copy())
                return

            if target <= 0:
                return

            # Prune with minimal and maximal possible sums
            if start + k_left - 1 > 9:
                return

            min_sum = (2 * start + (k_left - 1)) * k_left // 2
            max_sum = (19 - (k_left - 1)) * k_left // 2  # 9 + 8 + ... for k_left terms
            if target < min_sum or target > max_sum:
                return

            for x in range(start, 10):
                if x > target:
                    break
                path.append(x)
                dfs(x + 1, k_left - 1, target - x)
                path.pop()

        dfs(1, k, n)
        return res
