class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        nums = [i for i in range(1, 10)]

        def backtrack(index, current, sum):
            if sum > n:
                return
            if len(current) == k:
                if sum == n:
                    res.append(current[:])
                return

            for i in range(index, len(nums)):
                current.append(nums[i])
                backtrack(i + 1, current, sum + nums[i])
                current.pop()

        backtrack(0, [], 0)
        return res
