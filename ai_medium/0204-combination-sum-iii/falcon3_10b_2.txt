from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(start, comb, target):
            if len(comb) == k:
                if target == 0:
                    result.append(comb[:])
                return

            for i in range(start, 10):
                if i > target:
                    break
                comb.append(i)
                backtrack(i + 1, comb, target - i)
                comb.pop()
        
        result = []
        backtrack(1, [], n)
        return result
