from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(start, k, n, combo):
            if k == 0:
                if n == 0:
                    result.append(combo[:])
                return
            
            for i in range(start, 10):
                if i > n:
                    break
                combo.append(i)
                backtrack(i + 1, k - 1, n - i, combo)
                combo.pop()
        
        result = []
        backtrack(1, k, n, [])
        return result
