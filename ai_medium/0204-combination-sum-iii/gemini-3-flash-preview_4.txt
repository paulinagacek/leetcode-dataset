class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        results = []
        
        def backtrack(start: int, target: int, path: List[int]):
            # If the current combination has reached the target length k
            if len(path) == k:
                # If the current combination's sum equals n
                if target == 0:
                    results.append(list(path))
                return
            
            # Iterate through numbers from 'start' up to 9
            for i in range(start, 10):
                # Pruning: if the current number i is greater than the remaining sum target,
                # any further numbers will also exceed the target.
                if i > target:
                    break
                
                # Choose the number i
                path.append(i)
                # Recurse to find the next number in the combination
                # We start the next search from i + 1 to ensure each number is used only once
                backtrack(i + 1, target - i, path)
                # Backtrack: remove the last number to explore other possibilities
                path.pop()
        
        # Start the backtracking process from number 1 with target n
        backtrack(1, n, [])
        return results
