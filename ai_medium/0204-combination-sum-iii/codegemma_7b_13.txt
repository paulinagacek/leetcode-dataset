class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        def dfs(comb, start, total):
            if len(comb) == k and total == n:
                res.append(comb[:])
                return
            for i in range(start, 10):
                if total + i <= n:
                    comb.append(i)
                    dfs(comb, i + 1, total + i)
                    comb.pop()

        dfs([], 1, 0)
        return res
