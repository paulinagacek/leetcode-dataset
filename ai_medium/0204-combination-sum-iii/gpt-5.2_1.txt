class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(start: int, k_left: int, target: int) -> None:
            if k_left == 0:
                if target == 0:
                    res.append(path[:])
                return

            if start > 9:
                return

            if start + (start + k_left - 1) > 9:
                return

            min_sum = k_left * (2 * start + k_left - 1) // 2
            max_sum = k_left * (19 - k_left) // 2
            if target < min_sum or target > max_sum:
                return

            for x in range(start, 10):
                if x > target:
                    break
                path.append(x)
                dfs(x + 1, k_left - 1, target - x)
                path.pop()

        dfs(1, k, n)
        return res
