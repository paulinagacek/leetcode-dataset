class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def backtrack(start: int, k_left: int, target: int) -> None:
            if k_left == 0:
                if target == 0:
                    res.append(path.copy())
                return

            # Prune by min/max possible sums with k_left numbers from [start..9]
            if start > 9:
                return

            min_sum = k_left * (2 * start + (k_left - 1)) // 2
            max_sum = k_left * (2 * 9 - (k_left - 1)) // 2
            if target < min_sum or target > max_sum:
                return

            for num in range(start, 10):
                if num > target:
                    break
                path.append(num)
                backtrack(num + 1, k_left - 1, target - num)
                path.pop()

        backtrack(1, k, n)
        return res
