from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(start, total, path):
            if len(path) == k:
                if total == n:
                    result.append(path[:])
                return
            
            for i in range(start, 10):
                if i > n - total:
                    break
                path.append(i)
                backtrack(i + 1, total + i, path)
                path.pop()

        result = []
        backtrack(1, 0, [])
        return result
