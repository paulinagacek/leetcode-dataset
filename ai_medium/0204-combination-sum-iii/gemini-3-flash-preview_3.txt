class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        
        def backtrack(start: int, k: int, target: int, path: List[int]):
            if k == 0:
                if target == 0:
                    res.append(list(path))
                return
            
            if target <= 0:
                return

            for i in range(start, 10):
                # If the current number is greater than the target, 
                # no need to check further numbers in the range.
                if i > target:
                    break
                
                path.append(i)
                backtrack(i + 1, k - 1, target - i, path)
                path.pop()
        
        backtrack(1, k, n, [])
        return res
