class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(start: int, k_left: int, target: int) -> None:
            if k_left == 0:
                if target == 0:
                    res.append(path.copy())
                return

            if target <= 0:
                return

            min_sum = (2 * start + (k_left - 1)) * k_left // 2
            max_sum = (19 - k_left) * k_left // 2
            if target < min_sum or target > max_sum:
                return

            for num in range(start, 10):
                if num > target:
                    break
                path.append(num)
                dfs(num + 1, k_left - 1, target - num)
                path.pop()

        dfs(1, k, n)
        return res
