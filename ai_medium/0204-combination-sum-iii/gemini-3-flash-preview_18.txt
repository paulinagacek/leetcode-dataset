class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res = []
        
        def backtrack(start: int, path: List[int], target: int):
            # If the current combination has k numbers
            if len(path) == k:
                # Check if the sum equals the target n
                if target == 0:
                    res.append(list(path))
                return
            
            # If the combination exceeds k numbers or target is already exceeded
            if len(path) > k or target < 0:
                return

            # Try numbers from 'start' to 9
            for i in range(start, 10):
                # Optimization: if the current number is already larger than the remaining target,
                # there's no point in checking further numbers since they are all larger.
                if i > target:
                    break
                
                # Include i in the current combination
                path.append(i)
                # Move to the next number (i + 1) to ensure each number is used at most once
                backtrack(i + 1, path, target - i)
                # Backtrack: remove i from the current combination
                path.pop()

        backtrack(1, [], n)
        return res
