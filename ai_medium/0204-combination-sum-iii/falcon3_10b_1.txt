from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        result = []
        
        def backtrack(start: int, curr_sum: int, combo: List[int]):
            if len(combo) > k:
                return
            if curr_sum > n:
                return
            if len(combo) == k and curr_sum == n:
                result.append(combo[:])
                return
            
            for i in range(start, 10):
                backtrack(i + 1, curr_sum + i, combo + [i])
        
        backtrack(1, 0, [])
        return result
