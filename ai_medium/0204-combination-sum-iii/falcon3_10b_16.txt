from typing import List

class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        def backtrack(remain: int, start: int, path: list):
            if len(path) > k or remain < 0:
                return
            if len(path) == k and remain == 0:
                result.append(path[:])
                return
            for i in range(start, 10):
                backtrack(remain - i, i + 1, path + [i])

        result = []
        backtrack(n, 1, [])
        return result
