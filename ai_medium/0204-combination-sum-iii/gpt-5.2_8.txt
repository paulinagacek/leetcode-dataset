class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []
        path: List[int] = []

        def dfs(start: int, k_left: int, sum_left: int) -> None:
            if k_left == 0:
                if sum_left == 0:
                    res.append(path.copy())
                return

            if start > 9:
                return

            # Prune using min/max possible sums with remaining picks
            if start + k_left - 1 > 9:
                return

            min_sum = (start + (start + k_left - 1)) * k_left // 2
            max_sum = (9 + (9 - k_left + 1)) * k_left // 2
            if sum_left < min_sum or sum_left > max_sum:
                return

            for x in range(start, 10):
                if x > sum_left:
                    break
                path.append(x)
                dfs(x + 1, k_left - 1, sum_left - x)
                path.pop()

        dfs(1, k, n)
        return res
