class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        res: List[List[int]] = []

        def backtrack(start: int, remaining_k: int, remaining_sum: int, path: List[int]) -> None:
            if remaining_k == 0:
                if remaining_sum == 0:
                    res.append(path[:])
                return
            if remaining_sum <= 0:
                return

            for x in range(start, 10):
                if x > remaining_sum:
                    break
                if 10 - x < remaining_k - 1:
                    break
                path.append(x)
                backtrack(x + 1, remaining_k - 1, remaining_sum - x, path)
                path.pop()

        backtrack(1, k, n, [])
        return res
