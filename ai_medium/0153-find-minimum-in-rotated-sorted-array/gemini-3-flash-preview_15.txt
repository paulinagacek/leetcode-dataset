class Solution:
    def findMin(self, nums: List[int]) -> int:
        # Initialize binary search pointers
        left, right = 0, len(nums) - 1
        
        # Binary search for the pivot point (the minimum element)
        while left < right:
            mid = left + (right - left) // 2
            
            # If the middle element is greater than the rightmost element, 
            # the minimum element must be in the right half (excluding mid)
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the middle element is less than or equal to the rightmost element, 
            # the minimum element is either at mid or to the left of mid
            else:
                right = mid
                
        # When left == right, we have found the minimum element
        return nums[left]
