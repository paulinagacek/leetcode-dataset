class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If nums[mid] is greater than nums[right], the minimum
            # must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If nums[mid] is less than or equal to nums[right],
            # the minimum could be mid itself or in the left half.
            else:
                right = mid
                
        # After the loop, left == right, pointing to the minimum element.
        return nums[left]
