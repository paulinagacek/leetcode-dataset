class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = left + (right - left) // 2
            
            # If nums[mid] is greater than the element at right, 
            # the minimum element must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If nums[mid] is less than or equal to the element at right,
            # the minimum element is either at mid or in the left half of the array.
            else:
                right = mid
        
        # When left == right, we have found the minimum element's index.
        return nums[left]
