class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = left + (right - left) // 2
            
            # If the middle element is greater than the rightmost element,
            # it means the minimum element must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the middle element is less than the rightmost element,
            # the minimum element could be the middle element itself or something to its left.
            else:
                right = mid
        
        # When left == right, we've converged on the minimum element.
        return nums[left]
