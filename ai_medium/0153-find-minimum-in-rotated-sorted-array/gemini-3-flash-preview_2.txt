class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If the element at mid is greater than the element at right, 
            # it means the minimum element must be to the right of mid.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the element at mid is less than or equal to the element at right,
            # it means the minimum element is either at mid or to the left of mid.
            else:
                right = mid
        
        # After the loop, left == right, pointing to the minimum element.
        return nums[left]
