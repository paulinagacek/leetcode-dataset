class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If the value at mid is greater than the value at right, 
            # the minimum element must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # Otherwise, the minimum element is either at mid or to its left.
            else:
                right = mid
                
        return nums[left]
