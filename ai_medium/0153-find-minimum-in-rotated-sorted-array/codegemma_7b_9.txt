class Solution:
    def findMin(self, nums: List[int]) -> int:
        lo, hi = 0, len(nums) - 1
        while lo <= hi:
            mid = (lo + hi) // 2
            if mid > 0 and nums[mid] < nums[mid - 1]:
                return nums[mid]
            elif nums[mid] > nums[-1]:
                lo = mid + 1
            else:
                hi = mid - 1
        return nums[0]
