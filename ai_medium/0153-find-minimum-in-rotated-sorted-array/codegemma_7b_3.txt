class Solution:
    def findMin(self, nums: List[int]) -> int:
        l, r = 0, len(nums) - 1
        while l <= r:
            mid = (l + r) // 2
            next_mid = (mid + 1) % len(nums)
            prev_mid = (mid - 1 + len(nums)) % len(nums)
            if nums[mid] < nums[next_mid]:
                r = mid - 1
            elif nums[mid] < nums[prev_mid]:
                l = mid + 1
            else:
                return nums[mid]
