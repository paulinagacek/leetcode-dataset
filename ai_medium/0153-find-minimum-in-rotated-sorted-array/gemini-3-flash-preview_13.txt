class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = left + (right - left) // 2
            
            # If nums[mid] is greater than the last element, the minimum element
            # must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If nums[mid] is less than or equal to the last element, 
            # the minimum element is either nums[mid] itself or in the left half.
            else:
                right = mid
                
        # When left == right, the loop terminates and we've found the minimum.
        return nums[left]
