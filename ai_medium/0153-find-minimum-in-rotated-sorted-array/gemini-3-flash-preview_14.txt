class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If the middle element is greater than the rightmost element,
            # the minimum element must be in the right half of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the middle element is less than or equal to the rightmost element,
            # the minimum element must be the middle element or in the left half.
            else:
                right = mid
        
        # After the loop, left == right, which points to the minimum element.
        return nums[left]
