from typing import List

class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If the right half is sorted, minimum is on the left half or at mid
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the left half is sorted, minimum is on the right half
            else:
                right = mid
                
        return nums[left]
