class Solution:
    def findMin(self, nums: List[int]) -> int:
        left, right = 0, len(nums) - 1
        
        while left < right:
            mid = (left + right) // 2
            
            # If the middle element is greater than the rightmost element, 
            # the minimum element must be in the right part of the array.
            if nums[mid] > nums[right]:
                left = mid + 1
            # If the middle element is less than or equal to the rightmost element,
            # the minimum element could be the middle element itself or in the left part.
            else:
                right = mid
                
        # After the loop, left == right, pointing to the minimum element.
        return nums[left]
