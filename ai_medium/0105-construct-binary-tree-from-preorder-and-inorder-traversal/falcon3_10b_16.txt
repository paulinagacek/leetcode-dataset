class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        def build(pre_start, pre_end, in_start, in_end):
            if pre_start > pre_end:
                return None
            
            root_val = preorder[pre_start]
            index = in_order_map[root_val]
            
            root = TreeNode(root_val)
            root.left = build(pre_start + 1, pre_start + index - in_start, in_start, index - 1)
            root.right = build(pre_start + index - in_start + 1, pre_end, index + 1, in_end)
            
            return root
        
        in_order_map = {val: idx for idx, val in enumerate(inorder)}
        return build(0, len(preorder) - 1, 0, len(inorder) - 1)
