class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        inorder_map = {val: i for i, val in enumerate(inorder)}
        pre_iter = iter(preorder)
        
        def build(left, right):
            if left > right:
                return None
            
            # The first element in preorder traversal is always the root.
            # Using an iterator ensures we pick the next root for each subtree.
            root_val = next(pre_iter)
            root = TreeNode(root_val)
            
            # Root splits the inorder list into left and right subtrees.
            idx = inorder_map[root_val]
            
            # Recursively build the left subtree and then the right subtree.
            # It's important to build the left subtree first because the preorder 
            # sequence follows root -> left -> right.
            root.left = build(left, idx - 1)
            root.right = build(idx + 1, right)
            
            return root
            
        return build(0, len(inorder) - 1)
