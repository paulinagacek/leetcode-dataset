class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        in_map = {val: i for i, val in enumerate(inorder)}
        self.pre_idx = 0
        
        def array_to_tree(left: int, right: int) -> Optional[TreeNode]:
            if left > right:
                return None
            
            root_val = preorder[self.pre_idx]
            root = TreeNode(root_val)
            self.pre_idx += 1
            
            index = in_map[root_val]
            
            # The left subtree must be built before the right subtree 
            # because the preorder sequence processes root, then left, then right.
            root.left = array_to_tree(left, index - 1)
            root.right = array_to_tree(index + 1, right)
            
            return root
        
        return array_to_tree(0, len(inorder) - 1)
