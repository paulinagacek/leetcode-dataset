class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional['TreeNode']:
        if not preorder:
            return None

        idx = {v: i for i, v in enumerate(inorder)}
        n = len(preorder)

        def helper(pre_l: int, pre_r: int, in_l: int, in_r: int) -> Optional['TreeNode']:
            if pre_l > pre_r:
                return None
            root_val = preorder[pre_l]
            root_in = idx[root_val]
            left_size = root_in - in_l

            root = TreeNode(root_val)
            root.left = helper(pre_l + 1, pre_l + left_size, in_l, root_in - 1)
            root.right = helper(pre_l + left_size + 1, pre_r, root_in + 1, in_r)
            return root

        return helper(0, n - 1, 0, n - 1)
