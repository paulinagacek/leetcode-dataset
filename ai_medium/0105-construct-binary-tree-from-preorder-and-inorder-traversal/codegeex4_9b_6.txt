# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        def helper(in_left, in_right):
            nonlocal pre_idx
            # if there is no elements to construct subtrees
            if in_left == in_right:
                return None
            
            # select the current root
            root_val = preorder[pre_idx]
            root = TreeNode(root_val)
            
            # recursively construct the left subtree

                root.left = helper(in_left, in_index)
            
            # recursively construct the right subtree
            pre_idx += 1
            if in_index + 1 < in_right:
                root.right = helper(in_index + 1, in_right)

            return root
        
        # start from the first element in preorder
        pre_idx = 0
        
        # range of inorder indices
        in_start, in_end = 0, len(inorder)
