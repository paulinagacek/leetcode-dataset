class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        indexMap = {val: i for i, val in enumerate(inorder)}
        preorderIdx = 0
        
        def dfs(startIdx, endIdx):
            if startIdx > endIdx:
                return None
            
            val = preorder[preorderIdx]
            inorderIdx = indexMap[val]
            preorderIdx += 1
            
            root = TreeNode(val)
            root.left = dfs(startIdx, inorderIdx - 1)
            root.right = dfs(inorderIdx + 1, endIdx)
            
            return root
        
        return dfs(0, len(inorder) - 1)
