# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        def helper(in_left, in_right):
            nonlocal pre_idx
            if in_left > in_right:
                return None
            
            # Get the current node's value from preorder
            root_val = preorder[pre_idx]
            root = TreeNode(root_val)
            
            # Find the index of this value in inorder array
            idx = index_map[root_val]
            
            # Before incrementing pre_idx, construct the left and right subtree recursively
            root.left = helper(in_left, idx - 1)
            root.right = helper(idx + 1, in_right)
            
            pre_idx += 1
            return root
        
        # Index map to find the index of any value in inorder array 
        index_map = {val:idx for idx, val in enumerate(inorder)}
        pre_idx = 0
        
        return helper(0, len(inorder) - 1)
