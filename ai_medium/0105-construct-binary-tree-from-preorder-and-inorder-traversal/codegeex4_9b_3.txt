# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        if not preorder or not inorder:
            return None
        
        # Create the root node with the first element in preorder list
        root_val = preorder.pop(0)
        root = TreeNode(root_val)
        
        # Find the index of the root value in inorder list to split into left and right subtrees
        inorder_index = inorder.index(root_val)
        
        # Recursively construct the left subtree
        root.left = self.buildTree(preorder, inorder[:inorder_index])
        
        # Recursively construct the right subtree
        root.right = self.buildTree(preorder, inorder[inorder_index + 1:])
        
        return root
