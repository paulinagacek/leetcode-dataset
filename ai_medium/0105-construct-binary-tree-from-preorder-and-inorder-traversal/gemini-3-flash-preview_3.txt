class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        # Create a hash map to quickly find the index of a value in the inorder list
        inorder_idx_map = {val: i for i, val in enumerate(inorder)}
        
        # Use an iterator for preorder traversal to keep track of the root elements
        preorder_iter = iter(preorder)
        
        def helper(left_in, right_in):
            # Base case: if there are no elements to construct the subtree
            if left_in > right_in:
                return None
            
            # The next element in preorder traversal is the root of the current subtree
            root_val = next(preorder_iter)
            root = TreeNode(root_val)
            
            # Find the index of the root value in inorder traversal to split the subtrees
            mid_idx = inorder_idx_map[root_val]
            
            # Recursively build the left and right subtrees
            # Note: left subtree must be built first to match preorder order (root, left, right)
            root.left = helper(left_in, mid_idx - 1)
            root.right = helper(mid_idx + 1, right_in)
            
            return root
        
        return helper(0, len(inorder) - 1)
