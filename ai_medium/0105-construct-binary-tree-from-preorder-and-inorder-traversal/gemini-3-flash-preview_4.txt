class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        # Map each value to its index in inorder traversal for O(1) lookups
        in_map = {val: i for i, val in enumerate(inorder)}
        # Iterator to consume elements from preorder traversal in order
        pre_iter = iter(preorder)
        
        def helper(left, right):
            # Base case: no elements to process in this subtree
            if left > right:
                return None
            
            # The next value in the preorder iterator is the root of the current subtree
            root_val = next(pre_iter)
            root = TreeNode(root_val)
            
            # Find the index of this root in the inorder traversal to split left and right subtrees
            mid = in_map[root_val]
            
            # Recursively build the left and right subtrees.
            # In preorder (Root, Left, Right), the left subtree always comes immediately after the root.
            root.left = helper(left, mid - 1)
            root.right = helper(mid + 1, right)
            
            return root
            
        return helper(0, len(inorder) - 1)
