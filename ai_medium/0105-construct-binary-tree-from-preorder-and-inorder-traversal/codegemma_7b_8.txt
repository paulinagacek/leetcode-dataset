class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        def helper(start_pre, end_pre, start_in, end_in):
            if start_pre > end_pre or start_in > end_in:
                return None
            val = preorder[start_pre]
            index = inorder.index(val)
            left_size = index - start_in
            return TreeNode(val, helper(start_pre + 1, start_pre + left_size, start_in, index - 1), helper(start_pre + left_size + 1, end_pre, index + 1, end_in))
        return helper(0, len(preorder) - 1, 0, len(inorder) - 1)
