"""
# Definition for a Node.
class Node:
    def __init__(self, x: int, next: 'Node' = None, random: 'Node' = None):
        self.val = int(x)
        self.next = next
        self.random = random
"""

class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        
        old_to_new = {}
        cur = head

        # Step 1: Create new nodes and link them together
        while cur:
            new_node = Node(cur.val)
            old_to_new[cur] = new_node
            cur = cur.next
        
        # Step 2: Assign the 'random' pointers in the new list
        cur = head
        while cur:
            if cur.random:
                old_to_new[cur].random = old_to_new[cur.random]
            cur = cur.next
        
        # Step 3: Reconstruct the 'next' pointers in the new list
        new_head = old_to_new[head]
        cur_old, cur_new = head, new_head
        while cur_old.next:
            cur_new.next = old_to_new[cur_old.next]
            cur_old = cur_old.next
            cur_new = cur_new.next
        
        return new_head
