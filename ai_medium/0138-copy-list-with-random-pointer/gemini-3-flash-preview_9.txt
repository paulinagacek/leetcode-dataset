class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        
        # Mapping from original nodes to their corresponding new copies
        # Initialize with None: None to handle the end of the list and null random pointers
        old_to_new = {None: None}
        
        # First pass: Create all new nodes and store the mapping
        curr = head
        while curr:
            old_to_new[curr] = Node(curr.val)
            curr = curr.next
            
        # Second pass: Connect the next and random pointers of the new nodes
        curr = head
        while curr:
            new_node = old_to_new[curr]
            new_node.next = old_to_new[curr.next]
            new_node.random = old_to_new[curr.random]
            curr = curr.next
            
        # Return the head of the copied list
        return old_to_new[head]
