class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        
        # Dictionary to map original nodes to their corresponding copied nodes
        old_to_new = {}
        
        # First pass: create all nodes and store them in the dictionary
        curr = head
        while curr:
            old_to_new[curr] = Node(curr.val)
            curr = curr.next
            
        # Second pass: assign next and random pointers for the copied nodes
        curr = head
        while curr:
            copy_node = old_to_new[curr]
            # Use .get() to handle cases where next or random is None
            copy_node.next = old_to_new.get(curr.next)
            copy_node.random = old_to_new.get(curr.random)
            curr = curr.next
            
        return old_to_new[head]
