class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        
        # Dictionary to map original nodes to their corresponding new nodes
        # Initializing with None: None to handle pointers pointing to null
        old_to_new = {None: None}
        
        # First pass: Create all new nodes and store the mapping
        curr = head
        while curr:
            old_to_new[curr] = Node(curr.val)
            curr = curr.next
            
        # Second pass: Assign next and random pointers for the new nodes
        curr = head
        while curr:
            new_node = old_to_new[curr]
            new_node.next = old_to_new[curr.next]
            new_node.random = old_to_new[curr.random]
            curr = curr.next
            
        return old_to_new[head]
