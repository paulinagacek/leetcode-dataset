class Solution:
    def copyRandomList(self, head: 'Optional[Node]') -> 'Optional[Node]':
        if not head:
            return None
        
        # Use a dictionary to map old nodes to their corresponding new nodes
        old_to_new = {None: None}
        
        # First pass: Create all new nodes and store them in the dictionary
        curr = head
        while curr:
            old_to_new[curr] = Node(curr.val)
            curr = curr.next
            
        # Second pass: Set the next and random pointers for the new nodes
        curr = head
        while curr:
            new_node = old_to_new[curr]
            new_node.next = old_to_new[curr.next]
            new_node.random = old_to_new[curr.random]
            curr = curr.next
            
        return old_to_new[head]
