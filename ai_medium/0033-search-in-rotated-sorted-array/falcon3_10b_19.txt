class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left, right = 0, len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                return mid
            # Find the rotated part where nums[mid] < nums[0]
            if nums[mid] < nums[0]:
                if nums[mid] < target and nums[0] <= target:
                    left = mid + 1
                else:
                    right = mid - 1
            elif nums[0] <= target:
                right = mid - 1
            else:
                left = mid + 1
        return -1
