class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        # If either number is "0", the product is "0"
        if num1 == "0" or num2 == "0":
            return "0"
        
        # Initialize the result array with zeros.
        # The product of two numbers with lengths m and n will have at most m + n digits.
        m, n = len(num1), len(num2)
        res = [0] * (m + n)
        
        # Iterate from right to left through each digit of num1 and num2
        for i in range(m - 1, -1, -1):
            n1 = ord(num1[i]) - ord('0')
            for j in range(n - 1, -1, -1):
                n2 = ord(num2[j]) - ord('0')
                
                # Product of current digits
                mul = n1 * n2
                
                # Identify positions in the result array
                # The multiplication results for indices i and j contribute to indices i+j and i+j+1
                p1, p2 = i + j, i + j + 1
                
                # The current multiplication is added to the value already at res[p2]
                total = mul + res[p2]
                
                # Update res[p2] with the current digit and add the carry to res[p1]
                res[p2] = total % 10
                res[p1] += total // 10
                
        # Convert the result digits to characters and join them to form a string
        # Skip leading zeros that may exist in the result array
        ans = "".join(map(str, res))
        return ans.lstrip('0')
