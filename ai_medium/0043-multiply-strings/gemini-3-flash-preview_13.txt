class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        # If either number is "0", the product is "0"
        if num1 == "0" or num2 == "0":
            return "0"
        
        m, n = len(num1), len(num2)
        # The maximum possible number of digits for the product is m + n
        res = [0] * (m + n)
        
        # Multiply each digit of num1 by each digit of num2
        # Start from the end of both strings to emulate manual multiplication
        for i in range(m - 1, -1, -1):
            d1 = ord(num1[i]) - ord('0')
            for j in range(n - 1, -1, -1):
                d2 = ord(num2[j]) - ord('0')
                
                # The position in the res array corresponding to d1 * d2
                # is determined by the positions of d1 and d2 in their respective strings.
                p1, p2 = i + j, i + j + 1
                
                # Calculate the product of current digits and add it to existing values
                mul = d1 * d2
                total = mul + res[p2]
                
                # res[p2] stores the current digit, and its carry is added to res[p1]
                res[p2] = total % 10
                res[p1] += total // 10
        
        # Construct the final string result
        res_str = "".join(map(str, res))
        
        # The res array might have a leading zero if the product is smaller than m + n digits
        return res_str.lstrip('0')
