class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        if num1 == "0" or num2 == "0":
            return "0"
            
        n, m = len(num1), len(num2)
        res = [0] * (n + m)
        
        for i in range(n - 1, -1, -1):
            digit1 = ord(num1[i]) - ord('0')
            for j in range(m - 1, -1, -1):
                digit2 = ord(num2[j]) - ord('0')
                
                # The product of digits at i and j contributes to positions i+j and i+j+1
                mul = digit1 * digit2
                p1, p2 = i + j, i + j + 1
                
                total_sum = mul + res[p2]
                res[p2] = total_sum % 10
                res[p1] += total_sum // 10
        
        # Convert list of integers to string, skipping the leading zero if it exists
        start_index = 0
        while start_index < len(res) and res[start_index] == 0:
            start_index += 1
            
        return "".join(map(str, res[start_index:]))
