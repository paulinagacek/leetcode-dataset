class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        if num1 == "0" or num2 == "0":
            return "0"

        m, n = len(num1), len(num2)
        res = [0] * (m + n)

        for i in range(m - 1, -1, -1):
            a = ord(num1[i]) - 48
            carry = 0
            for j in range(n - 1, -1, -1):
                b = ord(num2[j]) - 48
                total = res[i + j + 1] + a * b + carry
                res[i + j + 1] = total % 10
                carry = total // 10
            res[i] += carry

        k = 0
        while k < len(res) and res[k] == 0:
            k += 1
        return "".join(str(d) for d in res[k:]) if k < len(res) else "0"
