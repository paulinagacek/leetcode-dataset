class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        # If either number is 0, the product is 0.
        if num1 == "0" or num2 == "0":
            return "0"
        
        m, n = len(num1), len(num2)
        # The maximum possible number of digits in the product is m + n.
        res = [0] * (m + n)
        
        # Multiply each digit from num1 with each digit from num2.
        for i in range(m - 1, -1, -1):
            d1 = ord(num1[i]) - ord('0')
            for j in range(n - 1, -1, -1):
                d2 = ord(num2[j]) - ord('0')
                
                # The position in the result for the product of num1[i] and num2[j] 
                # corresponds to the index i + j + 1.
                mul = d1 * d2
                total = mul + res[i + j + 1]
                
                # Update the current position with the remainder.
                res[i + j + 1] = total % 10
                # Add the carry to the next position.
                res[i + j] += total // 10
                
        # Skip the leading zeros in the result array.
        idx = 0
        while idx < len(res) and res[idx] == 0:
            idx += 1
            
        # Join the relevant digits from the result array to form the final string.
        return "".join(map(str, res[idx:]))
