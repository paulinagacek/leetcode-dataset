class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        # If either number is 0, the product is 0
        if num1 == "0" or num2 == "0":
            return "0"
        
        # Lengths of the input strings
        m, n = len(num1), len(num2)
        # The result of multiplying a number with m digits and a number with n digits
        # will have at most m + n digits.
        res = [0] * (m + n)
        
        # Iterate through num1 and num2 from right to left
        for i in range(m - 1, -1, -1):
            for j in range(n - 1, -1, -1):
                # Calculate product of digits and their position in the result array
                # Using ord() to comply with the requirement of not converting to int directly
                mul = (ord(num1[i]) - ord('0')) * (ord(num2[j]) - ord('0'))
                p1, p2 = i + j, i + j + 1
                
                # Add the product to the current value at position p2
                total = mul + res[p2]
                
                # Update p2 with the digit part and carry over to p1
                res[p2] = total % 10
                res[p1] += total // 10
                
        # Find the index of the first non-zero digit
        start = 0
        while start < len(res) and res[start] == 0:
            start += 1
            
        # Convert the integer array into a string
        return "".join(map(str, res[start:]))
