class Solution:
    def multiply(self, num1: str, num2: str) -> str:
        if num1 == "0" or num2 == "0":
            return "0"

        n, m = len(num1), len(num2)
        res = [0] * (n + m)

        for i in range(n - 1, -1, -1):
            a = ord(num1[i]) - 48
            for j in range(m - 1, -1, -1):
                b = ord(num2[j]) - 48
                res[i + j + 1] += a * b

        carry = 0
        for k in range(n + m - 1, -1, -1):
            total = res[k] + carry
            res[k] = total % 10
            carry = total // 10

        idx = 0
        while idx < len(res) - 1 and res[idx] == 0:
            idx += 1

        return "".join(chr(d + 48) for d in res[idx:])
