class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s:
            return ""
        n = len(s)
        res = s[0]
        for i in range(n):
            for j in range(i + 1, n):
                if s[i] == s[j]:
                    temp = s[i:j+1]
                    if temp == temp[::-1] and len(temp) > len(res):
                        res = temp
        return res
