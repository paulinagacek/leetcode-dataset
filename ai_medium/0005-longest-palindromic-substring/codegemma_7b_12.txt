class Solution:
    def longestPalindrome(self, s: str) -> str:
        dp = [[False] * len(s) for _ in range(len(s))]
        longest = ""
        for i in range(len(s)):
            for j in range(i + 1):
                if s[i] == s[j] and (j + 1 > i - 1 or dp[j + 1][i - 1]):
                    dp[j][i] = True
                    if len(s[j:i + 1]) > len(longest):
                        longest = s[j:i + 1]
        return longest
