class Solution:
    def longestPalindrome(self, s: str) -> str:
        def helper(s, i, j):
            while i >= 0 and j < len(s) and s[i] == s[j]:
                i -= 1
                j += 1
            return s[i+1:j]

        res = ""
        for i in range(len(s)):
            temp = helper(s, i, i)
            if len(temp) > len(res):
                res = temp
            temp = helper(s, i, i+1)
            if len(temp) > len(res):
                res = temp
        return res
