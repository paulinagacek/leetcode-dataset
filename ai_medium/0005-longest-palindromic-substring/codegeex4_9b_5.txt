class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s:
            return ""
        max_len = 0
        start = 0
        for i in range(len(s)):
            for j in range(i, len(s)):
                substr = s[i:j+1]
                if substr == substr[::-1] and len(substr) > max_len:
                    max_len = len(substr)
                    start = i
        return s[start:start+max_len]
