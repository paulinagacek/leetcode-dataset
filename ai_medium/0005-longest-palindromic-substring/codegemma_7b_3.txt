class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        dp = [[False] * n for _ in range(n)]
        longest = ""
        for i in range(n):
            dp[i][i] = True
            longest = s[i]
        for gap in range(1, n):
            for i in range(n - gap):
                if s[i] == s[i + gap] and dp[i + 1][i + gap - 1]:
                    dp[i][i + gap] = True
                    longest = s[i: i + gap + 1]
        return longest
