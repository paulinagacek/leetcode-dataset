class Solution:
    def longestPalindrome(self, s: str) -> str:
        def expand(s, i, j):
            while i >= 0 and j < len(s) and s[i] == s[j]:
                i -= 1
                j += 1
            return s[i+1:j]

        longest = ""

        for i in range(len(s)):
            odd = expand(s, i, i)
            even = expand(s, i, i+1)

            longest = max(longest, odd, even, key=len)

        return longest
