class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n <= 1:
            return s

        start = 0
        best_len = 1

        def expand(l: int, r: int):
            nonlocal start, best_len
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            l += 1
            r -= 1
            length = r - l + 1
            if length > best_len:
                best_len = length
                start = l

        for i in range(n):
            expand(i, i)       # odd length
            expand(i, i + 1)   # even length

        return s[start:start + best_len]
