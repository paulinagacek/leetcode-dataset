class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s: return ""

        start, maxLen = 0, 1

        for i in range(1, len(s)):
            if i - maxLen > 0 and s[i] == s[i - maxLen]:
                start = i - maxLen
                maxLen += 1
            elif i - maxLen > 0 and s[i] == s[i + maxLen]:
                start = i - maxLen
                maxLen = 1 if maxLen == 1 else maxLen - 1
            elif maxLen == 1:
                maxLen = 2

        return s[start:start+maxLen]
