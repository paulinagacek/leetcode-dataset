class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n <= 1:
            return s

        start = 0
        max_len = 1

        def expand(l: int, r: int):
            nonlocal start, max_len
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            l += 1
            r -= 1
            cur_len = r - l + 1
            if cur_len > max_len:
                max_len = cur_len
                start = l

        for i in range(n):
            expand(i, i)       # odd
            expand(i, i + 1)   # even

        return s[start:start + max_len]
