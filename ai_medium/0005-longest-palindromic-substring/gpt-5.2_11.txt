class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n < 2:
            return s

        start = 0
        max_len = 1

        def expand(l: int, r: int) -> None:
            nonlocal start, max_len
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            length = r - l - 1
            if length > max_len:
                max_len = length
                start = l + 1

        for i in range(n):
            expand(i, i)       # odd length
            expand(i, i + 1)   # even length

        return s[start:start + max_len]
