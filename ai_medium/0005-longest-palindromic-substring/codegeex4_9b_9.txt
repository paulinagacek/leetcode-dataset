class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s:
            return ""
        max_length = 0
        start = 0
        for i in range(len(s)):
            left = right = i
            while left >= 0 and right < len(s) and s[left] == s[right]:
                left -= 1
                right += 1
            if right - left + 1 > max_length:
                max_length = right - left + 1
                start = left
        return s[start:start + max_length]
