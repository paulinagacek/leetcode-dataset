class Solution:
    def longestPalindrome(self, s: str) -> str:
        if len(s) == 0:
            return ''
            
        longest = s[0]
        for i in range(len(s)):
            for j in range(i + 1, len(s) + 1):
                subStr = s[i:j]
                if subStr == subStr[::-1] and len(subStr) > len(longest):
                    longest = subStr
                    
        return longest
