class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s:
            return ""
        
        n = len(s)
        start, max_len = 0, 1
        
        for i in range(n):
            # Check for odd length palindromes (centered at i)
            # Initial boundaries for searching beyond the center character
            l, r = i - 1, i + 1
            while l >= 0 and r < n and s[l] == s[r]:
                if r - l + 1 > max_len:
                    max_len = r - l + 1
                    start = l
                l -= 1
                r += 1
            
            # Check for even length palindromes (centered between i and i + 1)
            l, r = i, i + 1
            while l >= 0 and r < n and s[l] == s[r]:
                if r - l + 1 > max_len:
                    max_len = r - l + 1
                    start = l
                l -= 1
                r += 1
                
        return s[start : start + max_len]
