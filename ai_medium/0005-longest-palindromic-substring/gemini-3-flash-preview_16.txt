class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n <= 1:
            return s
        
        start, max_len = 0, 1
        
        for i in range(n):
            # Check for odd length palindromes centered at i
            l, r = i, i
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            # Length of palindrome found is (r - 1) - (l + 1) + 1 = r - l - 1
            if r - l - 1 > max_len:
                start, max_len = l + 1, r - l - 1
            
            # Check for even length palindromes centered at i and i + 1
            l, r = i, i + 1
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            if r - l - 1 > max_len:
                start, max_len = l + 1, r - l - 1
        
        return s[start:start + max_len]
