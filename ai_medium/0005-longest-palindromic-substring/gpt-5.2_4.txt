class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n < 2:
            return s

        start = end = 0

        def expand(l: int, r: int) -> None:
            nonlocal start, end
            while l >= 0 and r < n and s[l] == s[r]:
                l -= 1
                r += 1
            l += 1
            r -= 1
            if r - l > end - start:
                start, end = l, r

        for i in range(n):
            expand(i, i)       # odd length
            expand(i, i + 1)   # even length

        return s[start:end + 1]
