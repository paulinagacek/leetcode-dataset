class Solution:
    def longestPalindrome(self, s: str) -> str:
        if not s or len(s) < 2:
            return s

        longest = ""
        for i in range(len(s)):
            # Odd-length palindrome centered at i
            left, right = i, i
            while left >= 0 and right < len(s) and s[left] == s[right]:
                longest = s[left:right+1] if len(s[left:right+1]) > len(longest) else longest
                left -= 1
                right += 1

            # Even-length palindrome centered at i and i+1
            left, right = i, i+1
            while left >= 0 and right < len(s) and s[left] == s[right]:
                longest = s[left:right+1] if len(s[left:right+1]) > len(longest) else longest
                left -= 1
                right += 1

        return longest
