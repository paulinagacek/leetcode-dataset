class Solution:
    def longestPalindrome(self, s: str) -> str:
        n = len(s)
        if n <= 1:
            return s
        
        start, max_len = 0, 1
        
        for i in range(n):
            # Case 1: Odd-length palindromes (expand from a single character center)
            # The character s[i] itself is the center.
            l, r = i - 1, i + 1
            while l >= 0 and r < n and s[l] == s[r]:
                current_len = r - l + 1
                if current_len > max_len:
                    start, max_len = l, current_len
                l -= 1
                r += 1
            
            # Case 2: Even-length palindromes (expand from a two-character center)
            # The gap between s[i] and s[i+1] is the center.
            l, r = i, i + 1
            while l >= 0 and r < n and s[l] == s[r]:
                current_len = r - l + 1
                if current_len > max_len:
                    start, max_len = l, current_len
                l -= 1
                r += 1
                
        return s[start:start + max_len]
