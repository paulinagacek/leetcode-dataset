from typing import List

class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequence_dict = {}
        result = []
        for i in range(len(s) - 9):
            sequence = s[i:i+10]
            if sequence not in sequence_dict:
                sequence_dict[sequence] = 1
            else:
                if sequence_dict[sequence] == 1:
                    result.append(sequence)
                sequence_dict[sequence] += 1
        return result
