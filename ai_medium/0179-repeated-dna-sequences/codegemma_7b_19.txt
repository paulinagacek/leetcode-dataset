from collections import Counter

class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        
        if len(s) < 10:
            return []
        
        sequences = [s[i:i+10] for i in range(len(s) - 9)]
        counts = Counter(sequences)
        return [seq for seq, count in counts.items() if count > 1]
