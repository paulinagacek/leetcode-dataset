from typing import List

class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequence_length = 10
        dna_dict = {}
        result = []
        
        for i in range(len(s) - sequence_length + 1):
            seq = s[i:i+sequence_length]
            if seq in dna_dict:
                dna_dict[seq] += 1
            else:
                dna_dict[seq] = 1
        
        for key, value in dna_dict.items():
            if value > 1:
                result.append(key)
        
        return result
