from typing import List
class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        count = {}
        result = []
        for i in range(len(s) - 9):
            kmer = s[i:i+10]
            if kmer not in count:
                count[kmer] = 1
            else:
                if count[kmer] == 1:
                    result.append(kmer)
                count[kmer] += 1
        return result
