class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequences = set()
        repeated = set()
        count = {}
        
        for i in range(len(s) - 9):
            seq = s[i:i+10]
            count[seq] = count.get(seq, 0) + 1
            if count[seq] == 2:
                repeated.add(seq)
        
        for seq in count:
            if count[seq] >= 2:
                sequences.add(seq)
        
        return list(sequences)
