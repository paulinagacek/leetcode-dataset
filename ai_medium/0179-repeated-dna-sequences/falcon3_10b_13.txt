class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequences = set()
        repeated = set()
        for i in range(len(s) - 9):
            seq = s[i:i+10]
            if s.count(seq) > 1:
                repeated.add(seq)
            sequences.add(seq)
        return list(repeated)
