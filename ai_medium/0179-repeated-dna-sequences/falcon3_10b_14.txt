class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        from collections import defaultdict
        d = defaultdict(int)
        result = []
        for i in range(len(s) - 9):
            seq = s[i:i+10]
            d[seq] += 1
        for seq, count in d.items():
            if count > 1:
                result.append(seq)
        return result
