from typing import List
from collections import defaultdict

class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        if len(s) <= 10:
            return []
        count = defaultdict(int)
        for i in range(len(s)-9):
            count[s[i:i+10]] += 1
        return [k for k,v in count.items() if v > 1]
