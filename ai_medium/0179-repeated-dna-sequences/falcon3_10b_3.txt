class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequences = set()
        seen = set()
        for i in range(len(s) - 9):
            seq = s[i:i+10]
            if seq in seen and seq not in sequences:
                sequences.add(seq)
            seen.add(seq)
        return list(sequences)
