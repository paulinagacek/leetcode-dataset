class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequences = set()
        counts = collections.Counter()
        for i in range(len(s) - 9):
            substring = s[i:i+10]
            counts[substring] += 1
            if counts[substring] == 2:
                sequences.add(substring)
        return list(sequences)
