class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        # Dictionary/Set to track seen substrings
        seen = set()
        # Set to track substrings that have appeared more than once
        repeated = set()
        
        # We only need to check substrings of length 10
        # The loop runs up to len(s) - 9 to ensure a 10-character window
        for i in range(len(s) - 9):
            substring = s[i : i + 10]
            if substring in seen:
                repeated.add(substring)
            else:
                seen.add(substring)
        
        # Convert the set of repeated sequences to a list
        return list(repeated)
