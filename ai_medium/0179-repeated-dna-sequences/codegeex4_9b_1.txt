from typing import List
class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        sequence_dict = {}
        result = []
        for i in range(len(s)-9):
            sequence = s[i:i+10]
            if sequence in sequence_dict:
                if sequence not in result:
                    result.append(sequence)
            else:
                sequence_dict[sequence] = 1
        return result
