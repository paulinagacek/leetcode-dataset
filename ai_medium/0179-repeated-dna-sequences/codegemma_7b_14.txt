class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        seen = set()
        result = set()
        
        for i in range(len(s) - 9):
            subsequence = s[i:i+10]
            if subsequence in seen:
                result.add(subsequence)
            seen.add(subsequence)
        
        return list(result)
