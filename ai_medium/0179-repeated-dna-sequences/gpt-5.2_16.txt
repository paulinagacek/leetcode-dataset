class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        n = len(s)
        if n < 10:
            return []

        enc = {'A': 0, 'C': 1, 'G': 2, 'T': 3}
        mask = (1 << 20) - 1

        x = 0
        for i in range(10):
            x = (x << 2) | enc[s[i]]

        seen = {x}
        repeated = set()
        for i in range(10, n):
            x = ((x << 2) & mask) | enc[s[i]]
            if x in seen:
                repeated.add(s[i - 9:i + 1])
            else:
                seen.add(x)

        return list(repeated)
