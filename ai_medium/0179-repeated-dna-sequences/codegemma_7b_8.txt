class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        dct = defaultdict(int)
        ans = set()
        for i in range(len(s) - 9):
            window = s[i:i + 10]
            dct[window] += 1
            if dct[window] == 2:
                ans.add(window)
        return list(ans)
