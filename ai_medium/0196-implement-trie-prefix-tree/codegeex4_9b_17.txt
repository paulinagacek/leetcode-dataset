class Trie:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.trie = {}


    def insert(self, word: str) -> None:
        """
        Inserts a word into the trie.
        """
        tree = self.trie
        for char in word:
            if char not in tree:
                tree[char] = {}
            tree = tree[char]
        tree['#'] = '#'  # Mark the end of a word


    def search(self, word: str) -> bool:
        """
        Returns if the word is in the trie.
        """
        tree = self.trie
        for char in word:
            if char not in tree:
                return False
            tree = tree[char]
        return '#' in tree


    def startsWith(self, prefix: str) -> bool:
        """
        Returns if there is any word in the trie that starts with the given prefix.
        """
        tree = self.trie
        for char in prefix:
            if char not in tree:
                return False
            tree = tree[char]
        return True
