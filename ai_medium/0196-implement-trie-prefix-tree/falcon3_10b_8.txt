class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for char in word:
            node = node.setdefault(char, {})
        node['#'] = True

    def search(self, word: str) -> bool:
        node = self.get_node(word)
        return bool(node) and '#' in node

    def startsWith(self, prefix: str) -> bool:
        return bool(self.get_node(prefix))

    def get_node(self, word: str) -> dict:
        node = self.root
        for char in word:
            if char not in node:
                return None
            node = node[char]
        return node
