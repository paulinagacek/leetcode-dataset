class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for ch in word:
            if ch not in node:
                node[ch] = {}
            node = node[ch]
        node['#'] = True

    def search(self, word: str) -> bool:
        node = self.find_node(word)
        return node and '#' in node

    def startsWith(self, prefix: str) -> bool:
        return bool(self.find_node(prefix))

    def find_node(self, word: str) -> dict:
        node = self.root
        for ch in word:
            if ch not in node:
                return None
            node = node[ch]
        return node
