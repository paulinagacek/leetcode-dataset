class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['#'] = True

    def search(self, word: str) -> bool:
        node = self.find_node(word)
        return node and '#' in node

    def startsWith(self, prefix: str) -> bool:
        return self.find_node(prefix) is not None

    def find_node(self, word: str) -> dict:
        node = self.root
        for char in word:
            if char not in node:
                return None
            node = node[char]
        return node
