class Trie:

    def __init__(self):
        self.trie = {}

    def insert(self, word: str) -> None:
        current_dict = self.trie
        for char in word:
            if char not in current_dict:
                current_dict[char] = {}
            current_dict = current_dict[char]
        current_dict['#'] = '#'

    def search(self, word: str) -> bool:
        current_dict = self.trie
        for char in word:
            if char not in current_dict:
                return False
            current_dict = current_dict[char]
        return '#' in current_dict

    def startsWith(self, prefix: str) -> bool:
        current_dict = self.trie
        for char in prefix:
            if char not in current_dict:
                return False
            current_dict = current_dict[char]
        return True
