class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['$'] = True  # Mark the end of a word

    def search(self, word: str) -> bool:
        node = self._search_node(word)
        return node and '$' in node

    def startsWith(self, prefix: str) -> bool:
        return self._search_node(prefix) is not None

    def _search_node(self, word: str) -> dict:
        node = self.root
        for char in word:
            if char not in node:
                return None
            node = node[char]
        return node
