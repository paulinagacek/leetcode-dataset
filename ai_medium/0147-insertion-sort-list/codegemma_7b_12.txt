class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        dummy = ListNode(0, head)

        curr = head.next
        while curr:
            prev = dummy
            target = curr.val

            while prev.next and prev.next.val < target:
                prev = prev.next

            if prev.next == curr:
                curr = curr.next
                continue

            temp = curr.next
            curr.next = prev.next
            prev.next = curr
            curr = temp

        return dummy.next
