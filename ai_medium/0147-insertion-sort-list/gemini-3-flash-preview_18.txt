class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # Create a dummy node to simplify inserting at the beginning of the sorted list
        dummy = ListNode(0)
        dummy.next = head
        
        # last_sorted is the last node of the sorted portion
        last_sorted = head
        # curr is the node to be inserted into the sorted portion
        curr = head.next
        
        while curr:
            if last_sorted.val <= curr.val:
                # If the current node is in the correct place relative to the sorted portion
                last_sorted = curr
            else:
                # Find the correct position to insert curr
                prev = dummy
                while prev.next.val < curr.val:
                    prev = prev.next
                
                # Remove curr from its current position
                last_sorted.next = curr.next
                # Insert curr after prev
                curr.next = prev.next
                prev.next = curr
            
            # Move to the next node in the original list
            curr = last_sorted.next
            
        return dummy.next
