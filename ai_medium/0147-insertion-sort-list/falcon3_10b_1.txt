# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        dummy = ListNode(0)
        dummy.next = head
        prev = dummy
        current = head
        
        while current and current.next:
            next_node = current.next
            
            if current.val <= next_node.val:
                current = next_node
                continue
                
            prev.next = next_node
            
            temp = dummy
            while temp.next.val < next_node.val:
                temp = temp.next
            
            next_node.next = temp.next
            temp.next = next_node
            current = prev
            
        return dummy.next
