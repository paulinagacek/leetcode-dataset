class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        dummy = ListNode(0)
        dummy.next = head
        last_sorted = head
        curr = head.next
        
        while curr:
            if curr.val >= last_sorted.val:
                last_sorted = curr
            else:
                prev = dummy
                while prev.next.val < curr.val:
                    prev = prev.next
                
                last_sorted.next = curr.next
                curr.next = prev.next
                prev.next = curr
            
            curr = last_sorted.next
            
        return dummy.next
