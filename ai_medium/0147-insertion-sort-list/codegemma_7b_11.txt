class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0)
        cur = head

        while cur:
            pre = dummy
            nxt = cur.next

            while pre.next and pre.next.val < cur.val:
                pre = pre.next

            cur.next = pre.next
            pre.next = cur
            cur = nxt

        return dummy.next
