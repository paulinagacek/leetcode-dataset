class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0)
        pre, cur = dummy, head
        
        while cur:
            tmp = cur.next
            pre = dummy
            
            while pre.next and pre.next.val < cur.val:
                pre = pre.next
            
            cur.next = pre.next
            pre.next = cur
            cur = tmp
            
        return dummy.next
