class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # dummy node helps in handling insertions at the head
        dummy = ListNode(0)
        dummy.next = head
        
        # last_sorted is the last node of the sorted part
        last_sorted = head
        # curr is the node to be inserted into the sorted part
        curr = head.next
        
        while curr:
            if last_sorted.val <= curr.val:
                # curr is already in the correct place
                last_sorted = last_sorted.next
            else:
                # Find the position where curr should be inserted
                prev = dummy
                while prev.next.val < curr.val:
                    prev = prev.next
                
                # Remove curr from the current position and insert it after prev
                last_sorted.next = curr.next
                curr.next = prev.next
                prev.next = curr
            
            # Move to the next node to be sorted
            curr = last_sorted.next
            
        return dummy.next
