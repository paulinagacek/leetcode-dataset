class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        dummy = ListNode(0)
        dummy.next = head
        
        sorted_head = head
        current = head.next
        
        while current:
            prev = sorted_head
            next_node = current.next
            
            while prev and prev.val < current.val:
                prev = prev.next
            
            current.next = prev.next
            
            if prev != sorted_head:
                prev.next = current
            else:
                sorted_head = current
            
            current = next_node
        
        return dummy.next
