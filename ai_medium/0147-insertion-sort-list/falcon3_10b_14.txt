# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head
        
        # Separate out the first node and set it as sorted_head
        sorted_head = ListNode(0)
        sorted_head.next = head
        sorted_tail = head
        
        while sorted_tail.next:
            current = sorted_tail.next
            
            if current.val >= sorted_tail.val:
                sorted_tail = sorted_tail.next
                continue
            
            # Find the correct position in sorted part
            prev = sorted_head
            while prev.next.val < current.val:
                prev = prev.next
            
            # Insert current node after prev
            sorted_tail.next = current.next
            current.next = prev.next
            prev.next = current
        
        return sorted_head.next
