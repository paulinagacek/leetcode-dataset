class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            # Calculate the width between the pointers
            width = right - left
            # The height of the water is limited by the shorter line
            h = min(height[left], height[right])
            # Update max_area if current area is larger
            current_area = width * h
            if current_area > max_area:
                max_area = current_area
            
            # Move the pointer pointing to the shorter line to potentially find a taller one
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
                
        return max_area
