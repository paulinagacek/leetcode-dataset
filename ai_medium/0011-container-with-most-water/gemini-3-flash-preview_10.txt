class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            # The height of the water is determined by the shorter of the two lines
            h = min(height[left], height[right])
            # The width is the distance between the two lines
            w = right - left
            # Calculate current area and update max_area if current is larger
            current_area = h * w
            if current_area > max_area:
                max_area = current_area
            
            # Move the pointer pointing to the shorter line to potentially find a taller container
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
                
        return max_area
