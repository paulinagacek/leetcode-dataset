class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            # The width of the container is the distance between pointers
            width = right - left
            # The height of the water is determined by the shorter of the two lines
            if height[left] < height[right]:
                current_height = height[left]
                current_area = width * current_height
                left += 1
            else:
                current_height = height[right]
                current_area = width * current_height
                right -= 1
            
            # Update the maximum area found so far
            if current_area > max_area:
                max_area = current_area
                
        return max_area
