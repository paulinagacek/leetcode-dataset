class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            h_left = height[left]
            h_right = height[right]
            
            # Calculate current area using the shorter line
            if h_left < h_right:
                current_area = h_left * (right - left)
                left += 1
            else:
                current_area = h_right * (right - left)
                right -= 1
            
            # Update max_area if current_area is larger
            if current_area > max_area:
                max_area = current_area
                
        return max_area
