class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_area = 0
        
        while left < right:
            # Calculate current area based on the shorter of the two lines
            h_left = height[left]
            h_right = height[right]
            current_height = h_left if h_left < h_right else h_right
            current_area = current_height * (right - left)
            
            # Update max_area if current area is larger
            if current_area > max_area:
                max_area = current_area
            
            # Move the pointer pointing to the shorter line to potentially find a taller one
            if h_left < h_right:
                left += 1
            else:
                right -= 1
                
        return max_area
