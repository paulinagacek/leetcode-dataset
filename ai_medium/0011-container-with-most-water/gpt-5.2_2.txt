class Solution:
    def maxArea(self, height: List[int]) -> int:
        l, r = 0, len(height) - 1
        best = 0
        while l < r:
            hl, hr = height[l], height[r]
            best = max(best, (r - l) * (hl if hl < hr else hr))
            if hl < hr:
                l += 1
            else:
                r -= 1
        return best
