class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def validate(node, low, high):
            # An empty node is valid
            if not node:
                return True
            
            # The current node's value must be strictly between low and high
            if not (low < node.val < high):
                return False
            
            # Recursively check the subtrees with updated constraints
            # Left child must be < current node val
            # Right child must be > current node val
            return (validate(node.left, low, node.val) and 
                    validate(node.right, node.val, high))
        
        # Initial call with the range of negative infinity to positive infinity
        return validate(root, float('-inf'), float('inf'))
