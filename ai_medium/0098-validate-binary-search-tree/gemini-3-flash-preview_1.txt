class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def validate(node, low, high):
            # An empty node is a valid BST
            if not node:
                return True
            
            # The current node's value must be between low and high limits
            if not (low < node.val < high):
                return False
            
            # Recursively check the subtrees with updated constraints
            # For the left child, the maximum allowed value is the current node's value
            # For the right child, the minimum allowed value is the current node's value
            return validate(node.left, low, node.val) and validate(node.right, node.val, high)
        
        return validate(root, float('-inf'), float('inf'))
