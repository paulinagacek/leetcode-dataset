# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def valid(node, lessThan, largerThan):
            if not node:
                return True
            if not (lessThan < node.val < largerThan):
                return False
            return valid(node.left, lessThan, node.val) and valid(node.right, node.val, largerThan)
        
        return valid(root, float('-inf'), float('inf'))
