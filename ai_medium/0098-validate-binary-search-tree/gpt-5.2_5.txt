class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        stack = [(root, float("-inf"), float("inf"))]
        while stack:
            node, lo, hi = stack.pop()
            if not node:
                continue
            if not (lo < node.val < hi):
                return False
            stack.append((node.right, node.val, hi))
            stack.append((node.left, lo, node.val))
        return True
