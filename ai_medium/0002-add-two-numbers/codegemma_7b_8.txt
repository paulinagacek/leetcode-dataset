class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode()
        p, q, curr = l1, l2, dummy
        carry = 0

        while p or q:
            sum = carry
            if p:
                sum += p.val
                p = p.next
            if q:
                sum += q.val
                q = q.next
            carry = sum // 10
            curr.next = ListNode(sum % 10)
            curr = curr.next

        if carry:
            curr.next = ListNode(carry)

        return dummy.next
