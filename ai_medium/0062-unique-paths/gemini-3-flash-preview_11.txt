class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # The problem can be solved using combinations.
        # To reach the bottom-right corner from the top-left corner in an m x n grid,
        # the robot must make exactly (m - 1) moves down and (n - 1) moves right.
        # The total number of moves is (m - 1) + (n - 1) = m + n - 2.
        # The number of unique paths is the number of ways to choose (m - 1) down moves 
        # out of the total (m + n - 2) moves.
        return math.comb(m + n - 2, m - 1)
