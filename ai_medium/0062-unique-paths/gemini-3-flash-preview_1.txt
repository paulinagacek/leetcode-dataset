class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # dp[j] will store the number of ways to reach a cell in the current row at column j.
        # Initialize the first row with 1s since there is only one way to reach each cell.
        dp = [1] * n
        
        # Iterate through each row starting from the second row.
        for i in range(1, m):
            # For each cell in the row (starting from the second column), 
            # the number of ways to reach it is the sum of ways to reach 
            # the cell above it (dp[j]) and the cell to its left (dp[j-1]).
            for j in range(1, n):
                dp[j] += dp[j - 1]
        
        # The result is stored in the last element of the dp array.
        return dp[-1]
