class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Create a 2D array to store the number of ways to reach each cell
        dp = [[1] * n for _ in range(m)]
        
        # Iterate over the grid starting from the second row and column
        for i in range(1, m):
            for j in range(1, n):
                # The number of ways to reach a cell is the sum of the ways to reach
                # the cell directly above it and the cell directly to the left of it
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        
        # The bottom-right corner will have the total number of unique paths
        return dp[m - 1][n - 1]
