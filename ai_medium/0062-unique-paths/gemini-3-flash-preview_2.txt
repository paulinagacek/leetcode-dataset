class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # dp array stores the number of unique paths to reach each cell in the current row
        dp = [1] * n
        
        # Iterate through each row starting from the second one (index 1)
        for i in range(1, m):
            # Iterate through each column starting from the second one (index 1)
            for j in range(1, n):
                # The number of paths to current cell dp[j] is the sum of:
                # 1. Paths from the cell above (currently stored in dp[j])
                # 2. Paths from the cell to the left (already updated dp[j-1])
                dp[j] += dp[j - 1]
                
        # Return the last element which represents the bottom-right corner
        return dp[n - 1]
