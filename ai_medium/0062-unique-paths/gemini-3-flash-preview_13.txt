class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # The number of unique paths is a combination problem.
        # To reach the bottom-right corner (m-1, n-1) from the top-left (0, 0):
        # - The robot must move Down exactly (m - 1) times.
        # - The robot must move Right exactly (n - 1) times.
        # The total number of steps is (m - 1) + (n - 1) = m + n - 2.
        # We need to choose the positions for the (m - 1) Down moves in the total moves.
        # This is equivalent to calculating "m + n - 2 choose m - 1".
        return math.comb(m + n - 2, m - 1)
