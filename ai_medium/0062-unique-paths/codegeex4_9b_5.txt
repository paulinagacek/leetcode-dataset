class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Create a 2D array to store the number of unique paths to each cell
        dp = [[0] * n for _ in range(m)]
        
        # Initialize the first row and the first column with 1s
        for i in range(m):
            dp[i][0] = 1
        for j in range(n):
            dp[0][j] = 1
        
        # Calculate the number of unique paths to each cell
        for i in range(1, m):
            for j in range(1, n):
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        
        # Return the number of unique paths to the bottom-right corner
        return dp[m - 1][n - 1]
