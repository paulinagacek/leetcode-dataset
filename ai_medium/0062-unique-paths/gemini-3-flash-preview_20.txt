class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Use a 1D array to store the number of ways to reach each cell in the current row
        # Initialize with 1s because there is only one way to reach any cell in the first row (moving Right)
        row = [1] * n
        
        # Iterate through the grid row by row, starting from the second row (index 1)
        for i in range(1, m):
            # Iterate through each column, starting from the second column (index 1)
            for j in range(1, n):
                # The number of ways to reach cell (i, j) is the sum of ways from the cell
                # above (current row[j]) and the cell to the left (newly updated row[j-1])
                row[j] += row[j - 1]
                
        # The last element in the array will contain the total paths to the bottom-right corner
        return row[n - 1]
