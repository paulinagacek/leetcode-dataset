class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Create a 2D array to store the number of ways to reach each cell
        paths = [[1] * n for _ in range(m)]
        
        # Iterate over the cells starting from the second row and column
        for i in range(1, m):
            for j in range(1, n):
                # The number of ways to reach a cell is the sum of the ways to reach the cell above and the cell to the left
                paths[i][j] = paths[i - 1][j] + paths[i][j - 1]
        
        # The bottom-right corner will have the total number of unique paths
        return paths[m - 1][n - 1]
