class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Create a 2D array to store the number of paths to each cell
        path_count = [[1] * n for _ in range(m)]
        
        # Iterate over each cell starting from (1,1)
        for i in range(1, m):
            for j in range(1, n):
                # The number of paths to a cell is the sum of the paths to the cell above and the cell to the left
                path_count[i][j] = path_count[i - 1][j] + path_count[i][j - 1]
        
        # The bottom-right corner contains the total number of unique paths
        return path_count[m - 1][n - 1]
