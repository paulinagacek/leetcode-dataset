class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Initialize a 2D array to store the number of paths to each cell
        dp = [[1] * n for _ in range(m)]
        
        # Iterate over each cell starting from (1,1)
        for i in range(1, m):
            for j in range(1, n):
                # The number of paths to a cell is the sum of the paths leading into it
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1]
        
        # Return the number of paths to the bottom-right corner
        return dp[m - 1][n - 1]
