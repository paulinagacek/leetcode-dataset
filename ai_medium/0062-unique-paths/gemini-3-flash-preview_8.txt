class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # The number of unique paths is the number of ways to choose (m - 1) down moves 
        # out of a total of (m - 1) + (n - 1) moves.
        # This is a combination problem: (m + n - 2) choose (m - 1).
        return math.comb(m + n - 2, m - 1)
