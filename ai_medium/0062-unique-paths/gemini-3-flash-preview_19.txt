class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # Use a 1D DP array to save space. dp[j] will store the number of 
        # unique paths to reach the current cell in column j.
        dp = [1] * n
        
        # Iterate through each row starting from the second row (index 1).
        for i in range(1, m):
            # For each cell, the number of paths is the sum of paths from 
            # the cell above (currently in dp[j]) and the cell to the left (dp[j-1]).
            for j in range(1, n):
                dp[j] += dp[j - 1]
                
        # The result is stored in the last element of the dp array.
        return dp[n - 1]
