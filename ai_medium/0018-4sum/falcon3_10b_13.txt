class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        def k_sum(nums, start, target, length, res):
            if length < 2 or start >= len(nums):
                return
            if length == 2:
                left, right = start, len(nums) - 1
                while left < right:
                    s = nums[left] + nums[right]
                    if s == target:
                        res.append([nums[left], nums[right]])
                        left += 1
                        while left < right and nums[left] == nums[left - 1]:
                            left += 1
                    elif s < target:
                        left += 1
                    else:
                        right -= 1
            else:
                for i in range(start, len(nums) - length + 1):
                    if i > start and nums[i] == nums[i - 1]:
                        continue
                    k_sum(nums, i + 1, target - nums[i], length - 1, res)

        nums.sort()
        res = []
        k_sum(nums, 0, target, 4, res)
        return [list(map(int, quad)) for quad in set(tuple(sorted(quad)) for quad in res)]
