class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        # Kadane's Algorithm
        # Time complexity: O(n)
        # Space complexity: O(1)
        
        # Initialize max_sum and current_sum with the first element
        max_sum = current_sum = nums[0]
        
        # Iterate through the array starting from the second element
        for i in range(1, len(nums)):
            # If current_sum becomes negative, it's better to start a new subarray 
            # at the current element. Otherwise, we extend the existing subarray.
            current_sum = max(nums[i], current_sum + nums[i])
            
            # Update the global maximum sum found so far
            if current_sum > max_sum:
                max_sum = current_sum
                
        return max_sum
