class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        # Initialize current_sum and max_sum with the first element of the array
        current_sum = max_sum = nums[0]
        
        # Iterate through the array starting from the second element
        for i in range(1, len(nums)):
            # If the current_sum becomes negative, discard it and start with the current element
            # Otherwise, add the current element to the current_sum
            current_sum = max(nums[i], current_sum + nums[i])
            
            # Update the global maximum sum found so far
            max_sum = max(max_sum, current_sum)
            
        return max_sum
