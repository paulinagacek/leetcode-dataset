class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        current_sum = nums[0]
        global_max = nums[0]
        for i in range(1, len(nums)):
            current_sum = max(current_sum + nums[i], nums[i])
            if current_sum > global_max:
                global_max = current_sum
        return global_max
