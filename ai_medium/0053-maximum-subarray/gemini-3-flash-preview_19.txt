class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        # Initialize max_sum as the first element to handle cases with only negative numbers
        max_sum = nums[0]
        current_sum = 0
        
        for n in nums:
            # Add the current element to the running sum
            current_sum += n
            
            # Update the maximum sum found so far
            if current_sum > max_sum:
                max_sum = current_sum
            
            # If the current sum becomes negative, it's better to start a new subarray
            # from the next element, so we reset current_sum to 0.
            if current_sum < 0:
                current_sum = 0
                
        return max_sum
