class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        # Initialize max_so_far and current_max with the first element of the array
        max_so_far = current_max = nums[0]
        
        # Iterate through the array starting from the second element
        for i in range(1, len(nums)):
            # Decide whether to extend the existing subarray or start a new one from the current element
            current_max = max(nums[i], current_max + nums[i])
            # Update the global maximum encountered so far
            max_so_far = max(max_so_far, current_max)
            
        return max_so_far
