class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        # Kadane's Algorithm: O(n) time complexity, O(1) space complexity
        max_current = max_global = nums[0]
        
        for i in range(1, len(nums)):
            # Determine whether to start a new subarray at the current element 
            # or extend the existing subarray ending at the previous element.
            max_current = max(nums[i], max_current + nums[i])
            
            # Update the global maximum if the current subarray sum is greater.
            if max_current > max_global:
                max_global = max_current
                
        return max_global
