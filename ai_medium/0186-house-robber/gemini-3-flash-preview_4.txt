class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        prev2 = 0
        prev1 = 0
        
        for num in nums:
            # At each house, the maximum amount we can have is either:
            # 1. The maximum amount from the previous house (skipping current house)
            # 2. The maximum amount from two houses ago + the amount in the current house
            current = max(prev1, prev2 + num)
            prev2 = prev1
            prev1 = current
            
        return prev1
