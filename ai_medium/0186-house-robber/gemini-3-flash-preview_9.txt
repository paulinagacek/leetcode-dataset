class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        # prev2 represents the max money robbed up to two houses ago
        # prev1 represents the max money robbed up to the previous house
        prev2, prev1 = 0, 0
        
        for num in nums:
            # For each house, we choose the maximum of:
            # 1. Robbing the current house + max money from 2 houses ago
            # 2. Skipping the current house and taking the max money from the previous house
            current = max(prev1, prev2 + num)
            prev2 = prev1
            prev1 = current
            
        return prev1
