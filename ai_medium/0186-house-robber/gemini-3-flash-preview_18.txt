class Solution:
    def rob(self, nums: List[int]) -> int:
        prev_max = 0
        curr_max = 0
        
        for num in nums:
            # At each house, we decide whether to rob it or skip it.
            # If we rob the current house, we add its value to the maximum amount
            # robbed up to the house before the previous one (prev_max).
            # If we skip it, the maximum amount remains the same as the house before (curr_max).
            temp = curr_max
            curr_max = max(prev_max + num, curr_max)
            prev_max = temp
            
        return curr_max
