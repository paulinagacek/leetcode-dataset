class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        prev_max = 0
        curr_max = 0
        
        for num in nums:
            # At each house, we decide to either:
            # 1. Take the current house and the max from two houses ago
            # 2. Skip the current house and keep the max from one house ago
            temp = curr_max
            curr_max = max(curr_max, prev_max + num)
            prev_max = temp
            
        return curr_max
