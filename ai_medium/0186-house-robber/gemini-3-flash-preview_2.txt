class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        prev_max = 0
        curr_max = 0
        
        for num in nums:
            # For each house, we decide whether to rob it or not.
            # If we rob it, we add its value to the max amount from two houses ago.
            # If we don't, the max amount remains the same as it was for the previous house.
            temp = curr_max
            curr_max = max(prev_max + num, curr_max)
            prev_max = temp
            
        return curr_max
