class Solution:
    def rob(self, nums: List[int]) -> int:
        prev_max = 0
        curr_max = 0
        
        for num in nums:
            # The maximum at the current house is either:
            # 1. The maximum from the previous house (skipping this house)
            # 2. The maximum from two houses ago + money in the current house
            temp = curr_max
            curr_max = max(prev_max + num, curr_max)
            prev_max = temp
            
        return curr_max
