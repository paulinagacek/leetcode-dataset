class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        
        prev_two, prev_one = nums[0], max(nums[0], nums[1])
        
        for i in range(2, len(nums)):
            current = max(prev_one, prev_two + nums[i])
            prev_two, prev_one = prev_one, current
        
        return prev_one
