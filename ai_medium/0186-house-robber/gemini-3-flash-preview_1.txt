class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        prev2 = 0
        prev1 = 0
        
        for num in nums:
            # For each house, decide whether to rob it or skip it.
            # If we rob it, we add its value to the maximum amount from two houses ago.
            # If we skip it, we take the maximum amount from the previous house.
            current = max(prev1, num + prev2)
            prev2 = prev1
            prev1 = current
            
        return prev1
