class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        prev2 = 0 # Represents the maximum value 2 houses back
        prev1 = 0 # Represents the maximum value 1 house back
        
        for num in nums:
            # At each house, we can either:
            # 1. Skip it: the max value remains the same as prev1
            # 2. Rob it: the max value is the current house amount plus prev2
            current = max(prev1, prev2 + num)
            prev2 = prev1
            prev1 = current
            
        return prev1
