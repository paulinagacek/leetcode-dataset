class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or k == 0:
            return head
        
        fast, slow = head, head
        n = 1
        
        while fast.next:
            fast = fast.next
            n += 1
        
        k %= n
        
        for _ in range(n - k):
            slow = slow.next
        
        fast.next = head
        head = slow.next
        slow.next = None
        
        return head
