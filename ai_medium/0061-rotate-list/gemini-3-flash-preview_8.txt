class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Step 1: Calculate the length of the list and find the tail node
        n = 1
        tail = head
        while tail.next:
            tail = tail.next
            n += 1
            
        # Step 2: Calculate the effective rotation
        k = k % n
        if k == 0:
            return head
            
        # Step 3: Connect the tail to the head to form a cycle
        tail.next = head
        
        # Step 4: Find the new tail (n - k positions from head)
        # The new tail will be at position (n - k - 1) from the start
        steps_to_new_tail = n - k
        new_tail = head
        for _ in range(steps_to_new_tail - 1):
            new_tail = new_tail.next
            
        # Step 5: The new head is the next node, break the cycle
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
