# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the linked list
        length = 1
        cur = head
        while cur.next:
            cur = cur.next
            length += 1
        
        # Create a circular linked list by connecting the last node to the head
        cur.next = head
        
        # Find the new tail, which is (length - k % length) steps from the current tail
        steps_to_new_tail = length - k % length
        new_tail = head
        for _ in range(steps_to_new_tail - 1):
            new_tail = new_tail.next
        
        # The new head is next to the new tail
        new_head = new_tail.next
        
        # Break the circular connection and return the new head
        new_tail.next = None
        
        return new_head
