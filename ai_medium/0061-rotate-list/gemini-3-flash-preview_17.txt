class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and locate the current tail
        length = 1
        curr = head
        while curr.next:
            curr = curr.next
            length += 1
        
        # Compute the effective number of rotations
        k = k % length
        if k == 0:
            return head
        
        # Connect the tail to the head to make it a circular linked list
        curr.next = head
        
        # Find the new tail: (length - k - 1) nodes from the start
        # Find the new head: (length - k) nodes from the start
        steps_to_new_tail = length - k - 1
        new_tail = head
        for _ in range(steps_to_new_tail):
            new_tail = new_tail.next
            
        # Set the new head and break the circular connection
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
