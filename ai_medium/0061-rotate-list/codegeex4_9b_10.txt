# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Step 1: Close the linked list into a circle
        old_tail = head
        n = 1
        while old_tail.next:
            old_tail = old_tail.next
            n += 1
        old_tail.next = head
        
        # Step 2: Find new tail position, which is (n - k % n - 1)th node starting from the head
        new_tail = head
        for _ in range(n - k % n - 1):
            new_tail = new_tail.next
        
        # Step 3: Set new head and break the circle
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
