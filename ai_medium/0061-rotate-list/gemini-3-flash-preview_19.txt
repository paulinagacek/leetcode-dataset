class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and find the tail node
        n = 1
        curr = head
        while curr.next:
            curr = curr.next
            n += 1
        
        # Determine the actual number of rotations needed
        k = k % n
        if k == 0:
            return head
        
        # Connect the tail to the head to make it a circular list
        curr.next = head
        
        # Find the new tail: (n - k - 1) steps from the current head
        new_tail_steps = n - k - 1
        new_tail = head
        for _ in range(new_tail_steps):
            new_tail = new_tail.next
        
        # The new head is the node following the new tail
        new_head = new_tail.next
        # Break the circular link
        new_tail.next = None
        
        return new_head
