class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and find the tail
        n = 1
        old_tail = head
        while old_tail.next:
            old_tail = old_tail.next
            n += 1
            
        # If k is a multiple of n, no rotation is needed
        k = k % n
        if k == 0:
            return head
            
        # Connect the tail to the head to form a circular list
        old_tail.next = head
        
        # Find the new tail: (n - k - 1) steps from the current head
        new_tail = head
        for _ in range(n - k - 1):
            new_tail = new_tail.next
            
        # The new head is the node following the new tail
        new_head = new_tail.next
        # Break the circle
        new_tail.next = None
        
        return new_head
