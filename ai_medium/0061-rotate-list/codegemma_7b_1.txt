class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or k == 0:
            return head

        # Calculate the length of the linked list
        length = 1
        temp = head
        while temp.next:
            length += 1
            temp = temp.next

        # Calculate the actual rotation step
        k = k % length

        # Close the linked list by connecting the tail node to the head node
        temp.next = head

        # Find the new tail node
        new_tail = head
        for _ in range(length - k - 1):
            new_tail = new_tail.next

        # Find the new head node
        new_head = new_tail.next

        # Disconnect the tail node from the new head node
        new_tail.next = None

        return new_head
