# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not k:
            return head
        
        # First, find the length of the linked list
        length = 0
        current = head
        while current:
            length += 1
            current = current.next
        
        # Calculate effective rotations needed (k % length)
        k = k % length
        if k == 0:
            return head
        
        # Find the new tail node and break the link
        slow, fast = head, head
        for _ in range(k):
            fast = fast.next
        
        while fast.next:
            slow = slow.next
            fast = fast.next
        
        # Rotate the list
        new_head = slow.next
        slow.next = None
        fast.next = head
        
        return new_head
