class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # 1. Calculate the length of the list and find the tail node
        length = 1
        tail = head
        while tail.next:
            tail = tail.next
            length += 1
            
        # 2. Determine the actual number of rotations needed
        k = k % length
        if k == 0:
            return head
        
        # 3. Connect the tail to the head to form a circular list
        tail.next = head
        
        # 4. Find the new tail node, which is at position (length - k - 1) from the start
        steps_to_new_tail = length - k - 1
        new_tail = head
        for _ in range(steps_to_new_tail):
            new_tail = new_tail.next
            
        # 5. The node following the new tail is the new head
        new_head = new_tail.next
        # Break the circular link
        new_tail.next = None
        
        return new_head
