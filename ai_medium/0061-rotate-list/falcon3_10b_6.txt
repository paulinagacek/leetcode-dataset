# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next

class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or k == 0:
            return head
        
        length = 0
        cur = head
        while cur:
            length += 1
            cur = cur.next
            
        k = k % length  # In case k > length, we want to rotate k places within the range of list length.
        
        if k == 0:
            return head
        
        cur = head
        for _ in range(length - k - 1):
            cur = cur.next
        
        new_head = cur.next  # New head after rotation
        cur.next = None     # Break the list at the current node to form a loop
        
        prev = new_head
        while prev.next:
            prev = prev.next
        
        prev.next = head
        
        return new_head
