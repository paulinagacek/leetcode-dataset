class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        if not head or k == 0:
            return head

        # Calculate the length of the linked list
        length = 1
        temp = head
        while temp.next:
            length += 1
            temp = temp.next

        # Adjust k based on the length of the linked list
        k = k % length

        # Close the loop and find the new tail
        temp.next = head
        new_tail = head
        for _ in range(length - k - 1):
            new_tail = new_tail.next

        # Get the new head and disconnect the tail
        new_head = new_tail.next
        new_tail.next = None

        return new_head
