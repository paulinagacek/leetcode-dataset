# Definition for singly-linked list.
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        # Find the length and last node of the list
        length = 0
        curr = head
        while curr:
            length += 1
            curr = curr.next

        # If list is empty or rotation factor is 0, return original
        if length == 0 or k == 0:
            return head

        # Convert k to effective rotation within the list length
        k = k % length
        if k == 0:
            return head

        # Find new tail and new head
        new_tail = head
        for _ in range(length - k - 1):
            new_tail = new_tail.next
        new_head = new_tail.next
        new_tail.next = None

        # Find the last node to connect to new_head
        curr = head
        while curr.next:
            curr = curr.next
        curr.next = new_head
        
        return new_head
