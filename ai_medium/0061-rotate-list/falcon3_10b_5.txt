# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head:
            return None

        # Find the length of the list and the last node
        length = 1
        last_node = head
        while last_node.next:
            last_node = last_node.next
            length += 1

        # Adjust k to be within the range [0, length)
        k %= length
        if k == 0:
            return head

        # Find the node which becomes the new tail (length - k - 1 steps from head)
        new_tail = head
        for _ in range(length - k - 1):
            new_tail = new_tail.next

        # Split the list at the new tail
        new_head = new_tail.next
        new_tail.next = None

        # Adjust the last node to point to the original head
        last_node.next = head

        return new_head
