class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and find the tail node
        length = 1
        tail = head
        while tail.next:
            tail = tail.next
            length += 1
        
        # Calculate the effective rotation
        k = k % length
        if k == 0:
            return head
        
        # Connect the tail to the head to make it circular
        tail.next = head
        
        # Find the new tail: it is at (length - k - 1)th position from head
        new_tail_steps = length - k
        new_tail = head
        for _ in range(new_tail_steps - 1):
            new_tail = new_tail.next
            
        # The new head is the node after the new tail
        new_head = new_tail.next
        # Break the circle
        new_tail.next = None
        
        return new_head
