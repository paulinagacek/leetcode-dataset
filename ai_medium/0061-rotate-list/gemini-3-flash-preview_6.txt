class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and find the tail node
        length = 1
        tail = head
        while tail.next:
            tail = tail.next
            length += 1
            
        # Determine the effective rotation amount
        k = k % length
        if k == 0:
            return head
            
        # Connect tail to head to form a circular list
        tail.next = head
        
        # Find the node that will become the new tail
        # The new tail is at index (length - k - 1)
        steps_to_new_tail = length - k - 1
        new_tail = head
        for _ in range(steps_to_new_tail):
            new_tail = new_tail.next
            
        # Break the circle and set the new head
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
