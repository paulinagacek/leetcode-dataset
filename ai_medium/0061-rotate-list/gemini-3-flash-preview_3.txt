class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # 1. Calculate the length of the list and find the tail node
        length = 1
        tail = head
        while tail.next:
            tail = tail.next
            length += 1
        
        # 2. Determine the actual number of rotations needed
        k = k % length
        if k == 0:
            return head
        
        # 3. Make the list circular
        tail.next = head
        
        # 4. Find the new tail: it is at (length - k - 1) steps from the head
        new_tail_steps = length - k - 1
        new_tail = head
        for _ in range(new_tail_steps):
            new_tail = new_tail.next
        
        # 5. The new head is the node after the new tail
        new_head = new_tail.next
        # Break the circular connection
        new_tail.next = None
        
        return new_head
