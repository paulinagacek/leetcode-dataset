class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        # Calculate the length of the list and find the tail
        n = 1
        tail = head
        while tail.next:
            n += 1
            tail = tail.next
            
        # Determine the effective number of rotations
        k = k % n
        if k == 0:
            return head
            
        # Connect the tail to the head to make it circular
        tail.next = head
        
        # Find the new tail: (n - k)th node from the beginning
        steps_to_new_tail = n - k
        new_tail = head
        for _ in range(steps_to_new_tail - 1):
            new_tail = new_tail.next
            
        # Set the new head and break the circular connection
        new_head = new_tail.next
        new_tail.next = None
        
        return new_head
