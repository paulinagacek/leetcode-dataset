# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k == 0:
            return head
        
        dummy = ListNode(0)
        dummy.next = head
        length = 0
        p = head
        while p:
            length += 1
            p = p.next
        
        k = k % length
        
        if k == 0:
            return head
        
        slow = fast = dummy
        for _ in range(k):
            fast = fast.next
        
        while fast.next:
            slow = slow.next
            fast = fast.next
        
        new_head = slow.next
        fast.next = dummy.next
        slow.next = None
        
        return new_head
