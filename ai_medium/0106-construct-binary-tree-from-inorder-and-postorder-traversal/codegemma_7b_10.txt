class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(instart, inend, pstart, pend):
            if pstart > pend:
                return None
            val = postorder[pend]
            idx = inorder.index(val)
            leftsize = idx - instart
            root = TreeNode(val)
            root.left = helper(instart, idx - 1, pstart, pstart + leftsize - 1)
            root.right = helper(idx + 1, inend, pstart + leftsize, pend - 1)
            return root

        return helper(0, len(inorder) - 1, 0, len(postorder) - 1)
