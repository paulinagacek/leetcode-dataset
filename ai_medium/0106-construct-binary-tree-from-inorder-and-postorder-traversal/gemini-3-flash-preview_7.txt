class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        # Mapping values to their indices in the inorder traversal for O(1) lookups
        idx_map = {val: i for i, val in enumerate(inorder)}
        
        def build(in_start, in_end):
            # Base case: if the current subtree range in inorder is empty
            if in_start > in_end:
                return None
            
            # The root of the current subtree is the last element in the postorder traversal
            root_val = postorder.pop()
            root = TreeNode(root_val)
            
            # Index of the current root in the inorder traversal
            idx = idx_map[root_val]
            
            # Postorder traversal is Left-Right-Root. Popping from the end 
            # results in processing the Root, then the Right subtree, 
            # then the Left subtree. This ordering is crucial.
            root.right = build(idx + 1, in_end)
            root.left = build(in_start, idx - 1)
            
            return root
        
        return build(0, len(inorder) - 1)
