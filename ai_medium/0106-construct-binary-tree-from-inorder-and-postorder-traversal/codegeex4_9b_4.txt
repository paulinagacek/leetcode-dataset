# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(in_left, in_right, post_root):
            if in_left > in_right:
                return None
            
            # The last element in the current postorder slice is the root of the subtree
            root_val = postorder[post_root]
            root = TreeNode(root_val)
            
            # Find the index of the root in inorder slice to divide into left and right subtrees
            inorder_root_index = index_map[root_val]
            
            # The number of nodes in the left subtree is the difference between the current root
            # index and the leftmost node index
            left_tree_size = inorder_root_index - in_left
            
            # Recursively build the left subtree
            root.left = helper(in_left, inorder_root_index - 1, post_root - 1 - in_right + inorder_root_index)
            
            # Recursively build the right subtree
            root.right = helper(inorder_root_index + 1, in_right, post_root - 1)
            
            return root
        
        # Create a hashmap to store the index of each value in the inorder list for quick lookup
        index_map = {val: i for i, val in enumerate(inorder)}
        
        # Start the recursive construction from the last element in postorder list
        return helper(0, len(inorder) - 1, len(postorder) - 1)
