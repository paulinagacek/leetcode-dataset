class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        # Map each value to its index in the inorder traversal for O(1) lookups
        idx_map = {val: i for i, val in enumerate(inorder)}
        
        def helper(in_left: int, in_right: int) -> Optional[TreeNode]:
            # Base case: if there are no elements in the current inorder range
            if in_left > in_right:
                return None
            
            # The root of the current subtree is the last element of the postorder list
            root_val = postorder.pop()
            root = TreeNode(root_val)
            
            # Get the index of the root in the inorder traversal to divide the tree
            index = idx_map[root_val]
            
            # Recursively build the right subtree first. This is necessary because we are
            # popping from the end of the postorder list, which follows the order:
            # [Left Subtree, Right Subtree, Root]. Popping gives Root, then Right, then Left.
            root.right = helper(index + 1, in_right)
            root.left = helper(in_left, index - 1)
            
            return root
            
        return helper(0, len(inorder) - 1)
