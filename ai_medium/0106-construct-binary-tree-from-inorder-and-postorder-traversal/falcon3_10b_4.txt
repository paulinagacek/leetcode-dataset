# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def build(in_start, in_end, post_start, post_end):
            if in_start > in_end or post_start > post_end:
                return None
            
            root_val = postorder[post_end]
            root = TreeNode(root_val)
            in_root_index = inorder_map[root_val]
            
            in_left_nodes = in_root_index - in_start
            left_subtree_size = in_left_nodes

            root.left = build(in_start, in_root_index - 1, post_start, post_start + left_subtree_size - 1)
            root.right = build(in_root_index + 1, in_end, post_end - 1, post_end - left_subtree_size)

            return root
        
        inorder_map = {val: idx for idx, val in enumerate(inorder)}
        return build(0, len(inorder) - 1, 0, len(postorder) - 1)
