class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        post_index = len(postorder) - 1
        in_start = 0
        in_end = len(inorder) - 1
        
        def helper(in_start, in_end):
            if in_start > in_end:
                return None
            val = postorder[post_index]
            index = inorder.index(val)
            
            post_index -= 1
            root = TreeNode(val)
            root.right = helper(index + 1, in_end)
            root.left = helper(in_start, index - 1)
            
            return root
        
        return helper(in_start, in_end)
