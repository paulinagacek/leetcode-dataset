class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        # Map values to their indices in the inorder traversal for O(1) lookups
        inorder_map = {val: i for i, val in enumerate(inorder)}
        
        def build(left: int, right: int) -> Optional[TreeNode]:
            if left > right:
                return None
            
            # The root of the current subtree is the last element in the current postorder segment
            root_val = postorder.pop()
            root = TreeNode(root_val)
            
            # Get the index of the root in the inorder traversal
            index = inorder_map[root_val]
            
            # Build the right subtree before the left subtree because the 
            # postorder traversal processes the root after its subtrees (Left -> Right -> Root). 
            # Popping from the end of postorder gives elements in the order: Root -> Right -> Left.
            root.right = build(index + 1, right)
            root.left = build(left, index - 1)
            
            return root
            
        return build(0, len(inorder) - 1)
