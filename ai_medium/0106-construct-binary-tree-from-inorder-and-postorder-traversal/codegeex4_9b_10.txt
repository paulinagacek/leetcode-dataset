# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(in_left, in_right, post_root):
            if in_left > in_right:
                return None
            
            # Get the root value from postorder
            root_val = postorder[post_root]
            
            # Create a new TreeNode with the root value
            root = TreeNode(root_val)
            
            # Find the index of the root value in inorder array
            inorder_index = index_map[root_val]
            
            # Calculate the size of left subtree
            left_tree_size = inorder_index - in_left
            
            # Recursively construct the right subtree first, because postorder is processed from right to left
            root.right = helper(inorder_index + 1, in_right, post_root - 1)
            root.left = helper(in_left, inorder_index - 1, post_root - 1 - (in_right - inorder_index))
            
            return root
        
        # Create a dictionary to map values to their indices
        index_map = {val:idx for idx, val in enumerate(inorder)}
        
        # Start from the last element in postorder array and traverse the inorder array
        return helper(0, len(inorder) - 1, len(postorder) - 1)
