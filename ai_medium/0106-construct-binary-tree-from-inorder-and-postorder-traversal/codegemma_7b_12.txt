class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(in_start, in_end, post_start, post_end):
            if post_start > post_end:
                return None

            val = postorder[post_end]
            index = inorder.index(val)

            left_size = index - in_start

            root = TreeNode(val)
            root.left = helper(in_start, index - 1, post_start, post_start + left_size - 1)
            root.right = helper(index + 1, in_end, post_start + left_size, post_end - 1)

            return root

        return helper(0, len(inorder) - 1, 0, len(postorder) - 1)
