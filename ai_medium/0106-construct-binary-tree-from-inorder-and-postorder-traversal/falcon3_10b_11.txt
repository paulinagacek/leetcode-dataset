class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def build(in_low, in_high, post_low, post_high):
            if in_low > in_high:
                return None
            
            root_val = postorder[post_high]
            root = TreeNode(root_val)
            in_root_index = index_map[root_val]
            
            left_subtree_size = in_root_index - in_low
            root.left = build(in_low, in_root_index - 1, post_low, post_low + left_subtree_size - 1)
            root.right = build(in_root_index + 1, in_high, post_high - left_subtree_size, post_high - 1)
            return root
        
        index_map = {val: i for i, val in enumerate(inorder)}
        
        return build(0, len(inorder) - 1, 0, len(postorder) - 1)
