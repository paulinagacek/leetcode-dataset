class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def construct_tree(in_start, in_end, post_start, post_end):
            if post_start > post_end: return None
            root_val = postorder[post_end]
            root = TreeNode(root_val)
            in_index = idx_map[root_val]
            left_subtree_size = in_index - in_start
            root.left = construct_tree(in_start, in_index-1, post_start, post_start + left_subtree_size - 1)
            root.right = construct_tree(in_index+1, in_end, post_end-1, post_start + left_subtree_size)
            return root
        
        idx_map = {val: idx for idx, val in enumerate(inorder)}
        return construct_tree(0, len(inorder)-1, 0, len(postorder)-1)
