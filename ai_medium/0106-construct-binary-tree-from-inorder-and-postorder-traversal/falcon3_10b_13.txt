class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(in_start, in_end, post_start, post_end):
            if in_start > in_end:
                return None
            
            root_val = postorder[post_end]
            root = TreeNode(root_val)
            
            in_root_idx = idx_map[root_val]
            left_subtree_size = in_root_idx - in_start
            
            root.left = helper(in_start, in_root_idx - 1, post_start, post_start + left_subtree_size - 1)
            root.right = helper(in_root_idx + 1, in_end, post_end - 1 - (in_end - in_root_idx), post_end - 1)
            
            return root
        
        idx_map = {val: idx for idx, val in enumerate(inorder)}
        return helper(0, len(inorder) - 1, 0, len(postorder) - 1)
