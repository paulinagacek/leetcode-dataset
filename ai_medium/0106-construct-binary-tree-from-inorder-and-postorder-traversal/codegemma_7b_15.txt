class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(instart, inend, poststart, postend):
            if poststart > postend:
                return None
            val = postorder[postend]
            index = inorder.index(val)
            left_nodes = index - instart
            root = TreeNode(val)
            root.left = helper(instart, index - 1, poststart, poststart + left_nodes - 1)
            root.right = helper(index + 1, inend, poststart + left_nodes, postend - 1)
            return root
        return helper(0, len(inorder) - 1, 0, len(postorder) - 1)
