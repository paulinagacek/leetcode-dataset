class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def build(in_start, in_end, post_end):
            if in_start > in_end:
                return None
            
            node = TreeNode(postorder[post_end])
            i = inorder.index(postorder[post_end])
            node.left = build(in_start, i-1, post_end - (in_end - i) - 1)
            node.right = build(i+1, in_end, post_end - 1)
            
            return node

        return build(0, len(inorder) - 1, len(postorder) - 1)
