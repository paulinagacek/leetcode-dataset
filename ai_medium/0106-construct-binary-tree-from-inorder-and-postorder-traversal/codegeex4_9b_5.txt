# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def buildTree(self, inorder: List[int], postorder: List[int]) -> Optional[TreeNode]:
        def helper(in_left, in_right, post_root):
            if in_left > in_right:
                return None
            
            # the last element in post order traversal is the root of current sub-tree
            root_val = postorder[post_root]
            root = TreeNode(root_val)
            
            # find the index of root in inorder array
            idx = idx_map[root_val]
            
            # calculate left and right subtree count for current root
            left_tree_size = idx - in_left
            
            # set up recursive calls by calculating appropriate postorder root index for each side of the tree (root)
            root.left = helper(in_left, idx - 1, post_root - 1 - in_right + idx)
            root.right = helper(idx + 1, in_right, post_root - 1)
            
            return root
        
        # build a hashmap to record value -> its index mapping for quick search
        idx_map = {val:idx for idx,val in enumerate(inorder)}
        
        # the last element of post order is root, so start from end of array
        post_root = len(postorder) - 1
        
        # call recursive helper function
        return helper(0, len(inorder) - 1, post_root)
