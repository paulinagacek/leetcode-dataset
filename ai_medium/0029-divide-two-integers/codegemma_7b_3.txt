class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        sign = (dividend < 0) is (divisor < 0)
        dividend = abs(dividend)
        divisor = abs(divisor)
        quotient = 0

        while dividend >= divisor:
            dividend -= divisor
            quotient += 1

        if not sign:
            quotient = -quotient

        if quotient > 2**31 - 1:
            quotient = 2**31 - 1
        elif quotient < -2**31:
            quotient = -2**31

        return quotient
