class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Define constants for 32-bit integer limits
        MAX_INT = 2**31 - 1
        MIN_INT = -2**31
        
        # Handle the overflow case specifically as per constraints
        if dividend == MIN_INT and divisor == -1:
            return MAX_INT
        
        # Determine the sign of the quotient
        negative = (dividend < 0) ^ (divisor < 0)
        
        # Work with absolute values to simplify the logic
        a, b = abs(dividend), abs(divisor)
        quotient = 0
        
        # Bit manipulation approach to perform division in O(log n) steps
        while a >= b:
            temp_divisor, multiple = b, 1
            # Find the largest multiple of the divisor using bit shifting
            while a >= (temp_divisor << 1):
                temp_divisor <<= 1
                multiple <<= 1
            
            # Subtract the found multiple from the dividend
            a -= temp_divisor
            quotient += multiple
            
        # Apply the determined sign to the quotient
        result = -quotient if negative else quotient
        
        # Ensure result stays within 32-bit signed integer limits
        return max(MIN_INT, min(MAX_INT, result))
