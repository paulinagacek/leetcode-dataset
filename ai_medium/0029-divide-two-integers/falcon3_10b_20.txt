class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        INT_MAX = 2 ** 31 - 1
        INT_MIN = -2 ** 31
        
        # Handle overflow cases
        if dividend == INT_MIN and divisor == -1:
            return INT_MAX
        
        # Make sure dividend is always positive
        divid = abs(dividend)
        divi = abs(divisor)
        result = 0
        temp = 0
        
        for i in range(32)[::-1]:
            if temp + (divi << i) <= divid:
                temp += divi << i
                result |= 1 << i
        
        # Adjust the sign of the result based on the signs of dividend and divisor
        if (dividend < 0) != (divisor < 0):
            result = -result
        
        return min(max(result, INT_MIN), INT_MAX)
