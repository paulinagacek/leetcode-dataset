class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Define constants for the 32-bit signed integer range
        MIN_INT = -2147483648
        MAX_INT = 2147483647

        # Handle overflow case: -2^31 / -1 = 2^31 (which is out of range)
        if dividend == MIN_INT and divisor == -1:
            return MAX_INT

        # Determine the sign of the quotient
        negative = (dividend < 0) != (divisor < 0)

        # Work with positive numbers to simplify the subtraction logic
        dividend, divisor = abs(dividend), abs(divisor)
        quotient = 0

        # Optimization: use bit manipulation to subtract multiples of the divisor
        while dividend >= divisor:
            temp_divisor = divisor
            multiple = 1
            # Double the divisor and the multiple as much as possible
            while dividend >= (temp_divisor << 1):
                temp_divisor <<= 1
                multiple <<= 1
            
            # Subtract the largest found multiple of the divisor
            dividend -= temp_divisor
            quotient += multiple

        # Apply the sign and ensure the result is within 32-bit integer limits
        if negative:
            quotient = -quotient
        
        return max(MIN_INT, min(MAX_INT, quotient))
