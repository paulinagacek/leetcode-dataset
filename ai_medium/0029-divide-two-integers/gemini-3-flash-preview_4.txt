class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Handle the overflow case where dividend is INT_MIN and divisor is -1
        if dividend == -2147483648 and divisor == -1:
            return 2147483647
        
        # Determine the sign of the quotient
        negative = (dividend < 0) ^ (divisor < 0)
        
        # Work with the absolute values of the dividend and divisor
        a, b = abs(dividend), abs(divisor)
        quotient = 0
        
        # Divide using bit manipulation (exponentially increase subtraction amount)
        while a >= b:
            temp_divisor, count = b, 1
            # Check if temp_divisor can be doubled without exceeding a
            while a >= (temp_divisor << 1):
                temp_divisor <<= 1
                count <<= 1
            
            # Subtract the largest power-of-2 multiple of divisor found
            a -= temp_divisor
            quotient += count
            
        # Apply the determined sign to the quotient
        return -quotient if negative else quotient
