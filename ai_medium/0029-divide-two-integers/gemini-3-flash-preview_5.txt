class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Define the limits for 32-bit signed integers
        INT_MIN = -2147483648
        INT_MAX = 2147483647

        # Handle overflow case for dividend = -2^31 and divisor = -1
        if dividend == INT_MIN and divisor == -1:
            return INT_MAX

        # Determine the sign of the result
        is_negative = (dividend < 0) ^ (divisor < 0)

        # Work with absolute values
        a = abs(dividend)
        b = abs(divisor)
        res = 0

        # Bitwise division approach
        while a >= b:
            temp = b
            multiple = 1
            # Double temp until it's larger than a
            while a >= (temp << 1):
                temp <<= 1
                multiple <<= 1
            
            # Subtract the largest found multiple from a
            a -= temp
            res += multiple

        # Apply the sign and return clamped to 32-bit range
        res = -res if is_negative else res
        return max(INT_MIN, min(INT_MAX, res))
