class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        s = (dividend < 0) is (divisor < 0)
        dividend, divisor = abs(dividend), abs(divisor)
        quotient = 0
        while dividend >= divisor:
            dividend -= divisor
            quotient += 1
        return max(-quotient if not s else quotient, -0x80000000)
