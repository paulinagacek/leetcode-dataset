class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        INT_MIN = -2**31
        INT_MAX = 2**31 - 1

        if dividend == 0:
            return 0
        if divisor == 1:
            return min(max(dividend, INT_MIN), INT_MAX)
        if divisor == -1:
            if dividend == INT_MIN:
                return INT_MAX
            return -dividend

        negative = (dividend < 0) ^ (divisor < 0)
        a = -dividend if dividend > 0 else dividend
        b = -divisor if divisor > 0 else divisor

        res = 0
        while a <= b:
            tmp = b
            multiple = 1
            while tmp >= (INT_MIN >> 1) and a <= (tmp << 1):
                tmp <<= 1
                multiple <<= 1
            a -= tmp
            res += multiple

        if not negative:
            res = -res
        if res < INT_MIN:
            return INT_MIN
        if res > INT_MAX:
            return INT_MAX
        return res
