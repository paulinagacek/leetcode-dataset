class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        if divisor == 0:
            return float('inf')
        
        sign = 1
        if (dividend < 0) ^ (divisor < 0):
            sign = -1
        
        dividend = abs(dividend)
        divisor = abs(divisor)
        quotient = 0
        
        while dividend >= divisor:
            dividend -= divisor
            quotient += 1
        
        quotient = sign * quotient
        
        if quotient > 2**31 - 1:
            return 2**31 - 1
        elif quotient < -2**31:
            return -2**31
        else:
            return quotient
