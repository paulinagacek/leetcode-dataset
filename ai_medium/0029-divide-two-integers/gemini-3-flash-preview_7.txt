class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Standard limits for 32-bit signed integers
        INT_MIN, INT_MAX = -2147483648, 2147483647
        
        # Handle the overflow case where dividend is -2^31 and divisor is -1
        if dividend == INT_MIN and divisor == -1:
            return INT_MAX
        
        # Determine the sign of the result
        negative = (dividend < 0) ^ (divisor < 0)
        
        # Use absolute values to simplify the calculation
        a, b = abs(dividend), abs(divisor)
        quotient = 0
        
        # Iterate from the largest potential power of 2 down to 0
        # This is essentially a binary long division
        for i in range(31, -1, -1):
            # Check if b * 2^i <= a using right shift to avoid potential overflow
            if (a >> i) >= b:
                # Add 2^i to the quotient
                quotient += (1 << i)
                # Subtract b * 2^i from a
                a -= (b << i)
        
        # Apply the determined sign to the quotient
        return -quotient if negative else quotient
