class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        
        result = 0
        dividend_is_negative = dividend < 0
        divisor_is_negative = divisor < 0
        
        dividend = abs(dividend)
        divisor = abs(divisor)
        
        while dividend >= divisor:
            dividend -= divisor
            result += 1
        
        if (dividend_is_negative and not divisor_is_negative) or (not dividend_is_negative and divisor_is_negative):
            result = -result
        
        if result > 2**31 - 1:
            return 2**31 - 1
        elif result < -2**31:
            return -2**31
        else:
            return result
