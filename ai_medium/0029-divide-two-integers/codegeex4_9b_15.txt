class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        
        # Edge case where result is outside the range of 32-bit signed integer
        if dividend == INT_MIN and divisor == -1:
            return INT_MAX
        
        # Determine the sign of the result
        sign = -1 if (dividend < 0) ^ (divisor < 0) else 1
        
        # Use absolute values to simplify the division process
        a, b = abs(dividend), abs(divisor)
        
        quotient = 0
        
        # Subtract multiples of divisor from dividend until dividend is smaller than divisor
        while a >= b:
            power = 1
            value = b
            
            # Double the value and power until it is the largest multiple less than or equal to dividend
            while value + value <= a:
                value += value
                power += power
            
            # Subtract the largest found multiple from dividend
            a -= value
            quotient += power
        
        # Apply the sign to the quotient and return
        return sign * quotient
