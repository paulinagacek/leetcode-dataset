class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        INT_MIN = -2**31
        INT_MAX = 2**31 - 1

        if dividend == 0:
            return 0
        if divisor == 1:
            return max(INT_MIN, min(INT_MAX, dividend))
        if divisor == -1:
            if dividend == INT_MIN:
                return INT_MAX
            return -dividend

        negative = (dividend < 0) ^ (divisor < 0)
        a = -dividend if dividend < 0 else dividend
        b = -divisor if divisor < 0 else divisor

        quotient = 0
        while a >= b:
            temp = b
            multiple = 1
            while (temp << 1) > 0 and (temp << 1) <= a:
                temp <<= 1
                multiple <<= 1
            a -= temp
            quotient += multiple

        if negative:
            quotient = -quotient

        if quotient < INT_MIN:
            return INT_MIN
        if quotient > INT_MAX:
            return INT_MAX
        return quotient
