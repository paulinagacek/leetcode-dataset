class Solution:
    def divide(self, dividend: int, divisor: int) -> int:
        # Define the limits for a 32-bit signed integer
        INT_MIN = -2147483648
        INT_MAX = 2147483647
        
        # Handle the specific case that causes overflow in the 32-bit range
        if dividend == INT_MIN and divisor == -1:
            return INT_MAX
        
        # Determine the sign of the quotient
        # Using XOR: True if signs are different, False if they are the same
        negative = (dividend < 0) ^ (divisor < 0)
        
        # Use absolute values for calculation to simplify the logic
        a, b = abs(dividend), abs(divisor)
        quotient = 0
        
        # Exponentially subtract divisor from dividend
        # This approach mimics binary long division for efficiency
        while a >= b:
            temp, multiple = b, 1
            # Double the divisor multiple (using bit shifts) until it is the 
            # largest multiple of b that is less than or equal to a
            while a >= (temp << 1):
                temp <<= 1
                multiple <<= 1
            
            # Subtract the found multiple from a and add the corresponding power of 2 to quotient
            a -= temp
            quotient += multiple
        
        # Apply the determined sign to the quotient
        result = -quotient if negative else quotient
        
        # Return the final clamped result
        return result
