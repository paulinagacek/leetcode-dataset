# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        if not root:
            return
        
        stack = []
        stack.append(root)
        
        while stack:
            node = stack.pop()
            
            # right should be processed before left
            if node.right:
                stack.append(node.right)
            if node.left:
                stack.append(node.left)
            
            # we insert a null node between elements to simulate the linked list structure
            if stack:
                stack[-1].left = None
                stack[-1].right = node
