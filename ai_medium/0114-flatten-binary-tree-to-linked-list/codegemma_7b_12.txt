class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        dummy = TreeNode(left=root)
        curr = dummy
        stack = [root]

        while stack:
            node = stack.pop()
            if node.right:
                stack.append(node.right)
            if node.left:
                stack.append(node.left)
            curr.right = node
            curr.left = None
            curr = curr.right
        root = dummy.right
