class Solution:
    def flatten(self, root: TreeNode) -> None:
        if not root: return
        stack = [root]
        prev = TreeNode(None)  # Helper node
        while stack:
            node = stack.pop()
            if node.right: stack.append(node.right)
            if node.left: stack.append(node.left)
            prev.right = node
            prev.left = None
            node.left = None
            node.right = prev
