# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        last = None
        
        def dfs(node):
            nonlocal last
            if not node: 
                return root
            
            dfs(node.right)
            node.right = last
            node.left = None
            last = node
            
            dfs(node.left)
        
        dfs(root)
