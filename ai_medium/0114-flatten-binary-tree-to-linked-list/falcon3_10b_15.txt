# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        if not root:
            return
        last = root
        while root and (root.left or root.right):
            if root.left:
                last_leftmost = root.left
                while last_leftmost.right:
                    last_leftmost = last_leftmost.right
                last_leftmost.right = root.right
                root.right = root.left
                root.left = None
            last = root.right
            root = root.right
        last.right = None
