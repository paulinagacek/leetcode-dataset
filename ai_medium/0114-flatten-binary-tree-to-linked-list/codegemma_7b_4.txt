class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        def helper(node):
            if not node:
                return None, None
            left, right = helper(node.left), helper(node.right)
            node.left = None
            if left:
                node.right = left[0]
                left[1].right = right[0] if right else None
            return node, right[1] if right else left[1]
        helper(root)
