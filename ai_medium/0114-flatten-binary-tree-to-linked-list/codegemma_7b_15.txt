class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        dummy = TreeNode()
        curr = dummy
        stack = [root]
        
        while stack:
            node = stack.pop()
            if node is None:
                continue
                
            curr.right = node
            curr.left = None
            stack.append(node.right)
            stack.append(node.left)
            
            curr = curr.right
