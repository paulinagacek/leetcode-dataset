class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        if not root:
            return
        dummy = TreeNode(val=0, left=root)
        pre_node = dummy
        stack = [root]

        while stack:
            curr = stack.pop()

            if curr.right:
                stack.append(curr.right)
            if curr.left:
                stack.append(curr.left)
            curr.left = None
            pre_node.right = curr
            pre_node = curr
