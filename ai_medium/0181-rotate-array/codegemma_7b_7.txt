class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        n = len(nums)
        k %= n  # Handle cases where k is larger than the array length

        # Reverse the entire array
        nums[:] = nums[::-1]

        # Reverse the first k elements
        nums[:k] = nums[n - k:]

        # Reverse the last n - k elements
        nums[k:] = nums[:n - k]
