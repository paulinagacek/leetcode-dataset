class Solution:
    def rotate(self, nums: List[int], k: int) -> None:
        n = len(nums)
        k %= n  # Handle cases where k is greater than n
        self.reverse(nums, 0, n - 1)  # Reverse the entire array
        self.reverse(nums, 0, k - 1)  # Reverse the first part
        self.reverse(nums, k, n - 1)  # Reverse the second part

    def reverse(self, nums: List[int], start: int, end: int) -> None:
        while start < end:
            nums[start], nums[end] = nums[end], nums[start]
            start += 1
            end -= 1
