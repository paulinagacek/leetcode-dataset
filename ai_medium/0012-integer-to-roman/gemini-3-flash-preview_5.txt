class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the values and their corresponding Roman numeral symbols
        # in descending order, including the subtractive forms.
        val_to_sym = [
            (1000, 'M'), (900, 'CM'), (500, 'D'), (400, 'CD'),
            (100, 'C'), (90, 'XC'), (50, 'L'), (40, 'XL'),
            (10, 'X'), (9, 'IX'), (5, 'V'), (4, 'IV'), (1, 'I')
        ]
        
        res = []
        for value, symbol in val_to_sym:
            if num == 0:
                break
            # Determine how many times the current value fits into the number
            count, num = divmod(num, value)
            # Append the symbol 'count' times to the result list
            res.append(symbol * count)
        
        # Join the list into the final Roman numeral string
        return "".join(res)
