class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the values and their corresponding Roman symbols in descending order,
        # including the subtractive forms (4, 9, 40, 90, 400, 900).
        val_sym = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        result = []
        # Iterate through the mapping, subtracting values and appending symbols.
        for value, symbol in val_sym:
            if num == 0:
                break
            # Calculate how many times this symbol should be appended.
            count = num // value
            if count > 0:
                result.append(symbol * count)
                num %= value
                
        return "".join(result)
