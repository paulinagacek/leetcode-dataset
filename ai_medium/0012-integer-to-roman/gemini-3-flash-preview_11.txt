class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the value and symbol mapping in descending order,
        # including the subtractive cases.
        val_to_symbol = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        result = []
        # Iterate through the list and subtract values from num until num reaches 0
        for val, symbol in val_to_symbol:
            if num == 0:
                break
            count = num // val
            result.append(symbol * count)
            num -= val * count
            
        return "".join(result)
