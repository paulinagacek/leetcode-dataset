class Solution:
    def intToRoman(self, num: int) -> str:
        # A list of tuples containing Roman numeral values and their symbols in descending order.
        # This includes both standard symbols and the subtractive forms allowed.
        val_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        result = ""
        
        # Iterate through the mapping from highest to lowest value.
        for value, symbol in val_map:
            # Determine how many times the current value fits into the remaining number.
            # Using num // value and then symbol * count is equivalent to a while loop.
            count = num // value
            if count > 0:
                result += symbol * count
                # Reduce the number by the value already accounted for.
                num %= value
                
        return result
