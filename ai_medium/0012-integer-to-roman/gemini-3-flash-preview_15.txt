class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of values to their corresponding Roman symbols
        # in descending order to handle the greedy subtraction logic.
        # This includes both the standard symbols and the special subtractive forms.
        val_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        roman_numeral = ""
        
        # Iterate through the mapped values and symbols
        for value, symbol in val_map:
            # While the remaining number is greater than or equal to the current Roman value
            while num >= value:
                # Append the corresponding symbol to the result
                roman_numeral += symbol
                # Subtract the value from the remaining number
                num -= value
                
        return roman_numeral
