class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the value-to-symbol mapping in descending order,
        # including the specific subtractive forms allowed.
        val_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        roman_num = ""
        
        # Iterate through the mapping and greedily subtract the largest 
        # possible values from the input number.
        for val, symbol in val_map:
            # Check how many times the current value symbol can fit into num
            while num >= val:
                roman_num += symbol
                num -= val
                
        return roman_num
