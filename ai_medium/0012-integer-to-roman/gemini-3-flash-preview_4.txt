class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the value-to-symbol mapping in descending order,
        # including both standard symbols and subtractive forms.
        val_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        roman_numeral = ""
        
        # Greedy approach: traverse the values from largest to smallest
        for val, symbol in val_map:
            # While the remaining number is greater than or equal to the current value,
            # append the symbol and subtract the value.
            while num >= val:
                roman_numeral += symbol
                num -= val
                
        return roman_numeral
