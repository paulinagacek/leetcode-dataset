class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the values and their corresponding Roman symbols in descending order,
        # including the specific subtractive forms allowed by the rules.
        val_map = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I")
        ]
        
        res = []
        for val, symbol in val_map:
            if num == 0:
                break
            # Determine how many times this symbol can be used
            count, num = divmod(num, val)
            # Append the symbol to the result 'count' times
            res.append(symbol * count)
            
        return "".join(res)
