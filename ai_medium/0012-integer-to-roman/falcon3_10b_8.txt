class Solution:
    def intToRoman(self, num: int) -> str:
        roman_map = [
            ("M", 1000), ("CM", 900), ("D", 500), ("CD", 400),
            ("C", 100), ("XC", 90), ("L", 50), ("XL", 40),
            ("X", 10), ("IX", 9), ("V", 5), ("IV", 4), ("I", 1)
        ]
        roman, i = '', 0
        while num > 0:
            count, factor = divmod(num, roman_map[i][1])
            roman += roman_map[i][0] * count
            num -= factor
            i += 1
        return roman
