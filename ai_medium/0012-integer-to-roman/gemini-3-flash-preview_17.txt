class Solution:
    def intToRoman(self, num: int) -> str:
        # Define a mapping of Roman numeral symbols and their values in descending order,
        # including the subtractive forms as specified by the rules.
        val_sym = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        res = ""
        # Iterate through the symbol-value pairs from highest to lowest.
        for value, symbol in val_sym:
            # While the current number is greater than or equal to the symbol value,
            # subtract the value from the number and append the symbol to the result.
            while num >= value:
                res += symbol
                num -= value
                
        return res
