class Solution:
    def intToRoman(self, num: int) -> str:
        roman_nums = ["M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"]
        arabic_nums = [1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1]
        result = ""
        i = 0
        while num > 0:
            while num >= arabic_nums[i]:
                result += roman_nums[i]
                num -= arabic_nums[i]
            i += 1
        return result
