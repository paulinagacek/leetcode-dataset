class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of decimal values to Roman symbols in descending order
        val_to_roman = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I")
        ]
        
        res = []
        for val, symbol in val_to_roman:
            # While the remaining number is larger than the current symbol value
            while num >= val:
                # Append the symbol to the result and subtract its value from num
                res.append(symbol)
                num -= val
        
        # Join the list of symbols into a single string
        return "".join(res)
