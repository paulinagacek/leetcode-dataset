class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the values and their corresponding Roman numeral symbols, 
        # including the subtractive forms as separate entries.
        roman_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        result = []
        
        # Iterate through the map from highest to lowest value.
        for value, symbol in roman_map:
            if num == 0:
                break
            # Determine how many times the symbol fits into the remaining number.
            count, num = divmod(num, value)
            # Append the symbol that many times to the result list.
            result.append(symbol * count)
            
        # Join the list into the final string.
        return "".join(result)
