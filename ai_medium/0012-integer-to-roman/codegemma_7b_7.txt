class Solution:
    def intToRoman(self, num: int) -> str:
        values = [1000,900,500,400,100,90,50,40,10,9,5,4,1]
        symbols = ["M","CM","D","CD","C","XC","L","XL","X","IX","V","IV","I"]
        roman = []
        i = 0
        while num:
            div = num // values[i]
            if div:
                roman.extend(symbols[i] * div)
                num -= values[i] * div
            i += 1
        return ''.join(roman)
