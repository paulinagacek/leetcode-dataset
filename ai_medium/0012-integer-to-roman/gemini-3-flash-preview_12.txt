class Solution:
    def intToRoman(self, num: int) -> str:
        # Define a list of tuples containing Roman numeral values and their symbols
        # Symbols are ordered from largest to smallest value
        val_map = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I")
        ]
        
        roman_numeral = ""
        
        # Iterate through the mapping and use divmod to calculate occurrences
        for value, symbol in val_map:
            if num == 0:
                break
            count, num = divmod(num, value)
            roman_numeral += symbol * count
            
        return roman_numeral
