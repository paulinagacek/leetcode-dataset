class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()
        
        def backtrack(start: int, current_target: int, path: List[int]):
            if current_target == 0:
                res.append(list(path))
                return
            
            for i in range(start, len(candidates)):
                if candidates[i] > current_target:
                    break
                
                path.append(candidates[i])
                # We pass 'i' as the next start index because the same number 
                # can be chosen an unlimited number of times.
                backtrack(i, current_target - candidates[i], path)
                path.pop()
                
        backtrack(0, target, [])
        return res
