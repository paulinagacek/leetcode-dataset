class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()
        
        def backtrack(remain: int, stack: List[int], start: int):
            if remain == 0:
                res.append(list(stack))
                return
            
            for i in range(start, len(candidates)):
                # If the candidate is greater than the remaining target, 
                # no need to check further because candidates are sorted.
                if candidates[i] > remain:
                    break
                
                stack.append(candidates[i])
                # We pass 'i' as the start index because we can reuse the same element.
                backtrack(remain - candidates[i], stack, i)
                stack.pop()
                
        backtrack(target, [], 0)
        return res
