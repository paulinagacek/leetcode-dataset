class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        dfs(candidates, target, 0, [], res)
        return res

def dfs(candidates, target, idx, path, res):
    if target == 0:
        res.append(path[:])
        return
    if idx >= len(candidates) or target < 0:
        return

    path.append(candidates[idx])
    dfs(candidates, target - candidates[idx], idx, path, res)

    path.pop()
    dfs(candidates, target, idx + 1, path, res)
