class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        def dfs(candidates, target, combo, start, result):
            if target < 0:
                return
            if target == 0:
                result.append(list(combo))
                return
            for i in range(start, len(candidates)):
                combo.append(candidates[i])
                dfs(candidates, target - candidates[i], combo, i, result)
                combo.pop()
        
        candidates.sort()
        result = []
        dfs(candidates, target, [], 0, result)
        return result
