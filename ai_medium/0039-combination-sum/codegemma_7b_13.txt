class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()

        def backtrack(combination, remained):
            if remained == 0:
                res.append(combination[:])
                return

            for candidate in candidates:
                if candidate <= remained:
                    combination.append(candidate)
                    backtrack(combination, remained - candidate)
                    combination.pop()

        backtrack([], target)
        return res
