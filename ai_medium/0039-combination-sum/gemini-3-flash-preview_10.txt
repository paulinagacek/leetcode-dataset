class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()
        
        def backtrack(rem: int, path: List[int], start: int):
            if rem == 0:
                res.append(list(path))
                return
            
            for i in range(start, len(candidates)):
                if candidates[i] > rem:
                    break
                
                path.append(candidates[i])
                # We use 'i' as the starting index because the same number 
                # can be chosen an unlimited number of times.
                backtrack(rem - candidates[i], path, i)
                path.pop()
                
        backtrack(target, [], 0)
        return res
