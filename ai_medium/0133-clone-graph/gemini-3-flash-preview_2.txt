class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return None
            
        # Dictionary to store mapping from original node to cloned node
        visited = {}
        
        def dfs(curr_node):
            # If the node has already been cloned, return its clone
            if curr_node in visited:
                return visited[curr_node]
            
            # Create a new clone for the node
            clone_node = Node(curr_node.val)
            visited[curr_node] = clone_node
            
            # Recursively clone all the neighbors
            for neighbor in curr_node.neighbors:
                clone_node.neighbors.append(dfs(neighbor))
                
            return clone_node
            
        return dfs(node)
