class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return None
        
        # Mapping from original node to its cloned counterpart
        clones = {}
        
        def dfs(curr_node):
            # If the node has already been cloned, return the existing clone
            if curr_node in clones:
                return clones[curr_node]
            
            # Create a clone for the current node (without neighbors initially)
            copy = Node(curr_node.val)
            clones[curr_node] = copy
            
            # Recursively clone all neighbors and add them to the current clone's neighbors list
            for neighbor in curr_node.neighbors:
                copy.neighbors.append(dfs(neighbor))
            
            return copy
        
        return dfs(node)
