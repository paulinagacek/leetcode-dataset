from typing import Optional
class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return node
        visited = {}
        def dfs(node):
            if node in visited:
                return visited[node]
            cloned_node = Node(node.val)
            visited[node] = cloned_node
            for neighbor in node.neighbors:
                cloned_node.neighbors.append(dfs(neighbor))
            return cloned_node
        return dfs(node)
