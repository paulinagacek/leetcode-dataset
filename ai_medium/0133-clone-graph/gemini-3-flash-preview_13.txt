class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return None
        
        # Mapping from original node to its cloned copy
        clones = {}
        
        def dfs(curr):
            # If the node has already been cloned, return its clone
            if curr in clones:
                return clones[curr]
            
            # Create a clone of the current node and add it to the mapping
            copy = Node(curr.val)
            clones[curr] = copy
            
            # Recursively clone all the neighbors and add them to the cloned node's neighbors list
            for neighbor in curr.neighbors:
                copy.neighbors.append(dfs(neighbor))
            
            return copy
        
        # Start the recursive cloning process from the given node
        return dfs(node)
