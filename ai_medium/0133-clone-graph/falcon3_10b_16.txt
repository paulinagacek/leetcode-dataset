"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

from typing import Optional, List, Dict

class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return None

        cloned_nodes = {}  # dictionary to store cloned nodes
        
        def dfs(node: 'Node') -> 'Node':
            if node in cloned_nodes:
                return cloned_nodes[node]
            
            cloned_node = Node(node.val)  # create cloned node
            cloned_nodes[node] = cloned_node  # store cloned node
            
            for neighbor in node.neighbors:
                cloned_neighbors = dfs(neighbor)  # recursively clone neighbors
                cloned_node.neighbors.append(cloned_neighbors)
            
            return cloned_node
        
        return dfs(node)
