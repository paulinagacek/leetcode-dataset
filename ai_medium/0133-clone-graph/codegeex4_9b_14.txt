"""
# Definition for a Node.
class Node:
    def __init__(self, val = 0, neighbors = None):
        self.val = val
        self.neighbors = neighbors if neighbors is not None else []
"""

from typing import Optional

class Solution:
    def cloneGraph(self, node: Optional['Node']) -> Optional['Node']:
        if not node:
            return None
        
        # Dictionary to keep track of visited nodes and their clones
        visited = {}
        
        def dfs(node):
            # If the node has already been visited, return its clone from the dictionary
            if node in visited:
                return visited[node]
            
            # Create a new node with the same value as the original node
            clone_node = Node(node.val)
            visited[node] = clone_node
            
            # Recursively clone each neighbor and add it to the neighbors list of the clone node
            for neighbor in node.neighbors:
                clone_node.neighbors.append(dfs(neighbor))
            
            return clone_node
        
        # Start the DFS traversal from the given node
        return dfs(node)
