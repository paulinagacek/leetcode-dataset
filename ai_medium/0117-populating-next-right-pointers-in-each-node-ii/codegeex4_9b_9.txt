"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return root
        
        # Initialize the current node as the root of the tree
        curr = root
        
        # Create a fake head for each level of the tree
        while curr:
            # dummy is going to be the head of the "linked list" that points to the next right nodes in the same level
            dummy = Node(0)
            # prev is going to track the end of the last node in the current level's linked list
            prev = dummy
            
            # Traverse through the tree at the current level
            while curr:
                if curr.left:
                    # Connect the previous node's next to the current node's left child
                    prev.next = curr.left
                    prev = prev.next
                if curr.right:
                    # Connect the previous node's next to the current node's right child
                    prev.next = curr.right
                    prev = prev.next
                
                # Move to the next node in the current level
                curr = curr.next
            
            # Start with the first node of the next level
            curr = dummy.next
        
        # Return the root of the tree with the updated next pointers
        return root
