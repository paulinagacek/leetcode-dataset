"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        # Initialize the current node to the root of the tree
        current = root
        
        # This loop will iterate while there is at least one level in the tree
        while current:
            # We will use a "dummy node" to help us build the next pointers for the child nodes
            dummy_root = Node(0)
            previous_child = dummy_root
            
            # Traverse the tree at the current level
            while current:
                # If current has a left child, attach it to the previous child's next
                if current.left:
                    previous_child.next = current.left
                    previous_child = previous_child.next
                
                # If current has a right child, attach it to the previous child's next
                if current.right:
                    previous_child.next = current.right
                    previous_child = previous_child.next
                
                # Move to the next node in the level
                current = current.next
            
            # The dummy_root's next will be the start of the next level
            current = dummy_root.next
        
        # Return the root of the tree with updated next pointers
        return root
