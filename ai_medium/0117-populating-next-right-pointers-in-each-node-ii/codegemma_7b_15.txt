class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return root
        
        queue = [root]
        
        while queue:
            level_size = len(queue)
            prev = None
            
            for _ in range(level_size):
                node = queue.pop(0)
                node.next = prev
                prev = node
                
                if node.right:
                    queue.append(node.right)
                if node.left:
                    queue.append(node.left)
    
        return root
