"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        dummy = Node(0)
        pre = dummy
        
        while root:
            while root: 
                if root.left: 
                    pre.next = root.left
                    pre = pre.next
                if root.right: 
                    pre.next = root.right
                    pre = pre.next
                root = root.next
            
            root = dummy.next
            dummy.next = None
            pre = dummy
            
        return root
