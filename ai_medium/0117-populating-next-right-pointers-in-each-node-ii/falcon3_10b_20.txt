# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None

        queue = [root]

        while queue:
            for i in range(len(queue)):
                node = queue.pop(0)

                # Set the next pointer for the current node
                if i < len(queue) - 1:
                    node.next = queue[0]
                else:
                    node.next = None

                # Add the children to the queue for the next level
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)

        return root
