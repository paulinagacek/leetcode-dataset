class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return root

        queue = [root]
        while queue:
            level_len = len(queue)
            for i in range(level_len):
                node = queue.pop(0)

                if i < level_len - 1:
                    node.next = queue[0]

                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)

        return root
