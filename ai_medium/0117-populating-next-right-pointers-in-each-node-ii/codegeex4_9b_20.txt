"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        # Initialize the current node as the root of the tree
        cur = root
        # Dummy node to act as a placeholder for the next level
        dummy = Node(0)
        
        while cur:
            # Pointer to track nodes at the current level
            pre = dummy
            
            # Traverse through the tree at the current level
            while cur:
                if cur.left:
                    pre.next = cur.left
                    pre = pre.next
                if cur.right:
                    pre.next = cur.right
                    pre = pre.next
                cur = cur.next
            
            # Move to the next level
            cur = dummy.next
            # Reset dummy node for the next level
            dummy.next = None
        
        return root
