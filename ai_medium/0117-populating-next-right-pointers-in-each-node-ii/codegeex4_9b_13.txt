class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        # Initialize the leftmost node at the current level
        leftmost = root
        while leftmost:
            # Initialize the temporary pointers for each level
            curr = leftmost
            # Nullify the leftmost pointer for the next level
            leftmost = None
            
            # Iterate through each level
            while curr:

                curr = curr.next
        
        return root
