"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None

        # Helper function to connect nodes at current level
        def connect_level(cur):
            if cur is None:
                return None
            left_most = cur
            while left_most:
                if left_most.left:
                    cur = left_most
                    while cur:
                        cur.left.next = cur.right
                        if cur.next:
                            cur.right.next = cur.next.left
                        cur = cur.next
                    left_most = left_most.left
                else:
                    left_most = left_most.next

        connect_level(root)
        return root
