class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        queue = [(root, 0)]
        prev_node = None
        prev_depth = None
        
        while queue:
            node, depth = queue.pop(0)
            
            if prev_depth is not None and prev_depth == depth:
                prev_node.next = node
            
            prev_depth = depth
            prev_node = node
            
            if node.left:
                queue.append((node.left, depth + 1))
            if node.right:
                queue.append((node.right, depth + 1))
        
        return root
