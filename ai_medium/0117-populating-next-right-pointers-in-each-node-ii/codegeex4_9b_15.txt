"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Node') -> 'Node':
        if not root:
            return None
        
        # Initialize the current node and level start node
        cur = root
        nxt_start = None
        prev = None
        
        while cur:
            # Process each node on the current level
            while cur:
                if cur.left:
                    if prev:
                        prev.next = cur.left
                    else:
                        nxt_start = cur.left
                    prev = cur.left
                    
                if cur.right:
                    if prev:
                        prev.next = cur.right
                    else:
                        nxt_start = cur.right
                    prev = cur.right
                
                # Move to the next node on the current level
                cur = cur.next
            
            # Move to the next level
            cur = nxt_start
            nxt_start = None
            prev = None
        
        return root
