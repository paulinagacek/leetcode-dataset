class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        parts: List[str] = []

        def backtrack(i: int, k: int) -> None:
            if k == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            remaining = n - i
            slots = 4 - k
            if remaining < slots or remaining > 3 * slots:
                return

            for length in range(1, 4):
                if i + length > n:
                    break
                if length > 1 and s[i] == "0":
                    break
                seg = s[i:i + length]
                if int(seg) > 255:
                    continue
                parts.append(seg)
                backtrack(i + length, k + 1)
                parts.pop()

        backtrack(0, 0)
        return res
