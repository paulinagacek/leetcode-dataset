class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []

        def valid(part: str) -> bool:
            if not part:
                return False
            if len(part) > 1 and part[0] == '0':
                return False
            if len(part) > 3:
                return False
            return int(part) <= 255

        def backtrack(i: int, parts: List[str]) -> None:
            remaining_parts = 4 - len(parts)
            remaining_chars = n - i
            if remaining_chars < remaining_parts or remaining_chars > 3 * remaining_parts:
                return

            if len(parts) == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                part = s[i:i + l]
                if valid(part):
                    parts.append(part)
                    backtrack(i + l, parts)
                    parts.pop()

        backtrack(0, [])
        return res
