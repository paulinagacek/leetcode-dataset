class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        def valid(part: str) -> bool:
            if not part:
                return False
            if len(part) > 1 and part[0] == '0':
                return False
            return int(part) <= 255

        def dfs(i: int, k: int, path: List[str]) -> None:
            if k == 4:
                if i == n:
                    res.append(".".join(path))
                return

            remaining = n - i
            min_need = (4 - k) * 1
            max_need = (4 - k) * 3
            if remaining < min_need or remaining > max_need:
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                part = s[i:i + l]
                if valid(part):
                    path.append(part)
                    dfs(i + l, k + 1, path)
                    path.pop()

        dfs(0, 0, [])
        return res
