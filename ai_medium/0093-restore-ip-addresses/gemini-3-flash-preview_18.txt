class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        res = []
        n = len(s)
        
        # Optimization: A valid IP has 4 parts, each 1-3 digits.
        # Min length: 4, Max length: 12.
        if n < 4 or n > 12:
            return res

        def backtrack(start_idx: int, path: List[str]):
            # If we have 4 segments, check if we've used all characters in s
            if len(path) == 4:
                if start_idx == n:
                    res.append(".".join(path))
                return
            
            # Try segments of length 1, 2, and 3
            for length in range(1, 4):
                if start_idx + length > n:
                    break
                
                segment = s[start_idx : start_idx + length]
                
                # Check for leading zeros: segment cannot be "0x" or "0xx"
                if len(segment) > 1 and segment[0] == '0':
                    continue
                
                # Check if segment value is within [0, 255]
                if int(segment) <= 255:
                    path.append(segment)
                    backtrack(start_idx + length, path)
                    path.pop()

        backtrack(0, [])
        return res
