class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        def backtrack(i: int, parts: List[str]) -> None:
            if len(parts) == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            remaining_parts = 4 - len(parts)
            remaining_digits = n - i
            if remaining_digits < remaining_parts or remaining_digits > 3 * remaining_parts:
                return

            for length in range(1, 4):
                if i + length > n:
                    break
                if length > 1 and s[i] == '0':
                    break
                seg = s[i:i + length]
                if int(seg) <= 255:
                    parts.append(seg)
                    backtrack(i + length, parts)
                    parts.pop()

        backtrack(0, [])
        return res
