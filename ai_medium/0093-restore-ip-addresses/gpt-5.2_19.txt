class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        def valid(part: str) -> bool:
            if not part:
                return False
            if len(part) > 1 and part[0] == '0':
                return False
            if len(part) > 3:
                return False
            return int(part) <= 255

        def backtrack(i: int, k: int, parts: List[str]) -> None:
            if k == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            remaining = n - i
            if remaining < (4 - k) or remaining > 3 * (4 - k):
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                part = s[i:i + l]
                if valid(part):
                    parts.append(part)
                    backtrack(i + l, k + 1, parts)
                    parts.pop()

        backtrack(0, 0, [])
        return res
