class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        parts: List[str] = []

        def dfs(i: int) -> None:
            remaining_parts = 4 - len(parts)
            remaining_chars = n - i
            if remaining_chars < remaining_parts or remaining_chars > 3 * remaining_parts:
                return

            if len(parts) == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                if l > 1 and s[i] == '0':
                    break
                seg = s[i:i + l]
                if int(seg) > 255:
                    continue
                parts.append(seg)
                dfs(i + l)
                parts.pop()

        dfs(0)
        return res
