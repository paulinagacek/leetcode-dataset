class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        def backtrack(index, dotCount, ipAddress):
            if dotCount == 4 and index == len(s):
                result.append(".".join(ipAddress))
                return

            for i in range(1, 4):
                if index + i <= len(s) and int(s[index:index + i]) <= 255 and (s[index] != "0" or i > 1):
                    ipAddress.append(s[index:index + i])
                    backtrack(index + i, dotCount + 1, ipAddress)
                    ipAddress.pop()

        result = []
        backtrack(0, 0, [])
        return result
