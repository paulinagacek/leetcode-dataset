class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        parts = [""] * 4

        def valid(seg: str) -> bool:
            if not seg:
                return False
            if len(seg) > 1 and seg[0] == '0':
                return False
            if len(seg) > 3:
                return False
            return int(seg) <= 255

        def dfs(i: int, k: int) -> None:
            remaining = n - i
            need = 4 - k
            if remaining < need or remaining > 3 * need:
                return
            if k == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                seg = s[i:i + l]
                if valid(seg):
                    parts[k] = seg
                    dfs(i + l, k + 1)

        dfs(0, 0)
        return res
