class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        def valid(seg: str) -> bool:
            if not seg:
                return False
            if len(seg) > 1 and seg[0] == '0':
                return False
            if len(seg) > 3:
                return False
            return int(seg) <= 255

        def backtrack(i: int, parts: List[str]) -> None:
            if len(parts) == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            remaining_parts = 4 - len(parts)
            remaining_chars = n - i
            if remaining_chars < remaining_parts or remaining_chars > remaining_parts * 3:
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                seg = s[i:i + l]
                if valid(seg):
                    parts.append(seg)
                    backtrack(i + l, parts)
                    parts.pop()

        backtrack(0, [])
        return res
