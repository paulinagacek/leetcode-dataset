from typing import List

class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        def is_valid(part):
            if len(part) > 1 and part[0] == '0':
                return False
            return 0 <= int(part) <= 255
        
        def backtrack(index, path):
            if index == len(s):
                if len(path) == 4:
                    yield '.'.join(path)
                return
            for i in range(1, 4):
                if index + i > len(s):
                    break
                part = s[index:index+i]
                if is_valid(part):
                    yield from backtrack(index+i, path + [part])
        
        return list(backtrack(0, []))
