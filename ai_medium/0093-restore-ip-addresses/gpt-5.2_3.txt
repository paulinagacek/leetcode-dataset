class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        parts: List[str] = []

        def valid_part(start: int, end: int) -> bool:
            # s[start:end] (end exclusive)
            if end - start > 1 and s[start] == '0':
                return False
            val = int(s[start:end])
            return val <= 255

        def backtrack(i: int, k: int) -> None:
            if k == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            remaining = 4 - k
            rem_len = n - i
            if rem_len < remaining or rem_len > 3 * remaining:
                return

            for length in range(1, 4):
                j = i + length
                if j > n:
                    break
                if valid_part(i, j):
                    parts.append(s[i:j])
                    backtrack(j, k + 1)
                    parts.pop()

        backtrack(0, 0)
        return res
