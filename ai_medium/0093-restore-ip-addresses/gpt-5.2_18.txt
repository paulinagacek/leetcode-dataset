class Solution:
    def restoreIpAddresses(self, s: str) -> List[str]:
        n = len(s)
        res: List[str] = []
        if n < 4 or n > 12:
            return res

        def valid(seg: str) -> bool:
            if not seg:
                return False
            if len(seg) > 1 and seg[0] == '0':
                return False
            val = int(seg)
            return 0 <= val <= 255

        def backtrack(i: int, parts: List[str]) -> None:
            parts_left = 4 - len(parts)
            remaining = n - i
            if remaining < parts_left or remaining > 3 * parts_left:
                return
            if len(parts) == 4:
                if i == n:
                    res.append(".".join(parts))
                return

            for l in range(1, 4):
                if i + l > n:
                    break
                seg = s[i:i + l]
                if valid(seg):
                    parts.append(seg)
                    backtrack(i + l, parts)
                    parts.pop()

        backtrack(0, [])
        return res
