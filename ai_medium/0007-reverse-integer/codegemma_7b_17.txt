class Solution:
    def reverse(self, x: int) -> int:
        min_int = -2**31
        max_int = 2**31 - 1
        reversed_x = 0
        while x:
            digit = x % 10
            reversed_x = reversed_x * 10 + digit
            x //= 10
            if reversed_x > max_int or (reversed_x == max_int and digit > 7):
                return 0
            if reversed_x < min_int or (reversed_x == min_int and digit < -8):
                return 0
        return reversed_x
