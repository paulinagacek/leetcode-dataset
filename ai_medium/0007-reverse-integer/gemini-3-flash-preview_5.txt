class Solution:
    def reverse(self, x: int) -> int:
        MAX_INT = 2**31 - 1
        MIN_INT = -2**31
        
        sign = -1 if x < 0 else 1
        x = abs(x)
        res = 0
        
        while x != 0:
            pop = x % 10
            x //= 10
            
            # Check for overflow before multiplying by 10
            # res * 10 + pop > MAX_INT
            if res > (MAX_INT - pop) // 10:
                return 0
                
            res = res * 10 + pop
            
        return res * sign
