class Solution:
    def reverse(self, x: int) -> int:
        sign = 1 if x >= 0 else -1
        res = 0
        limit = 2**31 - 1
        while x != 0:
            pop = x % 10
            x //= 10
            if res > (limit - pop) // 10:
                return 0
            res = res * 10 + pop
        return sign * res
