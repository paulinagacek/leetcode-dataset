class Solution:
    def reverse(self, x: int) -> int:
        INT_MAX = 2**31 - 1
        INT_MIN = -2**31
        str_x = str(x)
        if str_x[0] == '-':
            reversed_str_x = '-' + str_x[:0:-1]
        else:
            reversed_str_x = str_x[::-1]
        
        reversed_x = int(reversed_str_x)
        if reversed_x < INT_MIN or reversed_x > INT_MAX:
            return 0
        return reversed_x
