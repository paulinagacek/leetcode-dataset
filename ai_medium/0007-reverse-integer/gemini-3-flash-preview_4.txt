class Solution:
    def reverse(self, x: int) -> int:
        res = 0
        while x != 0:
            if x >= 0:
                pop = x % 10
                x //= 10
            else:
                pop = x % -10
                x = (x - pop) // 10
            
            if res > 214748364 or (res == 214748364 and pop > 7):
                return 0
            if res < -214748364 or (res == -214748364 and pop < -8):
                return 0
            
            res = res * 10 + pop
            
        return res
