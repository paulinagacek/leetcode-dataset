class Solution:
    def reverse(self, x: int) -> int:
       INT_MAX = 2**31 - 1
        INT_MIN = -2**31
        if x < 0:
            reverse_x = -int(str(abs(x))[::-1])
        else:
            reverse_x = int(str(x)[::-1])
        if reverse_x > INT_MAX or reverse_x < INT_MIN:
            return 0
        return reverse_x
