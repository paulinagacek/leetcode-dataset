class Solution:
    def reverse(self, x: int) -> int:
        # Determine the sign of the integer
        sign = -1 if x < 0 else 1
        # Reverse the string representation of the absolute value
        reversed_val = int(str(abs(x))[::-1])
        # Restore the sign
        res = sign * reversed_val
        
        # Check if the reversed integer is within the signed 32-bit range
        # -2^31 = -2147483648, 2^31 - 1 = 2147483647
        if res < -2**31 or res > 2**31 - 1:
            return 0
        
        return res
