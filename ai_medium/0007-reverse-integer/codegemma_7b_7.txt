class Solution:
    def reverse(self, x: int) -> int:
        negative = x < 0
        x = abs(x)
        reversed_digits = 0
        
        while x > 0:
            digit = x % 10
            reversed_digits = reversed_digits * 10 + digit
            x //= 10
        
        if reversed_digits > 2**31 - 1:
            return 0
        
        return -reversed_digits if negative else reversed_digits
