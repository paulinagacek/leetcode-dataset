class Solution:
    def reverse(self, x: int) -> int:
        rev = 0
        sign = 1 if x >= 0 else -1
        x = abs(x)
        while x:
            digit = x % 10
            if rev > (sys.maxsize // 10) or (rev == (sys.maxsize // 10) and digit > 7):
                return 0
            rev = rev * 10 + digit
            x //= 10
        return sign * rev
