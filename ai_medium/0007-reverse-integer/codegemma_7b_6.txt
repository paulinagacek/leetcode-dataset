class Solution:
    def reverse(self, x: int) -> int:
        is_negative = x < 0
        x = abs(x)
        reversed_number = 0

        while x:
            digit = x % 10
            reversed_number = reversed_number * 10 + digit
            x //= 10

        reversed_number = -reversed_number if is_negative else reversed_number

        if reversed_number < -2**31 or reversed_number > 2**31 - 1:
            return 0

        return reversed_number
