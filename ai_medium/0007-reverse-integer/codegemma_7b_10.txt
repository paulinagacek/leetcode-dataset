class Solution:
    def reverse(self, x: int) -> int:
        reversed_num = 0
        sign = 1 if x >= 0 else -1
        x *= sign
        while x:
            reversed_num = reversed_num * 10 + x % 10
            x //= 10
        reversed_num *= sign
        return reversed_num if reversed_num <= 2**31 - 1 else 0
