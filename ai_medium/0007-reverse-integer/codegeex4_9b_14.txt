class Solution:
    def reverse(self, x):
        if x < 0:
            reversed_num = int(str(-x)[::-1]) * -1
        else:
            reversed_num = int(str(x)[::-1])
        
        if reversed_num >= -(2**31) and reversed_num <= (2**31 - 1):
            return reversed_num
        else:
            return 0
