class Solution:
    def reverse(self, x: int) -> int:
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        res = 0
        sign = -1 if x < 0 else 1
        x *= sign

        while x:
            pop = x % 10
            x //= 10

            if res > INT_MAX // 10 or (res == INT_MAX // 10 and pop > 7):
                return 0

            res = res * 10 + pop

        res *= sign
        if res < INT_MIN or res > INT_MAX:
            return 0
        return res
