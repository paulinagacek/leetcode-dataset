class Solution:
    def reverse(self, x: int) -> int:
        rev = 0
        sign = 1 if x > 0 else -1
        x *= sign
        while x > 0:
            digit = x % 10
            rev = rev * 10 + digit
            x //= 10
        rev *= sign
        if -2**31 <= rev <= 2**31 - 1:
            return rev
        else:
            return 0
