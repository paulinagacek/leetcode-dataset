class Solution:
    def reverse(self, x: int) -> int:
        # Define the limits for a signed 32-bit integer
        MAX_INT = 2147483647
        
        res = 0
        # Determine the sign and work with the absolute value
        negative = x < 0
        x = abs(x)
        
        while x != 0:
            # Extract the last digit
            pop = x % 10
            x //= 10
            
            # Check for overflow before performing the operation
            # Since Python handles arbitrary precision, we simulate 
            # 32-bit constraints by checking if res exceeds 2147483647
            if res > MAX_INT // 10 or (res == MAX_INT // 10 and pop > 7):
                return 0
            
            # Build the reversed integer
            res = res * 10 + pop
            
        return -res if negative else res
