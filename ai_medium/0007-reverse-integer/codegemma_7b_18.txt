class Solution:
    def reverse(self, x: int) -> int:
        flag = 1 if x > 0 else -1
        x = abs(x)
        reverted = 0
        while x:
            reverted = reverted * 10 + x % 10
            x //= 10
        
        if reverted > (1 << 31) - 1:
            return 0
        
        return flag * reverted
