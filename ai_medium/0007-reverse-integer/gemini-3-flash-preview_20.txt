class Solution:
    def reverse(self, x: int) -> int:
        # Define the limits for a signed 32-bit integer
        MIN_INT = -2147483648
        MAX_INT = 2147483647
        
        # Determine the sign and work with the absolute value
        sign = -1 if x < 0 else 1
        x = abs(x)
        
        # Reverse the digits mathematically
        res = 0
        while x != 0:
            res = res * 10 + (x % 10)
            x //= 10
        
        # Apply the sign to the reversed result
        res *= sign
        
        # Check if the reversed integer is within the 32-bit range
        if res < MIN_INT or res > MAX_INT:
            return 0
            
        return res
