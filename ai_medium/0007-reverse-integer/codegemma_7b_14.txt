class Solution:
    def reverse(self, x: int) -> int:
        flag = 1
        if x < 0:
            flag = -1
            x *= -1
        result = 0
        while x:
            temp = x % 10
            result = result * 10 + temp
            x //= 10
        result *= flag
        if result < -2**31 or result > 2**31 - 1:
            return 0
        return result
