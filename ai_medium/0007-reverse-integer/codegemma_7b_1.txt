class Solution:
    def reverse(self, x: int) -> int:
        is_negative = x < 0
        x = abs(x)
        reversed_digits = []
        while x > 0:
            reversed_digits.append(x % 10)
            x //= 10

        reversed_number = 0
        multiplier = 1
        for digit in reversed_digits[::-1]:
            reversed_number += multiplier * digit
            multiplier *= 10

        if reversed_number > 2**31 - 1:
            return 0

        return -reversed_number if is_negative else reversed_number
