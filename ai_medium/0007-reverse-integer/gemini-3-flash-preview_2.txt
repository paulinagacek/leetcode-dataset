class Solution:
    def reverse(self, x: int) -> int:
        # Define the limits for a signed 32-bit integer
        MAX_INT = 2147483647  # 2^31 - 1
        MIN_INT = -2147483648 # -2^31
        
        res = 0
        sign = -1 if x < 0 else 1
        x = abs(x)
        
        while x != 0:
            # Extract the last digit
            pop = x % 10
            x //= 10
            
            # Check for potential overflow before performing the calculation.
            # We want to check if res * 10 + pop > MAX_INT.
            # To avoid storing a value that exceeds 32 bits, we rearrange the check.
            if res > (MAX_INT - pop) // 10:
                # If the input x is a valid signed 32-bit integer, its reversed
                # value can only exceed MAX_INT if it's strictly larger than 2^31-1.
                # The case where the result is exactly 2147483648 is only possible
                # if the original number was -8463847412, which is invalid input.
                return 0
            
            res = res * 10 + pop
            
        return res * sign
