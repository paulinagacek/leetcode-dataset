class Solution:
    def reverse(self, x: int) -> int:
        if x < 0:
            signal = -1
        else:
            signal = 1
        
        x_str = str(abs(x))
        reversed_x_str = x_str[::-1]
        
        reversed_x = int(reversed_x_str)
        
        if reversed_x > 2**31 - 1 or reversed_x < -(2**31):
            return 0
        else:
            return signal * reversed_x
