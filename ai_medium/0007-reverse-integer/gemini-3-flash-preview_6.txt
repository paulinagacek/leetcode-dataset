class Solution:
    def reverse(self, x: int) -> int:
        # Determine the sign and work with the absolute value
        sign = -1 if x < 0 else 1
        x = abs(x)
        
        res = 0
        while x:
            # Extract the last digit and move it to the reversed result
            res = res * 10 + (x % 10)
            x //= 10
        
        # Restore the sign
        res *= sign
        
        # Check if the reversed integer is within the 32-bit signed integer range
        # Signed 32-bit range: [-2^31, 2^31 - 1]
        if res < -2**31 or res > 2**31 - 1:
            return 0
        
        return res
