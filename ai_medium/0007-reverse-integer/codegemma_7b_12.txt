class Solution:
    def reverse(self, x: int) -> int:
        rev = 0
        while x:
            pop = x % 10
            x //= 10
            if rev > (2**31 - 1) // 10 or (rev == (2**31 - 1) // 10 and pop > 7):
                return 0
            if rev < -(2**31) // 10 or (rev == -(2**31) // 10 and pop < -8):
                return 0
            rev = rev * 10 + pop
        return rev
