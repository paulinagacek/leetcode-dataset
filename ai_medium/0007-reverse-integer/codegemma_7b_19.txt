class Solution:
    def reverse(self, x: int) -> int:
        res = 0
        while x:
            digit = x % 10
            if res > (2**31 - 1) // 10 or (res == (2**31 - 1) // 10 and digit > 7):
                return 0
            if res < -(2**31) // 10 or (res == -(2**31) // 10 and digit < -8):
                return 0
            res = res * 10 + digit
            x //= 10
        return res
