class Solution:
    def reverse(self, x: int) -> int:
        rev = 0
        sign = -1 if x < 0 else 1
        x = abs(x)

        while x != 0:
            pop = x % 10
            x //= 10
            reversed_num = rev * 10 + pop

            # Check for overflow
            if reversed_num < -2**31 or reversed_num > 2**31 - 1:
                return 0
            
            rev = reversed_num
        
        return sign * rev
