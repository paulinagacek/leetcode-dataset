class Solution:
    def reverse(self, x: int) -> int:
        sign = -1 if x < 0 else 1
        x = abs(x)
        rev = 0
        while x:
            pop = x % 10
            x //= 10
            rev = rev * 10 + pop
        rev *= sign
        return 0 if rev < -2**31 or rev > 2**31 - 1 else rev
