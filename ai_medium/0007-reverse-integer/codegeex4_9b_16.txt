class Solution:
    def reverse(self, x: int) -> int:
        if x >= 0:
            str_x = str(x)
            reversed_str_x = str_x[::-1]
            reversed_x = int(reversed_str_x)
        else:
            str_x = str(abs(x))
            reversed_str_x = str_x[::-1]
            reversed_x = -int(reversed_str_x)
        
        if reversed_x >= -(2**31) and reversed_x <= (2**31)-1:
            return reversed_x
        else:
            return 0
