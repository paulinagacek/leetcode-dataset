class Solution:
    def reverse(self, x: int) -> int:
        reversed_x = 0
        sign = 1 if x >= 0 else -1
        x = abs(x)
        while x > 0:
            reversed_x = reversed_x * 10 + x % 10
            x //= 10
        reversed_x *= sign
        return reversed_x if -2**31 <= reversed_x <= 2**31 - 1 else 0
