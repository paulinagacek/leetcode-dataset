class Solution:
    def reverse(self, x: int) -> int:
        sign = -1 if x < 0 else 1
        x = abs(x)
        rev = 0
        while x:
            pop = x % 10
            x //= 10
            new_rev = rev * 10 + pop
            # Check for overflow/underflow
            if new_rev > 2**31 - 1 or new_rev < -2**31:
                return 0
            rev = new_rev
        return sign * rev
