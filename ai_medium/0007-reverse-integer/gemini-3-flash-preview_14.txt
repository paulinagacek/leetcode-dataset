class Solution:
    def reverse(self, x: int) -> int:
        MAX_INT = 2**31 - 1
        MIN_INT = -2**31
        
        # Handle sign
        sign = 1 if x >= 0 else -1
        x = abs(x)
        
        # Reverse using string manipulation
        reversed_x = int(str(x)[::-1]) * sign
        
        # Check for 32-bit signed integer overflow
        if reversed_x < MIN_INT or reversed_x > MAX_INT:
            return 0
            
        return reversed_x
