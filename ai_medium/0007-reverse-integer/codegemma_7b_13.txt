class Solution:
    def reverse(self, x: int) -> int:
        reversed_num = 0
        sign = 1 if x >= 0 else -1
        x = abs(x)
        
        while x:
            digit = x % 10
            reversed_num = reversed_num * 10 + digit
            x //= 10
            
        reversed_num *= sign
        
        if reversed_num >= -2**31 and reversed_num <= 2**31 - 1:
            return reversed_num
        else:
            return 0
