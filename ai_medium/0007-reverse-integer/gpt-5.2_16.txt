class Solution:
    def reverse(self, x: int) -> int:
        INT_MIN, INT_MAX = -2**31, 2**31 - 1
        res = 0
        sign = -1 if x < 0 else 1
        x = -x if x < 0 else x

        while x:
            digit = x % 10
            x //= 10

            if res > INT_MAX // 10 or (res == INT_MAX // 10 and digit > 7):
                return 0

            res = res * 10 + digit

        res *= sign
        return res if INT_MIN <= res <= INT_MAX else 0
