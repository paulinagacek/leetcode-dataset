class Solution:
    def reverse(self, x: int) -> int:
        MAX_INT = 2147483647
        MIN_INT = -2147483648
        
        res = 0
        while x != 0:
            # Handle Python's modulo and division behavior for negative numbers
            if x > 0:
                pop = x % 10
                x //= 10
            else:
                pop = x % -10
                x = int(x / 10)
            
            # Check for overflow/underflow before updating res
            # MAX_INT // 10 is 214748364, MIN_INT // 10 (int division) is -214748364
            if res > 214748364 or (res == 214748364 and pop > 7):
                return 0
            if res < -214748364 or (res == -214748364 and pop < -8):
                return 0
            
            res = (res * 10) + pop
            
        return res
