class Solution:
    def reverse(self, x: int) -> int:
        if x < 0:
            signal = -1
            x = abs(x)
        else:
            signal = 1
        
        reversed_num = int(str(x)[::-1])
        
        if reversed_num > 2**31 - 1 or reversed_num < -(2**31):
            return 0
        else:
            return reversed_num * signal
