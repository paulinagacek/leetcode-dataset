from typing import List

class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        res = []
        self.helper(nums, 0, [], res)
        return res
    
    def helper(self, nums, index, subset, res):
        res.append(subset[:])
        for i in range(index, len(nums)):
            self.helper(nums, i + 1, subset + [nums[i]], res)
