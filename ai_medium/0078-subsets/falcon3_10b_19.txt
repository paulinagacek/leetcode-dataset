class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        result = []
        self.dfs(sorted(nums), 0, [], result)
        return result
        
    def dfs(self, nums, index, path, result):
        result.append(path[:])
        for i in range(index, len(nums)):
            path.append(nums[i])
            self.dfs(nums, i + 1, path, result)
            path.pop()
