class MinStack:

    def __init__(self):
        # Initialize a stack where each element is a tuple (value, current_minimum)
        self.stack = []

    def push(self, val: int) -> None:
        # If the stack is empty, the current value is also the minimum
        if not self.stack:
            self.stack.append((val, val))
        else:
            # Otherwise, calculate the minimum between the new value and the current minimum
            current_min = self.stack[-1][1]
            self.stack.append((val, min(val, current_min)))

    def pop(self) -> None:
        # Standard pop operation from the list
        self.stack.pop()

    def top(self) -> int:
        # Return the value of the top element (first element of the tuple)
        return self.stack[-1][0]

    def getMin(self) -> int:
        # Return the current minimum (second element of the tuple)
        return self.stack[-1][1]
