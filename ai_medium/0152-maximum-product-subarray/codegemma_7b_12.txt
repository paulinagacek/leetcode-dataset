class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_product = nums[0]
        min_product = nums[0]
        result = nums[0]

        for i in range(1, len(nums)):
            curr = nums[i]

            temp_max_product = max_product
            temp_min_product = min_product

            max_product = max(temp_max_product * curr, temp_min_product * curr, curr)
            min_product = min(temp_max_product * curr, temp_min_product * curr, curr)

            result = max(result, max_product)

        return result
