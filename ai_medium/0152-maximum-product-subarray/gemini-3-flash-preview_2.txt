class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        # Initialize the result, current maximum product, and current minimum product
        # with the first element of the array.
        res = cur_max = cur_min = nums[0]
        
        # Iterate through the array starting from the second element.
        for i in range(1, len(nums)):
            num = nums[i]
            
            # If the current number is negative, the potential maximum and minimum 
            # products ending at this position are swapped.
            if num < 0:
                cur_max, cur_min = cur_min, cur_max
            
            # Update the current maximum and minimum product ending at the current position.
            # cur_max tracks the highest product, and cur_min tracks the lowest product.
            cur_max = max(num, cur_max * num)
            cur_min = min(num, cur_min * num)
            
            # Update the overall maximum product found so far.
            res = max(res, cur_max)
            
        return res
