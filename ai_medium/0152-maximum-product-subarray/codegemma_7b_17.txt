class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_prod = nums[0]
        min_prod = nums[0]
        result = nums[0]

        for i in range(1, len(nums)):
            curr = nums[i]

            tmp_max = max_prod
            tmp_min = min_prod

            max_prod = max(curr, max_prod * curr, min_prod * curr)
            min_prod = min(curr, tmp_max * curr, tmp_min * curr)

            result = max(result, max_prod)

        return result
