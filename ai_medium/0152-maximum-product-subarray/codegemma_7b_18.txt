class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_positive = nums[0]
        max_negative = nums[0]
        result = nums[0]

        for num in nums[1:]:
            temp_max_positive = max(max_positive*num, max_negative*num, num)
            max_negative = min(max_positive*num, max_negative*num, num)
            max_positive = temp_max_positive
            result = max(result, max_positive)

        return result
