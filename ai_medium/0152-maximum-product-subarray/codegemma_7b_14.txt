class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_positive = nums[0]
        max_negative = nums[0]
        max_product = nums[0]

        for num in nums[1:]:
            temp_max_positive = max(max_positive * num, max_negative * num, num)
            temp_max_negative = min(max_positive * num, max_negative * num, num)

            max_positive = temp_max_positive
            max_negative = temp_max_negative

            max_product = max(max_product, max_positive)

        return max_product
