from typing import List

class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        max_product = nums[0]
        min_product = nums[0]
        result = nums[0]
        
        for i in range(1, len(nums)):
            temp_max = max(max_product * nums[i], nums[i], min_product * nums[i])
            temp_min = min(max_product * nums[i], nums[i], min_product * nums[i])
            
            max_product = temp_max
            min_product = temp_min
            
            result = max(result, max_product)
        
        return result
