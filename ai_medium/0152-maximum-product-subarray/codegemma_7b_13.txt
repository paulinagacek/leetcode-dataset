class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        min_val = nums[0]
        max_val = nums[0]
        result = nums[0]

        for i in range(1, len(nums)):
            temp_max = max_val
            temp_min = min_val

            max_val = max(temp_max * nums[i], temp_min * nums[i], nums[i])
            min_val = min(temp_max * nums[i], temp_min * nums[i], nums[i])

            result = max(result, max_val)

        return result
