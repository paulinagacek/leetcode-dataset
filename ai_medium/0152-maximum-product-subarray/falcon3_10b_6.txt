class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_product = nums[0]
        min_product = nums[0]
        result = nums[0]
        
        for num in nums[1:]:
            temp = max_product
            max_product = max(num, max_product * num, min_product * num)
            min_product = min(num, temp * num, min_product * num)
            result = max(result, max_product)
        
        return result
