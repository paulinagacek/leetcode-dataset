class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        res = nums[0]
        max_so_far = nums[0]
        min_so_far = nums[0]
        
        for i in range(1, len(nums)):
            curr = nums[i]
            
            if curr < 0:
                max_so_far, min_so_far = min_so_far, max_so_far
            
            max_so_far = max(curr, max_so_far * curr)
            min_so_far = min(curr, min_so_far * curr)
            
            res = max(res, max_so_far)
            
        return res
