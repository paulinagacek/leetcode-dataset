class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_pro = nums[0]
        min_pro = nums[0]
        result = nums[0]
        for i in range(1, len(nums)):
            tmp_max = max(max_pro*nums[i], min_pro*nums[i], nums[i])
            tmp_min = min(max_pro*nums[i], min_pro*nums[i], nums[i])
            max_pro, min_pro = tmp_max, tmp_min
            result = max(result, max_pro)
        return result
