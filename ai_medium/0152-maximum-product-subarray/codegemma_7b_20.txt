class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_product = nums[0]
        min_product = nums[0]
        result = nums[0]

        for i in range(1, len(nums)):
            curr = nums[i]

            temp_max = max(max_product * curr, min_product * curr, curr)
            temp_min = min(max_product * curr, min_product * curr, curr)

            max_product = temp_max
            min_product = temp_min

            result = max(result, max_product)

        return result
