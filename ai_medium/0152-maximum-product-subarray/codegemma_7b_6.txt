class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_positive = nums[0]
        min_negative = nums[0]
        max_product = nums[0]

        for num in nums[1:]:
            temp_max_positive = max(max_positive * num, min_negative * num, num)
            min_negative = min(max_positive * num, min_negative * num, num)
            max_product = max(max_product, temp_max_positive)

        return max_product
