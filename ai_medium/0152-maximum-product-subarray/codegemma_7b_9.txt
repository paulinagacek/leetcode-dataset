class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        min_so_far, max_so_far = nums[0], nums[0]
        result = nums[0]

        for i in range(1, len(nums)):
            current = nums[i]

            temp_min = min(min_so_far * current, max_so_far * current, current)
            temp_max = max(min_so_far * current, max_so_far * current, current)

            min_so_far, max_so_far = temp_min, temp_max

            result = max(result, max_so_far)

        return result
