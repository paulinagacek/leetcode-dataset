class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        max_prod = nums[0]
        min_prod = nums[0]
        result = nums[0]
        
        for i in range(1, len(nums)):
            temp_max = max(nums[i], max_prod * nums[i], min_prod * nums[i])
            temp_min = min(nums[i], max_prod * nums[i], min_prod * nums[i])
            max_prod = temp_max
            min_prod = temp_min
            
            result = max(result, max_prod)
        
        return result
