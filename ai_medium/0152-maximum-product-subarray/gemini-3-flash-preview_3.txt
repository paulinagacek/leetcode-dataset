class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        # Initialize the global maximum result, and the running maximum and minimum products
        # that end at the current position.
        res = nums[0]
        cur_max = nums[0]
        cur_min = nums[0]
        
        for i in range(1, len(nums)):
            n = nums[i]
            # If the current number is negative, the potential maximum and minimum
            # products swap when multiplied by this number.
            if n < 0:
                cur_max, cur_min = cur_min, cur_max
            
            # Update the running maximum and minimum by comparing the current number itself
            # with the product of the previous running maximum/minimum and the current number.
            cur_max = max(n, cur_max * n)
            cur_min = min(n, cur_min * n)
            
            # Update the global result if the current running maximum is larger.
            res = max(res, cur_max)
            
        return res
