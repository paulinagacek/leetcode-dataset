class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_end = min_end = ans = nums[0]
        for x in nums[1:]:
            if x < 0:
                max_end, min_end = min_end, max_end
            max_end = max(x, max_end * x)
            min_end = min(x, min_end * x)
            ans = max(ans, max_end)
        return ans
