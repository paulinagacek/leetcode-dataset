class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_prod, min_prod, res = nums[0], nums[0], nums[0]
        for i in range(1, len(nums)):
            tmp_max_prod, tmp_min_prod = max_prod, min_prod
            max_prod = max(tmp_max_prod * nums[i], tmp_min_prod * nums[i], nums[i])
            min_prod = min(tmp_max_prod * nums[i], tmp_min_prod * nums[i], nums[i])
            res = max(res, max_prod)
        return res
