class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_product = curr_max = curr_min = nums[0]
        for num in nums[1:]:
            temp = curr_max
            curr_max = max(num, curr_max * num, curr_min * num)
            curr_min = min(num, temp * num, curr_min * num)
            if curr_max > max_product:
                max_product = curr_max
        return max_product
