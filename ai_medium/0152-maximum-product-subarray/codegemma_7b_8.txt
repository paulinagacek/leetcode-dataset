class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        max_positive = nums[0]
        max_negative = nums[0]
        max_product = nums[0]
        
        for i in range(1, len(nums)):
            temp_max_positive = max(max_positive * nums[i], max_negative * nums[i], nums[i])
            temp_max_negative = min(max_positive * nums[i], max_negative * nums[i], nums[i])
            max_positive = temp_max_positive
            max_negative = temp_max_negative
            max_product = max(max_product, max_positive)
        return max_product
