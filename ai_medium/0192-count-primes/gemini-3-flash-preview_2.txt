class Solution:
    def countPrimes(self, n: int) -> int:
        if n < 3:
            return 0
        
        # Initialize a boolean list representing primality
        is_prime = [True] * n
        is_prime[0] = is_prime[1] = False
        
        # Sieve of Eratosthenes algorithm
        for i in range(2, int(n**0.5) + 1):
            if is_prime[i]:
                # Slice assignment is much faster in Python
                is_prime[i*i:n:i] = [False] * len(range(i*i, n, i))
        
        # Count the number of True values remaining in the list
        return is_prime.count(True)
