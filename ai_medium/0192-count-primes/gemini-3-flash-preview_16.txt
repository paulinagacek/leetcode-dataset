class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        
        primes = [True] * n
        primes[0] = primes[1] = False
        
        for p in range(2, int(n**0.5) + 1):
            if primes[p]:
                primes[p*p : n : p] = [False] * len(range(p*p, n, p))
                
        return sum(primes)
