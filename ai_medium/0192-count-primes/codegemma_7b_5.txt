class Solution:
    def countPrimes(self, n: int) -> int:
        def is_prime(num):
            if num <= 1:
                return False
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    return False
            return True

        primes = [num for num in range(n) if is_prime(num)]
        return len(primes)
