class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        
        primes = [True] * n
        primes[0] = primes[1] = False
        
        for i in range(2, int(n**0.5) + 1):
            if primes[i]:
                # Using slice assignment for faster execution in Python
                primes[i*i:n:i] = [False] * ((n - 1 - i*i) // i + 1)
        
        return sum(primes)
