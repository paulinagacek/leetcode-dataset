class Solution:
    def countPrimes(self, n: int) -> int:
        def isPrime(x):
            if x <= 1:
                return False
            for i in range(2, int(x**0.5) + 1):
                if x % i == 0:
                    return False
            return True

        count = 0
        for i in range(2, n):
            if isPrime(i):
                count += 1
        return count
