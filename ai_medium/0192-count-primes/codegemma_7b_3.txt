class Solution:
    def countPrimes(self, n: int) -> int:
        def isPrime(num):
            if num <= 1:
                return False
            for i in range(2, int(num**0.5) + 1):
                if num % i == 0:
                    return False
            return True

        count = 0
        for i in range(n):
            if isPrime(i):
                count += 1
        return count
