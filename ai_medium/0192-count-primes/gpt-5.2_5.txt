class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        is_prime = bytearray(b"\x01") * n
        is_prime[0:2] = b"\x00\x00"
        limit = int(n ** 0.5)
        for i in range(2, limit + 1):
            if is_prime[i]:
                start = i * i
                is_prime[start:n:i] = b"\x00" * (((n - 1 - start) // i) + 1)
        return sum(is_prime)
