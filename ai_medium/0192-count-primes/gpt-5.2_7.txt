class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        sieve = bytearray(b"\x01") * n
        sieve[0:2] = b"\x00\x00"
        limit = int(n ** 0.5) + 1
        for p in range(2, limit):
            if sieve[p]:
                start = p * p
                sieve[start:n:p] = b"\x00" * ((n - start - 1) // p + 1)
        return sum(sieve)
