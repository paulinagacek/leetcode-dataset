class Solution:
    def countPrimes(self, n: int) -> int:
        if n <= 2:
            return 0
        
        # Initialize a boolean array for the sieve
        is_prime = [True] * n
        is_prime[0] = is_prime[1] = False
        
        # Sieve of Eratosthenes
        for p in range(2, int(n**0.5) + 1):
            if is_prime[p]:
                # Using slice assignment for efficiency in Python
                # is_prime[p*p : n : p] = [False] * len(range(p*p, n, p))
                # Explicitly calculating the length of the slice:
                is_prime[p*p:n:p] = [False] * ((n - 1 - p*p) // p + 1)
                
        return sum(is_prime)
