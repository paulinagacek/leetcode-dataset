from typing import List

class Solution:
    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        if len(nums) == 2:
            return max(nums)
        
        def rob_linear(houses: List[int]) -> int:
            prev, curr = 0, 0
            for house in houses:
                prev, curr = curr, max(curr, prev + house)
            return curr
        
        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
