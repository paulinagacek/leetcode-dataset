from typing import List

class Solution:
    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        def rob_linear(nums: List[int]) -> int:
            prev, curr = 0, 0
            for num in nums:
                prev, curr = curr, max(num + prev, curr)
            return curr
        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
