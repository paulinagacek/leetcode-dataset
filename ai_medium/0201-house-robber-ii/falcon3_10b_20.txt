class Solution:
    def rob(self, nums: List[int]) -> int:
        def rob_linear(houses):
            dp_current = dp_previous = 0
            for house in houses:
                dp_current, dp_previous = max(dp_current, dp_previous + house), dp_current
            return dp_current

        if not nums:
            return 0
        elif len(nums) == 1:
            return nums[0]
        else:
            return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
