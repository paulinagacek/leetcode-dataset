class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums: return 0
        def rob_linear(houses):
            prev1, prev2 = 0, 0
            for num in houses:
                temp = max(prev2, prev1 + num)
                prev2, prev1 = prev1, temp
            return prev1

        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
