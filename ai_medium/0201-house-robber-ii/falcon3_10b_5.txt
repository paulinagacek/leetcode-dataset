class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        def rob_sequence(houses):
            dp = [0] * len(houses)
            dp[0] = houses[0]
            dp[1] = max(houses[0], houses[1])
            for i in range(2, len(houses)):
                dp[i] = max(dp[i-1], dp[i-2] + houses[i])
            return dp[-1]
        
        return max(rob_sequence(nums[:-1]), rob_sequence(nums[1:]))
