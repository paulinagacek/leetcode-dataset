from typing import List

class Solution:
    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        def solve(start, end):
            prev1, prev2 = 0, 0
            for i in range(start, end):
                temp = max(prev1, nums[i] + prev2)
                prev2 = prev1
                prev1 = temp
            return prev1
        return max(solve(0, len(nums) - 1), solve(1, len(nums)))
