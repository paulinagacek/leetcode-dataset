class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums: return 0
        n = len(nums)
        def dp(houses):
            length = len(houses)
            dp = [0] * (length + 1)
            dp[1] = houses[0]
            for i in range(2, length + 1):
                dp[i] = max(dp[i - 1], dp[i - 2] + houses[i - 1])
            return dp[length]
        return max(dp(nums[:-1]), dp(nums[1:]))
