class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0

        def rob_linear(nums):
            prev1, prev2 = 0, 0
            for num in nums:
                temp = max(prev1, prev2 + num)
                prev1, prev2 = prev2, temp
            return prev2

        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
