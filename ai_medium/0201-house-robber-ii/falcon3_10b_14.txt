class Solution:
    def rob(self, nums: List[int]) -> int:
        def helper(nums: List[int]) -> int:
            prev1 = prev2 = 0
            for num in nums:
                temp = max(prev2 + num, prev1)
                prev2 = prev1
                prev1 = temp
            return prev1
        
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        
        return max(helper(nums[:-1]), helper(nums[1:]))
