from typing import List

class Solution:
    def rob(self, nums: List[int]) -> int:
        def rob_linear(houses: List[int]) -> int:
            prev1, prev2 = 0, 0
            for money in houses:
                prev1, prev2 = prev2, max(prev2, prev1 + money)
            return prev2
        
        if len(nums) == 1:
            return nums[0]
        else:
            return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
