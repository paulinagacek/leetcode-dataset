class Solution:
    def rob(self, nums: List[int]) -> int:
        n = len(nums)
        if n == 1:
            return nums[0]

        def rob_linear(arr: List[int]) -> int:
            prev2 = 0
            prev1 = 0
            for x in arr:
                cur = max(prev1, prev2 + x)
                prev2, prev1 = prev1, cur
            return prev1

        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
