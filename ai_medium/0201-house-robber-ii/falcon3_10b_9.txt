class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums: return 0
        n = len(nums)
        if n == 1: return nums[0]
        
        def helper(start, end):
            dp = [0] * (end - start + 1)
            dp[0] = nums[start]
            for i in range(1, end - start + 1):
                dp[i] = max(dp[i-1], dp[i-2] + nums[start+i])
            return dp[-1]
        
        return max(helper(0, n-2), helper(1, n-1))
