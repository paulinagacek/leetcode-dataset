class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        
        def rob_linear(houses: List[int]) -> int:
            prev2, prev1 = 0, 0
            for amount in houses:
                # The recurrence relation: f(i) = max(f(i-1), f(i-2) + nums[i])
                current = max(prev1, prev2 + amount)
                prev2 = prev1
                prev1 = current
            return prev1
        
        # Because the houses are in a circle, we cannot rob both the first and the last house.
        # We solve the problem for two scenarios:
        # 1. Excluding the last house.
        # 2. Excluding the first house.
        # The answer is the maximum of these two linear robber scenarios.
        return max(rob_linear(nums[:-1]), rob_linear(nums[1:]))
