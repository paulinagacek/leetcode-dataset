class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        if not triangle:
            return 0
        dp = triangle[-1]
        for i in range(len(triangle) - 2, -1, -1):
            temp = [0] * (i + 1)
            for j in range(i):
                temp[j] = min(dp[j], dp[j + 1]) + triangle[i][j]
            dp = temp
        return dp[0]
