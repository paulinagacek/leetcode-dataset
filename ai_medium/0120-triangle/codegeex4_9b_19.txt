from typing import List

class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        if not triangle:
            return 0
        
        # Start from the second row and update each element with the minimum path sum to reach that element
        for i in range(1, len(triangle)):
            for j in range(len(triangle[i])):
                # If we are at the first column, we can only come from the above row
                if j == 0:
                    triangle[i][j] += triangle[i - 1][j]
                # If we are at the last column, we can only come from the left of the above row
                elif j == len(triangle[i]) - 1:
                    triangle[i][j] += triangle[i - 1][j - 1]
                else:
                    # Otherwise, take the minimum path sum from either directly above or from the left above
                    triangle[i][j] += min(triangle[i - 1][j], triangle[i - 1][j - 1])
        
        # The minimum path sum to reach the bottom row is the minimum value in the last row
        return min(triangle[-1])

# Example usage:
sol = Solution()
print(sol.minimumTotal([[2],[3,4],[6,5,7],[4,1,8,3]]))  # Output: 11
print(sol.minimumTotal([[-10]]))  # Output: -10
