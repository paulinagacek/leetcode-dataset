class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        if not triangle:
            return 0
        
        # Initialize a list to store the minimum path sum at each level
        dp = triangle[0]
        
        # Iterate through each level of the triangle starting from the second row
        for i in range(1, len(triangle)):
            # Create a new list for the current level
            temp = [0] * (i + 1)
            
            # Calculate the minimum path sum for each element in the current level
            for j in range(i + 1):
                if j == 0:
                    # If it's the first element, it can only be reached from the top
                    temp[j] = dp[j] + triangle[i][j]
                elif j == i:
                    # If it's the last element, it can only be reached from the left
                    temp[j] = dp[j - 1] + triangle[i][j]
                else:
                    # Otherwise, it can be reached from either the top or the left
                    temp[j] = min(dp[j], dp[j - 1]) + triangle[i][j]
            # Update dp to be the minimum path sum at the previous level
            dp = temp
        
        # The minimum path sum is the minimum value in dp
        return min(dp)
