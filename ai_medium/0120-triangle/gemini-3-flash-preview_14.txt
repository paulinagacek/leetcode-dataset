class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        # Use the bottom row as the starting point for our DP array
        n = len(triangle)
        dp = triangle[-1][:]
        
        # Iterate from the second to last row up to the top
        for row in range(n - 2, -1, -1):
            for i in range(row + 1):
                # The minimum path sum at current position is its value plus
                # the minimum of the two adjacent positions in the row below
                dp[i] = triangle[row][i] + min(dp[i], dp[i + 1])
                
        return dp[0]
