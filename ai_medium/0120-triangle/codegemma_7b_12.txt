class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        dp = triangle[-1]
        for i in range(n - 2, -1, -1):
            temp = [0] * (i + 1)
            for j in range(i):
                temp[j] = triangle[i][j] + min(dp[j], dp[j + 1])
            dp = temp
        return dp[0]
