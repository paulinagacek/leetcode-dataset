class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        # Use the last row as the initial DP state
        # The dp array stores the minimum path sum from the current cell down to the bottom
        dp = list(triangle[-1])
        
        # Iterate from the second-to-last row up to the top row
        for r in range(n - 2, -1, -1):
            # For each element in the row, update its minimum path sum
            for i in range(r + 1):
                # The minimum path sum from triangle[r][i] is its value 
                # plus the minimum of the two adjacent values in the row below
                dp[i] = triangle[r][i] + min(dp[i], dp[i + 1])
                
        # The first element of the dp array now contains the minimum path sum from top to bottom
        return dp[0]
