class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        if not triangle:
            return 0
        
        # Use a 1D array to store the minimum path sums from each point to the bottom.
        # Initialize it with the values of the last row.
        n = len(triangle)
        res = list(triangle[-1])
        
        # Move from the second-to-last row up to the top row.
        # For each node at (i, j), the minimum path sum to the bottom is 
        # triangle[i][j] + min(min_sum_at_node_directly_below, min_sum_at_node_diagonally_right).
        for i in range(n - 2, -1, -1):
            for j in range(i + 1):
                res[j] = triangle[i][j] + min(res[j], res[j + 1])
                
        # After processing row 0, the result will be stored at res[0].
        return res[0]
