class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        # We can use a 1D array to store the minimum path sums from the current row to the bottom.
        # Initialize it with the values of the bottom row of the triangle.
        dp = list(triangle[-1])
        
        # Iterate from the second-to-last row upwards to the top.
        for r in range(n - 2, -1, -1):
            for i in range(len(triangle[r])):
                # For each element, the minimum path sum is the value of the element 
                # plus the minimum of the two adjacent path sums from the row below.
                dp[i] = triangle[r][i] + min(dp[i], dp[i + 1])
                
        # The first element of dp will contain the minimum path sum from top to bottom.
        return dp[0]
