class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        # Use the last row of the triangle as the initial DP state
        # This provides O(n) extra space
        dp = list(triangle[-1])
        
        # Iterate from the second-to-last row up to the top row
        for i in range(n - 2, -1, -1):
            # For each element, find the minimum path sum to the bottom
            # by looking at its two adjacent neighbors in the row below
            for j in range(i + 1):
                dp[j] = triangle[i][j] + min(dp[j], dp[j + 1])
        
        # The result is stored at the first index of the dp array
        return dp[0]
