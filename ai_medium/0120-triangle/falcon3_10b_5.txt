class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        dp = triangle[-1].copy()
        for layer in range(len(triangle) - 2, -1, -1):
            for i in range(len(triangle[layer])):
                dp[i] = triangle[layer][i] + min(dp[i], dp[i + 1])
        return dp[0]
