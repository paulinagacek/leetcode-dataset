class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        # Use the last row as our starting DP state
        # Space complexity: O(n)
        dp = list(triangle[-1])
        
        # Iterate from the second-to-last row up to the top
        for row in range(n - 2, -1, -1):
            for col in range(len(triangle[row])):
                # The minimum path sum at index (row, col) is the value at that 
                # position plus the minimum of the two reachable sums below it.
                dp[col] = triangle[row][col] + min(dp[col], dp[col + 1])
        
        return dp[0]
