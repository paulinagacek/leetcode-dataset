class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m = len(obstacleGrid)
        n = len(obstacleGrid[0])
        
        # dp[j] will store the number of ways to reach column j in the current row
        dp = [0] * n
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                elif j > 0:
                    # The number of ways to reach (i, j) is the sum of ways to reach
                    # (i-1, j) (which is current dp[j]) and (i, j-1) (which is dp[j-1])
                    dp[j] += dp[j - 1]
                    
        return dp[n - 1]
