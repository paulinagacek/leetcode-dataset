class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m = len(obstacleGrid)
        n = len(obstacleGrid[0])
        
        # dp array stores the number of paths to reach cells in the current row
        dp = [0] * n
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                elif j > 0:
                    # Number of paths to (i, j) = paths from (i-1, j) + paths from (i, j-1)
                    # dp[j] currently holds paths from (i-1, j)
                    # dp[j-1] currently holds paths from (i, j-1)
                    dp[j] += dp[j-1]
                    
        return dp[n-1]
