class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or not obstacleGrid[0]:
            return 0
        
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        dp = [0] * n
        
        # Initial condition for the top-left cell
        dp[0] = 1 if obstacleGrid[0][0] == 0 else 0
        
        # Iteratively calculate the number of unique paths to each cell
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                elif j > 0:
                    # Current cell unique paths = paths from above (dp[j]) 
                    # + paths from the left (dp[j-1])
                    dp[j] += dp[j - 1]
        
        return dp[n - 1]
