class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m = len(obstacleGrid)
        n = len(obstacleGrid[0])
        
        # dp[j] will store the number of ways to reach cell in column j
        # of the current row being processed.
        dp = [0] * n
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                elif j > 0:
                    # Current dp[j] is the value from the cell above (i-1, j)
                    # dp[j-1] is the value from the cell to the left (i, j-1)
                    dp[j] += dp[j - 1]
                    
        return dp[n - 1]
