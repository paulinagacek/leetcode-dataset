from typing import List

class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        dp = [[0] * n for _ in range(m)]
        
        # If the starting point or ending point is blocked, no path exists
        if obstacleGrid[0][0] == 1 or obstacleGrid[m-1][n-1] == 1:
            return 0
        
        dp[0][0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] != 1:  # If not an obstacle
                    if i > 0:
                        dp[i][j] += dp[i-1][j]  # Paths from the top cell
                    if j > 0:
                        dp[i][j] += dp[i][j-1]  # Paths from the left cell
        
        return dp[m-1][n-1]
