class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        # If the grid is empty or the starting cell is an obstacle, no paths exist.
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        
        # We use a 1D DP array to store the number of ways to reach each cell in the current row.
        # This optimizes space from O(m*n) to O(n).
        dp = [0] * n
        
        # Base case: There is 1 way to reach the starting cell if it's not an obstacle.
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                # If the current cell is an obstacle, the number of ways to reach it is 0.
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                # If it's not an obstacle and not the first column, add the ways from the left cell.
                # dp[j] currently holds the value from the top cell (previous row).
                # dp[j-1] holds the value from the left cell (current row).
                elif j > 0:
                    dp[j] += dp[j - 1]
                # If j == 0, dp[0] remains its value unless blocked by an obstacle.
        
        # The result is the value in the last cell of the DP array.
        return dp[-1]
