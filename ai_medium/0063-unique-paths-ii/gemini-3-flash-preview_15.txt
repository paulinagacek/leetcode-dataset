class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid:
            return 0
        
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        dp = [0] * n
        
        # Initialize the starting point
        if obstacleGrid[0][0] == 0:
            dp[0] = 1
            
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    # If current cell is an obstacle, no paths can pass through it
                    dp[j] = 0
                elif j > 0:
                    # Number of paths to current cell is sum of paths from above 
                    # (which is already in dp[j]) and paths from left (dp[j-1])
                    dp[j] += dp[j - 1]
                # If j == 0 and not an obstacle, dp[j] remains dp[j] from the previous row
        
        return dp[-1]
