class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m = len(obstacleGrid)
        n = len(obstacleGrid[0])
        
        # dp[j] will store the number of ways to reach cell (i, j)
        # We optimize the space from O(m*n) to O(n)
        dp = [0] * n
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    dp[j] = 0
                elif j > 0:
                    # The current cell value is the sum of paths from the top (dp[j])
                    # and the paths from the left (dp[j-1])
                    dp[j] += dp[j-1]
        
        return dp[n - 1]
