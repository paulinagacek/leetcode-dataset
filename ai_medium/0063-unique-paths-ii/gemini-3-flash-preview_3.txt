class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or obstacleGrid[0][0] == 1:
            return 0
        
        m = len(obstacleGrid)
        n = len(obstacleGrid[0])
        
        # Use a 1D DP array to save space
        dp = [0] * n
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    # If there's an obstacle, no paths can pass through this cell
                    dp[j] = 0
                elif j > 0:
                    # The number of paths to cell (i, j) is the sum of paths
                    # from the cell above (dp[j]) and the cell to the left (dp[j-1])
                    dp[j] += dp[j-1]
                # If j == 0 and obstacleGrid[i][0] == 0, dp[0] remains its value 
                # from the previous row, representing the path from (i-1, 0).
                    
        return dp[n-1]
