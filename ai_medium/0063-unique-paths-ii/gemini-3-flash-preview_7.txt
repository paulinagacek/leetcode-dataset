class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        if not obstacleGrid or not obstacleGrid[0]:
            return 0
        
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        
        # We use a 1D DP array to store the number of paths to each cell in the current row.
        # dp[j] will store the number of paths to cell (i, j).
        dp = [0] * n
        
        # Initial base case: There is 1 way to be at the starting point if it's not an obstacle.
        dp[0] = 1
        
        for i in range(m):
            for j in range(n):
                if obstacleGrid[i][j] == 1:
                    # If the current cell is an obstacle, no paths can pass through it.
                    dp[j] = 0
                elif j > 0:
                    # If it's not an obstacle, the number of paths to (i, j) is the sum 
                    # of paths from the cell above (already in dp[j]) and the cell to the left (dp[j-1]).
                    dp[j] += dp[j-1]
                # If j == 0 and the cell is not an obstacle, dp[0] remains what it was 
                # (the number of paths from the cell directly above it).
                    
        return dp[-1]
