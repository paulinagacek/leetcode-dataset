class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]:
            return 0
        
        m, n = len(grid), len(grid[0])
        
        # We can update the grid in-place to store the minimum sum to reach each cell.
        for i in range(m):
            for j in range(n):
                if i == 0 and j == 0:
                    continue
                elif i == 0:
                    # Can only come from the left
                    grid[i][j] += grid[i][j - 1]
                elif j == 0:
                    # Can only come from above
                    grid[i][j] += grid[i - 1][j]
                else:
                    # Minimum of coming from above or from the left
                    grid[i][j] += min(grid[i - 1][j], grid[i][j - 1])
        
        return grid[m - 1][n - 1]
