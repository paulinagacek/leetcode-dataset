class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]:
            return 0
        
        m = len(grid)
        n = len(grid[0])
        
        # Modify the first row (only moves possible are from the left)
        for j in range(1, n):
            grid[0][j] += grid[0][j-1]
            
        # Modify the first column (only moves possible are from above)
        for i in range(1, m):
            grid[i][0] += grid[i-1][0]
            
        # Modify the rest of the grid
        for i in range(1, m):
            for j in range(1, n):
                # The value at (i, j) is the current cell value plus the 
                # minimum of the path from above or from the left
                grid[i][j] += min(grid[i-1][j], grid[i][j-1])
                
        return grid[m-1][n-1]
