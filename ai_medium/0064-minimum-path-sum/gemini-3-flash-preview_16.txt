class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        if not grid or not grid[0]:
            return 0
        
        m, n = len(grid), len(grid[0])
        
        # Use a 1D array to optimize space from O(m*n) to O(n)
        dp = [0] * n
        
        # Initialize the first element
        dp[0] = grid[0][0]
        
        # Fill the first row
        for j in range(1, n):
            dp[j] = dp[j-1] + grid[0][j]
            
        # Fill subsequent rows
        for i in range(1, m):
            # Update the first column element for the current row
            dp[0] += grid[i][0]
            for j in range(1, n):
                # The min path to (i, j) is grid[i][j] + min(path from above, path from left)
                # dp[j] currently holds the value from the row above (i-1, j)
                # dp[j-1] holds the value from the left (i, j-1)
                dp[j] = min(dp[j], dp[j-1]) + grid[i][j]
        
        return dp[-1]
