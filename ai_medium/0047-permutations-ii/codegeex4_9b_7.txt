from typing import List

class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start=0):
            if start == len(nums) - 1:
                res.append(nums[:])
            for i in range(start, len(nums)):
                if nums[start] != nums[i] or start == i:
                    nums[start], nums[i] = nums[i], nums[start]
                    backtrack(start + 1)
                    nums[start], nums[i] = nums[i], nums[start]

        res = []
        nums.sort()
        backtrack()
        return res
