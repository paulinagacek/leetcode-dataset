class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        ans, visited = [], [False] * len(nums)
        nums.sort()

        def backtrack(temp):
            if len(temp) == len(nums):
                ans.append(temp[:])
                return

            for i in range(len(nums)):
                if not visited[i]:
                    if i > 0 and nums[i] == nums[i - 1] and not visited[i - 1]:
                        continue

                    visited[i] = True
                    temp.append(nums[i])
                    backtrack(temp)
                    temp.pop()
                    visited[i] = False

        backtrack([])
        return ans
