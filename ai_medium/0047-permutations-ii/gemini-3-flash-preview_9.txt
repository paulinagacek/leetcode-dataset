class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        nums.sort()
        res = []
        used = [False] * len(nums)
        
        def backtrack(path: List[int]):
            if len(path) == len(nums):
                res.append(list(path))
                return
            
            for i in range(len(nums)):
                # If the current element is already used in the current path, skip it.
                if used[i]:
                    continue
                
                # If the current element is the same as the previous one and 
                # the previous one was not used in this path (meaning we just finished 
                # a branch with it), skip this duplicate to ensure uniqueness.
                if i > 0 and nums[i] == nums[i - 1] and not used[i - 1]:
                    continue
                
                used[i] = True
                path.append(nums[i])
                backtrack(path)
                path.pop()
                used[i] = False
                
        backtrack([])
        return res
