from typing import List

class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start=0):
            if start == len(nums) - 1:
                result.append(nums[:])
            else:
                for i in range(start, len(nums)):
                    if not used[i] and (i == start or nums[i] != nums[start]):
                        used[i] = True
                        nums[start], nums[i] = nums[i], nums[start]
                        backtrack(start + 1)
                        nums[start], nums[i] = nums[i], nums[start]
                        used[i] = False
        
        result = []
        used = [False] * len(nums)
        nums.sort()
        backtrack()
        return result
