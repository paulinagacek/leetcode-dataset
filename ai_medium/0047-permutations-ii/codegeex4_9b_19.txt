from typing import List

class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start):
            if start == len(nums) - 1:
                result.append(nums[:])
            else:
                for i in range(start, len(nums)):
                    if used[i] or (i > start and nums[i] == nums[start]):
                        continue
                    used[i] = True
                    nums[start], nums[i] = nums[i], nums[start]
                    backtrack(start + 1)
                    nums[start], nums[i] = nums[i], nums[start]
                    used[i] = False
        
        result = []
        nums.sort()
        used = [False] * len(nums)
        backtrack(0)
        return result
