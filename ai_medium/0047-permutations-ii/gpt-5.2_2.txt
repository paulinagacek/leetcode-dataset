class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        cnt = Counter(nums)
        n = len(nums)
        res: List[List[int]] = []
        path: List[int] = []

        def backtrack():
            if len(path) == n:
                res.append(path.copy())
                return
            for x in list(cnt.keys()):
                if cnt[x] == 0:
                    continue
                cnt[x] -= 1
                path.append(x)
                backtrack()
                path.pop()
                cnt[x] += 1

        backtrack()
        return res
