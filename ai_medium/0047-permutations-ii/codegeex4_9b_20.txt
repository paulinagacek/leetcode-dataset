from typing import List

class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start):
            if start == len(nums):
                result.append(nums[:])
            for i in range(start, len(nums)):
                if not used[i] and (i == start or nums[i] != nums[start-1] or used[i-1]):
                    used[i] = True
                    backtrack(start + 1)
                    used[i] = False

        nums.sort()
        result = []
        used = [False] * len(nums)
        backtrack(0)
        return result
