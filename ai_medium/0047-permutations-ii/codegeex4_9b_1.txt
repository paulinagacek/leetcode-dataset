from typing import List

class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start):
            if start == len(nums) - 1:
                res.append(nums[:])
                return
            used = set()
            for i in range(start, len(nums)):
                if nums[i] not in used:
                    used.add(nums[i])
                    nums[start], nums[i] = nums[i], nums[start]
                    backtrack(start + 1)
                    nums[start], nums[i] = nums[i], nums[start]

        res = []
        nums.sort()
        backtrack(0)
        return res
