class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1.0
        
        # Handle negative power
        if n < 0:
            x = 1 / x
            n = -n
        
        res = 1.0
        current_product = x
        
        # Binary exponentiation (Exponentiation by squaring)
        while n > 0:
            if n % 2 == 1:
                res *= current_product
            current_product *= current_product
            n //= 2
            
        return res
