class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n == 1:
            return x
        
        temp = self.myPow(x, n // 2)
        result = temp * temp
        
        if n % 2 == 1:
            result *= x
        
        if n < 0:
            result = 1 / result
        
        return result
