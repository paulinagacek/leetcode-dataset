class Solution:
    def myPow(self, x: float, n: int) -> float:
        result = 1.0
        sign = 1

        if n < 0:
            sign = -1
            n = abs(n)

        while n > 0:
            if n % 2 == 1:
                result *= x
            x *= x
            n //= 2

        if sign == -1:
            result = 1 / result

        return result
