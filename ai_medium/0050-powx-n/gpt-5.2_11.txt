class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1.0

        exp = n
        base = x
        if exp < 0:
            base = 1.0 / base
            exp = -exp

        result = 1.0
        while exp > 0:
            if exp & 1:
                result *= base
            base *= base
            exp >>= 1

        return result
