class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        elif n == 1:
            return x
        elif n == -1:
            return 1 / x
        else:
            if n > 0:
                return x * self.myPow(x, n - 1)
            else:
                return (1 / x) * self.myPow(x, -n)
