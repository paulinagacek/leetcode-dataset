class Solution:
    def myPow(self, x: float, n: int) -> float:
        positive = n >= 0
        n = abs(n)
        result = 1.0
        while n > 0:
            if n & 1:
                result *= x
            x *= x
            n >>= 1
        if not positive:
            result = 1 / result
        return result
