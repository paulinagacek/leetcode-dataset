class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n == 1:
            return x
        if n == -1:
            return 1 / x
        if n > 0:
            result = x
            for i in range(1, n):
                result *= x
            return result
        else:
            result = 1 / x
            n = -n
            for i in range(1, n):
                result /= x
            return result
