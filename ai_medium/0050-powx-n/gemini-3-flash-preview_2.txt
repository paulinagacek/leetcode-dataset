class Solution:
    def myPow(self, x: float, n: int) -> float:
        # If n is 0, any number (except potentially 0, but n > 0 if x is 0) raised to 0 is 1.0.
        if n == 0:
            return 1.0
        
        # Handle negative exponent by converting x to 1/x and n to -n.
        if n < 0:
            x = 1 / x
            n = -n
        
        # Binary Exponentiation (Exponentiation by Squaring) algorithm.
        # This provides an O(log n) time complexity.
        res = 1.0
        while n > 0:
            # If n is odd, multiply the result by the current base x.
            if n % 2 == 1:
                res *= x
            # Square the base and halve the exponent for the next iteration.
            x *= x
            n //= 2
            
        return res
