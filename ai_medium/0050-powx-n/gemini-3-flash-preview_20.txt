class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1.0
        
        # Handle negative powers
        if n < 0:
            x = 1 / x
            n = -n
        
        res = 1.0
        current_product = x
        
        # Iterative Binary Exponentiation (Exponentiation by Squaring)
        while n > 0:
            # If n is odd, multiply the result by the current product
            if n % 2 == 1:
                res *= current_product
            
            # Square the current product and reduce n by half
            current_product *= current_product
            n //= 2
            
        return res
