class Solution:
    def myPow(self, x: float, n: int) -> float:
        result = 1
        abs_n = abs(n)
        
        while abs_n > 0:
            if abs_n % 2 == 1:
                result *= x
            x *= x
            abs_n //= 2
        
        if n < 0:
            result = 1 / result
        
        return result
