class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        elif n == 1:
            return x
        elif n == -1:
            return 1/x
        elif n > 0:
            result = x
            for i in range(1, n):
                result *= x
            return result
        else:
            result = 1/x
            for i in range(1, -n, -1):
                result /= x
            return result
