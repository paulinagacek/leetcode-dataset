class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1.0
        
        # Handle negative n by taking the reciprocal of x and making n positive
        if n < 0:
            x = 1 / x
            n = -n
            
        result = 1.0
        current_product = x
        while n > 0:
            # If n is odd, multiply the result by the current power of x
            if n % 2 == 1:
                result *= current_product
            # Square the current power of x
            current_product *= current_product
            # Integer division of n by 2
            n //= 2
            
        return result
