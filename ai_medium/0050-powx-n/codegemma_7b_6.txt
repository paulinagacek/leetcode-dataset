class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n == 1:
            return x
        result = x
        abs_n = abs(n)
        for _ in range(abs_n - 1):
            result *= x
        if n < 0:
            result = 1 / result
        return result
