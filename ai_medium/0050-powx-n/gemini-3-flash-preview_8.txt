class Solution:
    def myPow(self, x: float, n: int) -> float:
        # Handle negative exponent
        if n < 0:
            x = 1 / x
            n = -n
        
        # Exponentiation by squaring
        res = 1.0
        while n:
            # If current exponent is odd, multiply the result by the current base
            if n & 1:
                res *= x
            # Square the base and halve the exponent
            x *= x
            n >>= 1
            
        return res
