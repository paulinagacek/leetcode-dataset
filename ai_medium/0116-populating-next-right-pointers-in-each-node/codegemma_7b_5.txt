def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
    if not root:
        return None

    queue = deque([root])

    while queue:
        level_size = len(queue)
        prev = None

        for _ in range(level_size):
            node = queue.popleft()

            if prev:
                prev.next = node

            if node.left:
                queue.append(node.left)

            if node.right:
                queue.append(node.right)

            prev = node

    return root
