"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        
        # Initialize the current node to the root of the tree
        curr = root
        
        # Traverse the tree level by level
        while curr.left:
            # Get a reference to the first node at the current level
            head = curr
            
            # Iterate through each node in the current level
            while curr:
                # Connect the left child to the right child
                curr.left.next = curr.right
                
                # If there is a next node, connect the right child to its next.left
                if curr.next:
                    curr.right.next = curr.next.left
                
                # Move to the next node in the level
                curr = curr.next
            
            # Move to the first node of the next level
            curr = head.left
        
        return root
