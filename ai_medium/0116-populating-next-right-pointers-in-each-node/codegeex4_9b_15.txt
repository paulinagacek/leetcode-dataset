"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        
        # Initialize the leftmost node of the current level
        leftmost = root
        
        while leftmost.left:
            # Iterate through each level of the tree
            head = leftmost
            
            while head:
                # Connect the next right nodes for all nodes in this level
                head.left.next = head.right
                
                if head.next:
                    head.right.next = head.next.left
                else:
                    head.right.next = None
                
                # Move to the next node on the same level
                head = head.next
            
            # Move to the leftmost node of the next level
            leftmost = leftmost.left
        
        return root
