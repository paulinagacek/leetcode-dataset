class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return root

        queue = [(root, None)]
        while queue:
            node, prev = queue.pop(0)
            if node.left:
                queue.append((node.left, node))
            if node.right:
                queue.append((node.right, node))

            if prev:
                node.next = prev
        return root
