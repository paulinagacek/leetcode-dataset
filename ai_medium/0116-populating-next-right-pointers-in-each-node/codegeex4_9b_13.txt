"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        
        # Initialize the current node as the root
        curr = root
        
        # Loop until there is no next level to connect
        while curr.left:
            # Initialize the first node of the next level
            head = curr.left
            
            # Connect all nodes of the next level
            while curr:
                curr.left.next = curr.right
                if curr.next:
                    curr.right.next = curr.next.left
                curr = curr.next
                
            # Move to the first node of the next level
            curr = head
        
        return root
