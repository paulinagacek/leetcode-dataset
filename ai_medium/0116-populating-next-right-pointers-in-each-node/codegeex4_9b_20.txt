"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: 'Node' = None, right: 'Node' = None, next: 'Node' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        
        # Initialize the current node as the root of the tree
        cur = root
        
        # Outer loop handles each level of the tree
        while cur:
            # This dummy node acts as a temporary parent node for the next Level
            dummy = Node(0)
            tail = dummy
            
            # Inner loop handles each node in the current level
            while cur:
                if cur.left:
                    # Connect tail's next to cur's left child
                    tail.next = cur.left
                    tail = tail.next
                
                if cur.right:
                    # Connect tail's next to cur's right child
                    tail.next = cur.right
                    tail = tail.next
                
                # Move to the next node in the current level
                cur = cur.next
            
            # Start with the first node in the next level
            cur = dummy.next
        
        # Return the root of the tree with all next pointers connected
        return root
