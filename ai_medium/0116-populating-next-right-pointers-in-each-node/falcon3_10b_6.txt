class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        
        queue = [root]

        while queue:
            prev = None
            new_level = []

            for node in queue:
                if node.left:
                    new_level.append(node.left)
                    new_level.append(node.right)

                if prev:
                    prev.next = node

                prev = node

            queue = new_level
        
        return root
