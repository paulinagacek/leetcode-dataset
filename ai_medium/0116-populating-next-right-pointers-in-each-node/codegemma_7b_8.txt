class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root: return root

        leftmost = root
        while leftmost:
            cur = leftmost

            while cur:
                if cur.left: cur.left.next = cur.right
                if cur.right and cur.next: cur.right.next = cur.next.left

                cur = cur.next

            leftmost = leftmost.left

        return root
