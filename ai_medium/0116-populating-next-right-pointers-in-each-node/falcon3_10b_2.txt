class Solution:
    def connect(self, root: 'Optional[Node]') -> 'Optional[Node]':
        if not root:
            return None
        queue = [(root, 0)]
        prev = None
        level = -1
        while queue:
            node, depth = queue.pop(0)
            if depth != level:
                level = depth
                prev = node
                node.next = None
            else:
                prev.next = node
                prev = node
            if node.left:
                queue.append((node.left, depth + 1))
            if node.right:
                queue.append((node.right, depth + 1))
        return root
