class LRUCache:

    def __init__(self, capacity: int):
        self.cache = {}  # dictionary to store key-value pairs
        self.capacity = capacity  # maximum number of items in the cache
        self.order = []  # list to maintain order of usage

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1  # key not found, return -1
        else:
            self.order.remove(key)  # remove key from current position
            self.order.append(key)  # add key to the end to mark it as recently used
            return self.cache[key]  # return the value

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            self.order.remove(key)  # remove key from current position if exists
        elif len(self.cache) == self.capacity:
            lru_key = self.order.pop(0)  # get the least recently used key
            del self.cache[lru_key]  # delete the least recently used key-value pair from cache
        self.cache[key] = value  # add/update key-value pair in cache
        self.order.append(key)  # add key to the end to mark it as recently used
