class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m = len(word1)
        n = len(word2)
        
        # Initialize a DP table with dimensions (m+1) x (n+1)
        # dp[i][j] will store the edit distance between word1[0...i-1] and word2[0...j-1]
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        # Base cases: converting a string to an empty string requires deletions
        for i in range(m + 1):
            dp[i][0] = i
            
        # Base cases: converting an empty string to a target string requires insertions
        for j in range(n + 1):
            dp[0][j] = j
            
        # Fill the DP table
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i - 1] == word2[j - 1]:
                    # Characters match, no new operation needed
                    dp[i][j] = dp[i - 1][j - 1]
                else:
                    # Characters don't match, consider three operations:
                    # dp[i-1][j]: Delete character from word1
                    # dp[i][j-1]: Insert character into word1
                    # dp[i-1][j-1]: Replace character in word1
                    dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1])
                    
        return dp[m][n]
