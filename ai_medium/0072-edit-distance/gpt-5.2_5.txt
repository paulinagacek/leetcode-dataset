class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        n, m = len(word1), len(word2)
        if m == 0:
            return n
        if n == 0:
            return m

        prev = list(range(m + 1))
        for i in range(1, n + 1):
            cur = [i] + [0] * m
            c1 = word1[i - 1]
            for j in range(1, m + 1):
                if c1 == word2[j - 1]:
                    cur[j] = prev[j - 1]
                else:
                    cur[j] = 1 + min(prev[j], cur[j - 1], prev[j - 1])
            prev = cur
        return prev[m]
