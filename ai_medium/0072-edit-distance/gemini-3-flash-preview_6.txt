class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        # dp[i][j] represents the minimum edit distance between word1[:i] and word2[:j]
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        # Base cases: converting empty string to a string or vice versa
        for i in range(m + 1):
            dp[i][0] = i
        for j in range(n + 1):
            dp[0][j] = j
            
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i - 1] == word2[j - 1]:
                    # Characters match, no operation needed for this position
                    dp[i][j] = dp[i - 1][j - 1]
                else:
                    # Choose minimum of insert, delete, or replace operation
                    dp[i][j] = 1 + min(
                        dp[i][j - 1],    # Insert
                        dp[i - 1][j],    # Delete
                        dp[i - 1][j - 1] # Replace
                    )
                    
        return dp[m][n]
