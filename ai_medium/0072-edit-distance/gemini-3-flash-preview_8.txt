class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        
        # Create a DP table with (m+1) rows and (n+1) columns
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        # Base cases: 
        # Converting word1 prefix to empty word2 requires deletions
        for i in range(m + 1):
            dp[i][0] = i
        # Converting empty word1 to word2 prefix requires insertions
        for j in range(n + 1):
            dp[0][j] = j
            
        # Fill the DP table
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i - 1] == word2[j - 1]:
                    # Characters match, no operation needed for this position
                    dp[i][j] = dp[i - 1][j - 1]
                else:
                    # Characters don't match, take the minimum of:
                    # dp[i-1][j] + 1: deletion
                    # dp[i][j-1] + 1: insertion
                    # dp[i-1][j-1] + 1: replacement
                    dp[i][j] = 1 + min(dp[i - 1][j],      
                                       dp[i][j - 1],      
                                       dp[i - 1][j - 1])  
        
        return dp[m][n]
