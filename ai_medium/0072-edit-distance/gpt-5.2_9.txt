class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        n, m = len(word1), len(word2)
        if m == 0:
            return n
        if n == 0:
            return m

        dp = list(range(m + 1))  # dp[j] = dist(word1[:i], word2[:j]) for current i
        for i in range(1, n + 1):
            prev_diag = dp[0]     # dp[j-1] from previous row
            dp[0] = i
            for j in range(1, m + 1):
                temp = dp[j]      # dp[j] from previous row
                if word1[i - 1] == word2[j - 1]:
                    dp[j] = prev_diag
                else:
                    dp[j] = 1 + min(prev_diag, dp[j], dp[j - 1])  # replace, delete, insert
                prev_diag = temp
        return dp[m]
