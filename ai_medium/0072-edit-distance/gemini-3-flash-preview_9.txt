class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        # Create a DP table with dimensions (m+1) x (n+1)
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        # Base case: converting word1 to an empty word2 (deletions)
        for i in range(m + 1):
            dp[i][0] = i
            
        # Base case: converting an empty word1 to word2 (insertions)
        for j in range(n + 1):
            dp[0][j] = j
            
        # Fill the DP table
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i-1] == word2[j-1]:
                    # If characters match, no operation is needed
                    dp[i][j] = dp[i-1][j-1]
                else:
                    # If characters don't match, consider insert, delete, and replace
                    dp[i][j] = 1 + min(dp[i-1][j],    # Delete
                                       dp[i][j-1],    # Insert
                                       dp[i-1][j-1])  # Replace
                                       
        return dp[m][n]
