class Solution:
    def minDistance(self, word1: str, word2: str) -> int:
        m, n = len(word1), len(word2)
        
        # Create a DP table with dimensions (m+1) x (n+1)
        dp = [[0] * (n + 1) for _ in range(m + 1)]
        
        # Base cases: 
        # Transforming word1[0...i] to an empty string requires i deletions
        for i in range(m + 1):
            dp[i][0] = i
        # Transforming an empty string to word2[0...j] requires j insertions
        for j in range(n + 1):
            dp[0][j] = j
            
        # Fill the DP table
        for i in range(1, m + 1):
            for j in range(1, n + 1):
                if word1[i - 1] == word2[j - 1]:
                    # Characters match, no operation needed for this step
                    dp[i][j] = dp[i - 1][j - 1]
                else:
                    # Characters don't match, take the minimum of 3 operations:
                    # dp[i-1][j] is deletion
                    # dp[i][j-1] is insertion
                    # dp[i-1][j-1] is replacement
                    dp[i][j] = 1 + min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1])
                    
        return dp[m][n]
