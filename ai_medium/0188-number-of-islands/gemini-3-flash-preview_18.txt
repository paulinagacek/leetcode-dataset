class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        m, n = len(grid), len(grid[0])
        num_islands = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    num_islands += 1
                    # Start a BFS to mark all connected lands
                    queue = deque([(i, j)])
                    grid[i][j] = '0'  # Mark the cell as visited by setting it to '0'
                    while queue:
                        r, c = queue.popleft()
                        # Explore all four directions
                        for nr, nc in ((r + 1, c), (r - 1, c), (r, c + 1), (r, c - 1)):
                            if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'  # Mark the cell as visited
                                queue.append((nr, nc))
                                
        return num_islands
