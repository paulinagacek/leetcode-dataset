class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        rows = len(grid)
        cols = len(grid[0])
        num_islands = 0
        
        def dfs(r, c):
            # Check boundaries and if the cell is water or already visited
            if r < 0 or c < 0 or r >= rows or c >= cols or grid[r][c] == '0':
                return
            
            # Mark the cell as visited by setting it to '0'
            grid[r][c] = '0'
            
            # Visit all adjacent neighbors
            dfs(r + 1, c)
            dfs(r - 1, c)
            dfs(r, c + 1)
            dfs(r, c - 1)
            
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == '1':
                    # Found a new island, start DFS to mark the entire island
                    num_islands += 1
                    dfs(r, c)
                    
        return num_islands
