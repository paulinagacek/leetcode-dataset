class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        m, n = len(grid), len(grid[0])
        num_islands = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    num_islands += 1
                    # Start iterative DFS
                    stack = [(i, j)]
                    grid[i][j] = '0'  # Mark as visited
                    while stack:
                        r, c = stack.pop()
                        # Check all 4 neighbors
                        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                            nr, nc = r + dr, c + dc
                            if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'  # Mark as visited
                                stack.append((nr, nc))
        
        return num_islands
