class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        rows = len(grid)
        cols = len(grid[0])
        num_islands = 0
        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == "1":
                    num_islands += 1
                    # Breadth First Search (BFS) to mark all connected lands as water
                    queue = deque([(r, c)])
                    grid[r][c] = "0" # Mark as visited immediately
                    while queue:
                        curr_r, curr_c = queue.popleft()
                        # Explore four adjacent directions: up, down, left, right
                        for dr, dc in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                            nr, nc = curr_r + dr, curr_c + dc
                            # Check bounds and if the neighbor is land
                            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == "1":
                                grid[nr][nc] = "0" # Mark as visited
                                queue.append((nr, nc))
                                
        return num_islands
