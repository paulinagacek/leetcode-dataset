class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid or not grid[0]:
            return 0

        m, n = len(grid), len(grid[0])
        islands = 0

        for i in range(m):
            for j in range(n):
                if grid[i][j] != '1':
                    continue
                islands += 1
                grid[i][j] = '0'
                q = deque([(i, j)])
                while q:
                    x, y = q.popleft()
                    if x > 0 and grid[x - 1][y] == '1':
                        grid[x - 1][y] = '0'
                        q.append((x - 1, y))
                    if x + 1 < m and grid[x + 1][y] == '1':
                        grid[x + 1][y] = '0'
                        q.append((x + 1, y))
                    if y > 0 and grid[x][y - 1] == '1':
                        grid[x][y - 1] = '0'
                        q.append((x, y - 1))
                    if y + 1 < n and grid[x][y + 1] == '1':
                        grid[x][y + 1] = '0'
                        q.append((x, y + 1))

        return islands
