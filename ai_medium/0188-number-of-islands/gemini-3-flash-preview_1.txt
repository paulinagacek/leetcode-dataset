class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        rows = len(grid)
        cols = len(grid[0])
        num_islands = 0
        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == '1':
                    # A new island is discovered
                    num_islands += 1
                    
                    # Perform BFS to sink the entire island
                    queue = deque([(r, c)])
                    grid[r][c] = '0'  # Mark the cell as visited
                    
                    while queue:
                        curr_r, curr_c = queue.popleft()
                        
                        # Explore all four neighbors
                        for dr, dc in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                            nr, nc = curr_r + dr, curr_c + dc
                            
                            # If the neighbor is within bounds and is land, visit it
                            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'  # Sink the neighbor
                                queue.append((nr, nc))
                                
        return num_islands
