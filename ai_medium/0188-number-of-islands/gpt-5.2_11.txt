class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid or not grid[0]:
            return 0

        m, n = len(grid), len(grid[0])
        islands = 0

        for i in range(m):
            for j in range(n):
                if grid[i][j] != '1':
                    continue

                islands += 1
                grid[i][j] = '0'
                q = deque([(i, j)])

                while q:
                    r, c = q.popleft()
                    if r > 0 and grid[r - 1][c] == '1':
                        grid[r - 1][c] = '0'
                        q.append((r - 1, c))
                    if r + 1 < m and grid[r + 1][c] == '1':
                        grid[r + 1][c] = '0'
                        q.append((r + 1, c))
                    if c > 0 and grid[r][c - 1] == '1':
                        grid[r][c - 1] = '0'
                        q.append((r, c - 1))
                    if c + 1 < n and grid[r][c + 1] == '1':
                        grid[r][c + 1] = '0'
                        q.append((r, c + 1))

        return islands
