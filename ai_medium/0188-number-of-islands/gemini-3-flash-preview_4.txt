class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        rows = len(grid)
        cols = len(grid[0])
        count = 0
        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == '1':
                    count += 1
                    # Start an iterative DFS to mark all connected land
                    stack = [(r, c)]
                    grid[r][c] = '0'  # Mark the starting cell as visited
                    while stack:
                        curr_r, curr_c = stack.pop()
                        # Explore the four adjacent directions
                        for dr, dc in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                            nr, nc = curr_r + dr, curr_c + dc
                            # If the neighbor is within bounds and is land, visit it
                            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'
                                stack.append((nr, nc))
                                
        return count
