class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        m = len(grid)
        n = len(grid[0])
        count = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    count += 1
                    # Perform Iterative DFS using a stack to mark the entire island
                    stack = [(i, j)]
                    grid[i][j] = '0'  # Mark the cell as visited by setting it to water
                    while stack:
                        r, c = stack.pop()
                        # Check all 4 adjacent neighbors (Up, Down, Left, Right)
                        for nr, nc in [(r - 1, c), (r + 1, c), (r, c - 1), (r, c + 1)]:
                            if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] == '1':
                                grid[nr][nc] = '0'  # Mark the cell as visited
                                stack.append((nr, nc))
                                
        return count
