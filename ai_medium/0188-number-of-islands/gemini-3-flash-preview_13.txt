class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        m, n = len(grid), len(grid[0])
        count = 0
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == '1':
                    count += 1
                    # Iterative DFS to mark the island cells as visited
                    stack = [(i, j)]
                    grid[i][j] = '0'
                    while stack:
                        r, c = stack.pop()
                        # Check all 4 adjacent neighbors
                        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                            nr, nc = r + dr, c + dc
                            if 0 <= nr < m and 0 <= nc < n and grid[nr][nc] == '1':
                                # Mark as visited immediately before adding to stack
                                grid[nr][nc] = '0'
                                stack.append((nr, nc))
        return count
