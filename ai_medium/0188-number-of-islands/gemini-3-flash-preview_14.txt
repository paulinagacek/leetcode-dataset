class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        if not grid:
            return 0
        
        rows = len(grid)
        cols = len(grid[0])
        island_count = 0
        
        for r in range(rows):
            for c in range(cols):
                if grid[r][c] == '1':
                    island_count += 1
                    # Start an iterative DFS to mark all connected land as visited
                    stack = [(r, c)]
                    grid[r][c] = '0'  # Mark the initial cell as visited
                    while stack:
                        curr_r, curr_c = stack.pop()
                        # Check all 4 neighbors
                        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                            nr, nc = curr_r + dr, curr_c + dc
                            if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == '1':
                                # Mark neighbor as visited before adding to stack to prevent redundant additions
                                grid[nr][nc] = '0'
                                stack.append((nr, nc))
                                
        return island_count
