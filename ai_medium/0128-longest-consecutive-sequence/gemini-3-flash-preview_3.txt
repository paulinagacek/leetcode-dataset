class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        num_set = set(nums)
        max_length = 0
        
        for num in num_set:
            # Check if current number is the start of a sequence
            if (num - 1) not in num_set:
                current_num = num
                current_streak = 1
                
                # Count the length of the sequence starting from this number
                while (current_num + 1) in num_set:
                    current_num += 1
                    current_streak += 1
                
                max_length = max(max_length, current_streak)
                
        return max_length
