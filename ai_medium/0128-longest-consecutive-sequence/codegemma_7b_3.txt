class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        seen = set(nums)
        longest = 0
        for num in seen:
            if num - 1 not in seen:
                current_length = 1
                while num + current_length in seen:
                    current_length += 1
                longest = max(longest, current_length)
        return longest
