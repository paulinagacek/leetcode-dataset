class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        num_set = set(nums)
        longest = 0

        for num in num_set:
            if num - 1 not in num_set:
                current_longest = 1
                while num + current_longest in num_set:
                    current_longest += 1
                longest = max(longest, current_longest)

        return longest
