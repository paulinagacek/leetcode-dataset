from typing import List

class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:
        # Create a graph representation of the courses and their prerequisites
        graph = [[] for _ in range(numCourses)]
        indegree = [0] * numCourses
        
        # Build the graph and compute indegrees of each node
        for dest, src in prerequisites:
            graph[src].append(dest)
            indegree[dest] += 1
        
        # Initialize the count of visited nodes and the order list
        count = 0
        order = []
        
        # Find all sources (nodes with zero indegree) and add them to the queue
        queue = [i for i in range(numCourses) if indegree[i] == 0]
        
        # Process the nodes in the queue, updating the graph and order list
        while queue:
            node = queue.pop(0)
            order.append(node)
            count += 1
            
            # Reduce the indegree of neighboring nodes and add them to the queue if their indegree becomes zero
            for neigh in graph[node]:
                indegree[neigh] -= 1
                if indegree[neigh] == 0:
                    queue.append(neigh)
        
        # If we have processed all nodes, return the order; otherwise, it's impossible to complete all courses
        return order if count == numCourses else []
