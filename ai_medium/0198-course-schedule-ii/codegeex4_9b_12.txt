from typing import List

class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:
        # Create a graph and an in-degree array
        graph = [[] for _ in range(numCourses)]
        in_degree = [0] * numCourses
        
        # Build the graph and fill in-degree array
        for dest, src in prerequisites:
            graph[src].append(dest)
            in_degree[dest] += 1
        
        # Initialize a queue with all courses having no prerequisites
        zero_indegree_queue = [i for i in range(numCourses) if in_degree[i] == 0]
        
        # To store the topological order
        topological_order = []
        
        while zero_indegree_queue:
            # Remove and get a vertex from zero_indegree_queue
            current_course = zero_indegree_queue.pop(0)
            
            # Add it to the topological sort
            topological_order.append(current_course)
            
            # Decrease in-degree of neighboring nodes by 1
            for neighbor in graph[current_course]:
                in_degree[neighbor] -= 1
                
                # If in-degree is now zero, add it to the queue
                if in_degree[neighbor] == 0:
                    zero_indegree_queue.append(neighbor)
        
        # If topological_order doesn't contain all courses, there is a cycle
        if len(topological_order) == numCourses:
            return topological_order
        else:
            return []
