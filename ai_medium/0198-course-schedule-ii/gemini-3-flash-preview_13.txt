class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:
        # Build adjacency list and calculate in-degree of each course
        adj = defaultdict(list)
        in_degree = [0] * numCourses
        
        for dest, src in prerequisites:
            adj[src].append(dest)
            in_degree[dest] += 1
            
        # Initialize queue with all courses that have no prerequisites (in-degree 0)
        queue = deque([i for i in range(numCourses) if in_degree[i] == 0])
        
        topological_order = []
        
        while queue:
            node = queue.popleft()
            topological_order.append(node)
            
            # Decrease the in-degree for each neighbor
            if node in adj:
                for neighbor in adj[node]:
                    in_degree[neighbor] -= 1
                    
                    # If in-degree becomes 0, add it to the queue
                    if in_degree[neighbor] == 0:
                        queue.append(neighbor)
        
        # Check if the topological sort includes all courses
        if len(topological_order) == numCourses:
            return topological_order
        else:
            # If there is a cycle, topological sort is impossible
            return []
