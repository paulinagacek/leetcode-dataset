from typing import List

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        result = []

        def backtrack(path):
            if len(path) == len(nums):
                result.append(path[:])
                return
            for num in set(nums) - set(path):
                path.append(num)
                backtrack(path)
                path.pop()

        backtrack([])
        return result
