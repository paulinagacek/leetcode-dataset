from typing import List

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(first = 0):
            # if all numbers are placed then return
            if first == n:  
                output.append(nums[:])
            for i in range(first, n):
                # place ith number first then permutations of rest of the numbers
                nums[first], nums[i] = nums[i], nums[first]
                backtrack(first + 1)
                # backtrack 
                nums[first], nums[i] = nums[i], nums[first]
        
        n = len(nums)
        output = []
        backtrack()
        return output
