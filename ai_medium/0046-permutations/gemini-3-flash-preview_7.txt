class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(first=0):
            # If all integers are used, add the current permutation to the result
            if first == n:
                output.append(nums[:])
                return
            
            for i in range(first, n):
                # Place the i-th integer first in the current permutation
                nums[first], nums[i] = nums[i], nums[first]
                # Continue generating permutations for the remaining integers
                backtrack(first + 1)
                # Backtrack to restore the original state for the next iteration
                nums[first], nums[i] = nums[i], nums[first]

        n = len(nums)
        output = []
        backtrack()
        return output
