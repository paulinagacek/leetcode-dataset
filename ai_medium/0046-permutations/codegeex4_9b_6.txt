from typing import List

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(first = 0):
            # if all numbers are placed then return a fresh list of current permutation
            if first == n:
                output.append(nums[:])
            for i in range(first, n):
                # place nums[i] at the first position and increase first to fix it
                nums[first], nums[i] = nums[i], nums[first]
                # use next available number to fill rest of positions
                backtrack(first + 1)
                # backtrack:
                # undo the previously done shift for nums[first]
                nums[first], nums[i] = nums[i], nums[first]
        
        n = len(nums)
        output = []
        backtrack()
        return output
