class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        result = []
        def backtrack(nums, current):
            if not nums:
                result.append(current[:])
                return
            for num in nums:
                current.append(num)
                nums.remove(num)
                backtrack(nums, current)
                nums.append(num)
                current.pop()
        backtrack(nums, [])
        return result
