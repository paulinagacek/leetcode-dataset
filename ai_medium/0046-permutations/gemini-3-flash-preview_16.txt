class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(start):
            # If all integers are used, add the current permutation to the result
            if start == len(nums):
                res.append(nums[:])
                return
            for i in range(start, len(nums)):
                # Swap the current element with the element at the 'start' index
                nums[start], nums[i] = nums[i], nums[start]
                # Recursively generate permutations for the remaining elements
                backtrack(start + 1)
                # Backtrack: restore the original order for the next iteration
                nums[start], nums[i] = nums[i], nums[start]
        
        res = []
        backtrack(0)
        return res
