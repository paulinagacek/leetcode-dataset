class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def backtrack(first=0):
            # If all integers are used up, add the current permutation to the result
            if first == len(nums):
                res.append(nums[:])
                return
            for i in range(first, len(nums)):
                # Place the i-th integer at the current 'first' position
                nums[first], nums[i] = nums[i], nums[first]
                # Recursively generate permutations for the remaining integers
                backtrack(first + 1)
                # Backtrack: restore the original order for the next iteration
                nums[first], nums[i] = nums[i], nums[first]
        
        res = []
        backtrack()
        return res
