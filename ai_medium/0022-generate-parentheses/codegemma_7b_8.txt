class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        def backtrack(index, left, right, parenthesis):
            if index == 2 * n:
                result.append("".join(parenthesis))
                return

            if left < n:
                parenthesis[index] = "("
                backtrack(index + 1, left + 1, right, parenthesis)

            if right < left:
                parenthesis[index] = ")"
                backtrack(index + 1, left, right + 1, parenthesis)

        result = []
        backtrack(0, 0, 0, [""] * 2 * n)
        return result
