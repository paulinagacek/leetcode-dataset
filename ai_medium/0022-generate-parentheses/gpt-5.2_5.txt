class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        res: List[str] = []
        path: List[str] = []

        def backtrack(opened: int, closed: int) -> None:
            if opened == n and closed == n:
                res.append("".join(path))
                return
            if opened < n:
                path.append("(")
                backtrack(opened + 1, closed)
                path.pop()
            if closed < opened:
                path.append(")")
                backtrack(opened, closed + 1)
                path.pop()

        backtrack(0, 0)
        return res
