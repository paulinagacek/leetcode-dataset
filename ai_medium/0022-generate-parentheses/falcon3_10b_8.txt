class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        result = []
        self.helper(result, "", n, n)
        return result
        
    def helper(self, result, curr, open_count, close_count):
        if len(curr) == 2 * n:
            result.append(curr)
        if open_count > 0:
            self.helper(result, curr + "(", open_count - 1, close_count)
        if close_count > open_count:
            self.helper(result, curr + ")", open_count, close_count - 1)
