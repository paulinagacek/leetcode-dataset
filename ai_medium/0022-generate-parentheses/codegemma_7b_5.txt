class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        def backtrack(index, open_count, close_count, str):
            if index == 2 * n:
                result.append(str[:])
                return
            if open_count < n:
                str.append('(')
                backtrack(index + 1, open_count + 1, close_count, str)
                str.pop()
            if close_count < open_count:
                str.append(')')
                backtrack(index + 1, open_count, close_count + 1, str)
                str.pop()

        result = []
        backtrack(0, 0, 0, [])
        return result
