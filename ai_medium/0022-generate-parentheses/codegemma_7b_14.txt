class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        def backtrack(index, open_par, close_par, result):
            if index == 2 * n:
                result.append(''.join(open_par + close_par))
                return

            if open_par <= n:
                open_par.append('(')
                backtrack(index + 1, open_par, close_par, result)
                open_par.pop()

            if close_par < open_par:
                close_par.append(')')
                backtrack(index + 1, open_par, close_par, result)
                close_par.pop()

        result = []
        backtrack(0, [], [], result)
        return result
