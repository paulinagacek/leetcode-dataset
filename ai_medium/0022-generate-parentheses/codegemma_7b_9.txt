class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        result = []
        
        def backtrack(open_par, close_par, curr):
            if open_par == close_par == n:
                result.append("".join(curr))
                return
            
            if open_par < n:
                curr.append("(")
                backtrack(open_par+1, close_par, curr)
                curr.pop()
            
            if close_par < open_par:
                curr.append(")")
                backtrack(open_par, close_par+1, curr)
                curr.pop()
        
        backtrack(0, 0, [])
        return result
