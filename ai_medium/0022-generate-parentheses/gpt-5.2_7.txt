class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        res: List[str] = []

        def backtrack(curr: List[str], open_cnt: int, close_cnt: int) -> None:
            if len(curr) == 2 * n:
                res.append("".join(curr))
                return
            if open_cnt < n:
                curr.append('(')
                backtrack(curr, open_cnt + 1, close_cnt)
                curr.pop()
            if close_cnt < open_cnt:
                curr.append(')')
                backtrack(curr, open_cnt, close_cnt + 1)
                curr.pop()

        backtrack([], 0, 0)
        return res
