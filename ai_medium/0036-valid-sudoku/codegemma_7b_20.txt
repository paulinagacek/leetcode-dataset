class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [set() for _ in range(9)]
        cols = [set() for _ in range(9)]
        subgrids = [set() for _ in range(9)]
        
        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    subgrid_idx = (i // 3) * 3 + j // 3
                    if (
                        board[i][j] in rows[i] or
                        board[i][j] in cols[j] or
                        board[i][j] in subgrids[subgrid_idx]
                    ):
                        return False
                    rows[i].add(board[i][j])
                    cols[j].add(board[i][j])
                    subgrids[subgrid_idx].add(board[i][j])
        
        return True
