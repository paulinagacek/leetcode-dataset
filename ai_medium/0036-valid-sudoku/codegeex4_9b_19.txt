class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [{} for _ in range(9)]
        cols = [{} for _ in range(9)]
        squares = [[{} for _ in range(3)] for _ in range(3)]

        for i in range(9):
            for j in range(9):
                num = board[i][j]
                if num != '.':
                    num = int(num)
                    rows[i][num] = rows[i].get(num, 0) + 1
                    cols[j][num] = cols[j].get(num, 0) + 1
                    squares[i // 3][j // 3][num] = squares[i // 3][j // 3].get(num, 0) + 1

                    if rows[i][num] > 1 or cols[j][num] > 1 or squares[i // 3][j // 3][num] > 1:
                        return False
        return True
