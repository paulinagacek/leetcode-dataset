class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [[0]*10 for _ in range(9)]
        cols = [[0]*10 for _ in range(9)]
        boxes = [[0]*10 for _ in range(9)]
        
        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    num = int(board[i][j])
                    box_idx = (i // 3) * 3 + (j // 3)
                    
                    if rows[i][num] or cols[j][num] or boxes[box_idx][num]:
                        return False
                    
                    rows[i][num] = cols[j][num] = boxes[box_idx][num] = 1
        
        return True
