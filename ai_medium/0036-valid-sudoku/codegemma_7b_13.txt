class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        seen = set()
        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    if (board[i][j], i, j) in seen:
                        return False
                    seen.add((board[i][j], i, j))
                    seen.add((board[i][j], i//3*3, j//3*3))
        return True
