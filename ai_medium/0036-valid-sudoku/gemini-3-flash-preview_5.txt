class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [set() for _ in range(9)]
        cols = [set() for _ in range(9)]
        boxes = [set() for _ in range(9)]

        for r in range(9):
            for c in range(9):
                val = board[r][c]
                if val == ".":
                    continue
                
                # Check rows
                if val in rows[r]:
                    return False
                rows[r].add(val)
                
                # Check columns
                if val in cols[c]:
                    return False
                cols[c].add(val)
                
                # Check 3x3 sub-boxes
                # Formula to determine the box index (0-8)
                box_idx = (r // 3) * 3 + (c // 3)
                if val in boxes[box_idx]:
                    return False
                boxes[box_idx].add(val)
                
        return True
