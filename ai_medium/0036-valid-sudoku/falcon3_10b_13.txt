class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [[0]*10 for _ in range(9)]
        cols = [[0]*10 for _ in range(9)]
        boxes = [[0]*10 for _ in range(9)]

        for i in range(9):
            for j in range(9):
                box_index = (i // 3) * 3 + j // 3
                if board[i][j] != '.':
                    num = int(board[i][j])
                    if rows[i][num] or cols[j][num] or boxes[box_index][num]:
                        return False
                    rows[i][num] = cols[j][num] = boxes[box_index][num] = True
        return True
