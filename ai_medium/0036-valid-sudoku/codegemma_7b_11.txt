class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows, cols, subs = [set() for _ in range(9)], [set() for _ in range(9)], [set() for _ in range(9)]

        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    row, col, sub = i, j // 3, i // 3 * 3 + j // 3

                    if (board[i][j] in rows[row] or
                        board[i][j] in cols[col] or
                        board[i][j] in subs[sub]):
                        return False

                    rows[row].add(board[i][j])
                    cols[col].add(board[i][j])
                    subs[sub].add(board[i][j])

        return True
