class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        rows = [set() for _ in range(9)]
        cols = [set() for _ in range(9)]
        boxes = [[set() for _ in range(3)] for __ in range(3)]
        
        for r in range(9):
            for c in range(9):
                if board[r][c] != '.':
                    num = int(board[r][c])
                    box_i, box_j = divmod(c, 3)
                    box_id = r // 3 * 3 + box_i
                    
                    if num in rows[r] or num in cols[c] or num in boxes[box_id][box_j]:
                        return False
                    
                    rows[r].add(num)
                    cols[c].add(num)
                    boxes[box_id][box_j].add(num)
        
        return True
