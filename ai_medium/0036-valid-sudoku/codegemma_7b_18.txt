class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:
        seen = set()

        # Check rows and columns
        for i in range(9):
            for j in range(9):
                if board[i][j] != '.':
                    if (board[i][j], i, j) in seen or (j, i, board[i][j]) in seen or (i // 3, j // 3, board[i][j]) in seen:
                        return False
                    seen.add((board[i][j], i, j))
                    seen.add((j, i, board[i][j]))
                    seen.add((i // 3, j // 3, board[i][j]))

        return True
