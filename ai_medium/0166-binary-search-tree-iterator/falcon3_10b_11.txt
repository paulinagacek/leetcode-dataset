class BSTIterator:

    def __init__(self, root: Optional[TreeNode]):
        self.stack = []
        self._inorder(root)

    def _inorder(self, node):
        while node:
            self.stack.append(node)
            node = node.left

    def next(self) -> int:
        topmost = self.stack.pop()
        if topmost.right:
            self._inorder(topmost.right)
        return topmost.val

    def hasNext(self) -> bool:
        return len(self.stack) > 0
