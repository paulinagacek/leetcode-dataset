class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()

        def dfs(index, current_sum, current_combination):
            if current_sum == target:
                res.append(current_combination[:])
                return

            if current_sum > target or index >= len(candidates):
                return

            # Explore the option of including the current candidate
            current_combination.append(candidates[index])
            dfs(index + 1, current_sum + candidates[index], current_combination)

            # Explore the option of excluding the current candidate
            current_combination.pop()
            dfs(index + 1, current_sum, current_combination)

        dfs(0, 0, [])
        return res
