class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        results = []
        
        def backtrack(start: int, current_target: int, path: List[int]):
            if current_target == 0:
                results.append(path)
                return
            
            for i in range(start, len(candidates)):
                # If the current number is greater than the remaining target,
                # no subsequent numbers will sum to target because the list is sorted.
                if candidates[i] > current_target:
                    break
                
                # Skip duplicate numbers at the same recursion level to avoid duplicate combinations.
                if i > start and candidates[i] == candidates[i-1]:
                    continue
                
                # Each number may only be used once, so the next start index is i + 1.
                backtrack(i + 1, current_target - candidates[i], path + [candidates[i]])
                
        backtrack(0, target, [])
        return results
