class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        results = []
        
        def backtrack(start: int, current_target: int, path: List[int]):
            if current_target == 0:
                results.append(list(path))
                return
            
            for i in range(start, len(candidates)):
                # Skip duplicate numbers at the same level of recursion
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                
                # If current candidate exceeds target, no need to check further due to sorted order
                if candidates[i] > current_target:
                    break
                
                path.append(candidates[i])
                # Recursively call with the next index (each number used once)
                backtrack(i + 1, current_target - candidates[i], path)
                path.pop()
                
        backtrack(0, target, [])
        return results
