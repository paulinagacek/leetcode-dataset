class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        res = []
        
        def backtrack(start: int, remain: int, path: List[int]):
            if remain == 0:
                res.append(path[:])
                return
            
            for i in range(start, len(candidates)):
                # If the current element is the same as the previous one at this depth, skip it
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                # If the current candidate exceeds the remaining target, no need to look further
                if candidates[i] > remain:
                    break
                
                path.append(candidates[i])
                # Move to the next index since each number can only be used once
                backtrack(i + 1, remain - candidates[i], path)
                path.pop()
                
        backtrack(0, target, [])
        return res
