class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        results = []
        
        def backtrack(start: int, target: int, path: List[int]):
            if target == 0:
                results.append(list(path))
                return
            
            for i in range(start, len(candidates)):
                # If current number is greater than target, no point in continuing
                # since the array is sorted.
                if candidates[i] > target:
                    break
                
                # If current number is the same as the previous one at this depth,
                # skip it to avoid duplicate combinations.
                if i > start and candidates[i] == candidates[i-1]:
                    continue
                
                path.append(candidates[i])
                # Move to the next index as each number can only be used once.
                backtrack(i + 1, target - candidates[i], path)
                path.pop()
        
        backtrack(0, target, [])
        return results
