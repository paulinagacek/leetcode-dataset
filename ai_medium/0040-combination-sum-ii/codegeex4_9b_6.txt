from typing import List

class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        def backtrack(remaining_target, start, path):
            if remaining_target == 0:
                result.append(path)
                return
            if remaining_target < 0:
                return
            
            for i in range(start, len(candidates)):
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                backtrack(remaining_target - candidates[i], i + 1, path + [candidates[i]])
        
        result = []
        candidates.sort()
        backtrack(target, 0, [])
        return result
