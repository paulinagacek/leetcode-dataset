class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        res = []
        
        def backtrack(start: int, target: int, path: List[int]):
            if target == 0:
                res.append(list(path))
                return
            
            for i in range(start, len(candidates)):
                # If the current number is greater than the target, no point in continuing
                if candidates[i] > target:
                    break
                
                # Skip duplicates at the same recursive level
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                
                path.append(candidates[i])
                # Move to next index because each number can be used only once
                backtrack(i + 1, target - candidates[i], path)
                path.pop()
        
        backtrack(0, target, [])
        return res
