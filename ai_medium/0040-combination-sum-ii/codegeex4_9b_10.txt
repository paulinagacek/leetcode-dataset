from typing import List

class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        result = []
        candidates.sort()
        self.helper(candidates, target, [], 0, result)
        return result
    
    def helper(self, nums: List[int], target: int, path: List[int], start: int, result: List[List[int]]):
        if target == 0:
            result.append(path[:])
        else:
            for i in range(start, len(nums)):
                if i > start and nums[i] == nums[i - 1]:
                    continue
                if nums[i] > target:
                    break
                self.helper(nums, target - nums[i], path + [nums[i]], i + 1, result)
