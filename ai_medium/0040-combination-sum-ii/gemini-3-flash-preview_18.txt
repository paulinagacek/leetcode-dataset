class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        result = []
        
        def backtrack(start: int, current_target: int, path: List[int]):
            if current_target == 0:
                result.append(path[:])
                return
            
            for i in range(start, len(candidates)):
                # Skip duplicate elements to avoid duplicate combinations
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                
                # If current candidate is greater than target, no point in checking further
                if candidates[i] > current_target:
                    break
                
                path.append(candidates[i])
                # Recursive call with the next index and reduced target
                backtrack(i + 1, current_target - candidates[i], path)
                path.pop()
                
        backtrack(0, target, [])
        return result
