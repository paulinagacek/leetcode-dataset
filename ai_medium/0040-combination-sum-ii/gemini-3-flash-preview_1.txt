class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates.sort()
        results = []
        
        def backtrack(start: int, target: int, current_path: List[int]):
            if target == 0:
                results.append(list(current_path))
                return
            
            for i in range(start, len(candidates)):
                # Skip duplicate elements at the same recursion level
                if i > start and candidates[i] == candidates[i - 1]:
                    continue
                
                # If current candidate exceeds target, no point in checking further (due to sorted order)
                if candidates[i] > target:
                    break
                
                current_path.append(candidates[i])
                # Move to next index because each element can be used only once
                backtrack(i + 1, target - candidates[i], current_path)
                current_path.pop()
        
        backtrack(0, target, [])
        return results
