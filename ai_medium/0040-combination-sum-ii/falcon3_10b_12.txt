from typing import List

class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        result = []
        candidates.sort()
        self.backtrack(candidates, target, 0, [], result)
        return result

    def backtrack(self, nums, target, start, path, result):
        if target < 0:
            return
        if target == 0:
            result.append(path[:])
            return
        for i in range(start, len(nums)):
            if i > start and nums[i] == nums[i - 1]:
                continue
            self.backtrack(nums, target - nums[i], i + 1, path + [nums[i]], result)
