class Solution:
    def jump(self, nums: List[int]) -> int:
        jumps, cur_end, cur_reach = 0, 0, 0
        for i in range(len(nums) - 1):
            cur_reach = max(cur_reach, i + nums[i])
            if i == cur_end:
                jumps += 1
                cur_end = cur_reach
        return jumps
