class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1:
            return 0
        
        jumps = 0
        current_end = 0
        furthest = 0
        
        for i in range(n - 1):
            # Update the furthest point reachable from the current index
            furthest = max(furthest, i + nums[i])
            
            # If we reach the end of the range reachable by the current number of jumps
            if i == current_end:
                jumps += 1
                current_end = furthest
                
                # If we can already reach the last index, we can stop
                if current_end >= n - 1:
                    break
        
        return jumps
