class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        if n < 2:
            return 0
        
        jumps = 0
        current_end = 0
        farthest = 0
        
        for i in range(n - 1):
            # Update the farthest point reachable from the current index i
            farthest = max(farthest, i + nums[i])
            
            # If we reach the end of the current jump's range
            if i == current_end:
                jumps += 1
                current_end = farthest
                
                # If the destination is already reachable, we can stop
                if current_end >= n - 1:
                    break
                    
        return jumps
