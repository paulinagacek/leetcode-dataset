class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        jumps = [0] * n
        for i in range(1, n):
            j = 0
            while j < i and i <= j + nums[j]:
                if jumps[i] == 0 or jumps[i] > jumps[j] + 1:
                    jumps[i] = jumps[j] + 1
                j += 1
        return jumps[-1]
