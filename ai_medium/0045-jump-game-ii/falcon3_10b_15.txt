from typing import List

class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        reachable = [False] * n
        reachable[0] = True
        
        steps = 0
        current_end = 0
        farthest = 0
        
        for i in range(n - 1):
            if not reachable[i]:
                continue
            
            if i + nums[i] >= n - 1:
                return steps + 1
            
            farthest = max(farthest, i + nums[i])
            if i == current_end:
                steps += 1
                current_end = farthest
        
        return steps
