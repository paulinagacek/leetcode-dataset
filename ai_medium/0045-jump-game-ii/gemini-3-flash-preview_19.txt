class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        # Minimum jumps to reach the last index.
        # Initialize jumps counter, the current reach limit, and the farthest reach possible.
        jumps = 0
        current_end = 0
        farthest = 0
        
        # Iterate through the array except for the last element.
        # We don't need to jump from the last element.
        for i in range(n - 1):
            # Update the farthest index we can reach from the current index.
            farthest = max(farthest, i + nums[i])
            
            # If we've reached the end of the range of the current jump,
            # we must take another jump to move forward.
            if i == current_end:
                jumps += 1
                current_end = farthest
                
                # If we can already reach the last index, we can break early.
                if current_end >= n - 1:
                    break
                    
        return jumps
