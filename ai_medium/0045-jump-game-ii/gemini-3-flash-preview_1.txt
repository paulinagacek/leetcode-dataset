class Solution:
    def jump(self, nums: List[int]) -> int:
        # Get the length of the input array
        n = len(nums)
        
        # If the array has only 1 element, we are already at the destination
        if n <= 1:
            return 0
        
        # Initialize variables to keep track of jumps and range coverage
        # jumps: minimum number of jumps to reach the current range
        # current_end: the farthest index reachable with the current number of jumps
        # farthest: the farthest index reachable with one additional jump
        jumps = 0
        current_end = 0
        farthest = 0
        
        # Iterate through the array until the second to last index
        # We don't need to jump from the last index
        for i in range(n - 1):
            # Update the farthest index we can reach from current position i
            farthest = max(farthest, i + nums[i])
            
            # If we have reached the end of the range of the current jump
            if i == current_end:
                # Increment jump count
                jumps += 1
                # Update the current range to the farthest point found so far
                current_end = farthest
                
                # If we can already reach the last index, we can stop
                if current_end >= n - 1:
                    break
                    
        return jumps
