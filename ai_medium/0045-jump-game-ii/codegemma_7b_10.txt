class Solution:
    def jump(self, nums: List[int]) -> int:
        jumps, curr_reach, max_reach = 0, 0, nums[0]
        for i in range(1, len(nums)):
            max_reach = max(max_reach, i + nums[i])
            if i == curr_reach:
                jumps += 1
                curr_reach = max_reach
        return jumps
