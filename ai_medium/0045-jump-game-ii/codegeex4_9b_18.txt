from typing import List

class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        maxReachable = 0
        jumps = 0
        end = 0
        
        for i in range(n - 1):
            if maxReachable >= i:
                maxReachable = max(maxReachable, i + nums[i])
                if i == end:
                    jumps += 1
                    end = maxReachable
                    
        return jumps
