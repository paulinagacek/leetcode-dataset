from typing import List

class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        jumps = 0
        end = 0
        maxReach = 0
        
        for i in range(n - 1):
            maxReach = max(maxReach, i + nums[i])
            
            if i == end:
                jumps += 1
                end = maxReach
                
                if end >= n - 1:
                    break
        
        return jumps
