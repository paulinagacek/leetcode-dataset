class Solution:
    def jump(self, nums: List[int]) -> int:
        n, end, max_reach, jump_count = len(nums), 0, 0, 0
        
        for i in range(n - 1):
            max_reach = max(max_reach, i + nums[i])
            if i == end:
                jump_count += 1
                end = max_reach
        return jump_count
