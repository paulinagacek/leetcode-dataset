from typing import List

class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        jumps = [0] * n
        for i in range(1, n):
            jumps[i] = float('inf')
            for j in range(i):
                if i <= j + nums[j] and jumps[i] > jumps[j] + 1:
                    jumps[i] = jumps[j] + 1
        return jumps[-1]
