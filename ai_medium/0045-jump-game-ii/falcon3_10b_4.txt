class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        steps = 0
        reach = 0
        farthest = 0
        
        for i in range(n - 1):
            farthest = max(farthest, i + nums[i])
            if i == reach:
                steps += 1
                reach = farthest
        return steps
