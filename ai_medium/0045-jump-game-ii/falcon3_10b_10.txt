class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        max_reach, steps, curr_max = 0, 0, 0
        for i in range(n - 1):
            max_reach = max(max_reach, i + nums[i])
            if i == curr_max:
                curr_max = max_reach
                steps += 1
        return steps
