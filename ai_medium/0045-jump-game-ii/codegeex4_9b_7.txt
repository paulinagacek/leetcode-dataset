from typing import List

class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        jumps = [0] * n
        
        for i in range(1, n):
            min_jumps = float('inf')
            for j in range(i):
                if j + nums[j] >= i:
                    min_jumps = min(min_jumps, jumps[j])
            jumps[i] = min_jumps + 1
        
        return jumps[n - 1]
