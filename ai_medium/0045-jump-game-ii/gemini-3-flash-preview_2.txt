class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1:
            return 0
        
        jumps = 0
        current_end = 0
        farthest = 0
        
        # We iterate to n - 1 because we don't need to jump if we are already at the last index.
        for i in range(n - 1):
            # Update the farthest index we can reach from the current position
            farthest = max(farthest, i + nums[i])
            
            # If we reach the boundary of the current jump reach, increment jumps
            if i == current_end:
                jumps += 1
                current_end = farthest
                
                # If the current_end already reaches or exceeds the last index, we can stop
                if current_end >= n - 1:
                    break
        
        return jumps
