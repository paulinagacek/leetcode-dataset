# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        # Morris traversal based solution
        first_elem, second_elem, prev_elem = None, None, TreeNode(float('-inf'))

        while root:
            if not root.left:
                # Visit node
                if prev_elem.val > root.val and not first_elem:
                    first_elem = prev_elem
                elif prev_elem.val > root.val and first_elem:
                    second_elem = root
                prev_elem = root
                root = root.right
            else:
                # Setup threads
                predecessor = root.left
                while predecessor.right and predecessor.right != root:
                    predecessor = predecessor.right

                if not predecessor.right:
                    predecessor.right = root
                    root = root.left
                else:
                    # Visit node
                    if prev_elem.val > root.val and not first_elem:
                        first_elem = prev_elem
                    elif prev_elem.val > root.val and first_elem:
                        second_elem = root
                    predecessor.right = None
                    prev_elem = root
                    root = root.right

        # Swap the values of the two nodes
        first_elem.val, second_elem.val = second_elem.val, first_elem.val
