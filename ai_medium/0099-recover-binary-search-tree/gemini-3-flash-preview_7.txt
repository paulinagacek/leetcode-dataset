class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        first = second = prev = None
        curr = root
        
        while curr:
            if curr.left:
                # Find the inorder predecessor of curr
                pre = curr.left
                while pre.right and pre.right is not curr:
                    pre = pre.right
                
                if pre.right is None:
                    # Create a temporary thread to the current node
                    pre.right = curr
                    curr = curr.left
                else:
                    # Thread already exists, destroy it and visit curr
                    pre.right = None
                    if prev and prev.val > curr.val:
                        if not first:
                            first = prev
                        second = curr
                    prev = curr
                    curr = curr.right
            else:
                # No left child, visit curr and move right
                if prev and prev.val > curr.val:
                    if not first:
                        first = prev
                    second = curr
                prev = curr
                curr = curr.right
        
        # Swap the values of the two nodes found
        if first and second:
            first.val, second.val = second.val, first.val
