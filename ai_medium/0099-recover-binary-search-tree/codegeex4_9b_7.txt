# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        first_elem = second_elem = prev Elem = None
        
        def inorder(node):
            nonlocal first_elem, second_elem, prev_elem
            if node is None:
                return 
            inorder(node.left)
            if prev_elem and node.val < prev_elem.val:
                second_elem = node
                if first_elem is None:
                    first_elem = prev_elem
            prev_elem = node
            inorder(node.right)

        inorder(root)

        first_elem.val, second_elem.val = second_elem.val, first_elem.val
