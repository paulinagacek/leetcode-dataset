class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        first = second = prev_node = None
        curr = root
        
        # Morris Inorder Traversal for O(1) space complexity
        while curr:
            if curr.left is None:
                # Comparison logic for identifying swapped nodes
                if prev_node and prev_node.val > curr.val:
                    if not first:
                        first = prev_node
                    second = curr
                prev_node = curr
                curr = curr.right
            else:
                # Find the inorder predecessor
                pre = curr.left
                while pre.right and pre.right != curr:
                    pre = pre.right
                
                if pre.right is None:
                    # Create temporary link to the successor
                    pre.right = curr
                    curr = curr.left
                else:
                    # Break the temporary link and visit curr
                    pre.right = None
                    if prev_node and prev_node.val > curr.val:
                        if not first:
                            first = prev_node
                        second = curr
                    prev_node = curr
                    curr = curr.right
        
        # Swap the values of the two nodes found
        if first and second:
            first.val, second.val = second.val, first.val
