# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        # Morris Inorder traversal to find the two swapped nodes
        node1 = node2 = prev = None
        while root:
            if root.left:
                ptr = root.left
                while ptr.right and ptr.right != root:
                    ptr = ptr.right
                if not ptr.right:
                    ptr.right = root
                    root = root.left
                else:
                    if prev and prev.val > root.val:
                        node2 = root
                        if not node1:
                            node1 = prev
                    prev = root
                    ptr.right = None
                    root = root.right
            else:
                if prev and prev.val > root.val:
                    node2 = root
                    if not node1:
                        node1 = prev
                prev = root
                root = root.right
        
        # Swap the values of the two nodes
        node1.val, node2.val = node2.val, node1.val
