# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        morgan = list()
        
        def inorder(node):
            if node is None:
                return
            inorder(node.left)
            morgan.append(node)
            inorder(node.right)
        
        inorder(root)
        a, b = morgan[0], morgan[-1]
        
        for i in range(1, len(morgan)):
            if morgan[i].val < morgan[i-1].val:
                b = morgan[i]
                break
        
        for i in range(len(morgan)-2, -1, -1):
            if morgan[i].val > morgan[i+1].val:
                a = morgan[i]
                break
        
        a.val, b.val = b.val, a.val
