class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        first = second = prev = None
        curr = root
        
        while curr:
            if curr.left:
                # Find the inorder predecessor of curr
                pred = curr.left
                while pred.right and pred.right != curr:
                    pred = pred.right
                
                if pred.right is None:
                    # Create a temporary link to the successor (current node)
                    pred.right = curr
                    curr = curr.left
                else:
                    # Link already exists, remove it and visit the current node
                    pred.right = None
                    if prev and curr.val < prev.val:
                        if not first:
                            first = prev
                        second = curr
                    prev = curr
                    curr = curr.right
            else:
                # No left child, visit the current node
                if prev and curr.val < prev.val:
                    if not first:
                        first = prev
                    second = curr
                prev = curr
                curr = curr.right
        
        # After traversal, swap the values of the identified nodes
        if first and second:
            first.val, second.val = second.val, first.val
