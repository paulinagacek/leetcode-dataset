class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        ops = {"+", "-", "*", "/"}
        
        for tok in tokens:
            if tok not in ops:
                stack.append(int(tok))
                continue
            
            b = stack.pop()
            a = stack.pop()
            
            if tok == "+":
                stack.append(a + b)
            elif tok == "-":
                stack.append(a - b)
            elif tok == "*":
                stack.append(a * b)
            else:  # "/"
                stack.append(int(a / b))  # truncates toward zero
        
        return stack[-1]
