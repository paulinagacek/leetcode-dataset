class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        for token in tokens:
            if token in "+-*/":
                b = stack.pop()
                a = stack.pop()
                result = {"+": a + b, "-": a - b, "*": a * b, "/": int(a / b)}.get(token)
                stack.append(result)
            else:
                stack.append(int(token))
        return stack.pop()
