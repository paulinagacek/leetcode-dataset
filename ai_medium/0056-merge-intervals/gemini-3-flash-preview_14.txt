class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        if not intervals:
            return []
        
        # Sort intervals by their start times
        intervals.sort(key=lambda x: x[0])
        
        merged = []
        for interval in intervals:
            # If merged is empty or the current interval does not overlap with the last,
            # append the current interval.
            if not merged or interval[0] > merged[-1][1]:
                merged.append(interval)
            else:
                # Otherwise, there is an overlap, so we merge the current and last intervals
                # by updating the end time of the last interval in the list.
                merged[-1][1] = max(merged[-1][1], interval[1])
        
        return merged
