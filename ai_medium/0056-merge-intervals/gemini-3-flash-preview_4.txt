class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        if not intervals:
            return []
        
        # Sort intervals by their start times
        intervals.sort(key=lambda x: x[0])
        
        merged = []
        for interval in intervals:
            # If the list of merged intervals is empty or the current interval 
            # does not overlap with the previous, simply append it.
            if not merged or interval[0] > merged[-1][1]:
                merged.append(interval)
            else:
                # Otherwise, there is an overlap, so we merge the current 
                # interval into the previous one by updating the end time.
                merged[-1][1] = max(merged[-1][1], interval[1])
        
        return merged
