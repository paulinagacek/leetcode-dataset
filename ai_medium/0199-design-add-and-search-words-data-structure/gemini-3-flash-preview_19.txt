class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['#'] = True

    def search(self, word: str) -> bool:
        def dfs(index, node):
            curr = node
            for i in range(index, len(word)):
                char = word[i]
                if char == '.':
                    for key, child in curr.items():
                        if key != '#' and dfs(i + 1, child):
                            return True
                    return False
                if char not in curr:
                    return False
                curr = curr[char]
            return '#' in curr
        
        return dfs(0, self.root)
