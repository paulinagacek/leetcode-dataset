class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class WordDictionary:

    def __init__(self):
        self.root = TrieNode()

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True

    def search(self, word: str) -> bool:
        self.res = False
        self._dfs(self.root, word, 0)
        return self.res

    def _dfs(self, node, word, index):
        if index == len(word):
            return node.is_end

        char = word[index]
        if char == '.':
            for child in node.children.values():
                if self._dfs(child, word, index + 1):
                    self.res = True
                    return True
        else:
            child = node.children.get(char)
            if child:
                if self._dfs(child, word, index + 1):
                    self.res = True
                    return True

        return False
