class WordDictionary:
    def __init__(self):
        self.trie = {"#": False}

    def addWord(self, word: str) -> None:
        node = self.trie
        for ch in word:
            if ch not in node:
                node[ch] = {"#": False}
            node = node[ch]
        node["#"] = True

    def search(self, word: str) -> bool:
        def dfs(i: int, node: dict) -> bool:
            if i == len(word):
                return node.get("#", False)

            ch = word[i]
            if ch == ".":
                for k, child in node.items():
                    if k != "#" and dfs(i + 1, child):
                        return True
                return False

            if ch not in node:
                return False
            return dfs(i + 1, node[ch])

        return dfs(0, self.trie)
