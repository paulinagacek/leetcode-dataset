class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['$'] = True

    def search(self, word: str) -> bool:
        def dfs(idx, node):
            curr = node
            for i in range(idx, len(word)):
                char = word[i]
                if char == '.':
                    for key, child_node in curr.items():
                        if key != '$' and dfs(i + 1, child_node):
                            return True
                    return False
                else:
                    if char not in curr:
                        return False
                    curr = curr[char]
            return '$' in curr
        
        return dfs(0, self.root)
