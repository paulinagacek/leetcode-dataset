class WordDictionary:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.trie = {}

    def addWord(self, word: str) -> None:
        """
        Adds a word into the data structure.
        """
        node = self.trie
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['#'] = '#'  # Mark the end of the word

    def search(self, word: str) -> bool:
        """
        Returns if the word is in the data structure. A word could contain the dot character '.' to represent any letter.
        """
        return self.searchHelper(word, self.trie)
        
    def searchHelper(self, word, node):
        for i in range(len(word)):
            char = word[i]
            if char == '.':
                # If current character is '.', check all possible children
                for key in node:
                    if self.searchHelper(word[i+1:], node[key]):
                        return True
                return False
            else:
                # Move to the next level of trie
                if char not in node:
                    return False
                node = node[char]
        # Check if it's a word ending node
        return '#' in node
