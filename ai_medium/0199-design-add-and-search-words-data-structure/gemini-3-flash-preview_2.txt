class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['#'] = True

    def search(self, word: str) -> bool:
        def dfs(index, node):
            for i in range(index, len(word)):
                char = word[i]
                if char == '.':
                    for child in node:
                        if child != '#' and dfs(i + 1, node[child]):
                            return True
                    return False
                if char not in node:
                    return False
                node = node[char]
            return '#' in node
        
        return dfs(0, self.root)
