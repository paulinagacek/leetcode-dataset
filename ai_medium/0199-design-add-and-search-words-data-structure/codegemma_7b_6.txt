class WordDictionary:
    def __init__(self):
        self.tree = {}

    def addWord(self, word: str) -> None:
        curr = self.tree
        for char in word:
            if char not in curr:
                curr[char] = {}
            curr = curr[char]
        curr['word'] = True

    def search(self, word: str) -> bool:
        def dfs(node, i):
            if i == len(word):
                return 'word' in node
            if word[i] == '.':
                for child in node:
                    if dfs(node[child], i + 1):
                        return True
            else:
                if word[i] in node:
                    return dfs(node[word[i]], i + 1)
            return False

        return dfs(self.tree, 0)
