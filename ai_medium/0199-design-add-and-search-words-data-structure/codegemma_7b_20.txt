class WordDictionary:

    def __init__(self):
        self.words = set()

    def addWord(self, word: str) -> None:
        self.words.add(word)

    def search(self, word: str) -> bool:
        for w in self.words:
            if len(w) == len(word):
                match = True
                for i in range(len(word)):
                    if word[i] != '.' and word[i] != w[i]:
                        match = False
                        break
                if match:
                    return True
        return False
