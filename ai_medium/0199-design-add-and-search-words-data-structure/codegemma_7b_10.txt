class WordDictionary:

    def __init__(self):
        self.words = set()

    def addWord(self, word: str) -> None:
        self.words.add(word)

    def search(self, word: str) -> bool:
        for w in self.words:
            if len(word) != len(w):
                continue
            if all(c == '.' or c == w[i] for i, c in enumerate(word)):
                return True
        return False
