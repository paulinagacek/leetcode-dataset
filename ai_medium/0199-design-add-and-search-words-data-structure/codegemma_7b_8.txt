class WordDictionary:

    def __init__(self):
        self.trie = {}

    def addWord(self, word: str) -> None:
        node = self.trie
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['$'] = True

    def search(self, word: str) -> bool:
        def dfs(node, word):
            if not word:
                return '$' in node
            if word[0] == '.':
                for child in node:
                    if dfs(node[child], word[1:]):
                        return True
                return False
            else:
                return word[0] in node and dfs(node[word[0]], word[1:])

        return dfs(self.trie, word)
