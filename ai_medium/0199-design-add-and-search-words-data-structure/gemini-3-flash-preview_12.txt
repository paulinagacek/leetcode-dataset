class WordDictionary:

    def __init__(self):
        self.root = {}

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node:
                node[char] = {}
            node = node[char]
        node['#'] = True

    def search(self, word: str) -> bool:
        word_len = len(word)
        
        def dfs(index, curr_node):
            node = curr_node
            for i in range(index, word_len):
                char = word[i]
                if char == ".":
                    # For wildcards, explore all possible children branches
                    for child_key in node:
                        if child_key != "#":
                            if dfs(i + 1, node[child_key]):
                                return True
                    return False
                else:
                    # For letters, follow the path if it exists
                    if char not in node:
                        return False
                    node = node[char]
            
            # Check if the current node marks the end of a valid word
            return "#" in node
            
        return dfs(0, self.root)
