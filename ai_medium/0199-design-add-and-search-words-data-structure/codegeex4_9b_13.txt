class WordDictionary:

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.words = []


    def addWord(self, word: str) -> None:
        """
        Adds a word into the data structure.
        """
        self.words.append(word)


    def search(self, word: str) -> bool:
        """
        Returns if the word is in the data structure. A word could contain the dot character '.' which means it can be a
        wildcard character.
        """
        from functools import lru_cache

        @lru_cache(None)
        def match(i, j):
            if j == len(word):
                return i == len(self.words[0])
            if i == len(self.words[0]):
                return False
            if word[j] == '.':
                return match(i + 1, j + 1) or match(i + 1, j)
            if self.words[0][i] == word[j]:
                return match(i + 1, j + 1)
            return False

        for i in range(len(self.words)):
            if match(0, 0):
                return True
        return False
