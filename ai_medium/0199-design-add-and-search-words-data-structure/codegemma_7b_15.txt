class WordDictionary:

    def __init__(self):
        self.trie = {}

    def addWord(self, word: str) -> None:
        cur = self.trie
        for char in word:
            if char not in cur:
                cur[char] = {}
            cur = cur[char]
        cur['$'] = True

    def search(self, word: str) -> bool:
        def dfs(index, node):
            if index == len(word):
                return '$' in node
            if word[index] == '.':
                for child in node:
                    if dfs(index + 1, node[child]):
                        return True
            else:
                if word[index] in node:
                    return dfs(index + 1, node[word[index]])
            return False

        return dfs(0, self.trie)
