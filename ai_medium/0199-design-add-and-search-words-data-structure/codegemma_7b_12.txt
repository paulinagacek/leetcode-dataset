class WordDictionary:

    def __init__(self):
        self.word_dict = {}

    def addWord(self, word: str) -> None:
        self.word_dict[word] = True

    def search(self, word: str) -> bool:
        if word in self.word_dict:
            return True

        for key in self.word_dict.keys():
            if len(key) == len(word):
                count = 0
                for i in range(len(key)):
                    if key[i] == word[i] or word[i] == '.':
                        count += 1
                if count == len(word):
                    return True

        return False
