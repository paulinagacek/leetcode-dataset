class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class WordDictionary:

    def __init__(self):
        self.root = TrieNode()

    def addWord(self, word: str) -> None:
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True

    def search(self, word: str) -> bool:
        self.dfs = 0  # to limit number of backtracking steps during DFS
        return self.dfs_search(self.root, word, 0)

    def dfs_search(self, node: TrieNode, word: str, index: int) -> bool:
        if self.dfs > len(word):
            return False  # limit backtracking to avoid infinite loop in cases like "a.b.c.d."
        if index == len(word):
            return node.is_end
        char = word[index]
        if char == '.':
            for child in node.children.values():
                if self.dfs_search(child, word, index + 1):
                    return True
            return False
        if char not in node.children:
            return False
        return self.dfs_search(node.children[char], word, index + 1)
        

# Example usage
wordDictionary = WordDictionary()
wordDictionary.addWord("bad")
wordDictionary.addWord("dad")
wordDictionary.addWord("mad")
print(wordDictionary.search("pad"))  # Output: False
print(wordDictionary.search("bad"))  # Output: True
print(wordDictionary.search(".ad"))  # Output: True
print(wordDictionary.search("b.."))  # Output: True
