class Solution:
    def numTrees(self, n: int) -> int:
        def count_trees(num):
            if num <= 1:
                return 1
            total = 0
            for i in range(num):
                left_trees = count_trees(i)
                right_trees = count_trees(num - i - 1)
                total += left_trees * right_trees
            return total
        
        return count_trees(n)
