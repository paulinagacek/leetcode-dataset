class Solution:
    def numTrees(self, n: int) -> int:
        dp = [0] * (n + 1)
        dp[0] = 1
        for nodes in range(1, n + 1):
            total = 0
            for left in range(nodes):
                total += dp[left] * dp[nodes - 1 - left]
            dp[nodes] = total
        return dp[n]
