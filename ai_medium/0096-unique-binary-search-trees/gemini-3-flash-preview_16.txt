class Solution:
    def numTrees(self, n: int) -> int:
        # The number of unique BSTs with n nodes is the n-th Catalan number.
        # G(n) = sum(G(i-1) * G(n-i)) for i from 1 to n.
        dp = [0] * (n + 1)
        dp[0] = 1
        dp[1] = 1
        
        for i in range(2, n + 1):
            for j in range(i):
                dp[i] += dp[j] * dp[i - 1 - j]
                
        return dp[n]
