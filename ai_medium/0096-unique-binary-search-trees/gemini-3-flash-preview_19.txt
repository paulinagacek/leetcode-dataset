class Solution:
    def numTrees(self, n: int) -> int:
        # G[i] will store the number of unique BST's for i nodes
        G = [0] * (n + 1)
        G[0] = 1
        G[1] = 1
        
        # We use the formula for Catalan numbers: G(n) = sum(G(i-1) * G(n-i)) for i from 1 to n
        for i in range(2, n + 1):
            for j in range(1, i + 1):
                G[i] += G[j - 1] * G[i - j]
                
        return G[n]
