class Solution:
    def numTrees(self, n: int) -> int:
        dp = [0] * (n + 1)
        dp[0] = 1
        dp[1] = 1
        
        for i in range(2, n + 1):
            for j in range(i):
                # dp[j] is the number of unique BSTs with j nodes in the left subtree
                # dp[i - 1 - j] is the number of unique BSTs with (i - 1 - j) nodes in the right subtree
                dp[i] += dp[j] * dp[i - 1 - j]
        
        return dp[n]
