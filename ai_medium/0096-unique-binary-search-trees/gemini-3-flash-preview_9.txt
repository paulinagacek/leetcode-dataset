class Solution:
    def numTrees(self, n: int) -> int:
        # dp[i] stores the number of unique BSTs that can be formed with i nodes
        dp = [0] * (n + 1)
        
        # Base cases: 
        # A tree with 0 nodes is empty (1 way)
        # A tree with 1 node has only the root (1 way)
        dp[0] = 1
        dp[1] = 1
        
        # Fill the dp table using the recursive formula:
        # G(n) = sum from i=1 to n of (G(i-1) * G(n-i))
        # where i is the root of the BST
        for nodes in range(2, n + 1):
            for root in range(1, nodes + 1):
                dp[nodes] += dp[root - 1] * dp[nodes - root]
                
        return dp[n]
