class Solution:
    def numTrees(self, n: int) -> int:
        # dp[i] stores the number of structurally unique BST's with i nodes.
        dp = [0] * (n + 1)
        dp[0] = 1
        
        # Iterate through each possible number of nodes from 1 to n.
        for i in range(1, n + 1):
            # For a BST with i nodes, pick each value j (from 0 to i-1) 
            # as the number of nodes in the left subtree.
            # Then there will be (i - 1 - j) nodes in the right subtree.
            for j in range(i):
                dp[i] += dp[j] * dp[i - 1 - j]
                
        return dp[n]
