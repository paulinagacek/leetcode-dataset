class Solution:
    dp = [0] * 20

    def numTrees(self, n: int) -> int:
        if self.dp[n] > 0:
            return self.dp[n]

        if n <= 1:
            return 1

        result = 0
        for i in range(1, n + 1):
            result += self.numTrees(i - 1) * self.numTrees(n - i)

        self.dp[n] = result
        return result
