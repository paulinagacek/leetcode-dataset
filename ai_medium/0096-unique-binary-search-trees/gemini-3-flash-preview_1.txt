class Solution:
    def numTrees(self, n: int) -> int:
        # dp[i] will store the number of unique BST's for i nodes
        dp = [0] * (n + 1)
        
        # Base case: A tree with 0 nodes or 1 node has only 1 unique structure
        dp[0] = 1
        
        # Iterate through each node count from 1 to n
        for i in range(1, n + 1):
            # For a tree with i nodes, pick each value j (from 0 to i-1) to be the number 
            # of nodes in the left subtree. Then the right subtree will have (i - 1 - j) nodes.
            for j in range(i):
                dp[i] += dp[j] * dp[i - 1 - j]
        
        return dp[n]
