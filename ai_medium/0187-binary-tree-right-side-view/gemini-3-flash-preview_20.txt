class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        res = []
        
        def dfs(node, depth):
            if not node:
                return
            
            # If this is the first time we've reached this depth, 
            # the node must be the rightmost one seen so far.
            if depth == len(res):
                res.append(node.val)
            
            # Prioritize the right child to ensure we see the right side first
            dfs(node.right, depth + 1)
            dfs(node.left, depth + 1)
            
        dfs(root, 0)
        return res
