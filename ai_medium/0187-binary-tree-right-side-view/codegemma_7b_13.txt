from collections import deque

class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        if not root:
            return []

        queue = deque([(root, 0)])
        result = []
        max_depth = -1

        while queue:
            node, depth = queue.popleft()

            if depth > max_depth:
                result.append(node.val)
                max_depth = depth

            if node.right:
                queue.append((node.right, depth + 1))
            if node.left:
                queue.append((node.left, depth + 1))

        return result
