class Solution:
    def rightSideView(self, root: Optional['TreeNode']) -> List[int]:
        if not root:
            return []
        
        res = []
        q = deque([root])
        
        while q:
            level_size = len(q)
            last_val = None
            for _ in range(level_size):
                node = q.popleft()
                last_val = node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.append(last_val)
        
        return res
