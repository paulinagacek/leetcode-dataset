class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        dp = [[] for _ in range(n+1)]
        dp[0] = [None]
        dp[1] = [TreeNode(1)]
        for i in range(2, n+1):
            for j in range(1, i+1):
                for left in dp[j-1]:
                    for right in dp[i-j]:
                        dp[i].append(TreeNode(j, left, right))
        return dp[n]
