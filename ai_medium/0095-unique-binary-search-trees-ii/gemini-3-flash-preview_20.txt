class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}

        def buildTrees(start, end):
            if start > end:
                return [None]
            if (start, end) in memo:
                return memo[(start, end)]
            
            all_trees = []
            for i in range(start, end + 1):
                # All possible left subtrees for root i
                left_trees = buildTrees(start, i - 1)
                # All possible right subtrees for root i
                right_trees = buildTrees(i + 1, end)
                
                # Combine each left and right subtree with root i
                for l in left_trees:
                    for r in right_trees:
                        current_tree = TreeNode(i)
                        current_tree.left = l
                        current_tree.right = r
                        all_trees.append(current_tree)
            
            memo[(start, end)] = all_trees
            return all_trees

        return buildTrees(1, n)
