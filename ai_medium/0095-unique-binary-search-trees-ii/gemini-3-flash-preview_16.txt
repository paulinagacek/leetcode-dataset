class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}

        def generate_subtree(start, end):
            if start > end:
                return [None]
            
            if (start, end) in memo:
                return memo[(start, end)]
            
            all_trees = []
            for i in range(start, end + 1):
                # Generate all possible left subtrees
                left_subtrees = generate_subtree(start, i - 1)
                
                # Generate all possible right subtrees
                right_subtrees = generate_subtree(i + 1, end)
                
                # Connect the current root node (i) with every combination of subtrees
                for left in left_subtrees:
                    for right in right_subtrees:
                        root = TreeNode(i)
                        root.left = left
                        root.right = right
                        all_trees.append(root)
            
            memo[(start, end)] = all_trees
            return all_trees

        return generate_subtree(1, n)
