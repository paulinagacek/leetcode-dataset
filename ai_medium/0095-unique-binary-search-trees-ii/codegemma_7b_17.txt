class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        return self.dfs(1, n)

    def dfs(self, start: int, end: int) -> List[Optional[TreeNode]]:
        if start > end:
            return [None]
        result = []
        for i in range(start, end + 1):
            left = self.dfs(start, i - 1)
            right = self.dfs(i + 1, end)
            for l in left:
                for r in right:
                    result.append(TreeNode(i, l, r))
        return result
