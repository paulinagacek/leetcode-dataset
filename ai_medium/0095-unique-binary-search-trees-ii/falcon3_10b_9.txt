# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        def generateTreesUtil(start, end):
            if start > end:
                return [None]
            all_trees = []
            for i in range(start, end+1):
                left_child_trees = generateTreesUtil(start, i-1)
                right_child_trees = generateTreesUtil(i+1, end)
                for l in left_child_trees:
                    for r in right_child_trees:
                        node = TreeNode(i)
                        node.left = l
                        node.right = r
                        all_trees.append(node)
            return all_trees
        
        return generateTreesUtil(1, n)
