class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}

        def generate_trees_range(start, end):
            if start > end:
                return [None]
            if (start, end) in memo:
                return memo[(start, end)]
            
            all_trees = []
            for i in range(start, end + 1):
                # Generate all possible left and right subtrees
                left_trees = generate_trees_range(start, i - 1)
                right_trees = generate_trees_range(i + 1, end)
                
                # Combine each left and right subtree with the root i
                for l in left_trees:
                    for r in right_trees:
                        current_tree = TreeNode(i)
                        current_tree.left = l
                        current_tree.right = r
                        all_trees.append(current_tree)
            
            memo[(start, end)] = all_trees
            return all_trees
        
        return generate_trees_range(1, n)
