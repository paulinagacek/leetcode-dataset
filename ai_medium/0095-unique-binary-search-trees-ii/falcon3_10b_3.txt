# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        def generateTreesHelper(start, end):
            if start > end:
                return [None]
            all_trees = []
            for i in range(start, end + 1):
                # All trees with root value `i`
                left_subtrees = generateTreesHelper(start, i - 1)
                right_subtrees = generateTreesHelper(i + 1, end)
                for l in left_subtrees:
                    for r in right_subtrees:
                        node = TreeNode(i)
                        node.left = l
                        node.right = r
                        all_trees.append(node)
            return all_trees
        
        if n == 0:
            return []
        return generateTreesHelper(1, n)
