class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}
        
        def generate(start, end):
            if start > end:
                return [None]
            if (start, end) in memo:
                return memo[(start, end)]
            
            res = []
            # Any value i in the range [start, end] can serve as the root.
            for i in range(start, end + 1):
                # Recursively generate all possible left and right subtrees.
                left_sub = generate(start, i - 1)
                right_sub = generate(i + 1, end)
                
                # Combine every left subtree with every right subtree for the current root.
                for l in left_sub:
                    for r in right_sub:
                        root = TreeNode(i, l, r)
                        res.append(root)
            
            memo[(start, end)] = res
            return res
            
        return generate(1, n)
