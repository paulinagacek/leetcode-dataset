class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        def generate(left, right):
            if left > right:
                return [None]

            ans = []
            for root in range(left, right + 1):
                for left_tree in generate(left, root - 1):
                    for right_tree in generate(root + 1, right):
                        node = TreeNode(root)
                        node.left = left_tree
                        node.right = right_tree
                        ans.append(node)
            return ans
        
        return generate(1, n)
