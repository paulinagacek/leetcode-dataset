class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}

        def buildTrees(start, end):
            if start > end:
                return [None]
            
            if (start, end) in memo:
                return memo[(start, end)]
            
            all_trees = []
            for i in range(start, end + 1):
                # Generate all possible left subtrees for the current root i
                left_trees = buildTrees(start, i - 1)
                # Generate all possible right subtrees for the current root i
                right_trees = buildTrees(i + 1, end)
                
                # Combine every left subtree with every right subtree for the current root i
                for l in left_trees:
                    for r in right_trees:
                        current_root = TreeNode(i)
                        current_root.left = l
                        current_root.right = r
                        all_trees.append(current_root)
            
            memo[(start, end)] = all_trees
            return all_trees

        return buildTrees(1, n)
