class Solution:
    def generateTrees(self, n: int) -> List[Optional[TreeNode]]:
        if n == 0:
            return []
        
        memo = {}

        def build_trees(start, end):
            if start > end:
                return [None]
            
            if (start, end) in memo:
                return memo[(start, end)]
            
            all_trees = []
            # Iterate through each number from start to end to use as the root
            for i in range(start, end + 1):
                # Generate all possible left and right subtrees
                left_trees = build_trees(start, i - 1)
                right_trees = build_trees(i + 1, end)
                
                # Combine each left and right subtree with the current root
                for left in left_trees:
                    for right in right_trees:
                        root = TreeNode(i)
                        root.left = left
                        root.right = right
                        all_trees.append(root)
            
            memo[(start, end)] = all_trees
            return all_trees

        return build_trees(1, n)
