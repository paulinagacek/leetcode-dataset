class Solution:
    def solve(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        if not board or not board[0]:
            return
        
        m, n = len(board), len(board[0])
        stack = []
        
        # Identify all 'O' cells on the boundary and mark them as 'S' (Safe from capture)
        for r in range(m):
            for c in range(n):
                if board[r][c] == 'O' and (r == 0 or r == m - 1 or c == 0 or c == n - 1):
                    board[r][c] = 'S'
                    stack.append((r, c))
        
        # Perform iterative DFS to find all 'O' cells connected to the boundary 'O's
        while stack:
            curr_r, curr_c = stack.pop()
            # Check the four adjacent directions
            for dr, dc in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                nr, nc = curr_r + dr, curr_c + dc
                # If an adjacent cell is 'O', it's also connected to the boundary
                if 0 <= nr < m and 0 <= nc < n and board[nr][nc] == 'O':
                    board[nr][nc] = 'S'
                    stack.append((nr, nc))
        
        # Final pass through the board:
        # 1. Flip all remaining 'O' cells (those not marked 'S') to 'X' (captured)
        # 2. Flip all 'S' cells back to 'O' (not surrounded)
        for r in range(m):
            for c in range(n):
                if board[r][c] == 'O':
                    board[r][c] = 'X'
                elif board[r][c] == 'S':
                    board[r][c] = 'O'
