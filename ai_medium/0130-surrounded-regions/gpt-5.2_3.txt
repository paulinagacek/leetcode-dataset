class Solution:
    def solve(self, board: List[List[str]]) -> None:
        if not board or not board[0]:
            return

        m, n = len(board), len(board[0])
        q = deque()

        def add(i: int, j: int) -> None:
            if 0 <= i < m and 0 <= j < n and board[i][j] == 'O':
                board[i][j] = 'T'
                q.append((i, j))

        for j in range(n):
            add(0, j)
            add(m - 1, j)
        for i in range(1, m - 1):
            add(i, 0)
            add(i, n - 1)

        while q:
            i, j = q.popleft()
            add(i - 1, j)
            add(i + 1, j)
            add(i, j - 1)
            add(i, j + 1)

        for i in range(m):
            row = board[i]
            for j in range(n):
                if row[j] == 'O':
                    row[j] = 'X'
                elif row[j] == 'T':
                    row[j] = 'O'
