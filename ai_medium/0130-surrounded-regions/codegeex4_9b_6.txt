class Solution:
    def solve(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        if not board or not board[0]:
            return
        
        m, n = len(board), len(board[0])
        
        def capture(r, c):
            if r < 0 or c < 0 or r >= m or c >= n or board[r][c] != 'O':
                return
            board[r][c] = 'T'
            capture(r + 1, c)
            capture(r - 1, c)
            capture(r, c + 1)
            capture(r, c - 1)
        
        # Start by capturing all 'O's that are connected to the top or bottom edge
        for i in range(m):
            capture(i, 0)
            capture(i, n - 1)
        
        # Capture all 'O's that are connected to the left or right edge
        for j in range(n):
            capture(0, j)
            capture(m - 1, j)
        
        # Replace all 'O' with 'X'
        for i in range(m):
            for j in range(n):
                if board[i][j] == 'O':
                    board[i][j] = 'X'
        
        # Recover 'O's that were not captured
        for i in range(m):
            for j in range(n):
                if board[i][j] == 'T':
                    board[i][j] = 'O'
