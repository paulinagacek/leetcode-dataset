class Solution:
    def solve(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        if not board or not board[0]:
            return
        
        m, n = len(board), len(board[0])
        queue = collections.deque()
        
        def mark_as_safe(r, c):
            if board[r][c] == 'O':
                board[r][c] = 'S'
                queue.append((r, c))
        
        # Identify 'O's on the boundary and mark them as safe ('S')
        for r in range(m):
            mark_as_safe(r, 0)
            if n > 1:
                mark_as_safe(r, n - 1)
        for c in range(1, n - 1):
            mark_as_safe(0, c)
            if m > 1:
                mark_as_safe(m - 1, c)
        
        # Perform BFS starting from safe boundary 'O's to find all connected 'O's
        while queue:
            r, c = queue.popleft()
            for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < m and 0 <= nc < n and board[nr][nc] == 'O':
                    board[nr][nc] = 'S'
                    queue.append((nr, nc))
        
        # Final pass: 
        # 'S' indicates an 'O' reachable from the boundary (safe).
        # Any remaining 'O' is surrounded and should be captured ('X').
        for r in range(m):
            for c in range(n):
                if board[r][c] == 'S':
                    board[r][c] = 'O'
                else:
                    board[r][c] = 'X'
