class Solution:
    def levelOrderBottom(self, root: Optional['TreeNode']) -> List[List[int]]:
        if not root:
            return []
        
        q = deque([root])
        res = deque()
        
        while q:
            level = []
            for _ in range(len(q)):
                node = q.popleft()
                level.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.appendleft(level)
        
        return list(res)
