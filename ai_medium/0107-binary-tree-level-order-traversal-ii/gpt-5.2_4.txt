class Solution:
    def levelOrderBottom(self, root: Optional['TreeNode']) -> List[List[int]]:
        if not root:
            return []
        
        q = deque([root])
        levels = deque()
        
        while q:
            size = len(q)
            level = []
            for _ in range(size):
                node = q.popleft()
                level.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            levels.appendleft(level)
        
        return list(levels)
