class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        spiral_order = []
        if not matrix or not matrix[0]:
            return spiral_order
        
        top, bottom, left, right = 0, len(matrix) - 1, 0, len(matrix[0]) - 1
        direction = 0
        
        while top <= bottom and left <= right:
            if direction == 0:  # moving right
                for i in range(left, right + 1):
                    spiral_order.append(matrix[top][i])
                top += 1
            elif direction == 1:  # moving down
                for i in range(top, bottom + 1):
                    spiral_order.append(matrix[i][right])
                right -= 1
            elif direction == 2:  # moving left
                for i in range(right, left - 1, -1):
                    spiral_order.append(matrix[bottom][i])
                bottom -= 1
            elif direction == 3:  # moving up
                for i in range(bottom, top - 1, -1):
                    spiral_order.append(matrix[i][left])
                left += 1
            
            direction = (direction + 1) % 4
        
        return spiral_order
