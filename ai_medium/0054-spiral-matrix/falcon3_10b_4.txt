class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        result = []
        while matrix:
            result += matrix.pop(0)
            if matrix and matrix[0]:
                for row in matrix:
                    result.append(row.pop())
            if matrix:
                result += list(reversed(matrix.pop()))
            if matrix and matrix[0]:
                for row in list(reversed(matrix)):
                    result.append(row.pop(0))
        return result
