class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        res = []
        if not matrix or not matrix[0]:
            return res
        
        m, n = len(matrix), len(matrix[0])
        upper_bound = lower_bound = 0
        left_bound = right_bound = 0
        
        while upper_bound < m and lower_bound < m and left_bound < n and right_bound < n:
            # Traverse from left to right.
            for i in range(left_bound, right_bound + 1):
                res.append(matrix[upper_bound][i])
            upper_bound += 1
            
            # Traverse downwards.
            for j in range(upper_bound, lower_bound + 1):
                res.append(matrix[j][right_bound])
            right_bound -= 1
            
            if upper_bound < m and lower_bound < m:
                # Traverse from right to left.
                for i in range(right_bound, left_bound - 1, -1):
                    res.append(matrix[lower_bound][i])
                lower_bound += 1
            
            if left_bound < n and right_bound < n:
                # Traverse upwards.
                for j in range(lower_bound, upper_bound - 1, -1):
                    res.append(matrix[j][left_bound])
                left_bound += 1
        
        return res
