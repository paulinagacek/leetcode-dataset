from typing import List

class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        res = []
        if not matrix or not matrix[0]:
            return res
        
        m, n = len(matrix), len(matrix[0])
        upper_bound = lower_bound = 0
        left_bound = right_bound = 0
        
        while len(res) < m * n:
            if upper_bound < m and left_bound < n:
                for j in range(left_bound, n):
                    res.append(matrix[upper_bound][j])
                upper_bound += 1
            
            if upper_bound < m and left_bound < n:
                for i in range(upper_bound, m):
                    res.append(matrix[i][n - 1])
                right_bound += 1
            
            if upper_bound < m and left_bound < n:
                for j in range(n - 1, left_bound - 1, -1):
                    res.append(matrix[m - 1][j])
                lower_bound += 1
            
            if upper_bound < m and left_bound < n:
                for i in range(m - 1, upper_bound - 1, -1):
                    res.append(matrix[i][left_bound])
                left_bound += 1
        
        return res
