class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Calculate the maximum length of a word in the dictionary to optimize the search
        max_len = 0
        for word in wordDict:
            max_len = max(max_len, len(word))
            
        for i in range(1, n + 1):
            # Check substrings s[j:i] starting from the end of the current prefix
            # Optimization: only check j such that the word length (i-j) is within bounds
            for j in range(i - 1, max(-1, i - max_len - 1), -1):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
                    
        return dp[n]
