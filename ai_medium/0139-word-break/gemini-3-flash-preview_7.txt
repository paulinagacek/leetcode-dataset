class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Optimization: track the maximum length of a word in the dictionary
        # to reduce the inner loop iterations.
        max_len = 0
        for word in wordDict:
            max_len = max(max_len, len(word))
            
        for i in range(1, n + 1):
            # Only check preceding indices j such that s[j:i] could be a word in wordDict
            for j in range(i - 1, max(-1, i - max_len - 1), -1):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
                    
        return dp[n]
