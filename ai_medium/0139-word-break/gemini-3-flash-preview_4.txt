class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        # dp[i] is True if s[0:i] can be segmented into words from wordDict
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Optimization: calculate the maximum length of a word in the dictionary
        max_len = max(len(word) for word in wordDict)
        
        for i in range(1, n + 1):
            # Check substrings s[j:i] for all j such that s[0:j] is segmentable
            # and the length (i-j) is within the maximum word length in the dictionary
            for j in range(max(0, i - max_len), i):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
                    
        return dp[n]
