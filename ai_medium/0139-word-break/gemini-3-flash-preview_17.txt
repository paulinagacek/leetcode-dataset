class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        # dp[i] is true if the prefix s[0:i] can be segmented into words from wordDict
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Iterate through every prefix length from 1 to n
        for i in range(1, n + 1):
            # Check every potential split point j for the current prefix s[0:i]
            for j in range(i):
                # If s[0:j] can be segmented and s[j:i] is a dictionary word, then s[0:i] can be segmented
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
        
        return dp[n]
