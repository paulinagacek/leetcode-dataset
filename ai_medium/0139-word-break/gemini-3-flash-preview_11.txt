class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        dp = [False] * (n + 1)
        dp[0] = True
        
        # We can optimize the inner loop by only checking up to the length 
        # of the longest word in the dictionary.
        max_len = max(len(word) for word in wordDict) if wordDict else 0
        
        for i in range(1, n + 1):
            # Check substrings s[j:i] where j is between (i - max_len) and (i - 1)
            for j in range(max(0, i - max_len), i):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
                    
        return dp[n]
