class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Optimization: track the maximum length of a word in the dictionary
        max_len = 0
        for word in wordDict:
            if len(word) > max_len:
                max_len = len(word)
                
        for i in range(1, n + 1):
            # Check substrings s[j:i] ending at i
            # j only needs to go back as far as the longest word in the dictionary
            for j in range(i - 1, max(-1, i - max_len - 1), -1):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
        
        return dp[n]
