class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        # Pre-calculate the maximum word length to prune search space
        max_len = max(len(word) for word in word_set)
        
        # dp[i] will be True if s[0:i] can be segmented into words in wordDict
        dp = [False] * (n + 1)
        dp[0] = True  # Base case: empty string can be considered segmented
        
        for i in range(1, n + 1):
            # Check substrings s[j:i] ending at i
            # Starting from the end of the substring (j = i-1) moving backwards
            for j in range(i - 1, -1, -1):
                # If the current word length exceeds the max length in dictionary, stop
                if i - j > max_len:
                    break
                
                # If s[0:j] can be segmented and s[j:i] is a valid word, then s[0:i] is segmentable
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
        
        return dp[n]
