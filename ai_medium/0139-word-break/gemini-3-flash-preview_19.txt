class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        dp = [False] * (n + 1)
        dp[0] = True
        
        # Optimization: track the maximum word length to narrow the inner loop search
        max_word_len = 0
        for word in wordDict:
            max_word_len = max(max_word_len, len(word))
            
        for i in range(1, n + 1):
            # Only check preceding positions within the range of the longest word
            for j in range(max(0, i - max_word_len), i):
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
                    
        return dp[n]
