class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        word_set = set(wordDict)
        n = len(s)
        # dp[i] is True if s[0...i-1] can be segmented into words from wordDict
        dp = [False] * (n + 1)
        dp[0] = True
        
        for i in range(1, n + 1):
            for j in range(i):
                # If s[0...j-1] is valid and s[j...i-1] is in dictionary, then s[0...i-1] is valid
                if dp[j] and s[j:i] in word_set:
                    dp[i] = True
                    break
        
        return dp[n]
