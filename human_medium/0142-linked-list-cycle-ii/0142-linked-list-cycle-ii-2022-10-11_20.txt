class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        fast ,slow = head ,head
    
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next

            if slow == fast:
                slow = head
                while fast != slow:
                    slow = slow.next
                    fast = fast.next         
                return fast 

        return None
