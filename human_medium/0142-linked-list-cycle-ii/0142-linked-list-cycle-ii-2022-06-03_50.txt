class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        if head is None or head.next is None or head.next.next is None:
            return None
        
        
        slow=head
        fast=head
        
        while fast.next is not None and fast.next.next is not None:
            slow=slow.next
            fast=fast.next.next
            
            if slow==fast:
                entry=head
                while entry!=slow:
                    slow=slow.next
                    entry=entry.next
                return entry
            
        return None
