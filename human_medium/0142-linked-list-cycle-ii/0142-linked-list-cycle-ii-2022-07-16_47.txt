class Solution:
    def detectCycle(self, head: ListNode) -> ListNode:
        if head==None or head.next==None:
            return None
        slow = head
        fast = head
        flag = 0
        while slow and fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            if slow==fast:
                break
        if fast==None or fast.next==None:
            return None
        slow = head
        pos = 0
        while fast!=slow:
            slow=slow.next
            fast=fast.next
        return slow
