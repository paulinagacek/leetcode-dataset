class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None: return None
        slow = head
        fast = head.next
        while slow is not fast:
            if fast is None or fast.next is None: return None
            slow = slow.next
            fast = fast.next.next
        slow = slow.next
        while slow != head:
            slow = slow.next
            head = head.next
        return slow
