class Solution:
    def detectCycle(self, head: ListNode) -> ListNode:

        slow, fast = head, head

        # check cycle exist or not
        while fast and fast.next:

            slow = slow.next
            fast = fast.next.next

            if slow == fast:
				# slow and fast meet somewhere in linked list, it has cycle
                break
				
        else:
			# no cycle
            return None


        # locate junction point from head node
        cur = head

        while cur != slow:
            cur = cur.next
            slow = slow.next
        
        return cur
