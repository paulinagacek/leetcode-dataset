class Solution(object):
    def findRepeatedDnaSequences(self, s):
        """
        :type s: str
        :rtype: List[str]
        """
        seq = set()
        res = set()
        for i in range(len(s)-9):
            if s[i:i+10] in seq:
                res.add(s[i:i+10])
            else:
                seq.add(s[i:i+10])
        return list(res)
