class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        ## RC ##
		## APPROACH : 1044. Longest Duplicate Substring ##
        ## TYPICAL SUBSTRING SEARCH PROBLEM ## WE HAVE MANY ALGORITHMS LIKE KMP, Rabin-Karp etc to solve this, Have a look at those.. ##
        
		## Time complexity : O((N\u2212L)L) ##
		## Space complexity : O((N\u2212L)L) ##
        
        L, n = 10, len(s)     
        seen, output = set(), set()
        for i in range(n - L + 1):
            tmp = s[i:i + L]
            if tmp in seen:
                output.add(tmp[:])
            seen.add(tmp)
        return output
