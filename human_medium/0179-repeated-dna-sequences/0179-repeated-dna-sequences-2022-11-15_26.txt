class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        seq = {}
        for i in range(len(s) - 9):
            if s[i:i + 10] not in seq:
                seq[s[i:i + 10]] = 0
            seq[s[i:i + 10]] += 1
        return [key for key, value in seq.items() if value > 1]
