class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:   
        if root==None:
            return []
        stack=[root]
        while(len(stack)>0):
            curr=stack.pop()
            if curr.right:
                stack.append(curr.right)
            if curr.left:
                stack.append(curr.left)
            if (len(stack)>0):
                curr.right=stack[-1]
            curr.left=None
