class Solution(object):
    def flatten(self, root):
        while root != None:
            if root.left == None:
                root = root.right 
            else:
                temp = root.right
                root.right = root.left
                root.left = None
                root = root.right
                right = root
                while right.right != None:
                    right = right.right
                right.right = temp
