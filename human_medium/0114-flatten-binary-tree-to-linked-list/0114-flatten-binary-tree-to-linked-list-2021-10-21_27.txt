class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:
        res=[]
        self.get(root,res)
        for i in res[1:]:
            root.right=i
            root.left=None
            root=root.right
    
    def get(self,root,res):
        if root==None:
            return
        res.append(root)
        self.get(root.left,res)
        self.get(root.right,res)
