class Solution(object):
    def flatten(self, root):
        """
        :type root: TreeNode
        :rtype: None Do not return anything, modify root in-place instead.
        """
        
        if not root:
            return None
        
        st = [root]
        dummy_head = TreeNode(0)
        it = dummy_head
        while st:
            cur = st.pop()
            if cur.right:
                st.append(cur.right)
            if cur.left:
                st.append(cur.left)
                
            it.right = cur
            it = it.right
            it.left = it.right = None
        return dummy_head.right
