class Solution:
    def flatten(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        # call helper function to reorganize the subtree
        self.helper(root)
    
    # reorganize function
    def helper(self, node):
        if node:
            if node.left:
                
                # helper function helps me reorganize the left subtree
                node.left = self.helper(node.left)
                
                # if there is a right subtree 
                if node.right:
                    
                    # reorganize the right subtree and store it in temporary pointer
                    tmp = self.helper(node.right)
                    
                    # replace the right subtree with the organized left subtree
                    node.right = node.left
                    
                    # create a current pointer point to the head of right subtree
                    cur = node.right
                    
                    # loop till the end of the organized right subtreee
                    while cur.right:
                        cur = cur.right
                    
                    # reconnect back the temporary pointer(previous organized right subtree) 
                    # to the end of current right subtree
                    cur.right = tmp
                    
                    # reset the left subtree since I already organized it and put it to the right subtree 
                    node.left = None
                    
                else:
                    # if there is no right subtree, then just organize the 
                    # left sutree and put it to the right subtree 
                    node.right = self.helper(node.left)
                    
                    # reset the left subtree
                    node.left = None
            else:
                # if there is no left subtree, then just reorganize the right subtree
                # and put it to the right
                node.right = self.helper(node.right)
                
        return node
