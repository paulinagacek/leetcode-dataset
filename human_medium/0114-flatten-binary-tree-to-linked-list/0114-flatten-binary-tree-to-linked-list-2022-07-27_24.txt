class Solution:
    def flatten(self, root: Optional[TreeNode]) -> None:

        prev= None
        
        def convert(root):
            nonlocal prev
            
            if not root:
                return 
            
            convert(root.right)  
            convert(root.left)
            
            root.right = prev
            root.left = None
            prev = root
            
            return 
    
        convert(root)
        return
