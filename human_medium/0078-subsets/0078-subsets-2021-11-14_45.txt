class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        
        ans = []

        
        def help(i,res):
            
            if i >= len(nums):
                ans.append(res)
                return
            
            help(i+1,res+[nums[i]])
            help(i+1,res)
            
            
            
        help(0,[])
