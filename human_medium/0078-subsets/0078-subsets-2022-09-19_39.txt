####### BACKTRACKING #########
    def subsets(self, nums: List[int]) -> List[List[int]]: 
        results = []
        n = len(nums)

        def dfs(index, subset):
            if index == n:
                results.append(subset.copy())
                return

            subset.append(nums[index])
            dfs(index + 1, subset)
            subset.pop()
            dfs(index + 1, subset)

        dfs(0, [])
        return results
