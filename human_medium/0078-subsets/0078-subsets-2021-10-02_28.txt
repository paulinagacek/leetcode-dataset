class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        
        res = []
		# for every element in the list, we have two choices: include it, or exclude it from the subset to be generated.
        
        def backtrack(subset = [], index = 0):
            
            if index == len(nums):
                res.append(subset)
                return
            
            backtrack(subset + [nums[index]], index + 1)  # choice 1: include the element in the current subset 
            backtrack(subset, index + 1)  # choice 2: exclude the element from the current subset
        
        backtrack()
        return res
