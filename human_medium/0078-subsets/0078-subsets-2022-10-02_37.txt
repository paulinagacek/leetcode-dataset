class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        n=len(nums)
        p=pow(2,n)
        a=[]
        for i in range(p):
            b=[]
            for j in range(n):
                if i & (1<<j):
                    b.append(nums[j])
            a.append(b)
        return a
