class Solution:
    def generateTrees(self, n: int) -> List[TreeNode]:
        if n == 0: return []
        def generate(lo, hi):
            if lo > hi: return [None]
            if lo == hi: return [TreeNode(val = lo)]
            ans = []
            for i in range(lo, hi + 1):
                root = TreeNode(val = i)
                for lefthead in generate(lo, i - 1):
                    root.left = lefthead
                    for righthead in generate(i + 1, hi):
                        root.right = righthead
                        ans.append(copy.copy(root))
                        # ans.append(root)
            return ans
        return generate(1, n)
