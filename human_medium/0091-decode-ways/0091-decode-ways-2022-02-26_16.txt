class Solution:
    def numDecodings(self, s: str) -> int:
        
        #If we don\'t have a string we return zero
        if not s:
            return 0
        
        #We initialize our dp array
        #dp Index: length up to this point when checking the s string
        #dp Value: Number of different ways we can decode up to this length of string which is determined by the index
        dp = [0] * (len(s) + 1)
        
        #Here, we\'re saying that a string of length zero can only be decoded 1 way. As an empty string
        dp[0] = 1
            
            
        #Now we iterate through dp i.e. the length of string s
        
        for i in range(1, len(s) + 1):
            
            #here, we\'re saying that as long as the previous number wasn\'t a 0, at the current length, there is atleast
            #the number of ways to decode as when length was length - 1
            if s[i - 1] != \'0\':
                dp[i] += dp[i - 1]
                
            #Condition 1: We\'re only checking for values when we have an i big enough
            #to support checking if a two digit number is within range
            #Condition 2:Checking to see if the number can be interpreted as a letter, i.e. 1 to 26
            #Condition 3: We are not accepting leading zeros as possible values letters that map to single digit values
            if i != 1 and s[i - 2: i] < \'27\' and s[i - 2: i] > \'09\':
                dp[i] += dp[i - 2]
        
        #Here, let\'s return dp where the length of the word we\'re checking is the length of the word
        return dp[len(s)]
