class Solution: 
    def numDecodings(self, s: str) -> int:   # Time: O(n) and Space: O(n)
	
        # Edge case check, when s is empty or the first element is 0
        if s is None or s[0] == \'0\':
            return 0
			
        dp = [1] * len(s)  # initializing dp list to store how many ways from 0 till that index i we can decode string s
        for i in range(1, len(s)):
		
            # Single digit, if its 0 then dp[i] = 0 else dp[i] = dp[previous i]
            dp[i] = 0 if int(s[i]) == 0 else dp[i - 1]
			
            # Two digit, if its in between 10 and 26 then move forward
            if 10 <= int(s[i - 1:i + 1]) <= 26:
                dp[i] += dp[i - 2 if i > 1 else 0]  # if i>1 dp[i] += dp[i-2] else dp[i] += dp[0]
				
        return dp[-1]  # Return the last element added as in this approach the max ways will be stored at the last index
