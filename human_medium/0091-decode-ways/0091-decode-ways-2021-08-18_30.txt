class Solution:
    def numDecodings(self, s: str) -> int:
        if s[0]==\'0\': return 0
        n = len(s)
        @lru_cache(None)
        def dp(i): 
            if i==n: return 1
            if s[i]==\'0\': return 0
            ans = dp(i+1)
            if i<n-1 and int(s[i:i+2])<27:
                ans += dp(i+2)
            return ans
        return dp(0)
