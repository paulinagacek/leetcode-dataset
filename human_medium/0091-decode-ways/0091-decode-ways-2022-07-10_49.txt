class Solution:
    def numDecodings(self, s: str) -> int:
        if s.startswith(\'0\'):
            return 0
        
        lookup = {str(i - 64) for i in range(65, 65 + 26)}
        curr = prev = 1
        
        for i in range(1, len(s)):
            temp = (
                prev * (s[i - 1:i + 1] in lookup) +  # Get number of ways up to last two letters
                curr * (s[i] in lookup)              # Get number of ways up to last one letter
            )
            curr, prev = temp, curr
        
        return curr
