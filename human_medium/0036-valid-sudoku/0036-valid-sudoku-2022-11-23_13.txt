class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:

        row, col, hse = [0]*9, [0]*9, [0]*9

        for i,j in product(range(9), range(9)):

            b = board[i][j]
            if not b.isdigit(): continue
            b, ij = 1<<int(b), 3*(i//3)+j//3

            if row[i]&b or col[j]&b or hse[ij]&b: return False

            row[i ]|= b
            col[j ]|= b
            hse[ij]|= b

        return True
