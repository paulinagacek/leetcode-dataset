class Solution:
    def isValidSudoku(self, board: List[List[str]]) -> bool:

        for i in range(9):
            nums = 0
            A = set()
            for j in range(9):
                if board[i][j].isnumeric():
                    A.add(board[i][j])
                    nums+=1
            if nums!=len(A):
                return False
        for i in range(9):
            nums = 0
            A = set()
            for j in range(9):
                if board[j][i].isnumeric():
                    A.add(board[j][i])
                    nums+=1
            if nums!=len(A):
                return False
        for i in range(3):
            for j in range(3):
                nums = 0
                A = set()
                for k in range(3):
                    for m in range(3):
                        if board[3*i+k][3*j+m].isnumeric():
                            A.add(board[3*i+k][3*j+m])
                            nums+=1

                if nums!=len(A):
                    return False
        return True
