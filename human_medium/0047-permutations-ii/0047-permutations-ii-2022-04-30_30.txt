class Solution:
	# non uniq permutation
    def permute(self, nums: List[int]) -> List[List[int]]:
        if len(nums) < 1:
            return [[]]
        return [[nums[i]]+p for i in range(len(nums)) \\
                        for p in self.permute(nums[:i]+nums[i+1:])]
        
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
		# drop duplication
        return list(set(tuple(p) for p in self.permute(nums)))
