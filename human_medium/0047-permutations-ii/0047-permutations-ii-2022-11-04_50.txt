class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:

        def permuteEnd(nums, start):
            if start == len(nums) - 1:
                return [nums.copy()]
            
            result = []
            used = set()
            for i in range(start, len(nums)):
                if nums[i] in used:
                    continue
                nums[start], nums[i] = nums[i], nums[start]
                result.extend(permuteEnd(nums, start + 1))
                nums[start], nums[i] = nums[i], nums[start]
                used.add(nums[i])
            return result

        return permuteEnd(nums, 0)
