stack = [(nums, [])] # (nums, path)
        res = []
        while stack: # -- O(V)
            nums, path = stack.pop()
            if not nums:
                if path not in res: # -- O(N!)
                    res.append(path)
            # children
            for i in range(len(nums)): # -- O(E)
                newNums = nums[:i] + nums[i+1:]
                newPath = path + [nums[i]]
                stack.append((newNums, newPath))
        return res
