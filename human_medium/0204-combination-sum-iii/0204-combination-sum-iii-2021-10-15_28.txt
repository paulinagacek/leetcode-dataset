class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        result, current = [] , []
        def dfs(start, target):
            if target == 0 and len(current) == k:
                result.append(current.copy())
                return
            if target < 0 or len(current) > k: return
            for i in range(start, 10):
                current.append(i)
                dfs(i + 1, target - i)
                current.pop()
        dfs(1, n)
        return result
