class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        
        a = [i+1 for i in range(9)]
        
        b = list(combinations(a,k))
        c = []
        for i in range(len(b)):
            if sum(b[i]) == n:
                c.append(list(b[i]))
                
        return c
