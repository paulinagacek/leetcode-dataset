class Solution(object):
    def combinationSum3(self, k, n):
		"""
		:type k: int
		:type n: int
		:rtype: List[List[int]]
		"""
		self.res = []

		def backtrack(curr, total):
			# 1. Backtrack function (exit condition)
			if total >= n:
				if total == n and len(curr) == k and curr not in self.res:
					self.res.append(curr.copy())
				return
				
			# 2. Backtrack function (loop)
			for i in range(1,10):
				if i not in curr:
					curr.add(i)
					total += i
					backtrack(curr, total)
					curr.remove(i)
					total -= i
		
		# 3. Commence backtracking
		backtrack(set(), 0)
		return self.res
