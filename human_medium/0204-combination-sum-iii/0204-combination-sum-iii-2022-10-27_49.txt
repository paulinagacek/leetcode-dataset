class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        ans = []
        
        def backtracking(start, res):
            if len(res) == k and sum(res) == n:
                ans.append(res[:])
                return
            
            for num in range(start, 10):
                if sum(res) + num > n:
                    return
                
                res.append(num)
                backtracking(num+1, res)
                res.pop()
       
        backtracking(1, [])
        return ans
