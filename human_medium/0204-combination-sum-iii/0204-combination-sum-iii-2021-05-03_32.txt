class Solution:
    def combinationSum3(self, k: int, n: int) -> List[List[int]]:
        
        res = []
        self.comb(k, n, res, 1, [])
        return res

    def comb(self, count, n, res, start, path):
        
        if count == 0:
            if n == 0:
                res.append(path)
            return
    
        for val in range(start, 10):
            self.comb(count-1, n-val, res, val+1, path+[val])
