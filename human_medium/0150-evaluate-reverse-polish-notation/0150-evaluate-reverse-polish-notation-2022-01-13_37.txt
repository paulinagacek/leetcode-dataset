class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []
        
        operators = {
            "+": Plus,
            "-": Minus,
            "*": Multi,
            "/": Div
        }
        
        for token in tokens:
            if token in operators:
                right = stack.pop()
                left = stack.pop()
                res = operators[token](left, right).evaluate()
                stack.append(res)
            else:
                stack.append(int(token))
        return stack[0]
