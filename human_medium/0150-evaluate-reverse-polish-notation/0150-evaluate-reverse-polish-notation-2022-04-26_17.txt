class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        
        stack = []
    
        for token in tokens:

            if token not in "+-/*":
                stack.append(int(token))
                continue

            number2 = stack.pop()
            number1 = stack.pop()

            number = 0
            if token == "+":
                number = number1 + number2
            elif token == "-":
                number = number1 - number2
            elif token == "*":
                number = number1 * number2
            else:
                number = int(number1 / number2)

            stack.append(number)

        return stack[0]
