class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        math = {"+": operator.add, "-": operator.sub,
                "*": operator.mul, "/": operator.truediv}
        stack = []
        for token in tokens:
            if token[-1].isdigit():
                stack.append(int(token))
            else:
                right, left = stack.pop(), stack.pop()
                stack.append(trunc(math[token](left, right)))
        return stack[0]
