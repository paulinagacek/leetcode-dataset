class Solution(object):
    def searchRange(self, nums, target):
        l1, r1 = 0, len(nums) - 1
        while l1 <= r1:
            m1 = (l1 + r1) / 2
            if nums[m1] < target: l1 = m1 + 1
            else: r1 = m1 - 1
        l2, r2 = 0, len(nums) - 1
        while l2 <= r2:
            m2 = (l2 + r2) / 2
            if target < nums[m2]: r2 = m2 - 1
            else: l2 = m2 + 1
		return (l1, r2) if l1 <= r2 else [-1, -1]
