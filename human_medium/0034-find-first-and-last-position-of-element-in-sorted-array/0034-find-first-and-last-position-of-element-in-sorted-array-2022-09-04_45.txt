class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        l, r = 0, len(nums) - 1
        res = []
        
        while l <= r:
            mid = (l+r) // 2
            if nums[mid] > target:
                r = mid - 1
            elif nums[mid] < target:
                l = mid + 1
            
            elif nums[mid] == target:
                #keep moving left and right pointers until nums[l] == nums[r] == target
                if nums[l] < target:
                    l = l + 1
                if nums[r] > target:
                    r = r -1 
                # if condition is true, append left and right
                # left will be the first occurance, right will be the last occurance
                if nums[l] == target and nums[r] == target:
                    res.append(l)
                    res.append(r)
                    break

        if len(res) == 0:
            return [-1, -1]
        else:
            return res
