class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        ans=[-1, -1]
        ans[0] = self.find_index(nums, target, \'l\')
        ans[1] = self.find_index(nums, target, \'r\')
        return ans
        
    def find_index(self, nums, target, method):    # method: leftmost or rightmost
        index, low, high = -1, 0, len(nums)-1
        while low <= high:
            mid = low + (high - low)//2
            if nums[mid] == target:
                index = mid
                if method == \'r\':           # if the selected method is rightmost
                    low = mid + 1
                elif method == \'l\':         # if the selected method is leftmost
                    high = mid - 1
            elif nums[mid] > target: 
                high = mid - 1
            else:
                low = mid + 1
        return index
