class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        # empty array, return [-1, -1]
        if not nums:
            return [-1, -1]
        
        # 1. find first target\'s index
        left, right = 0, len(nums) - 1
        while left < right:
            mid = left + (right - left) // 2
            if nums[mid] < target:
                left = mid + 1
            else:
                right = mid
        # target not found, return [-1, -1]
        if nums[left] != target:
            return [-1, -1]
        # save first target\'s index
        first = left
        
        # 2. find last target\'s index
        right = len(nums) - 1
        while left < right:
            mid = left + (right - left) // 2 + 1 # Make mid biased to the right
            if nums[mid] > target:
                right = mid - 1
            else:
                left = mid
        # save last target\'s index
        last = left
        
        return [first, last]
