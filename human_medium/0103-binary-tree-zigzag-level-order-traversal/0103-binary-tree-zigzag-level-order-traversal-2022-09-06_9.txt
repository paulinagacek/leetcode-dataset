if not root:
            return None
        q = [root] # create a queueand store root inside
        ans = []
        zig = True #A variable to control alternate reversing of levels
        while q:
            l = []
            for i in range(len(q)): #implementing level order
                node = q.pop(0)
                l.append(node.val)
                if node.left:  q.append(node.left)
                if node.right: q.append(node.right)
            if not zig: #reversing every next level
                l.reverse()
            zig = not zig
            ans.append(l)
        return ans
