def zigzagLevelOrder(self, root: TreeNode) -> List[List[int]]:
        
        if not root:
            return []
        
        levels = []
        def orderLevelTrav(root, level):
            if not root:
                return
            
            if len(levels) == level:
                levels.append([])
            
            levels[level].append(root.val)
             
            orderLevelTrav(root.left, level + 1)
            orderLevelTrav(root.right,level + 1)
            
        orderLevelTrav(root, 0)
        
        for i,level in enumerate(levels):
            if i % 2 != 0:
                levels[i] = level[::-1]
        return levels
