class Solution:
    def zigzagLevelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        
        if not root: return []
        
        q, res = deque([root]), []
        
        while q:
            level = []
            
            for i in range(len(q)):
                node = q.popleft()
                level.append(node.val)
                if node.left: q.append(node.left)
                if node.right: q.append(node.right)
                    
            if len(res) % 2 != 0:
                res.append(level[::-1])
            else:
                res.append(level)
                
        return res
