class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        
        # Quick response on simple cases
        
        if n == 0:
            return [ ]
        elif n == 1:
            return [ [1] ]
        
        
        # create a matrix of size n x n
        matrix = [ [ 0 for _ in range(n) ] for _ in range(n) ]
        
        left, right = 0, n-1
        top, bottom = 0, n-1
        
        val = 1
        
        # fill matrix with 1 ~ n^2 in spiral path
        while True:
            
            # top side go right
            for x in range(left, right+1):
                matrix[top][x] = val
                val += 1
            
            top += 1
            if top > bottom: break
            
            # right side go down
            for y in range(top, bottom+1):
                matrix[y][right] = val
                val += 1
            
            right -= 1
            if left > right: break
            
            # bottom side go left
            for x in range(right, left-1, -1):
                matrix[bottom][x] = val
                val += 1
            
            bottom -= 1
            if top > bottom: break
            
            # left side go up
            for y in range(bottom, top-1, -1):
                matrix[y][left] = val
                val += 1
            
            left += 1
            if left > right: break
                
        return matrix
