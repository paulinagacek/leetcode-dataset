class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        matrix = [[0] * n for i in range(n)]
        len_row, len_col = n, n
        start_row, end_row = 0, n - 1
        start_col, end_col = 0, n - 1
        direction = 1
        cnt = 1

        while len_row and len_col:
            curr_row, curr_col = start_row, start_col

            for c in range(len_col):
                curr_col = start_col + c * direction
                matrix[start_row][curr_col] = cnt
                cnt += 1

            start_col = curr_col
            start_row += direction
            len_row -= 1
            len_col -= 1

            for r in range(len_row):
                curr_row = start_row + r * direction
                matrix[curr_row][start_col] = cnt
                cnt += 1

            direction *= -1
            start_row = curr_row
            start_col += direction
        return matrix
