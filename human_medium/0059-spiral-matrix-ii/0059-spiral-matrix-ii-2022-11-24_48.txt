class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:

        adjlist = [[0]*n for _ in range(n)]
        total_layer = (n+1)//2
        count = 1

        for layer in range(total_layer):

            for right in range(layer, n-layer):
                adjlist[layer][right] = count
                count += 1

            for down in range(layer+1, n-layer):
                adjlist[down][n-layer-1] = count
                count += 1

            for left in range(layer+1, n-layer):
                adjlist[n-layer-1][n-left-1] = count
                count += 1

            for up in range(layer+1, n-layer-1):
                adjlist[n-up-1][layer] = count
                count += 1

        return adjlist
