class Solution:
    def generateMatrix(self, n: int) -> List[List[int]]:
        if n == 0:
            return [ ]
        elif n == 1:
            return [ [1] ]
        
        matrix = [ [ 0 for _ in range(n) ] for _ in range(n) ]
        
        left, right = 0, n-1
        top, bottom = 0, n-1
        
        num = 1
        while left <= right and top<=bottom:
                  
            for j in range(left, right+1):  #top row constant move from left--->right
                matrix[top][j] = num
                num += 1
            
            top += 1
            
            
            for i in range(top, bottom+1): # right column constant move from top ------>bottom
                matrix[i][right] = num
                num += 1
            
            right -= 1
           
            
           
            for j in range(right, left-1, -1): #bottom row constant move from right to left
                matrix[bottom][j] = num
                num += 1
            
            bottom -= 1
            
            
        
            for i in range(bottom, top-1, -1): #left column constant move from bottom to top
                matrix[i][left] = num
                num += 1
            
            left += 1
            
                
        return matrix
