import numpy as np

def generateMatrix(self, n):
	def spiral(grid, start=1):
		n = len(grid)
		
		# Breaking Condition:
		if n == 1:
			grid[0, 0] = start
		if n <= 1:
			return

		# Lace the frame with required numbers:
		for _ in range(4):
			grid[0, :-1] = np.arange(start, start + n - 1)
			start += n - 1
			grid = np.rot90(grid)

		# If inner matrix exist Recurse
		spiral(grid[1:-1, 1:-1], start)

	# Setup:
	grid = np.zeros((n, n), dtype=np.int)
	spiral(grid)
	return grid
