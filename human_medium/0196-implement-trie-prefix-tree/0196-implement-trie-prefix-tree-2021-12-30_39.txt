class Trie:
    def __init__(self):
        self.root = {}
    
    def insert(self, word: str) -> None:
        curr = self.root
        for c in word:
            if c not in curr:
                curr[c] = {}
            curr=curr[c]
        curr["^"]=True
        
    def traversal(self, word: str) -> bool:
        curr = self.root
        for c in word:
            if c not in curr: 
                return False
            curr=curr[c]
        return curr

    def search(self, word: str) -> bool:
        end = self.traversal(word)
        return "^" in end if end else False
        
    def startsWith(self, prefix: str) -> bool:
        return self.traversal(prefix)
