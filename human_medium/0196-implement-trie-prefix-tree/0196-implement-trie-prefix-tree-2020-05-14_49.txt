class Trie:

    def __init__(self):
        self.root = {}

    def insert(self, word: str) -> None:
        node = self.root
        for ch in word: node = node.setdefault(ch, {})
        node["$"] = word

    def search(self, word: str) -> bool:
        node = self.root
        for ch in word: 
            if ch not in node: return False 
            node = node[ch]
        return node.get("$")

    def startsWith(self, prefix: str) -> bool:
        node = self.root
        for ch in prefix: 
            if ch not in node: return False 
            node = node[ch]
        return True
