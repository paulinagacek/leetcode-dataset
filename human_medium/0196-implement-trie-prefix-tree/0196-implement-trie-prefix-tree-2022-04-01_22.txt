class Trie:

    def __init__(self):
        self.root = {}
        
    def insert(self, word: str) -> None:
        node = self.root
        for char in word:
            node = node.setdefault(char, {})
        node[\'is_end\'] = True
            
    def search(self, word: str) -> bool:
        node = self.root
        for char in word:
            if not char in node:
                return False
            node = node[char]
        return node.get(\'is_end\', False)

    def startsWith(self, prefix: str) -> bool:
        node = self.root
        for char in prefix:
            if not char in node:
                return False
            node = node[char]
        return True
