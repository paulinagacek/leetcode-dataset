class Trie:

    def __init__(self):
        self.root = {} 
        self.end = \'*\'

    def insert(self, word: str) -> None:
        node = self.root
        for l in word:
            if l not in node:
                node[l] = {}
                node = node[l]
            else: node = node[l]
        node[self.end] = True   

    def search(self, word: str) -> bool:
        node = self.root
        for l in word:
            if l not in node: return False
            else: node = node[l]
        return self.end in node
        
    def startsWith(self, prefix: str) -> bool:
        node = self.root
        for l in prefix:
            if l not in node: return False
            else: node = node[l]
        return True
