def twoSum(self, numbers: List[int], target: int) -> List[int]: 
        left = 0
        right = len(numbers) - 1
        
        sumLR = 0
        
        while left <= right:
            sumLR = numbers[left] + numbers[right]
            
            if sumLR > target:
                right -= 1
            elif sumLR < target:
                left += 1
            else:
                return [left+1, right+1]
