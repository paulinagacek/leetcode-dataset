class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        start = 0
        end = len(numbers) - 1
        res = []
        while start < end and len(res) == 0:
            if numbers[start] + numbers[end] < target:
                start = start + 1
            elif numbers[start] + numbers[end] > target:
                end = end - 1
            else:
                res.append(start + 1)
                res.append(end + 1)
        return res
