class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        s, e = 0, len(nums)-1
        while s<e:
            curr_sum = nums[s] + nums[e]
            if curr_sum == target:
                return [s+1, e+1]
            
            m = (s+e)//2
            if curr_sum > target:
                e = m - 1 if nums[s] + nums[m] > target else e - 1
            else:
                s = m + 1 if nums[m] + nums[e] < target else s + 1
