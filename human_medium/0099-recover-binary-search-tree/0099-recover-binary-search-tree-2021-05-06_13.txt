class Solution:
    def recoverTree(self, root: TreeNode) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        first = None
        sec = None
        prev = TreeNode(-math.inf)
        def check(root):
            nonlocal prev,first,sec
            if root==None:
                return
            check(root.left)
            if root.val<prev.val:
                if not first:
                    first = prev
                sec = root
            prev = root
            check(root.right)
        check(root)
        first.val,sec.val = sec.val,first.val
