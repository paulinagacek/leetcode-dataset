class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        node1 = node2 = prev = None        
        stack = []
        node = root
        while node or stack:
            if node:
                stack.append(node)
                node = node.left
            else:
                node = stack.pop()                
                if prev and prev.val > node.val:
                    if node1:
                        node1.val, node.val = node.val, node1.val
                        return 
                    else:
                        node1, node2 = prev, node
                        
                prev, node = node, node.right
                
        node1.val, node2.val = node2.val, node1.val
