class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        Since its a BST and exactly a pair has been swapped, we need to find the node and swap them.
        To do so we will recursively check for the condition between prevNode and current node.
        Once pointer are set at the required node just swap their values
        """
        startNode = lastNode = prevNode = None
        def dfs(root):
            nonlocal startNode, lastNode, prevNode
            if not root:
                return 
            dfs(root.left)
            if prevNode and prevNode.val > root.val:
                if not startNode:
                    startNode = prevNode
                lastNode = root
            prevNode = root
            dfs(root.right)
        dfs(root)
        if startNode and lastNode:
            startNode.val, lastNode.val = lastNode.val, startNode.val
        return root
