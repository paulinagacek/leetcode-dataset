class Solution:
    def __init__(self):
        self.k = 0
        
    def inOrder(self, root: Optional[TreeNode], arr: List[int]) -> None:
        if root is None:
            return
        self.inOrder(root.left, arr)
        arr.append(root.val)
        self.inOrder(root.right, arr)
        
    def compare(self, root: Optional[TreeNode], arr: List[int]) -> None:
        if root is None:
            return
        self.compare(root.left, arr)
        if root.val != arr[self.k]:
            root.val, arr[self.k] = arr[self.k], root.val
        self.k += 1
        self.compare(root.right, arr)
    
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        """
        Do not return anything, modify root in-place instead.
        """
        arr = []
        self.inOrder(root, arr)
        arr.sort()
        self.compare(root, arr)
