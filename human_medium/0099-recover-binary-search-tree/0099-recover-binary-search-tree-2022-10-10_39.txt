class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
        global first_node,second_node,prev
        first_node,second_node = None,None
        # first element will always be greater than its next element.
        # second element will always be smaller than its prev element.
        prev = TreeNode(-2147483648)
        
        def inorder(root: Optional[TreeNode]) -> None:
            global first_node,second_node,prev
            if(root == None):
                return
            
            inorder(root.left)

            if (first_node == None and prev.val > root.val):
                first_node = prev
            
            if (first_node != None and prev.val > root.val):
                second_node = root
            
            prev = root

            inorder(root.right)
        # Marked the faulty Nodes.
        inorder(root)
        # Once then faulty nodes are marked, Perform swap Operation. 
        first_node.val,second_node.val = second_node.val,first_node.val
