class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        if n == 0:
            return [\'\']
        
        res = [\'()\']
        for _ in range(1,n):
            tmp = []
            for x in res:
                for i in range(len(x)):
                    tmp.append(x[:i] + \'()\' + x[i:])
            res = list(set(tmp))
        return res
