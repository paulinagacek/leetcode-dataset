class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        if n == 0:
            return []
        else:
            x = [[] for i in range(n + 1)]
            x[0] = [""]
            for i in range(1, n + 1):
                x[i] += [\'(\' + s + \')\' + t for k in range(i) for s in x[i-k-1] for t in x[k]]
            return x[n]
