class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
		nums = set(nums)
        max_len = 0
        for num in nums:
            if num-1 not in nums:
                next_num = num + 1
                while next_num in nums:
                    next_num += 1
                max_len = max(max_len, next_num - num)
        return max_len
