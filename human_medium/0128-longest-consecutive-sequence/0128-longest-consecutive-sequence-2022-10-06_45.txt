class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if not nums:
            return 0
        U = Union()
        for i in nums:
            if i in U.parents:
                continue
            U.parents[i] = i
            U.size[i] = 1
            if i+1 in U.parents:
                U.union(i, i+1)
            if i-1 in U.parents:
                U.union(i, i-1)
        return max(U.size.values())
