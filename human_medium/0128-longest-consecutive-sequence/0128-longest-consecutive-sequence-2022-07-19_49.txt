class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:        
        nums_set = set(nums)
        
        longest = 0
        for num in nums:
            previous_num = num - 1
            sequence_length = 1
            
            # if the previous num is not in our set, this is the start of a sequence
            if previous_num not in nums_set:
                next_num = num + 1
                while next_num in nums_set: # keep counting sequence length until we hit a num not in set
                    sequence_length += 1
                    next_num += 1
                longest = max(longest, sequence_length)
                
        return longest
