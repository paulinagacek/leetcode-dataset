class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        numSet = set(nums)
        longest = 0
        for num in nums:
			# for an element to be the startof a sequence it should not have any element to the left of it, 
			# if it has then its not the start of the sequence and no need to compute the length of such sequences as it will always be less than the sequence it is subset of.
            if (num-1) not in numSet:
                length = 0
				# the length of the sequence needs to be calculated as long as there is a consecutive element present in the set.
                while (num+length) in numSet:
                    length +=1
                longest = max(length,longest)
        return longest
