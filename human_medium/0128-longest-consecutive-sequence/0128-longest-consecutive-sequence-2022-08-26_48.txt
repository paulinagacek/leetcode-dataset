class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        numSet = set(nums)
        maxCount = 0
        length = 0
        
        for i in nums:
            if i - 1 not in numSet:
                currentCount = 0
                while i + currentCount in numSet:
                    currentCount += 1
                maxCount = max(maxCount, currentCount)
        
        return maxCount
