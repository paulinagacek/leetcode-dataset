class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if len(nums)==0:return 0;
        x=list(set(nums))
        ans=1;fans=0
        x.sort()
        for i in range(1,len(x)):
            if x[i]-x[i-1]==1:ans+=1
            else:fans=max(ans,fans);ans=1
        ans=max(ans,fans)
        return ans;
