class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        
        c0 = 0  #count of number of zeroes
        c1 = 0  #count of number of ones
        c2 = 0  #count of number of twos
        
		#First, we will count number of 0\'s, 1\'s and 2\'s
		#represented by c0, c1 and c2 respectively
        for i in nums:
            if i == 0:
                c0 += 1
                
            elif i == 1:
                c1 += 1
                
            else:
                c2 += 1
        
		#Now, we will override the values with number of 0\'s -- 
		#so, index should go in inclusive range of [0, c0-1]
        for i in range(0, c0):
            nums[i] = 0 
        
		#After 0\'s, we will override 1\'s in the inclusive range of [c0, c0 + c1 - 1]
        for i in range(c0, c0 + c1):
            nums[i] = 1
        
		#Then, we will override the remaining indices with 2
        for i in range(c0 + c1, c0 + c1 + c2):
            nums[i] = 2
        return nums
