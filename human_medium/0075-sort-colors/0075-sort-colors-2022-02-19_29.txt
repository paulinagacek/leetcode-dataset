class Solution:
    def sortColors(self, nums: List[int]) -> None:
        n=len(nums)
        arr=nums
        for i in range(n-1):
            for j in range(0,n-i-1):
                if arr[j]>arr[j+1]:
                    arr[j],arr[j+1]=arr[j+1],arr[j]
        """
        Do not return anything, modify nums in-place instead.
        """
