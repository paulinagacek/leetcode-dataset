class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        zero = 0
        one = 0
        two = 0
        s = ""
        for i in range(len(nums)):
            if(nums[i] == 0): zero +=1
            elif(nums[i] == 1): one +=1
            elif(nums[i] == 2): two +=1
        s = "0"*zero + "1"*one + "2"*two
        for i in range(len(s)):
            nums[i] = s[i]
