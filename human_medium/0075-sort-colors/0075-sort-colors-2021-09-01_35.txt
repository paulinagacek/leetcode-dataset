class Solution:
    def sortColors(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        # invariant:
        # - elements left of zero_ptr are 0\'s
        # - elements right of two_ptr are 2\'s
        zero_ptr, two_ptr = 0, len(nums)-1
        i = 0
        while i <= two_ptr:
            if nums[i] == 1:
                i += 1
            elif nums[i] == 0:
                nums[i], nums[zero_ptr] = nums[zero_ptr], nums[i]
                zero_ptr += 1
                i += 1
            else:
                nums[i], nums[two_ptr] = nums[two_ptr], nums[i]
                two_ptr -= 1
                # cannot increment i here, what got swapped here 
                # may be a 2.
