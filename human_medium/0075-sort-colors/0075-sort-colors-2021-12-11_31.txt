class Solution:
    
    def sortColors(self, nums: List[int]) -> None:
        zero=-1
        one=-1
        for i in range(0,len(nums)):
            if nums[i] <2:
                one+=1
                
            if nums[i]==0:
                zero+=1
                nums[zero],nums[i]=nums[i],nums[zero]
                    
            if nums[i]==1:
                nums[one],nums[i]=nums[i],nums[one]
