class Solution(object):
    def connect(self, root):
        """
        :type root: Node
        :rtype: Node
        """
        # if root is Null or both left and right children are null return root
        if not root:
            return root
        if not root.left and not root.right:
            return root
        #if both left an dright children are present 
        root.left.next=root.right
        # if root has a next node
        if root.next:
            root.right.next=root.next.left
        # call connect on left and right children    
        self.connect(root.left)
        self.connect(root.right)
        return root
