class Solution:
    def connect(self, root: \'Optional[Node]\') -> \'Optional[Node]\':
        if not root:
            return
        
        c1, c2 = root.left, root.right
        
        while c1 and c2:
            c1.next = c2
            c1, c2 = c1.right, c2.left
        
        self.connect(root.left)
        self.connect(root.right)
        
        return root
