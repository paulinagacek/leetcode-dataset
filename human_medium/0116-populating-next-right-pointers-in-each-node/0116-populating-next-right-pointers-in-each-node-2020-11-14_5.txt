class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        
        if not root:
            return None
        
        q=collections.deque()
        
        q.append(root)
        
        while q:
            for i in range(1,len(q)):
                q[i-1].next=q[i]
            q[-1].next=None    
            newLevel=deque()
            for node in q:
                if node.left:
                    newLevel.append(node.left)
                if node.right:
                    newLevel.append(node.right)
                    
            q=newLevel
        return root
