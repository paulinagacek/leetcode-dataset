class Solution:
    def connect(self, root: \'Optional[Node]\') -> \'Optional[Node]\':
        if not root:
            return None
        
        queue = collections.deque([root])
        
        while queue:
            nxt = None
            
            for _ in range(len(queue)):
                curr = queue.popleft()
                curr.next, nxt = nxt, curr 
                                
                if curr.right:
                    #append right to left
                    queue.append(curr.right)
                    queue.append(curr.left)
        
        return root
