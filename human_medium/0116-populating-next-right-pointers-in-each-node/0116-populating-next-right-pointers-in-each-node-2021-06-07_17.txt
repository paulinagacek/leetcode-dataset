class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        if root==None:
            return
        temp = root
        q = []
        q.append(root)
        while q:
            size = len(q)
            for i in range(size):
                root = q.pop(0)
                
                if len(q)>0 and i<size-1:
                    root.next = q[0]
                else:
                    root.next = None
                
                if root.left:
                    q.append(root.left)
                if root.right:
                    q.append(root.right)
        return temp
