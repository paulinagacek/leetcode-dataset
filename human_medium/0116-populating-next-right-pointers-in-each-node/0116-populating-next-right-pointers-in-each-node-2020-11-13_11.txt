class Solution:
    def connect(self, root: Node) -> Union[Node, None]:
        if root is None:
            return root

        q = deque([root])
        while q:
            size = len(q)
            for i in range(size):
                node = q.popleft()

                if i + 1 < size:
                    node.next = q[0]

                if node.left is not None:
                    q.append(node.left)
                    q.append(node.right)

        return root
