def connect(self, root: \'Node\') -> \'Node\':
        if not root:
            return root
        Q = collections.deque([root])

        while Q:
            size = len(Q)
            # Iterate over all the nodes on the current level
            for i in range(size):
                node = Q.popleft()
                
                # The queue will contain nodes from 2 levels at most at any point in time. This check ensures we don\'t establish next pointers beyond the end of a level
				# we do next pointer redirection for first size-1 items popped from queue
                if i < size - 1:
                    node.next = Q[0]
                
                # Add the children, if any, to the back of the queue
                if node.left:
                    Q.append(node.left)
                    Q.append(node.right)
        
        return root
