class Solution:
    def countPrimes(self, n: int) -> int:
       
        prime = [True] * n
        
        if n < 2:
            return 0
        
        prime[0], prime[1] = False, False
        
        i = 2
        while i*i < n:
            if prime[i] == True:
                for j in range(i*i, n, i):
                    prime[j] = False
                    
            i += 1
            
        return prime.count(True)
