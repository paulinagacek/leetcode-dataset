class Solution:
    def countPrimes(self, n: int) -> int:
        c=0
        dp=[1 for i in range(n)]
        
        i=2
        while i*i<n:
            if dp[i]==0:
                i+=1
                continue
            else:
                for j in range(i*i,n,i):
                    dp[j]=0
                i+=1
        
        
        for i in range(2,n):
            if dp[i]==1:
                c+=1
        return c
