def countPrimes(self, n: int) -> int:
      
        if n <= 1:
            return 0
    
        ans = [True] * (n)
        
        ans[0] = False
        ans[1] = False
        
        for i in range(2, n):
            
            if ans[i] == True:
                j = 2
                while i*j<n:
                    ans[i*j] = False
                    j += 1
                    
        return sum(ans)
