class Solution:
    import math
    def countPrimes(self, n: int) -> int:
        if n <= 1:
            return 0
        prime_arr = [True] * n
        prime_arr[0] = False
        prime_arr[1] = False
        for i in range(2, ceil(n ** 0.5)):
            if prime_arr[i] is False:
                continue
            prime_arr[i*2::i] = [False] * len(prime_arr[i*2::i])
        return sum(prime_arr)
