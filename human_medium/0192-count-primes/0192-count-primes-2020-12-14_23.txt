class Solution:
    def countPrimes(self, n: int) -> int:
        sieve = [False] * (n + 1)
        i = 2
        while(i * i <= n):
            if not sieve[i]:
                st = i * i
                while(st <= n):
                    sieve[st] = True
                    st += i
            i += 1
        cnt = 0
        for i in range(2, len(sieve) - 1):
            if not sieve[i]:
                cnt += 1
        return cnt
