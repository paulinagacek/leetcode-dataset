class Solution(object):

    def countPrimes(self, n):
       
        if n <= 2:
            return 0
        primeNums = [2]
        for num in range(3, n):
            j = 0
            for j in primeNums:
                if num % j ==0:                    
                    break
                if j**2>num:
                    primeNums.append(num)
                    break
                       
        return len(primeNums)
