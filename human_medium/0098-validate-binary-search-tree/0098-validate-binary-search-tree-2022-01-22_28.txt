class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        path = []
        
        def dfs(node: Optional[TreeNode]) -> bool:
            if node is None:
                return

            dfs(node.left)
            path.append(node.val)
            dfs(node.right)

        dfs(root)
        return all(path[i] < path[i+1] for i in range(len(path) - 1))
