class Solution:
    def isValidBST(self, root: TreeNode) -> bool:
        temp = float("-inf")
        for val in self.getValues(root):
            if val > temp:
                temp = val
            else:
                return False
        return True

    def getValues(self, root, values=[]):
        if not root:
            return values
        self.getValues(root.left, values)
        values.append(root.val)
        self.getValues(root.right, values)
        return values
