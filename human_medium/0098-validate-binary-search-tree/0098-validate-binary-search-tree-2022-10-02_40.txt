class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        prev = None
        current = None
        stack = []
        node = root
        while stack or node:
            if node:
                stack.append(node)
                node = node.left
            else :
                node = stack.pop()
                prev, current = current, node.val
                node = node.right
            if prev is not None and current is not None and prev >= current:
                return False
        return True
