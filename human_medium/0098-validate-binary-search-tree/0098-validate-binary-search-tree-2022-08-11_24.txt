class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        
        def is_valid_root(node, parent_maximum, parent_minimum):
            if not node: return True
            
            # Check left side
            if not is_valid_root(node.left, node.val, parent_minimum): return False
            
            # Check right side
            if not is_valid_root(node.right, parent_maximum ,node.val): return False
            
            # Check constraints with current node
            if node.val >= parent_maximum or node.val <= parent_minimum: return False
            
            # Otherwise we are good
            return True
            
        
        return is_valid_root(root, float(\'inf\'), float(\'-inf\'))
