"""
As the BST is always sorted from left to right
@arr to put the values in its order in the tree
then we check if it is sorted
"""
self.arr = []

def helper(x):
	if not x: return

	helper(x.left)
	self.arr.append(x.val)
	helper(x.right)

helper(root)

for i in range(1, len(self.arr)):
	if self.arr[i] <= self.arr[i-1]: return False

return True
