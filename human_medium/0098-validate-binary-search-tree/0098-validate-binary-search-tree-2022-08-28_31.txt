class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        
        
        def bst(root,mini,maxi):
            if not root:
                return True
            
            if(mini!=None and root.val<=mini) or (maxi!=None and root.val>=maxi):
                return False
            
            if not bst(root.left,mini,root.val) or not bst(root.right,root.val,maxi):
                return False
            
            return True
           
        return bst(root,None,None)
