class Solution:
    def canJump(self, nums: List[int]) -> bool:
        def recurJump(index):
            if index>=len(nums)-1:
                return True
			#Checking all the possible values by backTracking starting from 1 , if the value is 0 , our code would return False as it would not enter the for loop
            for i in range(1,nums[index]+1):
                if(recurJump(index+i)):
                    return True
            return False
        return(recurJump(0))
