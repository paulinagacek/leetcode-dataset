class Solution:
    def canJump(self, nums: List[int]) -> bool:
        if (len(nums) == 1):
            return True
        first = len(nums) - 1
        for i in range(len(nums) - 2, -1,-1):
            j = nums[i]
            if (j + i >= first):
                first = i
        if (first == 0):
            return True
        return False
