class Solution:
    def canJump(self, nums: List[int]) -> bool:
        s=1
        n=len(nums)
        for i in range(0,n):
            s-=1
            if(i==n-1):
                return(True)
            if(s==0 and nums[i]==0):
                return(False)
            if(s<nums[i]):
                s=nums[i]
