class Solution:
    def canJump(self, nums: List[int]) -> bool:
        n = len(nums)
        dp = [0] * (n - 1)
        for i in range(n - 1):
            if i == 0:
                dp[i] = nums[0]
            else:
                dp[i] = max(dp[i - 1], i + nums[i])
        return all(dp[i] > i for i in range(n - 1))
