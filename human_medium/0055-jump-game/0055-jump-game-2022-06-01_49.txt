class Solution:
    def canJump(self, nums: List[int]) -> bool:
        if len(nums) == 1:
            return True
        
        if nums[0] == 0:
            return False
        
        reachable = 0
        for i in range(len(nums)):
            if reachable < i:
                return False
            reachable = max(reachable,i+nums[i])
        return reachable
