class Solution(object):
    def canJump(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        min_jump = 1
        
        for i in range(len(nums)-2,-1,-1):
            if nums[i] >= min_jump:
                min_jump = 1 
            else:
                min_jump += 1 
        
        return nums[0] >= min_jump if len(nums) > 1 else True
