class Solution(object):
    def canJump(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        n = len(nums)
        nums[n-1] = True
        l = n-1
        for i in reversed(range(n-1)):
            nums[i] = (nums[i+1] and nums[i]>0) or (nums[i] >= l - i)
            if nums[i]:
                l = i
        #print(nums)
        return nums[0]
