class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        candidates.sort()

        def dfs(cur_res, start, remaining_target):
            if 0 == remaining_target:
                res.append(cur_res)
                return
            for i in range(start, len(candidates)):
                if i > start and candidates[i] == candidates[i - 1]:  # skip duplicates
                    continue
                num = candidates[i]
                if num > remaining_target:  # stop early when the sum exceeds the target
                    break
                dfs(cur_res + [num], i + 1, remaining_target - num)
        
        dfs([], 0, target)
        return res
