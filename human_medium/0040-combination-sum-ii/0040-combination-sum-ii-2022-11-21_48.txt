class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:

        candidates.sort()
        res = []
        def dfs(position, sum, slate ,index_set):
            if sum > target:
                return 
            
            if sum == target:
                res.append(slate[:])

            for index in range(position, len(candidates)):
                if index > position and candidates[index] == candidates[index-1]:
                    continue
                if sum + candidates[index] > target:
                    break
                dfs(index+1, sum+candidates[index], slate+[candidates[index]],index_set )

        dfs(0, 0, [], set())
        return res
