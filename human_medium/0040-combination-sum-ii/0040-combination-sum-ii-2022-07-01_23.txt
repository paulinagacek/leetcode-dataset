class Solution:
    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:
        
        ## SORTING GIVEN ARRAY ###
        candidates.sort()
        finalAns=[]
        self.getCombinationSum2(candidates, 0, target, [], finalAns)
        return finalAns

        
    def getCombinationSum2(self, candidates: List[int], index: int, target: int, combinationList: List[int], finalAns: List[List[int]]):
        
        if target == 0:
            finalAns.append(combinationList[::])
            return
        
        for i in range(index, len(candidates)):
            if (i > index and candidates[i] == candidates[i-1]):
                continue
            if candidates[i]>target:
                break
                
            combinationList.append(candidates[i])
            self.getCombinationSum2(candidates, i+1, target-candidates[i], combinationList, finalAns)
            combinationList.pop()
        
    
#Time Complexity:
#----------------
#If all the elements in array are unique then (2^n)*k, k for adding combination to final answer.

#Space Complexity:
#----------------
#Assume average length k and x combinatins present so SC -> k*x
