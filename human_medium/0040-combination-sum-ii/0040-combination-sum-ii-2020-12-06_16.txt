nums.sort()
ans = []

def dfs(idx, path, target):

	if idx > len(nums):
		return

	if target == 0:
		ans.append(path)

	for i in range(idx, len(nums)):
		if i > idx and nums[i] == nums[i-1]: continue

		if nums[i] > target: return

		dfs(i+1, path+[nums[i]], target-nums[i])

dfs(0, [], target)
return ans
