# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class BSTIterator:

    def __init__(self, root: Optional[TreeNode]):
        self.curr = TreeNode(right=root)

    def next(self) -> int:
        node = self._next_morris()
        return node.val

    def hasNext(self) -> bool:
        return self.curr.right is not None
    
    def _next_morris(self):
        """
        Performs a single iteration of the Morris Traversal algorithm.
        When the next inorder node is found, the function returns.
        """
        self.curr = self.curr.right
        while self.curr:
            if not self.curr.left:
                return self.curr
            else:
                left_subtree = self.curr.left
                while left_subtree.right and left_subtree.right != self.curr:
                    left_subtree = left_subtree.right
                
                if not left_subtree.right:
                    left_subtree.right = self.curr
                    self.curr = self.curr.left
                else:
                    left_subtree.right = None
                    return self.curr
