class BSTIterator {
public:
    TreeNode* curr;
    stack<TreeNode*> st;
    BSTIterator(TreeNode* root) {
        this->curr = root;
    }
    
    int next() {
        while(curr){
            st.push(curr);
            curr = curr->left;
        }
        
        curr = st.top();
        st.pop();
        int res = curr->val;
        curr = curr->right;
        return res;
    }
    
    bool hasNext() {
        return !st.empty() || curr;
    }
};
