class BSTIterator:

    def __init__(self, root: Optional[TreeNode]):
        self.root = root
        self.stack = []
        while root:
            self.stack.append(root)
            root = root.left
        

    def next(self) -> int:
        if self.stack:
            x = self.stack[-1]
            z = self.stack.pop()
            
        if z.right:
            z = z.right
            while z:
                self.stack.append(z)
                z = z.left
        return x.val
    def hasNext(self) -> bool:
        if self.stack:
            return True
        return False
