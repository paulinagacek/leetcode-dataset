import queue
class BSTIterator:

    def __init__(self, root: TreeNode):
        self.q=queue.Queue()
        stack=[]
        
        
        while True:
            while root:
                stack.append(root)
                root=root.left
            if not stack:
                break
            node=stack.pop() 
            self.q.put(node.val)
            root=node.right
            

    def next(self) -> int:
        """
        @return the next smallest number
        """
        
        return self.q.get()
        

    def hasNext(self) -> bool:
        """
        @return whether we have a next smallest number
        """
        return not self.q.empty()
