class BSTIterator:

    def __init__(self, root: Optional[TreeNode]):
        self.tree = []
        
        def dfs(node):
            if not node: return
            dfs(node.left)
            self.tree.append(node.val)
            dfs(node.right)
        
        dfs(root)

    def next(self) -> int:
        return self.tree.pop(0)

    def hasNext(self) -> bool:
        return len(self.tree) > 0
