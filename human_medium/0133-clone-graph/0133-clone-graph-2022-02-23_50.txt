class Solution:
    def cloneGraph(self, node: \'Node\') -> \'Node\':
        hashmap = {}

        def dfs(node):
            if node in hashmap:
                return hashmap[node]

            newCopy = Node(node.val)
            hashmap[node] = newCopy

            for neighbor in node.neighbors:
                newCopy.neighbors.append(dfs(neighbor))

            return newCopy

        if node:
            return dfs(node)
        return None
