class Solution:
    visited = {}
    def cloneGraph(self, node: \'Node\') -> \'Node\':
        if not node:
            return None
        if node in self.visited:
            return self.visited[node]

        new_node = Node(node.val)
        self.visited[node] = new_node
        new_node.neighbors = []
        for v in node.neighbors:
            new_node.neighbors.append(self.cloneGraph(v))
        return new_node
