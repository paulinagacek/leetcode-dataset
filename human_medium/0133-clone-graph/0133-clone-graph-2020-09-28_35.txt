class Solution(object):
    def cloneGraph(self, node, visited = {}):
        """
        :type node: Node
        :rtype: Node
        """
        if not node:
          return node
        if node in visited:
          return visited[node]
        clone = Node(node.val)
        visited[node] = clone
        for i in node.neighbors:
          clone.neighbors.append(self.cloneGraph(i, visited))
        
        return clone
