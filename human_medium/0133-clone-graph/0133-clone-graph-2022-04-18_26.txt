class Solution:
    def cloneGraph(self, root: Optional[\'Node\']) -> Optional[\'Node\']:
        if not root: return None
        
        hmap = dict()

        def recur(node):
            if node in hmap:
                return hmap[node]
            new_node = Node(node.val)
            hmap[node] = new_node
            for x in node.neighbors:
                new_node.neighbors.append(recur(x))
            return new_node
        
        return recur(root)
