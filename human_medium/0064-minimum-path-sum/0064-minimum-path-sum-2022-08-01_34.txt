def minPathSum(self, grid: List[List[int]]) -> int:        
        for row in range(len(grid)):
            for col in range(len(grid[0])):
                if row == 0 and col == 0:
                    continue

                up_cell = grid[row - 1][col] if row - 1 >= 0 else float(\'inf\')
                left_cell = grid[row][col - 1] if col - 1 >= 0 else float(\'inf\')
                grid[row][col] = grid[row][col] + min(up_cell, left_cell)
        
        return grid[-1][-1]
