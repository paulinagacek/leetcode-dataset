class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        \'\'\'
        Approach:
        1. Compute the sum to reach untill each element in first column
        2. Compute the sum to reach untill each element in first row
        3. Then traverse from index 1,1 to untill end and only take the minimum path at each location 
        
        Minimum path comes from either from previous element(previous row and column).
        \'\'\'
        m, n = len(grid), len(grid[0]) 
        for i in range(1, m):
            grid[i][0] += grid[i-1][0]
        for j in range(1, n):
            grid[0][j] += grid[0][j-1]
        for i in range(1, m):
            for j in range(1, n):
                grid[i][j] += min(grid[i-1][j], grid[i][j-1])
        return grid[-1][-1]
