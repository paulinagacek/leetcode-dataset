class Solution:
    def solve(self,i,j,a):
        if i==0 and j==0:
            return a[0][0]
        if i<0 or j<0:
            return 1000000000000
        up=a[i][j]+self.solve(i-1,j,a)
        down=a[i][j]+self.solve(i,j-1,a)
        return min(up,down)
    
    def minPathSum(self, grid: List[List[int]]) -> int:
        m=len(grid)
        n=len(grid[0])
        return self.solve(m-1,n-1,grid)
