class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        return reduce(lambda acc, row: reduce(lambda acc1, zipped: (min(acc1[0], zipped[0]) + zipped[1], acc1[1] + [min(acc1[0], zipped[0]) + zipped[1]]), zip(acc, row), (10**10, []))[1], grid[1:], reduce(lambda res, el: res + [res[-1:][0] + el], grid[0][1:], [grid[0][0]]))[-1:][0]
