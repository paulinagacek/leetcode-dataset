class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
        dp = list(itertools.accumulate(grid[0]))
        n = len(grid[0])
        for row in grid[1:]:
            dp[0] += row[0]
            for i in range(1, n):
                dp[i] = row[i] + min(dp[i], dp[i-1])
            
        return dp[-1]
