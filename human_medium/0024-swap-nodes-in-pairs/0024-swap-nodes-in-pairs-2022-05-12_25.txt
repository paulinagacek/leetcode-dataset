class Solution:
    #T=O(n), S=O(1)
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        #sentinel node for edge cases
        dummy = ListNode(0, head)
        #init previous and current pointers
        prev, cur = dummy, dummy.next
        
        #edge case that there must be atleast two nodes to swap
        while cur and cur.next:
            #save pointers
            second = cur.next
            nextPair = cur.next.next
            
            #modify pointers
            second.next = cur
            cur.next = nextPair
            prev.next = second
            
            #shift pointers
            prev = cur
            cur = nextPair
        
        #return head
        return dummy.next
