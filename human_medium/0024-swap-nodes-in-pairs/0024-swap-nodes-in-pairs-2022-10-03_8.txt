class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # MAKE a dummy node
        dummy = prev = ListNode(0)
        prev.next = head
        
        # prev-a-b-c-d-e
        # prev-b-a-c
        while head and head.next:
            a = head 
            b = a.next 
            c = b.next 
            
            prev.next = b 
            b.next = a 
            a.next = c
            
            prev= a
            head= c
        
        
        return dummy.next
