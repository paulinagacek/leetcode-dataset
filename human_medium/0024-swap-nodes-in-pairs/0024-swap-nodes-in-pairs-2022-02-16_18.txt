class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        d=ListNode(0,head)
        prev=d
        cur=head
        while cur and cur.next:
            prev.next=cur.next
            cur.next=cur.next.next
            prev.next.next=cur
            cur=cur.next
            prev=prev.next.next
        return d.next
