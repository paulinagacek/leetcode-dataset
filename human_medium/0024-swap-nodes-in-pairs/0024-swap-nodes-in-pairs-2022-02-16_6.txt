class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head is None:
            return head
        
        node=head
        c=0
        while node:  #loop to check the length of linkedlist
            c+=1
            node=node.next
            
        node=head
        if c%2!=0:    #If the length of linkedlist is odd then this loop will be executed
            while node.next is not None:
                node.next.val, node.val=node.val, node.next.val  #swapping the values
                node=node.next.next
        else:
            while node is not None:
                node.val, node.next.val=node.next.val, node.val #swapping the values
                node=node.next.next
        return head
