class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head==None or head.next==None: 
            return head
        temp,temp2=head,head.next 
        temp.next,temp2.next=temp2.next,temp # For swapping the first two nodes
        t=temp
        head=temp2 #Storing value of temp2 in head
        if temp.next==None or temp2.next==None or temp.next.next==None or temp2.next.next==None:
            return head
        #Moving the nodes 2 steps in forward direction
        temp2,temp=temp.next.next,temp2.next.next 
        t.next=temp2
        while temp and temp2:
            temp.next,temp2.next=temp2.next,temp
            t=temp
            if temp.next==None or temp2.next==None or temp.next.next==None or temp2.next.next==None:
                break
            #Moving the nodes 2 steps in forward direction
            temp2,temp=temp.next.next,temp2.next.next 
            t.next=temp2
        return head
