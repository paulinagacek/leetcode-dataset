class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        if not head or head.next is None:
            return head
        
        prev = head
        temp = head.next.next
        
        while prev.next:
            prev.val, prev.next.val = prev.next.val, prev.val
            
            prev = temp
            if temp and temp.next: temp = temp.next.next
            else: break
                
        return head
