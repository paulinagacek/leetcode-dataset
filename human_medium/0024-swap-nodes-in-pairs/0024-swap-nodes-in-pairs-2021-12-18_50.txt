class Solution:
    
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head == None or head.next == None:
            return head
        
        next_call = head.next.next
        # swapping the nodes
        first = head
        second = head.next
        second.next = first
        first.next = self.swapPairs(next_call)
        return second
