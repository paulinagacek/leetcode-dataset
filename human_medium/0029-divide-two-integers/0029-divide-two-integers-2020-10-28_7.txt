var divide = function(a, b) {
    
	// step1
    let flag=0
    if(a<0){ if(b>0) flag = 1 }
    else if(b<0) flag = 1
    
	
	
	// step2
    if( a == -2147483648 && b == -1)  return 2147483647
    
	
	
    // step3 
    let res=0
    b = Math.abs(b)
    a = Math.abs(a)
    
    while (a >= b) {
      a -= b;
      res++;
    }
	
	
	
	// step4
    if (flag == 1 ) return (res*-1) // res = (-res)
    else return res
    
};
