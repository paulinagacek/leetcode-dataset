class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        if not head or k == 0:
            return head
        
        last = head
        count = 1
        while last.next:
            count += 1
            last = last.next
        
        last.next = head
        curr = head
        
        for _ in range(count - (k % count) - 1):
            curr = curr.next
        
        res = curr.next
        curr.next = None
        
        return res
