class Solution:
    def rotateRight(self, node: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not node or k == 0:
            return node
        k = self.findLength(node, k)
        if k == 0:return node
        slow, fast = node, node
        for _ in range(k):
            fast = fast.next

        while fast is not None and fast.next is not None:
            fast = fast.next
            slow = slow.next

        secondHalf = slow.next
        
        slow.next = None
        fast.next = node
        node = secondHalf

        return node
    
    def findLength(self, node, k):
        length = 0
        while node is not None:
            length += 1
            node = node.next

        return (k % length)
