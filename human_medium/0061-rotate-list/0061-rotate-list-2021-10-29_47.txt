class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head:
            return None
        n = 1
        curr = head
        while curr.next:
            n += 1
            curr = curr.next
        k = k%n
        if k == 0:
            return head
        curr.next = head
        for i in range(n-k):
            curr = curr.next
        new = curr.next
        curr.next = None
        return new
