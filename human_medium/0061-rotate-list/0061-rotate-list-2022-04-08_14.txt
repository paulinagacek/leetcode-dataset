class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        
        if k==0 or not head or not head.next:
            return head
        
        temp=head
        length=0
        
        while temp:
            temp=temp.next
            length+=1
        
        k=k%length
        if k==0:
            return head
        
        temp=head
        for i in range(length-k):
            prev=temp
            temp=temp.next
        
        resHead=temp
        prev.next=None
        
        while temp.next:
            temp=temp.next
        
        temp.next=head
        return resHead
