class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        l=0
        t=head
        if head is None:
            return head
        
        while t:
            t=t.next
            l+=1
            
        k = k%l
        s= head
        f=head
        for _ in range(k):
            f = f.next
        while f.next:
            f=f.next
            s=s.next
            
        
        f.next=head
        head = s.next
        s.next = None
        return head
