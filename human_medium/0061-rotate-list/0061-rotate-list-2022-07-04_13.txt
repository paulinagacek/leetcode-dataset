class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        
        if not head or not head.next: return head
        
        #first pass count length and make list circular
        N=1
        cur=head
        while cur.next:
            N+=1
            cur=cur.next
        cur.next=head
        
        M=N-k%N #list index of new head
        
        #second pass prev node point to none new node be head
        i=0
        cur=head
        while i<M:
            prev=cur
            cur=cur.next
            i+=1
        prev.next=None
        head=cur
        
        return head
