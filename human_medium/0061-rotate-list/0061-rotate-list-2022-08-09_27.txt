class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next or k==0: return head
        p=h=head
        length=1
        while p.next:  #count the length 
            length+=1
            p=p.next
            
        p.next=head    #to make cycle
        k=k%length     #to handle large k values
        
        cut=length-k   
        
        for _ in range(cut-1): h=h.next ##to cut the Linked list for common part
              
        head=h.next    #changing the head
        h.next=None    #assigning last to none
        
        return head
