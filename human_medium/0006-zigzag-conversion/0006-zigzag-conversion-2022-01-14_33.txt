class Solution:
        
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1: return s
		# Keeps track of current row
        count = 0
        direction = 1
        rows = ["" for i in range(numRows)]
		# Loop each index of the string
        for i in range(0, len(s)):
			# Add letter to correct row
            rows[count] += s[i]
			# Check whether the path should reverse
            if count == numRows - 1: direction = -1
            elif count == 0: direction = 1
            count += direction
		# Return rows joined in order.
        return "".join(rows)
