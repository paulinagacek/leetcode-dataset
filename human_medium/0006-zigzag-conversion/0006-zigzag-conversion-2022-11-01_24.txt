class Solution:
    def convert(self, s: str, numRows: int) -> str:

        solutionArray = [""] * numRows
        row = 0

        for ele in range(len(s)):
            
            if (numRows == 1): return s
            solutionArray[row] += s[ele]
            if (row == 0): step = 1
            elif (row == (numRows - 1)): step = -1

            row += step
        
        return \'\'.join(solutionArray)
