class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1:
            return s
        
        jump = 2 * (numRows-1) 
        r = len(s)
        res = ""
        
        for i in range(numRows): #Num of rows
            for j in range(i,len(s),jump): # For first and last rows it jumps 2 * (numRows-1) times
                res += s[j]
                if i > 0 and i < numRows-1 and j+jump-2*i  < len(s): 
				#For any middle rows it jumps the same amount but we need to take care the columns where only one character exists. 
				#For those characters we add them to the result string by jumping the same amount but 2 characters less everytime as we move down the row.
				#Ex. 1st char jump 8 spots, 2nd char jumps 8 spots but we add the current spot + jump - 2 * (current row) ...
                    res += s[j+jump-2*i]  #
                    
        return res
