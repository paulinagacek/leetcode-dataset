class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows == 1:
            return s
        res = [\'\' for _ in range(numRows)]
        flag = -1
        idx = 0
        for c in s:
            res[idx] += c
            if idx in [0, numRows-1]:
                flag = -flag
            idx += flag
        return \'\'.join(res)
