class Solution:
    def convert(self, s: str, numRows: int) -> str:
        if numRows==1:
            return s
        res=[]
        for i in range(numRows):
            res.append([])
        direction=1 # 1 for down and 0 for up
        i=0 # row index in which next letter is to be added
        for c in s:
            res[i].append(c)
            if (direction==1 and i==numRows-1): # last row downward
                i-=1
                direction=0
            elif (direction==0 and i==0): # first row upward
                i+=1
                direction=1
            elif (direction==1): # middle rows downward
                i+= 1
            else: # middle rows upward
                i-=1
        o=""
        for i in range(numRows):
            o+="".join(res[i])
        return o
