class Solution:
    def convert(self, s: str, numRows: int) -> str:
        
        if numRows == 1:
            return s
        
        rows = [\'\' for i in range(numRows)]
        cur_row = 0
        down = True
        for c in s:
            rows[cur_row] += c
            if down:
                cur_row += 1
            else:
                cur_row -= 1
            if cur_row in (0, numRows - 1):
                down = not down
        return \'\'.join(rows)
