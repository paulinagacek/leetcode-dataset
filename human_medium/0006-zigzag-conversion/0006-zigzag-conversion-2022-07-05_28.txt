def convert(self, s: str, numRows: int) -> str:
        if numRows == 1 or numRows >= len(s):
            return s
        
        res = [""] * numRows
        direction = -1
        row = 0
        for c in s:
            res[row] += c
            if row == 0 or row == numRows - 1:
                direction *= -1
            row += direction
        return "".join(res)
    
        """
        time: O(n) where n is len(s)
        space: O(n)
        """
