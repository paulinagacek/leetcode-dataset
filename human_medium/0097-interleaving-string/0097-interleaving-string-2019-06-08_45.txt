class Solution:
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        m, n, p = len(s1), len(s2), len(s3)
        if m + n != p:
            return False
        @functools.lru_cache(None)
        def dp(i, j, k):
            if k == p:
                return True
            return (i < m and s1[i] == s3[k] and dp(i+1, j, k+1)
                    or j < n and s2[j] == s3[k] and dp(i, j+1, k+1))
        return dp(0, 0, 0)
