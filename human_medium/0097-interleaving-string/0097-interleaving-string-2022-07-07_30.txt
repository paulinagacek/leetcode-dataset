class Solution:
    @lru_cache
    def isInterleave(self, s1: str, s2: str, s3: str) -> bool:
        
        if len(s1) + len(s2) != len(s3):
            return False
        
        if not s3 or s1 == s3 or s2 == s3:
            return True
        
        option1 = self.isInterleave(s1[1:], s2, s3[1:]) if s1 and s3 and s1[0] == s3[0] else False
        option2 = self.isInterleave(s1, s2[1:], s3[1:]) if s2 and s3 and s2[0] == s3[0] else False
        
        return option1 or option2
