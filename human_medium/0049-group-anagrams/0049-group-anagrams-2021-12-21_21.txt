class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        mapping = collections.defaultdict(list)
        res = []
        
        for s in strs:
            mapping[self.get_hash_using_sorting(s)].append(s)
        
        return list(mapping.values())
    
    #ensure anagrams have same hash
    def get_hash(self, s):
        count = [0]*26
        for c in s:
            count[ord(c) - ord(\'a\')] += 1
        
        return \'_\'.join(map(str, count))
    
    # Sorting can also be used to get a hash, but has worse complexity
    # Not recommended for large-sized s
    def get_hash_using_sorting(self, s):
        return \'\'.join(sorted(s))
