class Solution:
    # O(NK) Time complexity where K is the max length of the strings. Space complexity: O(NK)
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        ans = collections.defaultdict(list)
        for s in strs:
            count = [0] * 26
            for c in s:
                count[ord(c) - ord(\'a\')] += 1
            ans[tuple(count)].append(s)
        return ans.values()
