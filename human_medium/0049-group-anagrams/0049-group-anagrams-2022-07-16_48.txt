class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        strs_dict = dict()
        for word in strs:
            count = [0] * 26
            for char in word:
                count[ord(char) - ord(\'a\')] += 1
            i = tuple(count)
            if i in strs_dict:
                strs_dict[i].append(word)
            else:
                strs_dict[i] = [word]
        return list(strs_dict.values())
