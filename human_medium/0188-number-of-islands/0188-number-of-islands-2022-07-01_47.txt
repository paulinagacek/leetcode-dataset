class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        # if there are no elements in the grid return 0
        if len(grid)==0: return 0
        row,col=len(grid),len(grid[0])
        visited=set()
        count=0
        # define dfs to traverse the grid
        def dfs(r,c):
            # if row or column is invalid return
            if (r<0 or r>=row): return
            if (c<0 or c>=col): return
            # if the element is 0 or already visited return
            if grid[r][c]==\'0\': return
            if (r,c) in visited: return
            # else add the element in to visited
            visited.add((r,c))
            # run dfs for adjacent nodes
            dfs(r-1,c)
            dfs(r+1,c)
            dfs(r,c-1)
            dfs(r,c+1)
        # for each element in the grid    
        for r in range(row):
            for c in range(col):
                # if it\'s 1 and not already visited, its a new island
                if grid[r][c]!=\'0\' and (r,c) not in visited:
                    count+=1
                    dfs(r,c)
        # return count            
        return count
