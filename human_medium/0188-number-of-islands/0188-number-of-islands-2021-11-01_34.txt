class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        ROWS, COLS = len(grid),len(grid[0])
        soln=0
        
        #method to perform DFS. changes the value to I in grid so that the next iteration wont duplicate the count
        def capture(r,c):
            if r<0 or r>=ROWS or c<0 or c>=COLS or grid[r][c]!="1":
                return
            grid[r][c]=\'I\'
            capture(r+1,c)
            capture(r-1,c)
            capture(r,c+1)
            capture(r,c-1)
        
        #iterate through the grid and update self and neighbours and finally increment it to 1
        for i in range(ROWS):
            for j in range(COLS):
                if grid[i][j]=="1":
                    capture(i,j)
                    soln+=1
        return soln
