class DLLNode:
    def __init__(self, val: int = None, key: int = None, next = None, prev = None):
        self.val = val
        self.key = key
        self.next = next
        self.prev = prev

class DDLinkedList:
    def __init__(self):
        self.head = DLLNode()
        self.tail = DLLNode()
        self.head.next = self.tail
        self.tail.prev = self.head

    def add_tail(self, node):
        # --- prev --- tail
        # --- prev --- node --- tail
        prev = self.tail.prev
        prev.next = node
        node.prev = prev
        node.next = self.tail
        self.tail.prev = node

    def get_head(self):
        return self.head.next

    def remove(self, node):
        # prev --- node --- next
        prev = node.prev
        nxt = node.next
        prev.next = nxt
        nxt.prev = prev

class LRUCache:

    def __init__(self, capacity: int):
        self.dll = DDLinkedList()
        self.map = {}
        self.capacity = capacity
        self.size = 0

    def _move_to_back(self, node):
        self.dll.remove(node)
        self.dll.add_tail(node)

    def get(self, key: int) -> int:
        if key in self.map:
            node = self.map[key]
            self._move_to_back(node)
            return node.val
        return -1

    def put(self, key: int, value: int) -> None:
        if key in self.map:
            node = self.map[key]
            node.val = value
            self._move_to_back(node)
            return

        if self.size == self.capacity:
            node = self.dll.get_head()
            del self.map[node.key]
            node.key = key
            node.val = value
            self.map[key] = node
            self._move_to_back(node)
        else:
            node = DLLNode(value, key)
            self.map[key] = node
            self.dll.add_tail(node)
            self.size += 1
