class LRUCache:

    def __init__(self, capacity):
        self.cache = {}
        self.capacity = capacity

    def get(self, key):
        v = self.cache.get(key,-1)
        if not v is -1:
            del self.cache[key]
            self.cache[key] = v
        return v
            
    def put(self, key, value):
        if key in self.cache: 
            del self.cache[key]
        self.cache[key] = value
        
        if len(self.cache) > self.capacity:
            del self.cache[next(iter(self.cache))]
