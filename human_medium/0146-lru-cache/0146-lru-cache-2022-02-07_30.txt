class LRUCache:

    def __init__(self, capacity: int):
        self.dic = dict()
        self.capacity = capacity


    def get(self, key: int) -> int:
        if key in self.dic:
            value = self.dic[key]
            del self.dic[key]
            self.dic[key] = value
            return value

        return -1

    def put(self, key: int, value: int) -> None:
        if key in self.dic:
            del self.dic[key]

        self.dic[key] = value

        if len(self.dic) > self.capacity:
            oldest_key = next(iter(self.dic.keys()))
            del self.dic[oldest_key]
