def uniquePaths(self, m: int, n: int) -> int:
        if m<=0 or n<=0:
            return None
        if m==1 or n==1:
            return 1
        
        row_memory = [1]*n
        for i in range(1,m):
            for j in range(1,n):
                row_memory[j] = row_memory[j] + row_memory[j-1]
                
        return row_memory[-1]
