class Solution:
    def uniquePaths(self, m: int, n: int) -> int:
        # the robot needs to move down m-1 times, and right n-1 times
        # so move (m-1)+(n-1)=m+n-2 times in total
        # permute the m+n-2 steps, but insides the down or right steps respectively,
        # there is no order differencee
        # so divided by the number of permutation by down m-1 steps and right n-1 steps 
  
        # as you should know in math, the number of permutation of taking x elements from x elements is x!, in other words, the fatorial of x.
        
        def factorial(x):
            res = 1
            for k in range(1,x+1):
                res *= k
            return res
        
        ans = factorial(m+n-2) // (factorial(m-1) * factorial(n-1))
        return ans
