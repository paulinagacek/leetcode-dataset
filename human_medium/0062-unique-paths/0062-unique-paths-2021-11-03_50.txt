def uniquePaths(self, m: int, n: int,cnt=0) -> int:
        paths=[[1]*n]*m # initialize with 1 (mainly for 1st row and 1st column as base cases)
        
        for i in range(1,m):
            for j in range(1,n):
                paths[i][j] = paths[i-1][j] + paths[i][j-1] 
        
        return paths[m-1][n-1]
