class Solution:    
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        carry=0
        h=ListNode()
        head=h
        while l1 or l2 or carry:
            result=0
            if l1:
                result+=l1.val
                l1=l1.next
            if l2:
                result+=l2.val
                l2=l2.next
            if carry:
                result+=carry
                
            val=result%10
            carry=result//10
            head.next=ListNode(val)
            head=head.next
        return h.next
