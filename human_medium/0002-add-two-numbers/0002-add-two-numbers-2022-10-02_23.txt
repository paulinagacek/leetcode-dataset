class Solution:
        def addTwoNumbers(self, l1, l2):
                #create a arr(string) to store the values in l1
                arr = ""
                #iterate over the nodes in l1(linkedlist)
                while l1 != None:
                        #add each element to string(arr)
                        arr = arr + str(l1.val)
                        #change the l1 pointer to it\'s next
                        l1 = l1.next
                #after completing the iteration of l1(linkedlist)
                #reverse the string
                arr = arr[-1::-1]
                #create a arr2(string) to store the values in l2
                arr2 = ""
                #iterate over the nodes in l2(linkedlist)
                while l2 != None:
                        #add each element to string(arr2)
                        arr2 = arr2 + str(l2.val)
                        #change the l2 pointer to it\'s next
                        l2 = l2.next
                #after completing the iteration of l2(linkedlist)
                #reverse the string
                arr2 = arr2[-1::-1]
                #add two strings after changing them to int and store that value in variable(thoma)
                thoma = int(arr) + int(arr2)
                #change the variable thoma to string
                thoma  = str(thoma)
                #reverse the string thoma
                thoma = thoma[-1::-1]
                #create a node
                dummy = ListNode(0)
                #create a iterator and point to head
                head = dummy
                #iterate over the elements in string(thoma)
                for i in thoma:
                        #for each element create node with value of element 
                        newNode = ListNode(i)
                        #point head\'s next to newNode
                        head.next = newNode
                        #move head to it\'s next
                        head = head.next 
                #return dummy.next because first value is 0 
                return dummy.next
