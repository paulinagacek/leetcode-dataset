class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        # traverse l1 and concatenate the string val at each node
        d1 = \'\'
        while l1:
            d1 += str(l1.val)
            l1 = l1.next
        
        # traverse l2 and concatenate the string val at each node
        d2 = \'\'
        while l2:
            d2 += str(l2.val)
            l2 = l2.next
        
        # reverse the digits and convert from str to int
        d1 = int(d1[::-1])
        d2 = int(d2[::-1])
        
        # sum up the digits, convert from int to str, and reverse the digits
        digits = str(d1 + d2)[::-1]
        
        # construct a new linked list with each node[i] with val digits[i]
        head = ListNode(digits[0])
        node = head
        for i in range(1, len(digits)):
            node.next = ListNode(digits[i])
            node = node.next
        
        # return the head of the new linked list
        return head
