class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        # create a dummy node, start appending the sum
        dummy = ListNode()
        tail = dummy
        carry = 0

        while l1 or l2:
            l1val = l1.val if l1 else 0
            l2val = l2.val if l2 else 0
            tail.next = ListNode((carry + l1val + l2val)%10 )
            carry = (carry + l1val + l2val) // 10
            if l1:
                l1 = l1.next
            if l2:
                l2 = l2.next
            tail = tail.next
            
                
        # if we run out of both lists and there is a carry from the last node
        if carry>0 :
            tail.next = ListNode(carry)
        return dummy.next
