class ListNode:
	""" Given in the problem """
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
		
def make_list_from_ListNode(l1):
    list1 = []
    while l1: #is not None
        list1.append(l1.val)
        l1 = l1.next
    return list1

def make_ListNode_from_list(a_list):
    head = l3 = ListNode(a_list[0])
    for x in a_list[1:]:
        l3.next = l3 = ListNode(x)
    return head
