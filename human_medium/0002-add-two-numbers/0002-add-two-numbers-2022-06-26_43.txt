class Solution(object):
    def addTwoNumbers(self, l1, l2):
        """
        :type l1: ListNode
        :type l2: ListNode
        :rtype: ListNode
        """
        dummy= ListNode()
        cur=dummy
        carry=0
        while l1 or l2 or carry:          #taking carry in condition to cover edge cases where carry has to be inserted in a new node
            v1=l1.val if l1 else 0
            v2=l2.val if l2 else 0
            
            value=v1+v2+carry
            carry=value//10                   #Updating carry
            value=value%10                  #Updating the sum value which has to be updated excluding carry
            cur.next= ListNode(value)
            
            cur=cur.next
            l1=l1.next if l1 else None
            l2=l2.next if l2 else None
        return dummy.next
