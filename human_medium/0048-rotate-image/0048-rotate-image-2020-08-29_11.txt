class Solution:
    def rotate(self, matrix: list) -> None:
        l = len(matrix)
        for i in range(len(matrix) // 2):
            for j in range(i, len(matrix[i]) - i - 1):
                temp = matrix[i][j]
                matrix[i][j] = matrix[l - 1 - j][i]
                matrix[l - 1 - j][i] = matrix[l - 1 - i][l - 1 - j]
                matrix[l - 1 - i][l - 1 - j] = matrix[j][l - 1 - i]
                matrix[j][l - 1 - i] = temp
