class Solution:
    def rotate(self, matrix: List[List[int]]) -> None:
        """
        Do not return anything, modify matrix in-place instead.
        """
        row=len(matrix)
        col=len(matrix[0])
        
        #swap values (row,coloumn wise) upper half only
        for i in range(row):
            for j in range(i,col):
                matrix[i][j],matrix[j][i]=matrix[j][i],matrix[i][j]
        
        #reverse each row
        for i in range(row):
            matrix[i].reverse()
        return matrix
