class Solution:
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        n = len(triangle)
        dp = [[10e9 for _ in range(i+1)] for i in range(n)]
        dp[0][0] = triangle[0][0]
        for i in range(1, n):
            for j in range(i):
                dp[i][j] = dp[i-1][j] + triangle[i][j]
            for j in range(1, i+1):
                dp[i][j] = min(dp[i][j], triangle[i][j] + dp[i-1][j-1])
        return min(dp[-1])
