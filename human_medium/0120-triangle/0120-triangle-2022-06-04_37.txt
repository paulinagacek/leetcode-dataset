class Solution:
#     66ms 94.29%
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        if not triangle:
            return 0
        
        dp = [0] * (len(triangle) + 1)
        
        
        for i in reversed(range(len(triangle))):
            for idx, val in enumerate(triangle[i]):
                dp[idx] = min(dp[idx], dp[idx+1]) + val
                
        return dp[0]
