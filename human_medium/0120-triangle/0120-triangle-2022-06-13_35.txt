class Solution:
    from functools import lru_cache
    
    def minimumTotal(self, triangle: List[List[int]]) -> int:
        
        length = len(triangle)
        
        @lru_cache(maxsize = None)
        def traverse(row, idx):
            
            if row >= length or idx > row + 1:
                return 0

            elif idx == row + 1:
                return triangle[row][idx] + traverse(row+1, idx)
            else:
                return triangle[row][idx] + min(traverse(row+1, idx), traverse(row+1, idx+1))
    
        return traverse(0, 0)
