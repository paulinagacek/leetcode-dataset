class Solution:
    def grayCode(self, n: int) -> List[int]:
        def rechelper(n):
            if n ==1:
                return ["0", "1"]
            ans = rechelper(n-1)
            res = []
            for i in range(len(ans)):
                res.append(str(0)+ ans[i])
            for i in range(len(ans)-1, -1, -1):
                res.append(str(1) + ans[i])
            return res
        ls = rechelper(n)
        ls = [int(i,2) for i in ls]
        return ls
