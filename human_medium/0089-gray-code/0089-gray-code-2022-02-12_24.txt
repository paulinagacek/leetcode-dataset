class Solution:
    def grayCode(self, n: int) -> List[int]:
        if n == 1:
            return [0,1]
        else:
            tmp = self.grayCode(n-1)
            tmp1 = [item + 2**(n-1) for item in tmp]
            for i in range(1, 2**(n-1), 2):
                tmp.insert(i + (i//2) * 2, tmp1[i-1])
                tmp.insert(i + (i//2) * 2 + 1, tmp1[i])
            return tmp
