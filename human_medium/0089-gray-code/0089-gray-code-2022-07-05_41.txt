class Solution:
    def grayCode(self, n: int) -> List[int]:
        l = [\'0\',\'1\']
        if n == 1:
            return [0,1]
        for i in range(2,n+1):
            l_0 = l
            l_1 = l[::-1]
            for i in range(len(l_0)):
                l_0[i] = \'0\' + l_0[i]
            for j in range(len(l_1)):
                l_1[j] = \'1\' + l_1[j] 
            l = l_0 + l_1
        res = []
        b = 0
        for i in l:
            res.append(int(i,2))
        return res
