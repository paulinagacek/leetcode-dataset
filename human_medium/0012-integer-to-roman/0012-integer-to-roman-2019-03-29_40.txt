class Solution:
    def intToRoman(self, num: int) -> str:
        t = \'\'
        for i in X:
            tmp = num // i
            if tmp > 0:
                t += X[i] * tmp
                num -= (tmp * i)
        return t


X = {
    1000: \'M\',
    900: \'CM\',
    500: \'D\',
    400: \'CD\',
    100: \'C\',
    90: \'XC\',
    50: \'L\',
    40: \'XL\',
    10: \'X\',
    9: \'IX\',
    5: \'V\',
    4: \'IV\',
    1: \'I\',
}
