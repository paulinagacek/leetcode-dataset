def solution(height):
		i = 0
        j = len(height) - 1
        res = 0
        
        while i < j:
            temp_min = min(height[i], height[j])
            dist = j - i
            res = max(res, dist * temp_min)
            
            if temp_min == height[i]:
                i += 1
            else:
                j -= 1
        
        return res
