class Solution:
    def maxArea(self, A):
        left, right = 0, len(A) - 1
        res = 0
        while left < right:
            # print(left, right)
            res = max(res, min(A[left], A[right]) * (right - left))
            if A[left] < A[right]:
                left += 1
            else:
                right -= 1
        return res
