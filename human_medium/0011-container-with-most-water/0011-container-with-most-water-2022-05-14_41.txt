class Solution:
    def maxArea(self, height: List[int]) -> int:
        i=0
        j=len(height)-1
        maxWater=0
        while j>i:
            if(height[i]<height[j]):
                if(maxWater<(height[i]*(j-i))):
                    maxWater=height[i]*(j-i)
                i+=1
            else:
                if(maxWater<(height[j]*(j-i))):
                    maxWater=height[j]*(j-i)
                j-=1
        return maxWater
