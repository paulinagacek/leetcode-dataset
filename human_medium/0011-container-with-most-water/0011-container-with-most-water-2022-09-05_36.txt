class Solution:
    def maxArea(self, height: List[int]) -> int:
        max_volume = 0
        left, right = 0, len(height) - 1

        while left < right:
            new_volume = min(height[left], height[right]) * (right - left)
            max_volume = max(new_volume, max_volume)

            if height[left] < height[right]:
                left += 1
            else:
                right -= 1

        return max_volume
