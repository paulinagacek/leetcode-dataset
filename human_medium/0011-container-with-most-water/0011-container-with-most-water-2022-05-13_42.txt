class Solution:
    def maxArea(self, height: List[int]) -> int:
        
        left, right = 0, len(height) - 1
        res = min(height[right], height[left]) * (right - left)
        
        while left < right:
            
            if height[left] <= height[right]:
                left += 1
            else:
                right -= 1
            res = max(min(height[right], height[left]) * (right - left), res)
        
        return res
