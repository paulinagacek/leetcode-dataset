def connect(self, root: \'Optional[Node]\') -> \'Optional[Node]\':
	\'\'\'
	- level order traversal -> output list of node objects
	- simply iterate ifover nodes in each level and link it to its RHS neighbour
	\'\'\'
	# None root edge case
	if not root:
		return root

	# [STEP-1]: Generate levels
	from collections import deque
	q = deque()
	q.append(root)
	res = []
	while q:
		level = []
		for i in range(len(q)):
			node = q.popleft()
			level.append(node)

			if node.left:
				q.append(node.left)

			if node.right:
				q.append(node.right)         

		res.append(level)

	# [STEP-2]: For each level -> create a link between each node and its RHS neighbour 
	f_res = []
	for l in res:
		for i in range(len(l)):  
			node = l[i]

			if i != len(l)-1: # except the last node in a level
				nextNode = l[i+1]
			else:
				nextNode = None

			node.next = nextNode 
			# f_res.append(node.val)

	# print(f_res)
	return root
