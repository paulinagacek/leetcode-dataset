class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        parent = root
        while parent:
            child = dummy = Node()
            while parent: 
                if parent.left: child.next = child = parent.left
                if parent.right: child.next = child = parent.right
                parent = parent.next 
            parent = dummy.next
        return root
