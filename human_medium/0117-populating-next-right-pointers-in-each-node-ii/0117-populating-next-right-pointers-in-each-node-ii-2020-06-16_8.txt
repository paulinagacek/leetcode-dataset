class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        ## RC ##
        ## APPROACH : RECURSION ##
        ## Similar to Leetcode 116. Populating Next Right Pointers in Each Node ##    
        
		## TIME COMPLEXITY : O(N) ##
		## SPACE COMPLEXITY : O(1) ##
        
        ## EXAMPLE ##
        #                  1
        #                 / \\
        #       root --> 2   3
        #               / \\   \\
        # kid -->  (0) 4   5   7    temp = reference to start of this level
        # Now dry run the code, you will understand.
        
        self.root_node = root       # make a copy to return the result
        kid = temp = Node(0)        # kid is ptr for traversing the child level, temp is dummy ptr to just store kid starting position
        while root:
            while root:
                if root.left:
                    kid.next = root.left
                    kid = kid.next
                if root.right:
                    kid.next = root.right
                    kid = kid.next
                root = root.next
            root, kid = temp.next, temp
            kid.next = None         # Reset the chain for temp
        return self.root_node
