class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        if not root: return 
        stack=[root]
        while stack:
            temp=[]
            while stack:
                cur=stack.pop(0)
                if cur.left:
                    temp.append(cur.left)
                if cur.right:
                    temp.append(cur.right)
                if len(stack)==0:
                    cur.next=None
                else:
                    cur.next=stack[0]
            stack=temp
        return root
