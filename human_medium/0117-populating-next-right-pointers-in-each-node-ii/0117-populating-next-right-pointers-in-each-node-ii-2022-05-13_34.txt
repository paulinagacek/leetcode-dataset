class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        stack = [root]
        while stack:
            n = len(stack)
            for i in range(n):
                node = stack.pop(0)
                if node is None:
                    continue
                if i < n - 1:
                    node.next = stack[0]
                if node.left:
                    stack.append(node.left)
                if node.right:
                    stack.append(node.right)
        
        return root
