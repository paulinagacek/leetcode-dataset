class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        if not root:
            return root

        dummy = root
        stack = [dummy]

        while stack:
            N = len(stack)
            
            for i in range(1, N):
                stack[i-1].next = stack[i]
            
            tmp = stack.copy()
            stack.clear()

            for node in tmp:
                if node.left:   stack.append(node.left)
                if node.right:  stack.append(node.right)

        return root
