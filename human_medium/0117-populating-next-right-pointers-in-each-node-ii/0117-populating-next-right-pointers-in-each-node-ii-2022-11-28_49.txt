def connect(self, root: \'Node\') -> \'Node\':
        if not root: return
        q = collections.deque([root])
        while q:
            ll = len(q)
            for x in range(ll):
                node = q.popleft()
                if x < ll-1:
                    node.next = q[0]
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
        return root
