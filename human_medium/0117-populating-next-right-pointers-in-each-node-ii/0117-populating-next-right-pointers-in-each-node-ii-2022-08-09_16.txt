class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        if not root: return None
        
        queue = [root]
        
        #bfs
        
        while queue:
            L = len(queue)
            for i in range(L): 
                node = queue.pop(0)
                if (i + 1) == L: # if it\'s the right most node in current level
                    node.next = None
                else:
                    node.next = queue[0]  
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        return root
