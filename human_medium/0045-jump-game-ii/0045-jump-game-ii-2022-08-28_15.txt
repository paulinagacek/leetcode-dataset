class Solution:
    def jump(self, nums: List[int]) -> int:
        jumps = 0                                       # minimum number of jumps needed to reach goal
        l = r = 0                                       # initial window/portion pointers will [0]

        while r < len(nums)-1:
            # go through current window and find farthest index to form the new window
            farthest = 0
            for i in range(l, r+1):                     # right inclusive
                farthest = max(farthest, i + nums[i])

            l, r = r + 1, farthest                      # update window
            jumps += 1                                  # increment the jump
        
        return jumps
