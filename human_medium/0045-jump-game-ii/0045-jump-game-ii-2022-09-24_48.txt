class Solution {
public:
    int jump(vector<int>& nums) {
        
        
        int cur_reach = 0, reach = INT_MIN, count = 0;
        for(int i=0; i<nums.size();i++)
        {
            reach = max(reach, nums[i]);
            if(i == cur_reach && i != nums.size()-1)
            {
                count++;
                cur_reach = i + reach;
            }
            reach--;
        }
        
        return count;
    }
};
