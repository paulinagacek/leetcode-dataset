class Solution:
    def jump(self, nums: List[int]) -> int:
        
        @lru_cache(None)
        def min_jump(idx: int) -> int:
            if idx == len(nums) - 1:
                return 0
            if nums[idx] >= len(nums) - idx - 1:
                return 1
            if nums[idx] == 0:
                return math.inf
            return 1 + min([min_jump(i) for i in range(idx+1,idx+nums[idx]+1)])
        
        return min_jump(0)
