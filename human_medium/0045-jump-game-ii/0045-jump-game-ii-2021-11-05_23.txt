class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1 or nums[0] == 0:
            return 0
        jumps = 1
        current = nums[0]
        farthest = nums[0]
        for num in range(1, n):
            if num == n-1:
                break
            current -= 1
            farthest -= 1
            if farthest < nums[num]:
                farthest = nums[num]
            if current == 0:
                jumps += 1
                current = farthest
            # Negative scenario when we fail to reach the end, simply return -1
            if farthest == 0:
                return -1
        return jumps
