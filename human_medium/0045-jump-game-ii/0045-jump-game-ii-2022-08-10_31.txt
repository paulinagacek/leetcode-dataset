class Solution:
    def jump(self, nums: List[int]) -> int:
        N = len(nums)
        
        if N == 1:
            return 0
        
        dp = [0] * N
        
        for i in range(N - 2, -1, -1):
            if nums[i] == 0:
                dp[i] = float(\'inf\')
            elif nums[i] + i >= N - 1:
                dp[i] = 1
            else:
                dp[i] = min(dp[i + 1: (i + nums[i] + 1)]) + 1
        
        return dp[0]
