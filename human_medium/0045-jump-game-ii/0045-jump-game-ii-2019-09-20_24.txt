class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 1:
            return 0
        
        ladder = nums[0]    # keep track of the largest ladder you have
        stairs = nums[0]    # keep track of number of stairs in the current ladder
        jumps = 1
        
        for step in range(1, n):
            if step == n-1:        # if you reach the end of the ladder, 
                return jumps    # return the number jumps you made so far
            
            if step + nums[step] > ladder:      # if you can construct a ladder bigger than the
                ladder = step + nums[step]      # current one, construct it
                
            stairs -= 1     # climb up one stair
            
            if stairs == 0:             # if there are no stairs left
                jumps += 1               # make a jump to the next ladder
                stairs = ladder - step  # length of remaining stairs will be length of ladder minus the step you are on
        return jumps
