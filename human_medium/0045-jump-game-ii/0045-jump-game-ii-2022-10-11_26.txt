class Solution:
    def jump(self, nums: List[int]) -> int:
        n = len(nums)
        # max_reach at 0 index is (i + nums[i])
        max_reach = nums[0]
        # End pointer for 1 jump from i goes 0 to max_reach
        end = nums[0]
        # Initialy jump is 1 for at least 2 element
        jump = 1
        # Base Case if only one element then 0 jump
        if n==1:
            return 0
        # We should not go for end element because we already take one extra jump
        for i in range(1,n-1):
            # max_reach  should be max for minimizing the jump for every i until end pointer
            max_reach = max(max_reach, i+nums[i])
            # If i reached to end pointer then jump should increase & end pointer should be again max_reach
            if i==end:
                end = max_reach
                jump += 1
        return jump
