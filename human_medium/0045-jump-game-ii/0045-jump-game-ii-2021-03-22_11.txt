class Solution:
    def jump(self, array: List[int]) -> int:
        minJumps = [float("inf") for _ in array]
        minJumps[0] = 0
        for index in range(1, len(array)):
            for idx in range(index):
                if array[idx] >= index - idx:
                    minJumps[index] = min(minJumps[index], minJumps[idx] +1)
        return minJumps[-1]
