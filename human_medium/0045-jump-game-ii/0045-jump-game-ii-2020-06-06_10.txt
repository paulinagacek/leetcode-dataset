def jump(self, nums: List[int]) -> int:
        dp = []
        curr = 0
        prev = -1
        while curr < len(nums) - 1:
            max_reach = self.get_max_reach(nums[prev+1:curr+1])
            dp.append(max_reach)
            prev = curr
            curr += dp[-1]
            if curr == prev: return 0
        return len(dp)
    
    def get_max_reach(self, arr):
        out = 0
        for i,v in enumerate(arr):
            out = max(out, (i+1+v) - len(arr))
        return out
