class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        slow = fast = head

        for _ in range(n):
            fast = fast.next

        if not fast:
            return slow.next
        
        while fast:
            
            fast = fast.next
        
            if not fast:
                slow.next = slow.next.next
                return head
            
            slow = slow.next

        return None
