class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        # add a dummy node at the beginning of the linked list
        dummy = ListNode(0, head) # val doesn\'t matter
        l = dummy
        r = head
        # use loop to move the r pointer and let the distance between l and r pointers to be n+1
        while n > 0 and r:
            r = r.next
            n -= 1
        # move l and r pointers simutaneously until r reach the end (null)
        while r:
            l = l.next
            r = r.next
        # let the pointer points to (n+1)-th node to "remove" n-th node    
        l.next = l.next.next
        return dummy.next
