def removeNthFromEnd(self, head: ListNode, n: int) -> ListNode:
    
        temp = temp2 = head
        
        for i in range(n):
            temp = temp.next
        
        if not temp:  # cases like [1] 2
            return head.next
        
        while temp.next:
            temp = temp.next
            temp2 = temp2.next 
        
        temp2.next = temp2.next.next
      
        return head
