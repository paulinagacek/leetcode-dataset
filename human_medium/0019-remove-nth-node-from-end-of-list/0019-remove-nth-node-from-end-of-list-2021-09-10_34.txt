class Solution:
    def removeNthFromEnd(self, head: ListNode, n: int) -> ListNode:
        prev_slow = None
        next_slow = head.next
        
        slow = head
        
        fast = head
        
        while n > 0:
            fast = fast.next
            n -= 1
        
        while fast:
            fast = fast.next
            
            prev_slow = slow
            slow = next_slow
            
            next_slow = next_slow.next
        
        if slow == head:
            head = next_slow
        else:
            prev_slow.next = next_slow
        
        return head
