class Solution:
    def removeNthFromEnd(self, head: Optional[ListNode], n: int) -> Optional[ListNode]:
        # T: O(n), S: O(1) (one pass)
        dummy = ListNode(-1, head)
        p = curr = dummy
        while n > 0:
            p = p.next
            n -= 1
        while p.next:
            p = p.next
            curr = curr.next
        curr.next = curr.next.next
        return dummy.next
