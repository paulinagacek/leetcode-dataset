class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        maxsum = -float(\'inf\')
        
        cursum = 0
        n = len(nums)
        for i in range(n):
            cursum = nums[i]
            maxsum = max(maxsum,cursum)
            for j in range(i+1,n):
                cursum+=nums[j]
                maxsum = max(maxsum,cursum)
        return maxsum
