class Solution(object):
    def maxSubArray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        maxcurr = nums[0]
        maxglobal = nums[0]
        for i in range(1,len(nums)):
            maxcurr = max(nums[i], maxcurr + nums[i]) 
            maxglobal = max(maxcurr, maxglobal)
        return maxglobal
