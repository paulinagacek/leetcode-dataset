class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int maxSum = INT_MIN, n = nums.size();
        for(int i=0; i<n; i++){
            int curSum = 0;
            for(int j=i; j<n; j++){
                curSum+= nums[j];
                maxSum = max(curSum, maxSum);
            }
        }   
        return maxSum;
    }
};
