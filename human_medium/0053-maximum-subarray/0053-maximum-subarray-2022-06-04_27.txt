class Solution:
    def maxSubArray(self, nums: List[int]) -> int:

        # Initially the current sum and the maximum sum of any given array will be it\'s first element
        currentSum = nums[0]
        maxSum = nums[0]

        for i in range(1, len(nums)):   # Since we initialized with the first element of the given array we\'ll iterate the array from index-1 upto the end of the array

            if currentSum >=0:
                currentSum += nums[i]   #If the current Sum is positive then add the current element to it
            else:
                currentSum = nums[i]    # Otherwise the negative sum is of no use to us so we\'ll start the current sum all over again with the current indexed element

            maxSum = max(currentSum, maxSum)    # We\'ll check which sum is greater whether it\'s the currentSum or the maxSum then update the maxSum basically it\'s doing -> if currentSum > maxSum: if yes then maxSum = currentSum
            
        return maxSum
