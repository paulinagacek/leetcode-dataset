class Solution:
    def rob(self, nums: List[int]) -> int:
        def simple_rob(nums):
            dp = [0 for i in range(len(nums) + 2)]
            for i in range(len(nums)):
                dp[i] = max(dp[i-1], dp[i-2] + nums[i])
            return dp[len(nums) - 1]
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        return max(simple_rob(nums[1:]), simple_rob(nums[:-1]))
