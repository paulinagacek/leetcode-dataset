class Solution:
    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        def solve(nums):
            rob1, rob2 = 0, 0
            for n in nums:
                temp = max(n+rob1, rob2)
                rob1 = rob2
                rob2 = temp
            return rob2
        s1 = solve(nums[:len(nums)-1])
        s2 = solve(nums[1:])
        return max(s1, s2)
