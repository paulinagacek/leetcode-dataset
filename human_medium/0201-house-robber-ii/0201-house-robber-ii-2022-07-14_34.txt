class Solution:
    """
    approach: 
    this is similar to house robber problem
    since houses are circular, hence house 1 and house n are adjacent, and hence both cannot be
    robbed
    so we can get maximum value by robbing house 1 to house (n-1)
    and also, get the maximum value by robbing house 2 to house n
    we return the maximum obtained in both the cases
    """
    def rob(self, nums: List[int]) -> int:
        if len(nums) == 1:
            return nums[0]
        
        elif len(nums) == 2:
            return max(nums[0], nums[1])
        
        # dp1 to get max value till house1 to house (n-1)
        # dp2 to get max value till house2 to house (n)
        dp1 = [nums[0], max(nums[0], nums[1])]
        dp2 = [nums[1], max(nums[1], nums[2])]
        
        # update dp1 array
        for i in range(2, len(nums)-1):
            dp1.append(max(dp1[-1], dp1[-2]+nums[i]))
        
        # update dp2 array
        for i in range(3, len(nums)):
            dp2.append(max(dp2[-1], dp2[-2]+nums[i]))
        
        return max(dp1[-1], dp2[-1])
