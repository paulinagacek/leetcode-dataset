class Solution:
    def reverse(self, x: int) -> int:
        x = \',\'.join(str(x)).split(\',\')
        sign = x.pop(0) if not x[0].isalnum() else None
        x = \'\'.join(x[::-1])
        res = sign+x if sign else x
        res = int(res)
        return res if -1*pow(2, 31) < res < pow(2, 31)-1 else 0
