class Solution:
    def reverse(self, x: int) -> int:
        pn = 1 
        if x < 0 :
            pn = -1
            x *= -1
        x = int(str(x)[::-1]) * pn
 #^Convert integer into string and reverse using slicing and convert it back to integer
        return 0 if x < 2**31 * -1 or x > 2**31 -1 else x
