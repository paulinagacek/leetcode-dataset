class Solution:
    def reverse(self, x: int) -> int:
        nums=0
        max_int = 2**31-1
        min_int = -2**31
        y = abs(x)
        while(y>0):
            nums = nums*10 + y%10
            y //= 10
        if x<0:
            nums *= -1
        if nums > max_int or nums < min_int:
            return 0
        return nums
