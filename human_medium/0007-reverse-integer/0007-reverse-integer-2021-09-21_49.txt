class Solution(object):
    def reverse(self, x):
        rev=0
        neg = False
        if x < 0:
            x *= -1
            neg = True
        while x!=0:
            rev=(rev*10)+(x%10)
            x=x/10
            if rev>=pow(2,31)-1:
                return 0
        if neg:
            return rev * -1
        else:
            return rev
