class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort()
        
        result = [intervals[0]]
        for start, end in intervals:
            if start > result[-1][1]:
                result.append([start, end])
            elif end > result[-1][1]:
                result[-1][1] = end
               
        return result
