class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        intervals.sort(key=lambda x: x[0])
        lenintrvl = len(intervals)
        res = []
        i = 1
        while i < lenintrvl:
            lc , lr = intervals[i-1]
            rc , rr = intervals[i]
            if rc<=lr:
               intervals[i-1] = [min(lc,lr, rc,rr), max(lc,lr, rc,rr)]
               intervals.pop(i)
               lenintrvl -= 1
            else:
               i += 1
        return intervals
