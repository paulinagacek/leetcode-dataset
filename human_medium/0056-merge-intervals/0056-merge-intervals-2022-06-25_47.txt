class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
		# python will sort lists by first element by default
        intervals.sort()
        i = 0
        while i+1 < len(intervals):
			# if two intervals can be merged
            if intervals[i][1] >= intervals[i+1][0]:
			# merge two intervals into interval i
                intervals[i][1] = max(intervals[i+1][1],intervals[i][1])
				# remove interval i+1
                intervals.pop(i+1)
            else:
			# move forward
                i+=1
            
        return intervals
