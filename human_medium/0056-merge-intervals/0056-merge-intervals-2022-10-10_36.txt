class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        ans=[]
        pre=intervals[0]
		
        for cur in range(1,len(intervals)):
            if intervals[cur][0]<=pre[1]:
                pre[1]=max(pre[1],intervals[cur][1])
            else:
                ans.append(pre)
                pre=intervals[cur]
        ans.append(pre)
        return ans
