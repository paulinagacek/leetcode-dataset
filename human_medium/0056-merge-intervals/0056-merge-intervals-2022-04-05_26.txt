class Solution:
    def merge(self, intervals: List[List[int]]) -> List[List[int]]:
        
        intervals.sort()
        res = [intervals[0]]
        
        for i in range(1, len(intervals)):
            interval = intervals[i]
            if interval[0] > res[-1][1]:
                res.append(interval)
                continue
            res[-1][1] = max(res[-1][1], interval[1])
        
        return res
