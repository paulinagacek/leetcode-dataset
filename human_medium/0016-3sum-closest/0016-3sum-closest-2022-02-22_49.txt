class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        
        res = sum(nums[0:3]) # set the result to the sum of the first three numbers
        nums.sort() # Sort the array, we need this to break down the problem to a two sum problem

        # Here we will iterate through the array, if it confuses you why we don\'t need to go to the end
        # you can loop the entire array, but you need 3 numbers so if try to find 3 numbers when they dont exist
        # while searching nums[len(nums) - 2:] you are just doing extra work
        #
        # [-4,-1,1,2]
        for i in range(len(nums) - 2):
            # Keep track of our left and right pointers for the right side of the array where
            # we are searching for two numbers to add to the current index to sum to target (or get close)
            #
            # This is what the iterations will look like:
            #
            # [-4] => [-1,1,2]
            #           L   R
            # [-1] => [1,2]
            #          L R
            #
            left = i + 1
            right = len(nums) - 1
            while left < right:
                
                # Get the sum of the index plus two that you found using the pointers
                three_sum = nums[i] + nums[left] + nums[right]
                
                # Since the array is sorted, we know that if we are at a number > target
                # if we move the right pointer to the left, we should get a smaller number
                # bringing us closer to our goal, target
                #
                # You can reverse this logic for the left, if we are at a number less than target
                # we need to move the left pointer to the right to find a larger number to bring us to our goal
                #
                # If we hit the target, we can just return the target since we know thats the best answer
                if three_sum > target:
                    right -= 1
                elif three_sum < target:
                    left += 1
                else:
                    return target
                
                # To find the distance between your 3sum and the target you can use the absolute value of the
                # difference abs(target - three_sum), you just need to update the value of your result if the
                # distance is shorter than the current answer you have
                if abs(target - three_sum) < abs(res - target):
                    res = three_sum
                    
        return res
