class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        closeness = inf
        res = None
        n = len(nums)
        
        for i in range(n - 2):
            j = i + 1
            k = n - 1
            
            while j < k:
                currSum = nums[i] + nums[j] + nums[k]
                if abs(currSum - target) < closeness:
                    closeness = abs(currSum - target)
                    res = currSum
                
                if currSum == target: return target
                if currSum < target: j += 1
                else: k -= 1
        
        return res
