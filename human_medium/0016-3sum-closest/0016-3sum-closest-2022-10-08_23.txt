class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        ans = nums[0]+nums[1]+nums[n-1]
        
        for i in range(n-2):
            left, right = i+1, n-1
            while left<right:
                Sum = nums[left]+nums[right]+nums[i]
                if Sum==target:
                    return Sum
                if abs(target-Sum)<abs(target-ans):
                    ans = Sum
                if Sum<target:
                    left+=1
                else:
                    right-=1
        return ans
