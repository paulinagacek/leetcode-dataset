class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)

        closest = float(\'inf\')
        stop = False
        
        for i in range(n-2):
            
            if i > 0 and nums[i] == nums[i-1]:
                continue
            
            start, end = i+1, n-1
            
            minSum = nums[i] + nums[start] + nums[start + 1]
            maxSum = nums[i] + nums[end] + nums[end - 1]
            
            if target <= minSum:
                closest = min(closest, minSum, key=lambda x: abs(target-x))
                return closest
            elif target >= maxSum:
                closest = min(closest, minSum, key=lambda x: abs(target-x))
            else:
                while start < end:
                    curSum = nums[i] + nums[start] + nums[end]
                    prevDiff = abs(target - closest)

                    if (abs(target - curSum) <= prevDiff):
                        closest = curSum

                    if curSum > target:
                        end -= 1
                    elif curSum < target:
                        start += 1
                    else:
                        return curSum

        return closest
