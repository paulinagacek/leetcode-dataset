class Solution:
    def dfs(self,obstacleGrid,i,j,n,m):
        if i>=n or j>=m or obstacleGrid[i][j]==1:
            return 0
        if i==n-1 and j==m-1:
            return 1
        res = 0
        obstacleGrid[i][j]=1
        res = self.dfs(obstacleGrid,i+1,j,n,m) + self.dfs(obstacleGrid,i,j+1,n,m)
        obstacleGrid[i][j]=0
        return res
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        n,m = len(obstacleGrid),len(obstacleGrid[0])
        return self.dfs(obstacleGrid,0,0,n,m)
