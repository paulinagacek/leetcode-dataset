class Solution:
    # top down dp
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        m, n = len(obstacleGrid), len(obstacleGrid[0])
        
        # memo for dp
        memo = [[None for _ in range(n)] for __ in range(m)]
        
        # edge case
        if obstacleGrid[0][0] == 1:
            return 0
        
        # we want to fill first row and column with 0 if we find an obstacle from that point to the end of the row or column. This way we can return 0 to indicate that a path is not possible because an obstacle blocks it
        row_fill = 1
        col_fill = 1
        for r in range(m):
            if obstacleGrid[r][0] == 1:
                row_fill = 0
            obstacleGrid[r][0] = row_fill
            
        for c in range(1, n):
            if obstacleGrid[0][c] == 1:
                col_fill = 0
            obstacleGrid[0][c] = col_fill

        # recursive function for top down dp
        def dp(r, c):
            # if we first row or col we return 1 if there is no blocking path
            if r == 0 or c == 0:
                return obstacleGrid[r][c]
            # if we reach an obstacle there is no possible path
            elif obstacleGrid[r][c] == 1:
                return 0
            # update memo if not cached
            if memo[r][c] is None:
                memo[r][c] = dp(r, c-1) + dp(r-1, c)
            return memo[r][c]
    
        return dp(m - 1, n - 1)
