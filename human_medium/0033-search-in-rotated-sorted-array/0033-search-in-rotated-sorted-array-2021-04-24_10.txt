class Solution:
    def search(self, nums: List[int], target: int) -> int:
        if not nums:
            return -1
        min_value_index = self.find_min_value_index(nums)
        result = self.binary_search(nums, target, min_value_index, len(nums)-1)
        if result == -1 and min_value_index != 0:
            result = self.binary_search(nums, target, 0, min_value_index-1)
        return result
               
    def binary_search(self, nums, target, start, end):
        while start <= end:
            mid = start + (end - start) // 2
            if target == nums[mid]:
                return mid
            if target < nums[mid]:
                end = mid - 1
            else:
                start = mid + 1
        return -1
        
    def find_min_value_index(self, nums):
        start = 0
        end = len(nums) - 1
        while start < end:
            mid = start + (end - start) // 2
            if nums[mid] > nums[end]:
                # go right
                start = mid + 1
            else:
                # go left and do not eliminate mid
                end = mid
        return start
