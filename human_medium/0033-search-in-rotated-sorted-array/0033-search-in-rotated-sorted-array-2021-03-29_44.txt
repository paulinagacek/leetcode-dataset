class Solution:
    def search(self, nums: List[int], target: int) -> int:
		N=len(nums)
        start=0
        end=N-1
        while start<=end:
            mid=(start+end)//2
            if target==nums[mid]:
                return mid
            # first half order
            if nums[mid]>=nums[start]:
                if nums[mid]>target>=nums[start]:
                    end=mid-1
                else:
                    start=mid+1
            # second half order
            else:
                if nums[mid]<target<=nums[end]:
                    start=mid+1
                else:
                    end=mid-1
        return -1
