class Solution:
    def search(self, nums: List[int], target: int) -> int:
        lower = 0
        higher = len(nums) - 1

        while lower <= higher:
            mid = (higher + lower) // 2

            if nums[mid] == target:
                return mid

            if nums[mid] >= nums[lower]:
                if target > nums[mid] or target < nums[lower]:
                    lower = mid + 1
                else:
                    higher = mid - 1
            else:
                if target < nums[mid] or target > nums[higher]:
                    higher = mid - 1
                else:
                    lower = mid + 1

        return -1
