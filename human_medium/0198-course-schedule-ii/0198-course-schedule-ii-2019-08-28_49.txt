class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> bool:
        todo = {i:set() for i in range(numCourses)}
        graph = defaultdict(set)
        for i,j in prerequisites:
            todo[i].add(j)
            graph[j].add(i)
        q = deque([])
        res = []
        while q:
            n = q.popleft()
            for i in graph[n]:
                todo[i].remove(n)
                if todo[i] == 0:
                    q.append(i)
            todo.pop(n)
            res.append(n)
                    
        if not todo:
            return res
        else:
            return []# if todo has items left, it means it can\'t be finished.
