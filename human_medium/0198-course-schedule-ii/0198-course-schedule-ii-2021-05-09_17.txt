class Solution:
    def findOrder(self, numCourses: int, prerequisites: List[List[int]]) -> List[int]:
        g = {i: set() for i in range(numCourses)}
        for a, b in prerequisites: 
            g[a].add(b)
        try:
            return list(TopologicalSorter(g).static_order())
        except:
            return []
