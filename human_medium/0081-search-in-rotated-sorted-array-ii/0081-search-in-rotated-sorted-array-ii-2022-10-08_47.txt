class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        l, r = 0, len(nums) - 1
        while l < r:
            while nums[l] == nums[r] and l < r:
                l += 1
            mid = (l + r) // 2
            if nums[l] <= nums[mid] < target or  nums[mid] < target < nums[l] or target < nums[l] <= nums[mid]:
                l = mid + 1
            else:
                r = mid
        return True if l in range (0, len(nums)) and nums[l] == target else False
