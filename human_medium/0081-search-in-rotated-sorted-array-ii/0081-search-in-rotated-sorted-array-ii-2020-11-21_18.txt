class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        left,right = 0,len(nums)-1
        while left <= right:
            mid = left + (right-left)//2
            if nums[mid] == target or nums[left] == target or nums[right] == target:
                return True
            if nums[left] < nums[right]:
                if nums[mid] < target:
                    left = mid + 1
                else:
                    right = mid - 1
            elif nums[left] >= nums[right] :
                right -= 1
        return False
