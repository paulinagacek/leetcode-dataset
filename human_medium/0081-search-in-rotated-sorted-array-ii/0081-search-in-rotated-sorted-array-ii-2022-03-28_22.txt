class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        l, r = 0, len(nums) - 1
        while l <= r:
            m = l + (r - l) // 2
            # Case 1: nums[m] is our target. Return True
            if nums[m] == target:
                return True
            # Case 2: nums[m] < nums[r] which means the values
            # with m <= index <= r are sorted. If the target
            # is between nums[m] and nums[r], we have to search
            # in this half otherwise in the other half
            elif nums[m] < nums[r]:
                if nums[m] < target <= nums[r]:
                    l = m + 1
                else:
                    r = m - 1
            # Case 3: nums[m] > nums[r] which means the other
            # half is sorted. If target is bigger than nums[r]
            # but lower than nums[m] we have to search in the
            # left half (the ordered one) otherwise on the right half
            elif nums[m] > nums[r]:
                if nums[r] < target < nums[m]:
                    r = m - 1
                else:
                    l = m + 1
            # Case 4: nums[m] == nums[r]. We reduce r by one.
            # We can not do r = m - 1 because nums[m] == nums[r]
            # doesn\'t imply that all the values in-between are
            # equal to nums[m] and nums[r]. Eg the 2 in
            # [1,1,1,1,1,1,1,2,1,1,1]
            else:
                r -= 1
        return False
