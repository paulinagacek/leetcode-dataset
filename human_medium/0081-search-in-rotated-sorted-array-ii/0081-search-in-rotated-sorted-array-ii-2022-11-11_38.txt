class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        start, end = 0, len(nums) - 1
        while start <= end:
            mid = (start + end) // 2
            if nums[mid] == target or nums[start] == target:
                return True
            if nums[start] < target < nums[mid]:
                end = mid - 1
            elif nums[mid] < target < nums[end]:
                start = mid + 1
            else:
                start += 1
        return False
