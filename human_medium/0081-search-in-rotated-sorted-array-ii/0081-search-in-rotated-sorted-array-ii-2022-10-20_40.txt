def search(self, nums: List[int], target: int) -> bool:
        first = 0
        last = len(nums) - 1
        while first <= last:
            mid = (first+ last) // 2
            
            if nums[mid] == target:
                return True
            
            # tricky part
            # if out first element is equal to the last one
            if nums[first] == nums[last]:
                last -= 1
            
            # check if left side of the array is sorted
            elif nums[first] <= nums[mid]:
                if nums[first] <= target and nums[mid] >= target:
                    last = mid -1
                else:
                    first = mid + 1
            else:
                if nums[mid] <= target and nums[last] >= target:
                    first = mid + 1
                else:
                    last = mid - 1
        return False
