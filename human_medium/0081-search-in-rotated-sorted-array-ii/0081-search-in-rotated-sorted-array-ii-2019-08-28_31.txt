l, r = 0, len(nums)-1
        
        while l<=r:
            mid = l + (r-l)/2
            if nums[mid]==target:
                return True
            elif nums[mid]>target:
                if nums[l]<=target<=nums[mid]:
                    r -=1
                else:
                    l += 1
            else:
                if nums[mid]<=target<=nums[r]:
                    l +=1
                else:
                    r -=1
                    
        return False
