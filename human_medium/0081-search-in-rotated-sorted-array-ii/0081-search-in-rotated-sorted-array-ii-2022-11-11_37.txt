class Solution:
    def search(self, nums: List[int], target: int) -> int:
        start = 0
        end = len(nums) - 1
        
        # Skips any duplicate numbers from the start that equal nums[end]
        if(nums[0] == nums[-1]):
            if(nums[0] == target):
                return True
            while(start < len(nums) and nums[start] == nums[0]):
                start += 1
        
        
        # Apply the solution to Search in Rotated sorted array
        while(start < end):
            mid = (start + end)//2
            if(nums[mid] == target):
                return True
            if(nums[start] <= nums[mid]):
                if(nums[start] <= target and target < nums[mid]):
                    end = mid - 1
                else:
                    start = mid + 1
            else:
                if(nums[mid] < target and target <= nums[end]):
                    start = mid + 1
                else:
                    end = mid - 1
        
        return -1
