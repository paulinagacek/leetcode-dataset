class Solution:
    def findMin(self, a: List[int]) -> int:
        h=len(a)-1
        if h==0 or a[0]<a[-1]:
            return a[0]
        elif a[-1]<a[-2]:
            return a[h]
        l=0
        while l<h:
            m=(l+h)//2
            if a[m]<a[m-1]:
                return a[m]
            elif a[m]>a[0]:
                l=m+1
            else:
                h=m
        return a[l]
