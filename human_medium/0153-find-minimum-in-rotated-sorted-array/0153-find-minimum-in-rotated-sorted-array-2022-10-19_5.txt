class Solution {
    public int FindMin(int[] nums) {
        int li = 0;             //low index
        int hi = nums.Length-1; //high index
      
        while(hi>li)
        {
            int midi = li+(hi-li)/2;
            
            if (nums[midi] < nums[hi]) 
            {
                hi = midi;
            } 
            else 
            {
                li = midi+1;
            }
        }
        
        return nums[li];
    }
}
