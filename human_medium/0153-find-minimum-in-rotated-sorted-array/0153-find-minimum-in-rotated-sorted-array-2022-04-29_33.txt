class Solution(object):
    def findMin(self, nums):
        l = 0
        r = len(nums)-1
        #special case --> array is not rotated 
        if(nums[0]<=nums[len(nums)-1]):
            return nums[0]
        while(l<=r):
            mid  = (l+r)//2
            if(nums[mid]<nums[mid-1]):
                return nums[mid]
            elif(nums[mid]<nums[0]):
                r = mid-1
            else:
                l = mid+1
