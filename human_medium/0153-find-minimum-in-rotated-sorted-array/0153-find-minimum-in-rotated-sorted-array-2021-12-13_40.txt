class Solution:
    def findMin(self, nums: List[int]) -> int:
            
        start=0
        end=len(nums)-1
        small=nums[0]
        
        if end==1:
            return small if small <nums[1] else nums[1]
        
        while start<=end:
            mid=start+(end-start)//2
            
            if nums[mid]>small:
                start=mid+1
                
            else:
                small=nums[mid]
                end=mid-1
                
        return small
