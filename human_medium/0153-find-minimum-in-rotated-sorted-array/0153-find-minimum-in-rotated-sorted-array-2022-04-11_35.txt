class Solution:
    def findMin(self, nums: List[int]) -> int:
        n = len(nums)
        # print(n, nums)
        if n==1:
            return nums[0]
        if n==2:
            return min(nums[0], nums[1])
        mid = n//2
        m1 = self.findMin(nums[:mid])
        m2 = self.findMin(nums[mid:])
        return min(m1, m2)
