class Solution:
    # merge
    def findMin(self, nums: List[int]) -> int:
        if len(nums) <= 1:
            return nums[0]
        mid = len(nums) // 2
        left = self.findMin(nums[:mid])
        right = self.findMin(nums[mid:])
        return min(left, right)

    # binary search
    def findMin2(self, nums: List[int]) -> int:
        a, b = 0, len(nums)-1
        while a < b:
            mid = (a+b)//2
            if nums[mid] > nums[b]:
                a = mid+1 # search second half
            else:
                b = mid # search first half
        return nums[a]
