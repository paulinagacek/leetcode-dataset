class Solution:
    def findMin(self, nums: List[int]) -> int:
        start, end = 0, len(nums) - 1
        while(start <= end):
            mid = (start + end)//2
            if(nums[mid] < nums[start]):
                end = mid
            elif(nums[end] < nums[mid]):
                start = mid + 1
            else:
                return nums[start]
