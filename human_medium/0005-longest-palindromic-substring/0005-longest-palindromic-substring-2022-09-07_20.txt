class Solution:
    def is_palindrome(self, string):
        return string == string[::-1]
    
    def longestPalindrome(self, s: str) -> str:
        if self.is_palindrome(s):
            return s
        
        current_max_length = 1
        starting_index = 0

        for x in range(1, len(s)):
            one_more_letter_word = s[x - current_max_length:x + 1]
            two_more_letters_word = s[x - current_max_length - 1:x + 1]
            
            if self.is_palindrome(two_more_letters_word) and x - current_max_length - 1 >= 0:
                starting_index = x - current_max_length - 1
                current_max_length += 2
            elif self.is_palindrome(one_more_letter_word):
                starting_index = x - current_max_length
                current_max_length += 1       
                
        return s[starting_index:starting_index + current_max_length]
