class Solution:
    def copyRandomList(self, head: \'Node\') -> \'Node\':
        if not head :
            return
        curr=head
        
        while curr!=None:
            newNode=Node(curr.val)
            newNode.next=curr.next
            curr.next=newNode
            curr=curr.next.next
        
        
        curr=head
        
        while curr!=None:
            if curr.random:
                curr.next.random =curr.random.next
            curr=curr.next.next
        
        
        curr=head
        dummy=curr.next
        
        while curr.next!=None:
            temp=curr.next
            curr.next=curr.next.next
            curr=temp
        
        return dummy
