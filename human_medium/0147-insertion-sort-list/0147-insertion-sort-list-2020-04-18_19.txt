class Solution:
    def insertionSortList(self, head: ListNode) -> ListNode:
        if not head: return
        
        ret = ListNode(0)
        ret.next = head
        
        pre, curr = head, head.next
                
        while(curr):
            ## if curr.val < pre_node.val: iterate pointer from beginning until val is big than curr.val            
            if curr.val < pre.val:
                ptr = ret
                while(ptr and ptr.next and ptr.next.val < curr.val):
                    ptr = ptr.next
                    
                ## insert curr at proper position
                ptr.next, curr.next, pre.next= curr, ptr.next, curr.next
        
                ## move to next node
                curr = pre.next
            
            else:
                pre, curr = curr, curr.next
            
        return ret.next
