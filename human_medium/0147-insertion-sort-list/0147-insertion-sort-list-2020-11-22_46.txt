class Solution:
    def isort(self,prev,curr,head):
        if curr==None:
            return head
        if prev==head:
            if prev.val<=curr.val:
                prev=prev.next
                curr=prev.next
            else:
                prev.next=curr.next
                curr.next=prev
                head=curr
                curr=prev.next
            return self.isort(prev,curr,head)
        else:
            if prev.val<=curr.val:
                prev=prev.next
                curr=prev.next
            else:
                temp=head
                if temp.val>=curr.val:
                    prev.next=curr.next
                    curr.next=temp
                    head=curr
                else:    
                    while temp!=prev:
                        if temp.next.val>=curr.val:
                            prev.next=curr.next
                            nex=temp.next
                            temp.next=curr
                            curr.next=nex
                            break
                        temp=temp.next
                curr=prev.next
            return self.isort(prev,curr,head)
    
    
    def insertionSortList(self, head: ListNode) -> ListNode:
        if head==None or head.next==None:
            return head
        return self.isort(head,head.next,head)
