class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        res = [head.val]
        pivot = head.val
        temp = head.next
        
        while temp != None:
            if temp.val < pivot:
                for y in range(len(res)):
                    if temp.val < res[y]:
                        val = res[y:]
                        res[y] = temp.val
                        res[y+1:] = val
                        break
            else:
                res.append(temp.val)
                pivot = temp.val
            temp = temp.next
        
        head = ListNode(val=res[0])
        temp = head
        for i in range(1, len(res)):
            temp.next = ListNode(res[i])
            temp = temp.next
        return head
