class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head:
            return None
        arr = []
        while head:
            arr.append(head.val)
            head = head.next
        
        for i in range(1, len(arr)):
            key = arr[i]
            j = i-1
            while j >= 0 and key < arr[j] :
                    arr[j + 1] = arr[j]
                    j -= 1
            arr[j + 1] = key

        arr = arr[::-1]
        res =None
        for i in arr:
            res = ListNode(val= i,next=res)    
        return res
