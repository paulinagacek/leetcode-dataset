class Solution:
    def insertionSortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        arr =[]

        while head :
            arr.append(head.val)
            head = head.next 
        if not arr:
            return None
        arr.sort()
        for i in range(len(arr)):
            arr[i] = ListNode(arr[i])

        for i in range(len(arr)-1):
            arr[i].next = arr[i+1]
        arr[len(arr)-1].next = None
        return arr[0]
