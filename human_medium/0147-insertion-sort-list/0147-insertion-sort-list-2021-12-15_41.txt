class Solution(object):
    def insertionSortList(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        a = []
        while head:  # convert into list
            a.append(head.val)
            head = head.next
        b = sorted(a)
        
        dummy = ListNode()
        tail = dummy
        for k in range(len(b)):  # convert back
            tail.next = ListNode(b[k])
            tail = tail.next
        return dummy.next
