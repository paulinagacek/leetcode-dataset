class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        if root is None:
            return []
        queue=[root]
        res=[]
        while queue:
            n=len(queue)
            count=0
            for i in range(n):
                node=queue.pop(0)
                count+=1
                if count==n:
                    res.append(node.val)
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
        
        return res
