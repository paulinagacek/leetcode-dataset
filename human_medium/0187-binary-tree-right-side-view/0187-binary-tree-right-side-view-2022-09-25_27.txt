class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        
        return self.get_rv(root)
    
    def get_rv(self,root):
        
        if not root:
            return []
        
        q = [root]
        output = []
        
        while q:
            
            length = len(q)
            output.append(q[-1].val)
            
            for _ in range(length):
                
                node = q.pop(0)
                
                if node.left:
                    q.append(node.left)
                    
                if node.right:
                    q.append(node.right)
                    
        return output
