class Solution(object):
    def rightSideView(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        levels = {} #dictionary to store rightmost number in each level
        level = 0
        q = deque()
        q.append(root)
        if root is None:
            return []
        while q:
            sz = len(q)
            for i in range(0, sz):
                cur = q.popleft()
                if level not in levels:
                    levels[level] = cur.val
                #Note we put the right child node to the queue first
                if cur.right:
                    q.append(cur.right)
                if cur.left:
                    q.append(cur.left)
            level += 1
        return levels.values()
