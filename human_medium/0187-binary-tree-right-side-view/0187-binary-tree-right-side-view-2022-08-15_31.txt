def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        """
        I will try to solve this problem using dfs now
        """
        ans = []
        def dfs(root, level):
            if root is None:
                return
            if level > len(ans):
                # need to consider the node at that level
                ans.append(root.val)
            dfs(root.right, level+1)
            dfs(root.left, level+1)
        dfs(root, 1)
