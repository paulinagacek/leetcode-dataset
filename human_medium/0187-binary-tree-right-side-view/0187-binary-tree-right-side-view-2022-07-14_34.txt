class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        result=[]
        
        def func(root,level):
            if root is None:
                return 
            if len(result)==level:
                result.append(root.val)
            func(root.right,level+1)
            func(root.left,level+1)

        
        func(root,0)
        
        return result
