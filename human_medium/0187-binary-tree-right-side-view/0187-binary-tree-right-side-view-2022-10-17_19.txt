class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        if not root: return []
        q,ans = [root],[]
        while q:
            n = len(q)
            for i in range(n):
                curr = q.pop(0)
                if i==n-1:
                    ans.append(curr.val)
                if curr.left:
                    q.append(curr.left)
                if curr.right:
                    q.append(curr.right)
        return ans
