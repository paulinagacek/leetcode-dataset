class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        if not root:
            return []
        l=[root]
        level=[]
        answer=[root.val]
        while l:
            for each in l:
                if each.left:
                    level.append(each.left)
                if each.right:
                    level.append(each.right)
                print(level)
            if level:
                answer.append(level[-1].val)
            l=level
            level=[]
        return answer
