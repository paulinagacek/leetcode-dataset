class Solution:
    def rightSideView(self, root: TreeNode) -> List[int]:
        if not root:
            return []
        
        res = []
        deque = collections.deque([root])
        
        while deque:
            size = len(deque)
            
            for i in range(size):
                node = deque.popleft()
                if i == size - 1:
                    res.append(node.val)
                
                if node.left:
                    deque.append(node.left)
                
                if node.right:
                    deque.append(node.right)
            
        return res
