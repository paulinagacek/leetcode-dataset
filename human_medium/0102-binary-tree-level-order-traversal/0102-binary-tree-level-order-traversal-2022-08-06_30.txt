class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        res = []                                # list containing level-lists

        # BFS - Breadth First Search
        def traverse(node, i):
            if node:
                if i < len(res):
                    res[i].append(node.val)     # append the value in current level-list
                else:
                    res.append([node.val])      # add list for new level
                traverse(node.left, i+1)        # repeat for left child with level+1
                traverse(node.right, i+1)       # repeat for right child with level+1
        
        traverse(root, 0)                       # call BFS with root and level 0
        return res
