class Solution:
    def levelOrder(self, root: TreeNode) -> List[List[int]]:
        if not root:
            return []
        else:
            queue=[root]
            nextqueue=[]
            level=[]
            result=[]
            while queue != []:
                for root in queue:
                    level.append(root.val)
                    if root.left is not None:
                        nextqueue.append(root.left)
                    if root.right is not None:
                        nextqueue.append(root.right)
                        
                result.append(level)
                level=[]
                queue=nextqueue
                nextqueue=[]
        return result
