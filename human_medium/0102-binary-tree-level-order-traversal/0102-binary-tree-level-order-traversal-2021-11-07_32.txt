class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        from collections import deque
        q=collections.deque()
        q.append(root)
        result=[]
        
        while q:
            l=len(q)
            level=[]
            for i in range(l):
                node=q.popleft()
                if node:
                    level.append(node.val)
                    q.append(node.left)
                    q.append(node.right)
            if level:
                    result.append(level)
        return result
