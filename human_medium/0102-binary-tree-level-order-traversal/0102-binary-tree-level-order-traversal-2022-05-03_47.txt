class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        d = {}
        def rec(root,level):
            if not root:
                return
            d[level] = d.get(level,[]) + [root.val]
            rec(root.left,level+1)
            rec(root.right,level+1)
        rec(root,0)
        return list(d.values())
