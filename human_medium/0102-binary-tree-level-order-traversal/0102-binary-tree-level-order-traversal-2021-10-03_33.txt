class Solution {
    public List<List<Integer>> levelOrder(TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        level(result,root,0);
        return result;
    }
    private void level(List<List<Integer>> result, TreeNode root, int height){
        if(root == null)    return;
        if(height >= result.size())   result.add(new LinkedList<Integer>());
        result.get(height).add(root.val);
        level(result , root.left , height+1);
        level(result , root.right , height+1);
    }
}
