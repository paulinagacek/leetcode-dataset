def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root:
            return root
        q,result = [],[]
        q.append(root)
        while len(q):
            l = []
            for _ in range(len(q)):
                node = q.pop(0)
                l.append(node.val)
                
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
                    
            result.append(l)
        return result
