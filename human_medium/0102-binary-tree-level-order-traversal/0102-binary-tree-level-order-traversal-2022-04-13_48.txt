class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root:
            return []
        queue = [root]
        res = []
        
        while len(queue) > 0:
            val = []
            for i in range(len(queue)):
                node = queue.pop(0)
                val.append(node.val)
                if node.left is not None:
                    queue.append(node.left)
                if node.right is not None:
                    queue.append(node.right)
                    
            res.append(val)
                
        return res
