class Node:
    def __init__(self, val):
        self.h = {}
        self.isTerminal=False
        self.val = val
        
class WordDictionary(object):

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.head = Node(None)

    def addWord(self, word):
        """
        :type word: str
        :rtype: None
        """
        root = self.head
        for ch in word:
            if not ch in root.h:
                nn = Node(ch)
                root.h[ch] = nn
            root = root.h[ch]
        root.isTerminal = True

    def search(self, word):
        """
        :type word: str
        :rtype: bool
        """
        def dfs(root, word):
            for i, ch in enumerate(word):
                if ch==".":
                    for _ch in root.h:
                        if dfs(root.h[_ch], word[i+1:]):
                            return True                        
                if not ch in root.h:
                    return False
                root = root.h[ch]
            return root.isTerminal
        
        root = self.head
        return dfs(root, word)
