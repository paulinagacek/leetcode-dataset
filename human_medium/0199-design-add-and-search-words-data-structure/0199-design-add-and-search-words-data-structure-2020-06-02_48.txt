class WordDictionary:

    def __init__(self):
        self.trie = {}

    def addWord(self, word: str) -> None:
        cur = self.trie
        for ch in list(word):
            if ch in cur:
                cur = cur[ch]
            else:
                cur[ch] = {}
                cur = cur[ch]
        cur[\'*\'] = True
        
    def search(self, word: str) -> bool:
        def _search(subWord, trie):
            if not subWord:
                return True if (\'*\' in trie.keys()) else False
            
            if not trie:
                return False

            if subWord[0] in trie:
                return _search(subWord[1:], trie[subWord[0]])
            elif subWord[0] == \'.\':
                for ch in trie.keys():
                    if ch == \'*\':
                        continue
                    if _search(subWord[1:], trie[ch]):
                        return True
            return False
        
        return _search(word, self.trie)
