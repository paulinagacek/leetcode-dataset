class Solution:
    def countAndSay(self, n: int) -> str:
        return self.countAndSayHelper(n)
    
    def countAndSayHelper(self, n:int) -> str:
        if n == 1:
            return "1"            
        
        prev = self.countAndSayHelper(n-1)
        return self.countAndSayPrev(prev)
    
    def countAndSayPrev(self, st: str) -> str:
        count = 1
        prev = st[0]
        output:List[int] = []
                        
        i = 1
        while i <len(st):
            if st[i] == prev:
                count += 1
            else:
                output.append(str(count))
                output.append(st[i-1])
                prev = st[i]
                count = 1
            i += 1
            
        if i==len(st):
            output.append(str(count))
            output.append(prev)
                
        return \'\'.join(output)
