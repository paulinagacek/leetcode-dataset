class Solution:
    def countAndSay(self, n: int) -> str:
        # base Case
        if n == 1:
            return "1"
        
        nums = self.countAndSay(n - 1)
        output = ""
        
        currCount = 0
        currNum = nums[0]

        for num in nums:
            if num == currNum:
                currCount += 1
            else:
                output += str(currCount) + currNum
                currNum = num
                currCount = 1
            
        return output + str(currCount) + currNum
