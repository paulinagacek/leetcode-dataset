class Solution:
    def countAndSay(self, n: int) -> str:
        ans = "1"
        for _ in range(1, n):
            ans = self.process_num(ans)
        
        return ans
        
    def process_num(self, s : str):
        ans = ""
        cnt = 1  #to keep count of current digit
		
		# Iterate from index 1 to the last index, 
		# check equality of current digit to the previous digit 
        for i in range(1, len(s)):
            if s[i] == s[i-1]:
                cnt += 1
            else:
                ans += str(cnt) + s[i-1]  # add count, digit to the result 
                cnt = 1 #reset counter
        
		# Add the last digit and its count to result
        ans += str(cnt) + s[-1]
        return ans
