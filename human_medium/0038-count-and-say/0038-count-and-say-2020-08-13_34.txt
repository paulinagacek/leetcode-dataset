class Solution:
    def countAndSay(self, n: int) -> str:
        assert n > 0 # n == 1 is the base case
		# base case
        if n == 1:
            return \'1\'
		# recursive case
        ret = self.countAndSay(n - 1)
        arr = []
        last_char = ret[0] # track the last character checked
        cnt = 1 # and track how many there are in a row
        for i in range(1, len(ret)):
            if ret[i] == last_char:
                cnt += 1
            else:
                arr.append((str(cnt), last_char))
                last_char = ret[i]
                cnt = 1
        if last_char == ret[-1]: # check to see if the last streak never broke, as that wouldn\'t get caught in the if in the loop
            arr.append((str(cnt), last_char))
        return \'\'.join(\'\'.join(i) for i in arr) # since each tuple is (length, char), you can just join it, and then join that to the return value
