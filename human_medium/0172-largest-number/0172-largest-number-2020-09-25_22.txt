def largestNumber(self, nums: List[int]) -> str:
        
        i = 0
        nums = [str(i) for i in nums]
        nums_len = len(nums)
        
        while i < nums_len:
            for j in range(i + 1, nums_len):
                if nums[i] + nums[j] < nums[j] + nums[i]:
                    nums[i], nums[j] = nums[j], nums[i]
            i += 1
            
        return "0" if nums[0] == "0" else "".join(nums)
