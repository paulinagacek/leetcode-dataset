class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        nums = list(map(str, nums))
        
        maxlength = max(map(len, nums))
        
        nums.sort(key=lambda x:x*(maxlength // len(x) + 1), reverse=True)
        
        return \'\'.join(nums) if int(nums[0]) != 0 else \'0\'
