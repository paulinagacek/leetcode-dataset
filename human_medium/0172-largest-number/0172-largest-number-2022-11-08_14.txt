class Solution:
    def largestNumber(self, nums: List[int]) -> str:
        def compare(x, y):
            if x + y > y + x:
                return -1
            return 1
        
        arr = [str(num) for num in nums]
        arr.sort(key = cmp_to_key(compare))
        
        ans = "".join(arr)
        ans = ans.lstrip("0")
        if ans == "":
            return "0"
        return ans
