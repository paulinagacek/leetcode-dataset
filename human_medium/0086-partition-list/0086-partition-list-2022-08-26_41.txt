class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        list1, list2 = ListNode(), ListNode()
        curr1, curr2 = list1, list2
        
        while head:
            if head.val < x:
                curr1.next = head
                curr1 = curr1.next
            else:
                curr2.next = head
                curr2 = curr2.next
            head = head.next
        
        curr1.next = list2.next    # concatenate two lists
        curr2.next = None   # make the tail of the second list point to null
        return list1.next
