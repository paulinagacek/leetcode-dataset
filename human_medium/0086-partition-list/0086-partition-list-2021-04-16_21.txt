def partition(self, head: ListNode, x: int) -> ListNode:
	dummy_1 = ListNode(0, head)
	dummy_2 = ListNode(0, head)

	great, small, h = dummy_1, dummy_2, head

	while h:
		if h.val<x:
			small.next = h
			small = small.next
		else:
			great.next = h
			great = great.next
		h = h.next

	great.next = None
	small.next = dummy_1.next
	return dummy_2.next
