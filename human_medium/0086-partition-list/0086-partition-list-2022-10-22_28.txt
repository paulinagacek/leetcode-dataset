class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        dummy1 = ListNode(-1)
        dummy2 = ListNode(-1)
        
        p1 = dummy1
        p2 = dummy2
        
        p = head
        
        while p:
            if p.val < x:
                p1.next = p
                p1 = p1.next
            else:
                p2.next = p
                p2 = p2.next
                
            tmp = p.next
            p.next = None
            p = tmp
        p1.next = dummy2.next
        return dummy1.next
