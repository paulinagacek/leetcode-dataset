class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        if not head:
            return None
        head1, head2 = ListNode(-1), ListNode(-1)
        dummy1, dummy2 = head1, head2
        while head:
            if head.val < x:
                head1.next = head
                head1 = head1.next
            else:
                head2.next = head
                head2 = head2.next
            head = head.next
            
        head2.next = None
        head1.next = dummy2.next
        return dummy1.next
