def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        node, node2 = ListNode(), ListNode()
        ptr1 = head
        ret_node2 = node2
        ret_node = node
        while ptr1:
            if ptr1.val < x:
                node.next = ptr1
                node = node.next
            else:
                node2.next = ptr1
                node2 = node2.next
            ptr1 = ptr1.next
        node2.next=None
        node.next = ret_node2.next
        
        return ret_node.next
