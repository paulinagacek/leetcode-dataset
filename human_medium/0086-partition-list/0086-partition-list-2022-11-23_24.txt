class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        left, right = ListNode(), ListNode()
        l_cur, r_cur = left, right
        while head:
            if head.val < x:
                l_cur.next = head
                l_cur = l_cur.next
            else:
                r_cur.next = head
                r_cur = r_cur.next
            head = head.next
        r_cur.next = None
        l_cur.next = right.next
        return left.next
