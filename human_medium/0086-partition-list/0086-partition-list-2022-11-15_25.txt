class Solution:
    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        # Create dummy nodes for lists of values smaller / larger than x 
        dummy1, dummy2 = ListNode(-1), ListNode(-1)
        p, p1, p2 = head, dummy1, dummy2
        
        # traverse head using p
        while p:
            if p.val >= x:
                p2.next = p
                p2 = p2.next
            else:
                p1.next = p
                p1 = p1.next
            
            # disconnect self.next of each node 
            temp = p.next
            p.next = None
            p = temp
            
        p1.next = dummy2.next
        
        return dummy1.next
