class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        total_profit = 0
        i = 0
        j = 0
        
        while j < len(prices):
            if prices[j] <= prices[i]:
                i = j
            else:
                while j < len(prices)-1 and prices[j+1] > prices[j]:
                    j += 1
                profit = prices[j] - prices[i]
                total_profit += profit
                i = j + 1
            j += 1
        return total_profit
