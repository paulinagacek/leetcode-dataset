class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        
        tot_profit = 0
        buy_price = prices[0]
        
        for i in range(1,len(prices)):
            if prices[i] > buy_price:
                tot_profit += prices[i] - buy_price
            buy_price = prices[i]
        return tot_profit
