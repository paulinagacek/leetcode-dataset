class Solution {
    public int maxProfit(int[] prices) {
        // Initialize the max profit...
        int maximumProfit = 0;
        // Traverse all the element through loop...
        for(int i=1; i<prices.length; i++) {
            // Set the difference so that we can get the max profit...
            int diff = prices[i]-prices[i-1];
            // If the difference is positive value, add to profit and increse it...
            if(diff > 0){
                maximumProfit += diff;
            }
        }
        return maximumProfit;      // Return the max profit...
    }
}
