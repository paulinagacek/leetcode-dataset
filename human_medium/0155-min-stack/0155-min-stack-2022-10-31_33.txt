class MinStack {
//PLEASE UPVOTE IF YOU LIKE THE APPROACH,THANKS!!
public:
  stack<pair<int,int>>st;//created pair to store the value and minimum value
    int mn;
    MinStack() {
        mn=INT_MAX;
    }
    void push(int val) {
        if(st.size()==0){
            mn=val;
        }
        else{
            mn=st.top().second;
        }
        mn=min(mn,val); //updated minimum value
        st.push({val,mn}); //push the value and the minimum value
    }
    
    void pop() {
        st.pop();
    }
    
    int top() {
        return st.top().first; //as first holds the value
    }
    
    int getMin() {
        return st.top().second; //as the second holds the minimum value
    }
};
