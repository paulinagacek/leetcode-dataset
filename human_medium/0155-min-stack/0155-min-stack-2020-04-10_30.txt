class MinStack:
    def __init__(self):
        """
        initialize your data structure here.
        """
        self.stack=[]

    def push(self, x: int) -> None:
        if len(self.stack)==0:
            min_x=x
        else:
            min_x=min(self.stack[-1][1],x)
        self.stack.append([x,min_x])

    def pop(self) -> None:
        self.stack.pop()

    def top(self) -> int:
        return self.stack[-1][0]

    def getMin(self) -> int:
        return self.stack[-1][1]
