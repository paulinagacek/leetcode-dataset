class MinStack {

    // stack element is an array with length of 2
    // [number, current Min Number in the stack]
    Stack<Integer[]> stack;

    public MinStack() {
        stack = new Stack<>();
    }
    
    public void push(int val) {
        if (stack.isEmpty()) {
            stack.push(new Integer[]{val, val});
            return;
        }
        int curMin = stack.peek()[1];
        // if this val is smaller then curMin
        // we update curMin by just write the val in the second index
        if (val < curMin) { 
            stack.push(new Integer[]{val, val});
        } else {
            stack.push(new Integer[]{val, curMin});
        }
    }
    
    public void pop() {
        stack.pop();
    }
    
    public int top() {
        return stack.peek()[0];
    }
    
    public int getMin() {
        return stack.peek()[1];
    }
}
