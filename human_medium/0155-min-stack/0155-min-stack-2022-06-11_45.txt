class MinStack:
    def __init__(self):
        self.st = []
        self.mins = []

    def push(self, val: int) -> None:
        self.st.append(val)
        if not self.mins or self.mins[-1] >= val:
            self.mins.append(val)

    def pop(self) -> None:
        if self.mins[-1] == self.st.pop():
            self.mins.pop()
        
    def top(self) -> int:
        return self.st[-1]

    def getMin(self) -> int:
        return self.mins[-1]
