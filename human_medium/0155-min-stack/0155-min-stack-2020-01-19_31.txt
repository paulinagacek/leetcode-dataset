class Node:
    def __init__(self, val, current_min):
        self.val = val
        self.next = None
        self.current_min = current_min
        
class MinStack:

    def __init__(self):
        self.head = None
        self.current_min = None
        

    def push(self, x: int) -> None:
        node = Node(x, x)
        
        if self.head == None:
            self.head = node
        else:
            if self.head.current_min < x:
                node.current_min = self.head.current_min
                
            node.next = self.head
            self.head = node

    def pop(self) -> None:
        if self.head:
            self.head = self.head.next

    def top(self) -> int:
        if self.head:
            return self.head.val
        
        return None

    def getMin(self) -> int:
        if self.head:
            return self.head.current_min
        
        return None
