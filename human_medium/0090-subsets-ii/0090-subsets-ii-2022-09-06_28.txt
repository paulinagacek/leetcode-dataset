class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        ans = set()
        for i in range(1<<len(nums)):
            sublist = ()
            for j in range(len(nums)):
                if (i&(1<<j)): #to check if ith bit of number i is set.
                    sublist += (nums[j],)
            ans.add(tuple(sorted(sublist)))
        return ans
