class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        s = set()
        nums.sort()
        for i in range(2**len(nums)):
            ans = []
            for j in range(len(nums)):
                if i & (1<<j):
                    ans.append(nums[j])
            s.add(tuple(ans))
        return list(s)
