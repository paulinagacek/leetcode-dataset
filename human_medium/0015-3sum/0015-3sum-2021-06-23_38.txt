def threeSum(self, nums):
	"""
	:type nums: List[int]
	:rtype: List[List[int]]
	"""
	ans = []
	nums.sort()
	for i in range(len(nums)-2):
		a = nums[i]
		start = i + 1
		end = len(nums) - 1
		while start < end:
			b = nums[start]
			c = nums[end]
			if a + b + c == 0:
				if [a, b, c] not in ans:
					ans.append([a, b, c])
				start += 1
				end -= 1
			elif a + b + c > 0:
				end -= 1
			else:
				start += 1
	return ans
