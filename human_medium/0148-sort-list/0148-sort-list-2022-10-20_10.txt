# ---------------------------- 1st Approach using extra array T : O(nlogn) and S : O(n) ----------------------------
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        nums = []
        curr = head
        while curr:
            nums.append(curr.val)
            curr = curr.next
            
        nums.sort()                                # sorting use O(nlogn) time complexity
        curr = head
        for i in nums:
            curr.val = i
            curr = curr.next
            
        return head
		
# As we are using extra array nums storing each value node into array and sorting nums and again
# sorted nums is stored as new linked list, that\'s why space complexity - O(n)
	
# ---------------------------- 2nd Approach Merge Sort T : O(nlogn) and S : O(logn) ----------------------------
    def sortList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # base case 
        if not head  or not head.next:
            return head
        
        # split the list into 2 halfs
        left = head
        right = self.getMid(head)
        tmp = right.next
        right.next = None
        right = tmp
        
        left = self.sortList(left)
        right = self.sortList(right)
        
        return self.mergeLists(left, right)
    
    # using slow and fast pointers and return slow
    def getMid(self, head):
        slow, fast = head, head.next
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            
        return slow
    
    # merge the left and right lists
    def mergeLists(self, list1, list2):
        tail = dummy = ListNode()
        
        while list1 and list2:
            if list1.val < list2.val:
                tail.next = list1
                list1 = list1.next
            else:
                tail.next = list2
                list2 = list2.next
                
            tail = tail.next
        
        if list1:
            tail.next = list1
        else:
            tail.next = list2
            
        return dummy.next

# Recursion used extra space for handling multiple return statements
# in the call stack that\'s why space complexity O(logn)
