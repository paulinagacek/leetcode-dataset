class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        n = len(s)
        wordDict = set(wordDict)
        
        # dp[i] means if s[:i] meets
        dp = [False for _ in range(n+1)]
        dp[0] = True
        
        for i in range(1, n+1):
            for j in range(i):
                sub_string = s[j:i]
                if dp[j] and sub_string in wordDict:
                    dp[i] = True
					break
        return dp[-1]
