class Solution(object):
    def __init__(self):
        self.cache = dict()
        
    def wordBreak(self, target, wordDict):
        """
        :type s: str
        :type wordDict: List[str]
        :rtype: bool
        """
        if target == \'\':
            return True

        if target in self.cache:
            return self.cache[target]

        for word in wordDict:
            if target.startswith(word):
                self.cache[target] = self.wordBreak(target[len(word):], wordDict)
                if self.cache[target]:
                    return True

        self.cache[target] = False
        return self.cache[target]
