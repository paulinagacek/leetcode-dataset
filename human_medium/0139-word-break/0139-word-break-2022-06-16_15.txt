class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        wordDict = set(wordDict)
        dp = {}
        
        def getResult(s,index):
            if not s:
                return True
            if index in dp:
                return dp[index]

            for i in range(len(s)+1):
                k = s[:i]
                if k in wordDict:
                    if getResult(s[i:],index+i):
                        dp[index] = True
                        return dp[index]
            dp[index] = False
            return dp[index]
            
        return getResult(s,0)
