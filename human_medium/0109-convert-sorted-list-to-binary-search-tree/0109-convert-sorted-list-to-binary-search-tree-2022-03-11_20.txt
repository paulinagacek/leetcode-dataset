class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        itr=head
        lst=[]
        while itr:
            lst.append(itr.val)
            itr=itr.next
        def helper(lst):
            if not lst:
                return None
            mid=len(lst)//2
            node=TreeNode(lst[mid])
            node.left=helper(lst[:mid:])
            node.right=helper(lst[mid+1::])
            return node
        node=helper(lst)
        return node
