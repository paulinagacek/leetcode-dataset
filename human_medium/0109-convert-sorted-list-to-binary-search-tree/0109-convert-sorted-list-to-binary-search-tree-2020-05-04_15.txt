class Solution:
    def sortedListToBST(self, head: ListNode) -> TreeNode:
        arr=[]
        node=head
        while node:
            arr.append(node.val)
            node=node.next
        #print (arr)    
        def construct(arr,start,end):
            if start<=end:
                mid=(start+end)//2
                node=TreeNode(arr[mid])
                node.left=construct(arr,start,mid-1)
                node.right=construct(arr,mid+1,end)
                return node
        return construct(arr,0,len(arr)-1)
