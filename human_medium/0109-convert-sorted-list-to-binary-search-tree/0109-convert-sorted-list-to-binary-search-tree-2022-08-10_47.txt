class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        
        nums = []
        while head:
            nums.append(head.val)
            head = head.next
                
        
        def helper(nums):
            if not nums:
                return None

            mid = len(nums)// 2

            root = TreeNode(nums[mid])

            root.left = helper(nums[ : mid])
            root.right = helper(nums[mid+1 : ])

            return root
        
        return helper(nums)
