class Solution:
    def sortedListToBST(self, head: Optional[ListNode]) -> Optional[TreeNode]:
        nums = []
        while head:
            nums.append(head.val)
            head = head.next
        
        def ToBST(nums):
            if not nums:
                return None
            
            mid = len(nums)//2
            node = TreeNode(nums[mid])
            node.left = ToBST(nums[:mid])
            node.right = ToBST(nums[mid+1:])
            return node
        
        return ToBST(nums)
