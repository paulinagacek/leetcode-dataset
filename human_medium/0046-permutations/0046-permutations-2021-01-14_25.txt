class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []

        def helper(vals, cands):
            if len(vals) == len(nums):
                res.append(vals)
                return
            for i in range(len(cands)):
                helper(vals + [cands[i]], cands[:i] + cands[i+1:])
                
        helper([], nums)
        return res
