class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res = []

        def backtrack(pointer):
            if pointer == len(nums):
                res.append(nums.copy())

            for i in range(pointer, len(nums)):
                nums[pointer], nums[i] = nums[i], nums[pointer]
                backtrack(pointer + 1)
                nums[pointer], nums[i] = nums[i], nums[pointer]
        backtrack(0)
        
        return res
