class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        
        permutations = [nums]
        
        for i in range(len(nums)):
            for num in permutations:
                tmp = [num[i]] + num[:i] + num[i + 1:]
                if tmp not in permutations:
                    permutations.append(tmp)
                
        return permutations
