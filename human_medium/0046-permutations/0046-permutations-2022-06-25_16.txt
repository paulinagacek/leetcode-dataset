class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        rst = []
        
        def backtrack(cur):
            if len(cur) == len(nums):
                rst.append(cur[:])
                return
            
            for n in nums:
                if n not in cur:
                    cur.append(n)
                    backtrack(cur)
                    cur.pop()
        
        backtrack([])
        return rst
