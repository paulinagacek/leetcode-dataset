class Solution {

    // helper
    public List<List<Integer>> recursive(List<Integer> numsLst, List<List<Integer>> res, List<Integer> path) {
        
        if (numsLst.isEmpty()) {
            res.add(new ArrayList(path));
        }
        else {
            for (int i=0; i<numsLst.size(); i++) {
                List<Integer> newNumsLst = new ArrayList(numsLst);
                newNumsLst.remove(i);
                path.add(numsLst.get(i));
                recursive(newNumsLst, res, path);
                path.remove(path.size() - 1);
            }
        }
        return res;
    }
    
    
    public List<List<Integer>> permute(int[] nums) {
        // a turn around is to instantiate vars here - equivalent to Pythonic optional vars 
        List<List<Integer>> res = new ArrayList();
        List<Integer> path = new ArrayList();
        
        // convert array to list
        List<Integer> numsLst = new ArrayList<Integer>();
        for (int num: nums) {
            numsLst.add(num);
        }
        
        // call helper
        return recursive(numsLst, res, path);
        
    }
}
