class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        res=[]
        track=[]
        nums.sort()
        def perm(nums,track):
            if len(track)==len(nums):
                res.append(track[:])
                return
            
            for i in range(0,len(nums)):
                if nums[i] in track: continue # If the element in track already, then we just avoid that element and move further.
                track.append(nums[i])
                perm(nums,track)
                track.pop()
        perm(nums,[])
        return res
