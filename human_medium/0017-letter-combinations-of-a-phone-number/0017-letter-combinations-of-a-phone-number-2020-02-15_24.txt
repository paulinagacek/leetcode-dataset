class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if not digits or len(digits) == 0:
            return []
        letters = ["", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"]
        return list(self.combine(letters, digits, "", 0))

    def combine(self, letters: List[str], digits: str, prefix: str, index: int):
        if index  == len(digits):
            yield prefix
        else:
            for letter in letters[int(digits[index])]:
                yield from self.combine(letters, digits, prefix + letter, index + 1)
