class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        if len(digits)==0:
            return []
        d={"2":"abc","3":"def","4":"ghi","5":"jkl","6":"mno","7":"pqrs","8":"tuv","9":"wxyz"}
        result=[i for i in d[digits[0]]]
        for i in digits[1:]:
            result=[k+j for k in result for j in d[i]]
        return result
