class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        self.res = []
        self.map = {"2":"abc","3":"def","4":"ghi","5":"jkl","6":"mno","7":"pqrs","8":"tuv","9":"wxyz"}
        self.len = len(digits)
        def helper(curr_digit,curr_str):
		# curr_digit points to index in digits 
		# curr_str is the string formed till now
            if curr_digit==self.len:
                self.res.append(curr_str)  # Case that we have added a letter corresponding each digit( for digit in digits)
                return 
            for i in self.map[digits[curr_digit]]: 
                helper(curr_digit+1,curr_str+i)   #We go to next index and add corresponding letter to curr_str
            
        helper(0,"")
        return self.res if digits else []
