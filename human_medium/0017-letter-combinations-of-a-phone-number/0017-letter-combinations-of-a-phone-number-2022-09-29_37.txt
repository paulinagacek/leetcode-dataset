class Solution:
    def letterCombinations(self, digits: str) -> List[str]:
        d_map = {"2": ["a", "b", "c"], "3": ["d", "e", "f"],
                "4": ["g", "h", "i"], "5": ["j", "k", "l"], "6": ["m", "n", "o"],
                "7": ["p", "q", "r", "s"], "8": ["t", "u", "v"], "9": ["w", "x", "y", "z"]}
        c = [d_map[i] for i in digits]
        return ["".join(i) for i in product(*c)] if c else []
