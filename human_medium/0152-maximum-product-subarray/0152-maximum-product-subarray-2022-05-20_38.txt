class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        pos = nums[0]
        neg = nums[0]
        best = nums[0]
        for i in range(1,len(nums)):
            prevMax = max(pos*nums[i],neg*nums[i],nums[i])
            prevMin = min(pos*nums[i],neg*nums[i],nums[i])
            pos = prevMax
            neg = prevMin
            best = max(prevMax,best)
        return best
