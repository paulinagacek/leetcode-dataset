class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        mini=[nums[0]]
        maxi=[nums[0]]
        n=len(nums)
        for i in range(1,n):
            if(nums[i]>=0):
                maxi.append(max(maxi[i-1]*nums[i],nums[i]))
                mini.append(min(mini[i-1]*nums[i],nums[i]))
            else:
                maxi.append(max(mini[i-1]*nums[i],nums[i]))
                mini.append(min(maxi[i-1]*nums[i],nums[i]))
        return(max(maxi))
