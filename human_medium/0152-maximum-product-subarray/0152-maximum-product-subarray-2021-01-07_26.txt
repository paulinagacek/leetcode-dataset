class Solution:
    def maxProduct(self, nums: List[int]) -> int:
        
        curr_max = curr_min = all_max = nums[0]
        for ele in nums[1:]:  
            curr_max, curr_min = max(curr_max * ele, curr_min * ele, ele), min(curr_max * ele, curr_min * ele, ele)
            all_max = max(all_max, curr_max)
            
        return all_max
