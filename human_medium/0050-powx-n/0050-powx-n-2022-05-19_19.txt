class Solution:
    def myPow(self, x: float, n: int) -> float:
        @functools.lru_cache(None)
        def pow(x, n):
            if n == 1:
                return x
            return pow(x, n // 2) * pow(x, n - n // 2)
        if n > 0:
            return pow(x, n)
        elif n == 0:
            return 1
        else:
            return 1 / pow(x, -n)
