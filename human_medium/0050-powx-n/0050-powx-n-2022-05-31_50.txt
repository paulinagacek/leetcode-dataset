    @cache
    def myPow(self, x: float, n: int) -> float:
                
        if n == 0:
            return 1
        if n == 1:
            return x
        if n == 2:
            return x*x
        if n == -1:
            return 1/(x)
        if n == -2:
            return 1/(x*x)
        
        return self.myPow(x, n - n//2) * self.myPow(x, n - ceil(n/2))
