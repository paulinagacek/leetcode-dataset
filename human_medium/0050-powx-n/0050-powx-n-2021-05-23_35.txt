class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n < 0:
            x, n = 1/x, -n
        prod, remain = 1/x, 1
        while n > 1:
            if n % 2:
                remain *= prod
            prod = prod * prod
            n = n // 2
        return 1/prod/remain
