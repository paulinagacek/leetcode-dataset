class Solution:
    def myPow(self, x: float, n: int) -> float:
        return 1 / self.bigPow(x, -n) if n < 0 else self.bigPow(x, n)
    
    def bigPow(self, x: float, n: int) -> float:
        if n == 0:
            return 1
        if n & 1:
            return x * self.bigPow(x, n - 1)
        else:
            res = self.bigPow(x, n // 2)
            return res * res
