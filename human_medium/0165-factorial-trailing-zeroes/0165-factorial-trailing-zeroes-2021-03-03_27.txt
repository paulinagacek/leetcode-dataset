class Solution:
    def trailingZeroes(self, n: int) -> int:
        factorial = 1
        zero_count = 0
        for i in range(1,n+1):
            factorial = factorial * i # This will give me the factorial after the loop ends
        factorial_string = str(factorial) # converting the factorial into string so I can iterate through the numbers

        for i in range(len(factorial_string)-1, 0, -1): # I am looping it backwards and comparing the last digit to 0
            while factorial_string[i] == "0": # comparing last digit to 0 and increasing the count if its 0. Else breaking the loop
                zero_count +=1
                i = i -1 # If i do not decrement i here, this while loop will go into infinite loop 
            break
        return zero_count
