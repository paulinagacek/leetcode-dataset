class Solution:
    def trailingZeroes(self, n: int) -> int:
        val = 1
        for i in range(1,n+1):
            val = val*i
        if n==0:
            return 0
        else:
            str1 = str(val)
            count= 0
            length = len(str1)
            for i in range(length-1,-1,-1):
                if str1[i]=="0":
                    count +=1
                else:
                    break
            return count
