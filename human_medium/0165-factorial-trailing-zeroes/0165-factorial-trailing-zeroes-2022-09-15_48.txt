class Solution:
    def trailingZeroes(self, n: int) -> int:
        fact = 1
        count = 0
        for i in range(1, n+1):
            fact = fact * i
        l = list(str(fact))
        r = l[::-1]
        for i in range(len(r)):
            if(r[i] == str(0)):
                count += 1
            else:
                break
        return count
