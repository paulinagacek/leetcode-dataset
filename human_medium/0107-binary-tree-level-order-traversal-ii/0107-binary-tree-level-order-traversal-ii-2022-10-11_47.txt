class Solution(object):
    def levelOrderBottom(self, root):
        """
        :type root: TreeNode
        :rtype: List[List[int]]
        """
        if root is None:
            return []
        q=deque([root])
        ans=[]
        while q:
            cl=[]
            for i in range(len(q)):
                curr=q.popleft()
                cl.append(curr.val)
                if curr.left is not None:
                    q.append(curr.left)
                if curr.right is not None:
                    q.append(curr.right)
            ans.append(cl)
        return ans[::-1]
