class Solution:
    def levelOrderBottom(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root: return []
        res=[[root.val]]
        q=deque()
        q.append(root)
        while q:
            temp=[]
            for i in range(len(q)):
                curr=q.popleft()
                if curr.left:
                    temp.append(curr.left.val)
                    q.append(curr.left)
                if curr.right:
                    temp.append(curr.right.val)
                    q.append(curr.right)
            if len(temp)>0:
                res.append(temp)
        return res[::-1]
