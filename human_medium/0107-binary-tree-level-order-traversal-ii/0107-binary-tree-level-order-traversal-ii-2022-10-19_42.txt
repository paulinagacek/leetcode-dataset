class Solution:
    def levelOrderBottom(self, root: Optional[TreeNode]) -> List[List[int]]:
        if not root:
            return None

        q = deque([root])
        res = []

        while q:
            temp = []
            length = len(q)
            #poping the previous depth nodes and appending the next depth nodes
            for i in range(length):
                node = q.popleft()
                temp.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            res.append(temp)
        return res[:: -1]
