class Solution:
    def levelOrderBottom(self, root: TreeNode) -> List[List[int]]:
        if root is None:
            return root
        queue = []
        queue.append(root)
        result = []
        while len(queue) > 0:
            ans = []
            l = len(queue)
            for i in range(l):
                node = queue.pop(0)
                ans.append(node.val)
                if node.left != None:
                    queue.append(node.left)
                if node.right != None:
                    queue.append(node.right)
            result.append(ans)
        return result[::-1]
