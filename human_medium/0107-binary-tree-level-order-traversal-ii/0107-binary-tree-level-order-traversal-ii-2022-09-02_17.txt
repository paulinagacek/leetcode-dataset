class Solution(object):
    def levelOrderBottom(self, root):
        """
        :type root: TreeNode
        :rtype: List[List[int]]
        """
        levelnode=defaultdict(list)
        def dfs(node,level):
            if not node:
                return
            
            levelnode[level].append(node.val)
            dfs(node.left,level+1)
            dfs(node.right,level+1)
        dfs(root,0)
        return levelnode.values()[::-1]
