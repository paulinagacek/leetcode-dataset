class Solution:
    def levelOrderBottom(self, root: TreeNode) -> List[List[int]]:
        ans = []
        if not root:
            return ans
        que = [root]
        
        while que:
            level = []
            for i in range(len(que)):
                node = que.pop(0)
                level.append(node.val)
                for nei in [node.left, node.right]:
                    if nei:
                        que.append(nei)
            
            ans = [level] + ans
        return ans
