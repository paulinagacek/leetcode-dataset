class Solution:
    def levelOrderBottom(self, root: Optional[TreeNode]) -> List[List[int]]:
        result = []
        if root is None:
            return result
        
        q = deque()
        q.append(root)

        while len(q) > 0:
            level_list = []
            count = len(q)

            while count > 0:
                node = q.popleft()
                level_list.append(node.val)
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
                count -= 1
            result.append(level_list)
        
        return result[::-1]
