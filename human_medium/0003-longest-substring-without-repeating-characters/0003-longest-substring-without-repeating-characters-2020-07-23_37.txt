class Solution:
    def lengthOfLongestSubstring(self, s):
        str_1, str_2 = \'\', \'\'
        for i in range(len(s)):
            if s[i] in str_1:
                if len(str_1) > len(str_2): str_2 = str_1
                str_1 = str_1[str_1.index(s[i])+1:] + s[i]
            else: str_1 += s[i]
        return max(len(str_1), len(str_2))
