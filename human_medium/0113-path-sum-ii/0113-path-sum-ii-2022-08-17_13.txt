class Solution(object):
    def pathSum(self, root, targetSum):
        if not root: return []
        res = []
        
        def dfs(node, path):
            if sum(path) == targetSum and not node.left and not node.right:
                res.append(path)
                return
            if node.left:
                dfs(node.left, path+[node.left.val])
            if node.right:
                dfs(node.right, path+[node.right.val])
                
        dfs(root, [root.val])
        return res
