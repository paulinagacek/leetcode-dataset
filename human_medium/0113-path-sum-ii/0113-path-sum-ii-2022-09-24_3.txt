def binaryTreePaths(self, root):
        result = []
        def dfs(root, path):
            nonlocal result
            if not root:
                return []
            path += [root.val]           
            if not root.left and not root.right:
                result.append(path.copy())
            dfs(root.left, path)
            dfs(root.right, path)
            path.pop()
            return result    
        return dfs(root, [])
