class Solution(object):
    def pathSum(self, root, sum):
        """
        :type root: TreeNode
        :type sum: int
        :rtype: List[List[int]]
        """
        
        # DFS solution
        if not root:
            return [] 
        if root.left is None and root.right is None:
            if sum == root.val:
                return [[root.val]]
            else:
                return []

        res = [] 
        left, right = [], []
        sum -= root.val
        
        if root.left:
            left = self.pathSum(root.left, sum)
        if root.right:
            right = self.pathSum(root.right, sum)
            
        for l in left:
            res.append([root.val] + l)
        for r in right:
            res.append([root.val] + r)
            
        return res
