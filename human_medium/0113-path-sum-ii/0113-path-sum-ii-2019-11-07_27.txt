class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> List[List[int]]:
        if not root:
            return []
        
        def dfs(root: TreeNode, sum: int, path_sum: List[int], res: List[List[int]]) -> None:
            if not (root.left or root.right) and sum == root.val:
                path_sum.append(root.val)
                res.append(path_sum)
            
            if root.left:
                dfs(root.left, sum - root.val, path_sum + [root.val], res)
            
            if root.right:
                dfs(root.right, sum - root.val, path_sum + [root.val], res)
            
        res = []
        dfs(root, sum, [], res)
        return res
