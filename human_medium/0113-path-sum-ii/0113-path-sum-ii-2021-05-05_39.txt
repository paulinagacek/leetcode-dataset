class Solution:
    def pathSum(self, root: TreeNode, targetSum: int) -> List[List[int]]:
        
        def helper(node, target_sum, current_path, result_path):
            
            if node is None: return []
            
            current_path.append(node.val)
            if node.left is None and node.right is None and node.val == target_sum:
                result_path.append(list(current_path))
            
            helper(node.left, target_sum - node.val, current_path, result_path)
            helper(node.right, target_sum - node.val, current_path, result_path)
            
            current_path.pop()
            
            return result_path
        
        return helper(root, targetSum, [], [])
