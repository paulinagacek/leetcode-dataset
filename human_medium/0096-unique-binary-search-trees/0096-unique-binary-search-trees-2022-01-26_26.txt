class Solution:
    def dp(self, n, dp):
        if n <= 1:
            return 1

        if n in dp:
            return dp[n]

        ans = 0
        for i in range(1, n + 1):
            ans += self.dp(i - 1, dp) * self.dp(n - i, dp)

        dp[n] = ans
        return ans

    def numTrees(self, n):
        dp = {}
        return self.dp(n, dp)
