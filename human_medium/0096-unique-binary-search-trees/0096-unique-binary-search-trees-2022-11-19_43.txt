class Solution:
    def numTrees(self, n: int) -> int:
        DP = [0] * (n + 1)
        DP[0] = DP[1] = 1
        for i in range(2, n + 1):
            # compute the number of BST with i nodes.
            # we just care about the structural difference.
            for j in range(i):
                DP[i] += DP[j] * DP[i - j - 1]
        return DP[n]
