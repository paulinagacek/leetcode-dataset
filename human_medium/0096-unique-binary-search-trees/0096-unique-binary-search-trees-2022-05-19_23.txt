class Solution:
    def numTrees(self, n: int) -> int:

        cache = [1] * (n + 1)

        for i in range(1, n + 1):
            remaining_nodes = i - 1

            total = 0
            for j in range(remaining_nodes + 1):
                left = j
                right = remaining_nodes - j
                total = total + (cache[left] * cache[right])
            cache[i] = total

        return (cache[n])
