class Solution:
    def numTrees(self, n: int) -> int:
        
        memo = collections.defaultdict(int)
        
        memo[0] = 1
        memo[1] = 1
        memo[2] = 2
        
        for i in range(3, n+1):
            for j in range(i):
                memo[i] += memo[j]*memo[i-j-1]
                
        return memo[n]
