class Solution:
    def numTrees(self, n: int) -> int:
        numTree = [1] * (n + 1) 
        # start at range 2 as basecase 0 & 1 will = to 1 
        for nodes in range(2, n + 1): 
            total = 0
            for root in range(1, nodes + 1): 
                left = root - 1
                right = nodes - root
                total += numTree[left] * numTree[right]
            # store the value in our list basically cacheing 
            numTree[nodes] = total
        # return last index(n) as our result 
        return numTree[n]
