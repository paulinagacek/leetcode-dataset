class Solution:
    def numTrees(self, n: int) -> int:
       
        dp = [0 for i in range(n+1)]
        dp[1] = 1
        dp[0] = 1
        for i in range(1,n+1):
            ans = 0
            for j in range(1,i+1):
                left = dp[j-1]
                right = dp[i-j]
                ans += left*right
            dp[i] = ans
        return dp[n]
