class Solution:
    def numTrees(self, n: int) -> int:
        treeCount = {0:1} #For memoization
        
        for count in range(1, n+1): #Bottom Up DP on number of nodes
            left, right, temp = 0, count - 1, 0
			
            #left -> number of nodes to the left
			#right -> number of nodes to the right
			#temp -> number of BSTs able to formed w/ count nodes
			
            while left <= right: 
				#If left > right, we aren\'t taking advantage of the symmetry
				
                if left == right: #We are at the last value, only count this one once
                    temp += treeCount[left] * treeCount[right]
					
                else: #Otherwise, we can count it twice
                    temp += (treeCount[left] * treeCount[right]) * 2 #Symmetry in root choice around n/2
                left += 1
                right -= 1 #Proceed to the next node, updating left and right counts
                
            treeCount[count] = temp
            
        return treeCount[n]
