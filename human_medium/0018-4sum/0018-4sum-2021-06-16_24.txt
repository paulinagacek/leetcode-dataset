class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        
        l = len(nums)
        if l < 4:
            return []

        # sort
        nums.sort()
        
        # calc the min sum
        min_sum = nums[0] + nums[1] + nums[2] + nums[3]
        # If the min sum value is greater than the target value, then return
        if min_sum > target:
            return []
        
        # calc the max sum
        max_sum = nums[-1] + nums[-2] + nums[-3] + nums[-4]
        # If the max sum value is less than the target value, then return
        if max_sum < target:
            return []
        
        res = []
		# to be 3sum
        for i in range(l):
            for k in range(i+1, l):
                left = k + 1
                right = l - 1
                
                while left < right:
                    # calc the result
                    result = nums[i] + nums[k] + nums[left] + nums[right] - target
                    # If the value is greater than 0
                    if result > 0:
                        right -= 1
					# if the value is less than 0
                    elif result < 0:
                        left += 1
                    else:
                        once = [nums[i], nums[k], nums[left], nums[right]]
                        if once not in res:
                            res.append(once)
                        
                        while left < right and nums[left] == nums[left + 1]:
                            left += 1
                        while left < right and nums[right] == nums[right - 1]:
                            right -= 1
                        
                        left += 1
                        right -= 1
        return res
