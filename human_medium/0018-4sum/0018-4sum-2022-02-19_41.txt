def fourSum(self, nums: List[int], target: int) -> List[List[int]]:
        nums.sort()
        quads = []
        
        for left in range(len(nums)):
            for right in range(left+1, len(nums)):
                start = right+  1
                end = len(nums) - 1
                while start < end:
                    s = nums[left] + nums[right] + nums[end] + nums[start]
                    if s == target:
                        quads.append([nums[left], nums[right], nums[start], nums[end]])
                        start += 1
                    elif s < target:
                        start += 1
                    else:
                        end -= 1
                        
        #remove dupicated answers
        quads = self.removeDuplicates(quads)
        return quads
    
    def removeDuplicates(self, arr):
        temp = []
        for each in range(len(arr)):
            right = arr[each]
            if right not in temp:
                temp.append(right)
        return temp
