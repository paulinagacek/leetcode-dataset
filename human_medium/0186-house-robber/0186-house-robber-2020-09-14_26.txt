class Solution:
    def rob(self, nums: List[int]) -> int:
        if  len(nums) == 0:
	        return 0
    
        even_houses = 0
        odd_houses = 0
 
        for i in range(len(nums)):
            if i % 2 == 0:
                
                even_houses += nums[i]
                even_houses = even_houses if even_houses > odd_houses else odd_houses
		    
            else:
                
                odd_houses += nums[i]
                odd_houses = even_houses if even_houses > odd_houses else odd_houses
		    
        return  even_houses if even_houses > odd_houses else odd_houses
