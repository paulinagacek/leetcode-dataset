class Solution:
    def rob(self, nums: List[int]) -> int:
        if not nums:
            return 0
        if len(nums)<3:
            return max(nums)
        
        dp = [0 for i in range(len(nums))]
        dp[0], dp[1]= nums[0], nums[1]
        t = max(nums[0], nums[1])
        
        for i,n in enumerate(nums[2:], start=2):
            dp[i] = max(dp[i-2], dp[i-3])+n
            t = max(t, dp[i])
        return t
