class Solution(object):
    def rob(self, nums):
        if len(nums) == 1:
            return nums[0]
        if len(nums) > 2:
            nums[2] += nums[0]
        for i in range(3, len(nums)):
            nums[i] += max(nums[i-2], nums[i-3])
        return max(nums[len(nums)-1], nums[len(nums)-2])
