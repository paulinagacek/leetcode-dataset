def rob(self, nums: List[int]) -> int:
        
        def choose(nums):
            for i in range(1, len(nums)):
                self.dp[i] = max(nums[i]+self.dp[i-2], self.dp[i-1])
            return self.dp[i]
           
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
		self.dp = [0]*(len(nums))
        self.dp[0] = nums[0]
        return choose(nums)
