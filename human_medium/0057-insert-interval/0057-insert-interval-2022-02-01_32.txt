class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        intervals=intervals+[newInterval]
        
		#Code for  Q 56: Merge Intervals
        intervals.sort()
        ans=[intervals[0]]
        for i in intervals:
            if i[0] <= ans[-1][1]:
                ans[-1][1]=max(ans[-1][1],i[1])
            else:
                ans.append(i)     
        return ans
