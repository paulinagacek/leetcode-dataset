class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        if intervals == []:
            return [newInterval]
        # mark indices of old intervals intersected with newInterval (the name is just a analogy)
		infected = []
		# mark where we should put newInterval if none of old intervals intersect
        left = -1
        for i in range(len(intervals)):
            if intervals[i][1] < newInterval[0]:
                left = i
			# old interval i is infected if intersected with newInterval
            if newInterval[0] <= intervals[i][0] <= newInterval[1] or newInterval[0] <= intervals[i][1] <= newInterval[1] or intervals[i][0] <= newInterval[0] <= intervals[i][1] or intervals[i][0] <= newInterval[1] <= intervals[i][1]:
                infected.append(i)
        # if never intersects, just insert newInterval
		if infected == []:
            return intervals[:left+1] + [newInterval] + intervals[left+1:]
        # if intersection exists, merge them and insert back into intervals
		else:
            return intervals[:infected[0]] + [[min(intervals[infected[0]][0],newInterval[0]), max(intervals[infected[-1]][1],newInterval[1])]] + intervals[infected[-1]+1:]
