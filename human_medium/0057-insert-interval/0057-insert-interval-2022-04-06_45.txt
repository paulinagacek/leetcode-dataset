class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        i = 0
        result = []
        num_intervals = len(intervals)
        
        # Find insertion point & add intervals up until then
        while i < num_intervals and intervals[i][1] < newInterval[0]:
            result.append(intervals[i])
            i += 1

        # Accumulate intervals to be merged
        while i < num_intervals and newInterval[1] >= intervals[i][0]:
            newInterval[0] = min(intervals[i][0], newInterval[0])
            newInterval[1] = max(intervals[i][1], newInterval[1])
            i += 1
        
        # Insert the accumulated new interval
        result.append(newInterval)
        
        # Insert the rest of the intervals
        while i < num_intervals:
            result.append(intervals[i])
            i += 1
        
        return result
