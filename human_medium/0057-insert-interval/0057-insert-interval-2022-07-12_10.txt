class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        
        if len(intervals) < 2 and len(newInterval) == 0: # if there is only one element in the provided list and if htere is no element in newInterval
            return intervals # then just return interval. 
      
        intervals.append(newInterval) # merging both the lists.
        
        # Below is the same program for merge interval, as we have to return a list with overlape. 
        
        intervals.sort()# Sorting the list, if we dont then we wont be able to check for the overlapping intervals.
        start = intervals[0][0] # saving first element of the sublist in a variable for comparing it. 
        end = intervals[0][1] # saving second element of the sublist in a variable for comparing it. 
        res = [] # crerating a empty list for saving the result. 
        
        for i in range(1, len(intervals)):# traversing the list from 1 as 0th element of the list is allocated to the variables, start & end.
            interval = intervals[i]# assigning i`th element to the a variable for performing the logic.
            if interval[0]<=end: # overlapping intervals
                end = max(interval[1], end)
            else:  # non-overlapping interval, add the previous interval and reset
                res.append([start, end])
                start = interval[0] # updating start variable with current interval element.
                end = interval[1]# updating end variable with current interval element.
        res.append([start, end])  # add the last interval
        return res
