class Solution:
    def insert(self, intervals: List[List[int]], newInterval: List[int]) -> List[List[int]]:
        
        # This will be our result array         
        res = []
        
        # iterate through all the intervals        
        for i in range(len(intervals)):
            
            # Set current interval to a variable 
            curr = intervals[i]
            
            # The logic goes like this: 
            # If we have an interval that is in a range before the current interval:
            #   - Append the new interval to the result and add the rest of the intervals list and return
            #
            # The next scenrio is that we have an interval that overlaps with the current interval
            # If you can imagine this as a line with points representing the interval 
            # you would have something like this:
            #
            #         *---------*
            #     *---------*
            #   ====================
            #     *-------------*
            #
            # You can see that to "merge" these lines you just take the minimum point of each and
            # the maximum point of each, this is how we contruct a larger interval, so we can continue
            # doing this until we either find a place for the interval, or it gets added to the end.
            #
            
            if newInterval[0] < curr[0] and newInterval[1] < curr[0]:
                res.append(newInterval)
                res.extend(intervals[i:])
                return res
            elif newInterval[0] <= curr[0] or newInterval[0] <= curr[1]:
                newInterval[0] = min(newInterval[0], curr[0])
                newInterval[1] = max(newInterval[1], curr[1])
            else:
                # Here we append the current interval if theres no overlaps, we only care about overlaps
                res.append(curr)
        
        # if we had nowhere to put our new interval then it belongs in the end of the result
        res.append(newInterval)
        return res
