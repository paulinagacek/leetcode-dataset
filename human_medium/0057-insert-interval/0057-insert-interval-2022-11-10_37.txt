class Solution(object):
    def insert(self, intervals, newInterval):
        """
        :type intervals: List[List[int]]
        :type newInterval: List[int]
        :rtype: List[List[int]]
        """
        if len(newInterval) == 0:
            return intervals
        if len(intervals) == 0:
            return [newInterval]

        # simple way to find which index to insert your new interval
        # check the first element of current jth index interval and 
        # newInterval. When newInterval[0] is less jth index interval
        # you\'ve found the j index to insert at with built in array 
        # .insert function.
        j = 0
        while j < len(intervals) and newInterval[0] > intervals[j][0]:
            j += 1
        intervals.insert(j, newInterval)


        # merge intervals algorithm without sorting needed
        merged = [intervals[0]]
        lastInterval = intervals[0]

        for i in range(1, len(intervals)):
            currInterval = intervals[i]
            if currInterval[0] <= lastInterval[1]:
                lastInterval[0] = min(currInterval[0], lastInterval[0])
                lastInterval[1] = max(currInterval[1], lastInterval[1])
                merged[-1] = lastInterval
            else:
                lastInterval = currInterval
                merged.append(lastInterval)
        return merged
