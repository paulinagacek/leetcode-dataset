class Solution:
    def partition(self, s: str) -> List[List[str]]:
        res=[]
        self.helper(res,s,[])
        return res
    
    def helper(self,res,s,path):
        if len(s)==0:
            res.append(path)
            return
        for i in range(1,len(s)+1):
            if s[:i]==s[:i][::-1]:
                self.helper(res,s[i:],path+[s[:i]])
