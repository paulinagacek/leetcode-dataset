class Solution:
    def intToRoman(self, num: int) -> str:
        # Define mapping of values to Roman symbols in descending order
        # including the special subtractive cases
        mapping = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        result = []
        
        for value, symbol in mapping:
            if num == 0:
                break
            # Determine how many times this symbol/group fits into the current number
            count, num = divmod(num, value)
            result.append(symbol * count)
            
        return "".join(result)
