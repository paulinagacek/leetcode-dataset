class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of values to Roman symbols in descending order
        # including the specific subtractive forms allowed.
        mapping = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I")
        ]
        
        roman_parts = []
        
        for value, symbol in mapping:
            # Determine how many times the symbol fits into the current number
            if num == 0:
                break
            count = num // value
            if count > 0:
                roman_parts.append(symbol * count)
                num -= value * count
                
        return "".join(roman_parts)
