class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of values to Roman symbols in descending order
        # including the special subtractive cases (4, 9, 40, etc.)
        mapping = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"),
            (1, "I")
        ]
        
        result = []
        for value, symbol in mapping:
            # Determine how many times the current symbol/pair fits into num
            count = num // value
            if count > 0:
                result.append(symbol * count)
                num %= value
                
        return "".join(result)
