class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of values to Roman numeral symbols in descending order
        # This includes both the standard symbols and the special subtractive forms
        value_map = [
            (1000, "M"), (900, "CM"), (500, "D"), (400, "CD"),
            (100, "C"), (90, "XC"), (50, "L"), (40, "XL"),
            (10, "X"), (9, "IX"), (5, "V"), (4, "IV"), (1, "I")
        ]
        
        result = []
        
        for value, symbol in value_map:
            # Determine how many times this symbol can fit into the current number
            count = num // value
            if count > 0:
                result.append(symbol * count)
                # Reduce the number by the total value added to the result
                num %= value
                
        return "".join(result)
