class Solution:
    def intToRoman(self, num: int) -> str:
        # Define the mapping of values to Roman symbols in descending order
        # including the subtractive cases for 4 and 9 patterns.
        val_map = [
            (1000, "M"),
            (900, "CM"),
            (500, "D"),
            (400, "CD"),
            (100, "C"),
            (90, "XC"),
            (50, "L"),
            (40, "XL"),
            (10, "X"),
            (9, "IX"),
            (5, "V"),
            (4, "IV"),
            (1, "I")
        ]
        
        roman_numeral = ""
        
        for value, symbol in val_map:
            # Determine how many times the symbol fits into the current number
            count = num // value
            if count > 0:
                roman_numeral += symbol * count
                num %= value
                
        return roman_numeral
