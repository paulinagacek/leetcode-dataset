from typing import List

class Solution:
    def maxArea(self, height: List[int]) -> int:
        left = 0
        right = len(height) - 1
        max_water = 0
        
        while left < right:
            # Calculate the width between the lines
            width = right - left
            
            # The height of the container is determined by the shorter line
            current_height = min(height[left], height[right])
            
            # Calculate area
            current_area = width * current_height
            
            # Update maximum area found so far
            max_water = max(max_water, current_area)
            
            # Move the pointer of the shorter line inward to try and find a taller line
            if height[left] < height[right]:
                left += 1
            else:
                right -= 1
                
        return max_water
