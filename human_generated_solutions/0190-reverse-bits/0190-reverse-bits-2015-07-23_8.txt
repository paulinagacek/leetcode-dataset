class Solution:
    def reverseBits1(self, n):
        res = 0
        for _ in range(32):
            res = res*2 + (n & 1)
            n >>= 1
        return res

    def reverseBits(self, n):
        res, i = 0, 31
        while n:
            res += (n & 1) << i
            n >>= 1
            i -= 1
        return res
