class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        slow = head
        fast = head
        isCycle = False
        
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next
            
            if slow == fast:
                isCycle = True
                break
        
        if not isCycle: return None
        
        check = head
        while check != slow:
            slow = slow.next
            check = check.next
        
        return check

# Time: O(N)
# Space: O(1)
