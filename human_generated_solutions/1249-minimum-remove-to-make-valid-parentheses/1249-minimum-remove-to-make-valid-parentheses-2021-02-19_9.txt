class Solution:
    def minRemoveToMakeValid(self, s: str) -> str:
        s = list(s)
        left_parr = 0
        left_ind = []
        for i in range(len(s)):
            if s[i] == \')\' and left_parr == 0:
                s[i] = \'\'
            elif s[i] == \'(\':
                left_parr += 1
                left_ind.append(i)
            elif s[i] == \')\' and left_parr > 0:
                left_parr -= 1
                left_ind.pop()

        if len(left_ind) > 0:
            for j in left_ind:
                s[j] = \'\'

        return \'\'.join(s)
