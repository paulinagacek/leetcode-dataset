class Solution:
    def reverseVowels(self, s: str) -> str:
        vowels = set(list("aeiouAEIOU"))
        s = list(s)
        i, j = 0, len(s) - 1
        while i < j:
            if s[j] not in vowels:
                j -=1
            elif s[i] not in vowels:
                i +=1
            else:
                s[i], s[j] = s[j], s[i]
                i += 1
                j -= 1
        return \'\'.join(s)
