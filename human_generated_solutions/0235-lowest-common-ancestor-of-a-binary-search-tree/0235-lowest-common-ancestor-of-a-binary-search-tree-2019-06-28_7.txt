class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        while True:
            if p.val <= root.val <= q.val or q.val <= root.val <= p.val:
                return root 
            if p.val > root.val and q.val > root.val:
                root = root.right
            elif p.val < root.val and q.val < root.val:
                root = root.left
