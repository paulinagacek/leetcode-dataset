class Solution:
    def maxProductDifference(self, nums: List[int]) -> int:
        min1 = min2 = inf
        max1 = max2 = 0
        for n in nums:
            if n < min1:
                min2 = min1
                min1 = n
            elif n < min2:
                min2 = n
            if n > max2:
                max1 = max2
                max2 = n
            elif n > max1:
                max1 = n
        return max1 * max2 - min1 * min2
