class Solution(object):
    def checkRecord(self, s):
        isLate = False
        absent = 0

        for i in range(0,len(s)) :
            if s[i] == \'A\':
                absent += 1
            
            if absent >= 2 :
                return False
            
            if s[i] == \'L\' and i <= len(s)-3 and s[i+1] == \'L\' and s[i+2] == \'L\':
                return False

        return True
