class Solution:
    def checkPowersOfThree(self, n: int) -> bool:
        initial_number=n
        res=[]
        p=[]
        while n>0:
            res.append(int(n%3))
            n=int(n/3)
*         for i in range(len(res)):
            if res[i]==1:
                p.append(3**i)
        if sum(p)==initial_number:
            return True
        else:
            return False
