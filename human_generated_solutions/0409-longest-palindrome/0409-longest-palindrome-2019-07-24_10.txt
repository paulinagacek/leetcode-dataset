class Solution:
    def longestPalindrome(self, s: str) -> int:
        if not s:return 0
        d={}
        for i in s:
            d[i]=d.get(i,0)+1
        res=0
        o=0
        for i in d:
            if d[i]%2==0:
                res+=d[i]
            else:
                res+=d[i]-1
                o+=1
        return res+(o>0)
