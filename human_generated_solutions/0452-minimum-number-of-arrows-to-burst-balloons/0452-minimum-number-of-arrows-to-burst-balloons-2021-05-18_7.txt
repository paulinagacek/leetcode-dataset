class Solution:
    def findMinArrowShots(self, intervals: List[List[int]]) -> int:
        n = len(intervals)
        if n <= 1:
            return n

        intervals.sort(key=lambda x: x[1])
        selected = [intervals[0]]
        for i in range(1,n):
            if intervals[i][0] > selected[-1][1]:
                selected.append(intervals[i])

        return len(selected)
