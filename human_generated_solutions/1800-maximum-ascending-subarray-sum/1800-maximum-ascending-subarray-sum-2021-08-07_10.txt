class Solution:
    def maxAscendingSum(self, nums: List[int]) -> int:
        if not nums:
            return 0
        
        currentSum = nums[0]
        maxSum = currentSum
        
        for i in range(1, len(nums)):
            if nums[i] > nums[i - 1]:
                currentSum += nums[i]
                maxSum = max(maxSum, currentSum)
            else:
                currentSum = nums[i]
        
        return maxSum
