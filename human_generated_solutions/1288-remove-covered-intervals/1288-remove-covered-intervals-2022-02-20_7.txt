class Solution(object):
    def removeCoveredIntervals(self, intervals):
        """
        :type intervals: List[List[int]]
        :rtype: int
        """
        n = len(intervals)
        c = n
        for i in range(n):
            for j in intervals[:i] + intervals[i+1:]:
                if intervals[i][0] >= j[0] and intervals[i][1] <= j[1]:
                    c-=1
                    break
        return c
