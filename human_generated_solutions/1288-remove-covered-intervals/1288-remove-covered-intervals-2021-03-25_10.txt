class Solution:
    def removeCoveredIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort(key=lambda x: (x[0], -x[1],))
        output = len(intervals)
        end = intervals[0][1]
        
        for i in range(1, len(intervals)):
            if intervals[i][1] <= end:
                output -= 1
            else:
                end = intervals[i][1]
                
        return output
