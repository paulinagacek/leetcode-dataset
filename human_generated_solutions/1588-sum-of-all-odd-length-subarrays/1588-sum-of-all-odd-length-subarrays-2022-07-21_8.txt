class Solution:
    def sumOddLengthSubarrays(self, arr: List[int]) -> int:
        if len(arr) < 3: # if we are having only 2 elements then we cant perform anything with them, rather just provide there sum.
            return sum(arr)
       
        prefix_sum = [0] * (len(arr) + 1) # this will create a list of multiple zeros in it till the len(arr) + 1, i.e. [0, 0, 0, 0, 0, 0]
        for index, value in enumerate(arr):
            prefix_sum[index+1] = prefix_sum[index] + value
        # the above loop will give us a list of prefix_sum operation on arr. i.e. [0, 1, 5, 7, 12, 15]
        
        result = 0
        
        for start in range(1, len(arr)+1):
            for end in range(start, len(arr)+1, 2):
                result += (prefix_sum[end] - prefix_sum[start-1])
