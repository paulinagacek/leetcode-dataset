class Solution:
    def leastBricks(self, wall: List[List[int]]) -> int:
        sumMap, maxSum = {}, 0
        for i in range(len(wall)):
            currSum = 0
            for j in range(len(wall[i])-1):
                currSum += wall[i][j]
                if currSum not in sumMap:
                    sumMap[currSum] = 0
                sumMap[currSum] += 1
                maxSum = max(maxSum, sumMap[currSum])
        return len(wall) - maxSum
