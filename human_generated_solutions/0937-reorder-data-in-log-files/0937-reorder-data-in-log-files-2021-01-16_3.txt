def reorderLogFiles(self, logs: List[str]) -> List[str]:
        
        nums = list()
        alp = list()
        
        for entry in logs:
            idf, log = entry.split(\' \',1)
            
            if log[0].isdigit():
                nums.append(entry)
            else:
                alp.append(entry)
        
        alp.sort(key = lambda x: (x.split(\' \',1)[1], x.split(\' \',1)[0]))
        
        return alp+nums
