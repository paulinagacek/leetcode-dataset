class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        
        l, r = 0, len(nums)-1
        
        while l <= r:
            
            mid = l + (r - l) // 2 # this is how we calculate middle of the array
            
            if nums[mid] == target: # if the middle index value of array is  equal to target then return the index. 
                return mid
            elif target < nums[mid]: # if the target value is lower then middle index
                r = mid - 1 # then remove all the right hand side elements.
            else: 
                l = mid + 1 # if target is greater then the middle, then remove the smaller elements which are on left side. 

        return l # index where target element can be stored
