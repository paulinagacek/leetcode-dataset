class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        N = len(nums)-1
        l,r = 0,N
        best_pos = 0
        
        
        while l<=r:
            mid = (l+r)//2
            
            val = nums[mid]
            
            if val == target:
                return mid
            
            if target < val:
                best_pos = mid
                r = mid-1
            else:
                best_pos = mid+1
                l = mid+1
                
        return best_pos
