class Solution:
    def reverseString(self, s: List[str]) -> None:            
        # recursive solution
        def helper(l,r,s):
            if l>=r:
                return 
            else:
                s[l],s[r] = s[r],s[l]
                helper(l+1,r-1,s) 
            
        helper(0,len(s)-1,s)
