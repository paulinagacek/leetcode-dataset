class Solution:
    def mergeInBetween(self, list1: ListNode, a: int, b: int, list2: ListNode) -> ListNode:
        node1 = None
        node2 = None
        
        temp = list1
        for i in range(b+1):
            if i == a-1:
                node1 = temp
            if i == b:
                node2 = temp.next
                
            temp = temp.next
        
        node1.next = list2
        
        temp = list2
        while temp.next:
            temp = temp.next
        temp.next = node2
        
        return list1
