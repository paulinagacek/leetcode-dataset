# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def mergeInBetween(self, list1, a, b, list2):
        """
        :type list1: ListNode
        :type a: int
        :type b: int
        :type list2: ListNode
        :rtype: ListNode
        """
        temp = list1
        count = 0
        if list1 == None:
            return list1
        while temp:
            if count == a-1:
                left = temp
            if count == b+1:
                right = temp
            temp = temp.next
            count += 1
        temp = list2
        left.next = temp
        while temp:
            if temp.next == None:
                break
            temp = temp.next
        temp.next = right
        return list1
