def detect_capital_use(W: str) -> bool:
    """
    Detect Capital

    time: O(n)
    space: O(1)

    :param str W:
    :return bool:
    """
    N = len(W)

    is_capital = lambda c: ord("A") <= ord(c) <= ord("Z")

    for i in range(N - 1):
        if i == 0 and is_capital(W[i]):
            continue
        elif is_capital(W[i]) != is_capital(W[i + 1]):
            return False

    return True
