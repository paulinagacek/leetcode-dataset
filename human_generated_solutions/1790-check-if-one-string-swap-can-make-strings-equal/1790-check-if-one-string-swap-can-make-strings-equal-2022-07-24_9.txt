class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        # Counting variable
        different_char_count = 0
        
        # FOR loop to go through the indices of each string making sure they match
        for i in range(len(s1)) : 
            if s1[i] != s2[i] :
                different_char_count += 1
        # END FOR loop
        # Check the number of characters that are different.
        return True if different_char_count <= 2 and sorted(s1) == sorted(s2) else False
