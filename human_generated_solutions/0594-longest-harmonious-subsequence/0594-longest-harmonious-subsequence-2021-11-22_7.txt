class Solution:
    def findLHS(self, nums: List[int]) -> int:
       
        if not nums or len(nums) == 1: 
            return 0
        
        ##2 pointer    
        nums.sort()
        p1,p2 = 0,1
        res = 0
        while p1<len(nums) and p2<len(nums):

            if nums[p2]- nums[p1] == 0:
                p2 += 1
            elif nums[p2]- nums[p1] ==1:
                res = max(res, p2-p1+1)
                p2 += 1
            else:
                p1 += 1
        return res
