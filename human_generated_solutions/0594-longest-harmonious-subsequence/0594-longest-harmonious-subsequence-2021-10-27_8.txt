class Solution:
    def findLHS(self, nums: List[int]) -> int:
        d={}
        for i in nums:
            if i not in d:
                d[i]=0
            d[i]+=1
        
        ans=0
        for i,j in d.items():
            if i+1 in d:
                ans=max(ans,d[i]+d[i+1])
            if i-1 in d:
                ans=max(ans,d[i]+d[i-1])
            
        return ans
