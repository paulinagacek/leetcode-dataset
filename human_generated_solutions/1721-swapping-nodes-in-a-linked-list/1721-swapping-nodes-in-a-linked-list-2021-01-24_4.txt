class Solution:
    def swapNodes(self, head, k):
        
        n  =  head
        for _ in range(k-1):
            n  =  n.next
        
        a  =  n
        b  =  head
        
        while n.next:
            b  =  b.next
            n  =  n.next
        
        a.val, b.val  =  b.val, a.val
        
        return head
