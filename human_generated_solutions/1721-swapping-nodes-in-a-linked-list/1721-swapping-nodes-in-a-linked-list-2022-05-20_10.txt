# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapNodes(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        # searching kth node
        first=head
        for i in range(k-1):
            first=first.next
        
        #searching kth node form the end
        last=head
        fast=first.next
        while fast:
            last=last.next
            fast=fast.next
        
        #swapping their values
        last.val,first.val=first.val,last.val

        return head
