class Solution:
    def summaryRanges(self, nums: List[int]) -> List[str]:
        ans, n = [], len(nums)
        
        start = 0
        for i in range(1,n+1):
            if i == n or nums[i] != nums[i-1]+1:
                ans.append(self.helper(nums[start],nums[i-1]))
                start = i
        return ans
    
    def helper(self, a, b):
        return str(a) if a==b else str(a)+"->"+str(b)
