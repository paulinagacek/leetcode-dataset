class Solution:
    def validMountainArray(self, arr: List[int]) -> bool:
        if len(arr) < 3: return False; # mountain array cant exist
        l = 0; # one pointer at beginning
        r = len(arr) - 1 # one pointer at the end
        while l + 1 < len(arr) - 1 and arr[l] < arr[l + 1]:  # until l pointer doesnt reach end and the element at l < element at l+1
            l += 1 # increment l
        while r - 1 > 0 and arr[r] < arr[r - 1]: # same logic but reverse way
            r -= 1 # decrement r
        return l == r # check if l == r, if yes then its mountain array
        # note : if l and r cross each other then not mountain array
