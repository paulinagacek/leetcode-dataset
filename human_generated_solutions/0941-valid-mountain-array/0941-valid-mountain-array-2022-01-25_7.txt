class Solution(object):
    def validMountainArray(self, arr):
        """
        :type arr: List[int]
        :rtype: bool
        """
        index=len(arr)
        i=0
        for i in range(0,index-1): #index-1 to avoid index out of bounds
            if arr[i+1]>arr[i]:#strictly increasing
                continue
            else:
                break
        if i==0 or i==index-1:
            return False
        for j in range(index-1,i,-1):
            if arr[j-1]>arr[j]: #strictly decreasing
                continue
            else:
                return False
        return True #control comes here only if both the for loops are traversed.
