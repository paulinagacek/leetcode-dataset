class Solution:
    def bagOfTokensScore(self, tokens: List[int], power: int) -> int:
        N, score, max_score = len(tokens), 0, 0
        tokens.sort()
        left, right = 0, N - 1
        while left <= right:
            if power >= tokens[left]:
                score += 1
                power -= tokens[left]
                left += 1
                max_score = max(max_score, score)
            elif score >= 1:
                score -= 1
                power += tokens[right]
                right -= 1
            else:
                break
        return max_score
