class Solution:
    def newBST(self, root, sumSoFar):
        if root == None:
            return sumSoFar
        
        newSum = self.newBST(root.right, sumSoFar)
        root.val = root.val + newSum
        return self.newBST(root.left, root.val)
        
    
    def convertBST(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        self.newBST(root,0)
        return root
