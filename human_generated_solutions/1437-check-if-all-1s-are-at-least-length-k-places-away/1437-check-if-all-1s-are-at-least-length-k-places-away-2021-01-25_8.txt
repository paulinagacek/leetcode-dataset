w: two pointer + greedy
h: we only need to check the distance between previous 1 and current

class Solution:
    def kLengthApart(self, nums: List[int], k: int) -> bool:
        left = 0
        flag = 0
           
        res = float(inf)
        
        for idx, num in enumerate(nums):
            if num == 1:
                if flag == 0: # first time we meet 1
                    flag = 1
                    left = idx
                else:
                    dist = idx - left - 1
                    left = idx
                    res = min(res, dist)
                    if res < k:
                        return False
        
        
        return True
