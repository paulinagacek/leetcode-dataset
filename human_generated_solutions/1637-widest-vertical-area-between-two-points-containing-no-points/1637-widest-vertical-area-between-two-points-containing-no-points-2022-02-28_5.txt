class Solution:
    def maxWidthOfVerticalArea(self, points: List[List[int]]) -> int:
        
        widest = 0
        
        points.sort()
        
        for i in range(len(points) - 1):
            if points[i+1][0] - points[i][0] > widest:
                widest = points[i+1][0] - points[i][0]
                
        return widest
