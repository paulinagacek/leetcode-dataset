class Solution:
    def findRestaurant(self, list1: List[str], list2: List[str]) -> List[str]:
        n, m = len(list1), len(list2)
        list1Hash = {restaurant: i for i, restaurant in enumerate(list1)}
        
        minIndexSum = n + m
        for j, restaurant in enumerate(list2):
            i = list1Hash.get(restaurant, n)
            if i < n and i + j < minIndexSum:
                minIndexSum = i + j
                minIndexRestaurants = [restaurant]
            elif i + j == minIndexSum:
                minIndexRestaurants.append(restaurant)
                
        return minIndexRestaurants
