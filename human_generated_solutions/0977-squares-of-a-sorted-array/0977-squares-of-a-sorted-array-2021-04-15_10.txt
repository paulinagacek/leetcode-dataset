class Solution(object):
    def sortedSquares(self, nums):
        stack = []
        output = []
        for i in range(len(nums)):
            if nums[i] < 0:
                stack.append(-nums[i])
            else:
                while stack and stack[-1] < nums[i]:
                    output.append(stack.pop() ** 2)
                output.append(nums[i] ** 2)
        while stack:
            output.append(stack.pop() ** 2)
        return output
