class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        """
        [3,2,2,3],val = 3
        
        [2,2,3,3]
        """
        e = 0
        for i in range(len(nums)):
            if nums[i] != val:
                nums[e], nums[i] = nums[i],nums[e]
                e += 1
        return e
