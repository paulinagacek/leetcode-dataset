class Solution:
    def numSubarraysWithSum(self, nums: List[int], goal: int) -> int:
        def atMostGoal(goal):
            if goal < 0:
                return 0
            res, l, s = 0, 0, 0
            for r in range(len(nums)):
                s += nums[r]
                while s > goal:
                    s -= nums[l]
                    l += 1
                res += r - l + 1
            return res
        return atMostGoal(goal) - atMostGoal(goal - 1)
