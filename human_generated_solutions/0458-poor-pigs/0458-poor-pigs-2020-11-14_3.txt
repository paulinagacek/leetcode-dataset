# (Time, Space) Complexity : O(1), O(1)
#
# 1. Pig is a bit, x pigs mean x bits
# 2. Round is the highest value in a bit, q round means q+1 carry to next bit
# 3. There are n buckets. There are p//m rounds
# 4. Now we have a pig register! It have to record number n.
#    (q+1)^x >= n
#    x*log(q+1) >= log(n)
#    x >= log(n) / log(p//m+1)
#    minimum x = ceil(log(n) / log(p//m+1))
class Solution:
    def poorPigs(self, buckets: int, minutesToDie: int, minutesToTest: int) -> int:
        return ceil(log(buckets) / log(minutesToTest//minutesToDie + 1))
