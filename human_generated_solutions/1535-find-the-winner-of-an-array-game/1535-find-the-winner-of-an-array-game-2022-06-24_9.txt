class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        n = len(arr)
        winningInteger = arr[0]
        winCount = 0
        for i in range(1, n):
            if winningInteger > arr[i]:
                winCount += 1
            else:
                winningInteger = arr[i]
                winCount = 1
            if winCount == k:
                return winningInteger
        return winningInteger
