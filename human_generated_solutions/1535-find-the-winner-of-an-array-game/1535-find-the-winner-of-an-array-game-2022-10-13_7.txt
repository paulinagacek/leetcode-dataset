from collections import deque

class Solution:
    def getWinner(self, arr: List[int], k: int) -> int:
        number = arr[0]
        streak = 0

        for i in range(1, len(arr)):
            if arr[i] > number:
                streak = 1
                number = arr[i]
            else:
                streak += 1
            
            if streak == k:
                return number
        
        return number
