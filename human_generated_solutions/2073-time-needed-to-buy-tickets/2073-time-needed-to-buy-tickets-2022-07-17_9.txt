class Solution(object):
    def timeRequiredToBuy(self, tickets, k):
        """
        :type tickets: List[int]
        :type k: int
        :rtype: int
        """
        seconds = 0
        while tickets[k]!=0:
            
            for i in range(len(tickets)):
                if tickets[i]!=0 and tickets[k]!=0:
                    tickets[i] = tickets[i]-1
                    seconds +=1
        return seconds
