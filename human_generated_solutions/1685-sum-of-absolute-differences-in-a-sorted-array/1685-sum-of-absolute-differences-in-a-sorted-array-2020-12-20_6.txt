class Solution:
    def getSumAbsoluteDifferences(self, nums: List[int]) -> List[int]:
        n = len(nums)
        s = sum(nums)
        
        result = [0] * n        
        left  = 0
        for i, num in enumerate(nums):
            right = s - left - num
            result[i] = -left + right + num * (2*i - n + 1)
            left += num
            
        return result
