from collections import Counter
class Solution:
    def minSetSize(self, arr: List[int]) -> int:
        dic=Counter(arr)
        target=len(arr)//2
        sum=0
        c=1
        values=sorted(list(dic.values()),reverse=True)
        for value in values:
            if sum+value >=target:
                return c
            else:
                c+=1
                sum+=value
        return c
