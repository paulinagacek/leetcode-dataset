class Solution(object):
    def minSetSize(self, arr):
        result = 0
        current_number_of_removed_items = 0
        half_array_size = len(arr) / 2
        numbers_counter = Counter(arr)
        
        for number, count in numbers_counter.most_common():
            result += 1
            current_number_of_removed_items += count
            
            if current_number_of_removed_items >= half_array_size:
                return result
