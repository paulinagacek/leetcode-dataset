class Solution:
    
    def isContiguousPath(self, h:Optional[ListNode], r:Optional[TreeNode]) ->bool:
        if not h: return True
        if not r: return False
        if h.val == r.val:
            return self.isContiguousPath(h.next, r.left) or \\
                    self.isContiguousPath(h.next, r.right)
            
        return False


    def isSubPath(self, h: Optional[ListNode], r: Optional[TreeNode]) -> bool:
        if not h: return True
        if not r: return False
        if h.val == r.val and self.isContiguousPath(h, r): return True
                
        return self.isSubPath(h, r.left) or \\
                self.isSubPath(h, r.right)
