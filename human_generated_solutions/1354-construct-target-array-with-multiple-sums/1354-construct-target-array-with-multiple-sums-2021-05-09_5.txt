class Solution:
    def isPossible(self, target: List[int]) -> bool:
        if len(target) <=1:
            return target[0] == 1
        summ = sum(target)
        maxNum = max(target)
        maxIndex = target.index(maxNum)
        
        while maxNum > 1:
            remainingSum = summ - maxNum
#           use modulo instead of minus to repeadly substract from the large number
            numToReplace = maxNum % remainingSum
            if remainingSum == 1:   # happens when list has two elemnents
                return True
#             return False if remaining sum can\'t be substracted from largest number or number to be replaced is zero
            if maxNum < remainingSum or numToReplace == 0:
                return False
            target[maxIndex] = numToReplace #replace the number by new number
            summ = remainingSum + numToReplace #calculate the new summ after changing the maxNum 
            maxNum = max(target) 
            maxIndex = target.index(maxNum)
        return True
