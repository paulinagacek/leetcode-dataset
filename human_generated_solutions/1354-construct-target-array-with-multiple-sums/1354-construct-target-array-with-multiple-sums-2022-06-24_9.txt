class Solution:
    def isPossible(self, target: List[int]) -> bool:
        max_heap = [-n for n in target]
        heapq.heapify(max_heap)
        total = sum(target)
        while max_heap[0] != -1:
            largest = -heapq.heappop(max_heap)
            if total >= 2 * largest or total == largest:
                return False
            largest_ele_in_prev_step = largest % (total - largest)
            if not largest_ele_in_prev_step:
                largest_ele_in_prev_step = total - largest
            
            total = total - largest + largest_ele_in_prev_step
            heapq.heappush(max_heap, -largest_ele_in_prev_step)
        return True
