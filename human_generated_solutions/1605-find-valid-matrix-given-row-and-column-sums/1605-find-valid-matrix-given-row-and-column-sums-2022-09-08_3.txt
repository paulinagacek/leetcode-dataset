class Solution:
    def restoreMatrix(self, rowSum: List[int], colSum: List[int]) -> List[List[int]]:
        n = len(rowSum)
        m = len(colSum)
        
        ans = [[0 for j in range(m)] for i in range(n)]
        
        for i in range(n):
            for j in range(m):
                val = min(rowSum[i], colSum[j])
                rowSum[i] -= val
                colSum[j] -= val
                ans[i][j] = val
        
        return ans
