class Solution:
    def chalkReplacer(self, chalk: List[int], k: int) -> int:
        """
        k % prefix evaluates to first value smaller than prefix we will reach when we 
        keep doing k -= prefix * nice math trick *

        """
        chalks = k
        i = 0
        n = len(chalk)
        if n == 1:
            return 0
        prefix = sum(chalk)
        k = k % prefix
        for i in range(n):
            if chalk[i] > k:
                return i
            k -= chalk[i]
