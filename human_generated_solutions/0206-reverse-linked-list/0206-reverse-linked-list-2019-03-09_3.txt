# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
	def reverseList_i(self, head: ListNode) -> ListNode:  # iterative
	# O(n) Time / O(1) Space
		prev = None
		cur = head
		while cur:
			tmp_next = cur.next
			cur.next = prev
			prev = cur
			cur = tmp_next
		return prev
    
    def reverseList_r(self, head: ListNode) -> ListNode: # recursive
	# O(n) Time / O(n) Space
        if not head or not head.next: return head
        p = self.reverseList(head.next)
        head.next.next = head
        head.next = None
        return p
