class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head==None or head.next==None: return head
        prev,curr,temp=head,head.next,head.next.next
        prev.next=None
        while temp:
            curr.next=prev
            prev=curr
            curr=temp
            temp=temp.next
        curr.next=prev
        return curr
