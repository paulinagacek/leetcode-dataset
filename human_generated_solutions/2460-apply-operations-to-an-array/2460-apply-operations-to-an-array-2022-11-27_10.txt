class Solution:
    def applyOperations(self, nums: List[int]) -> List[int]:
        lst = []
        for i in range(len(nums) - 1):
            if nums[i] == nums[i+1]:
                nums[i] *= 2
                nums[i+1] = 0
        zeroCount = nums.count(0)
        lst = [x for x in nums if x != 0]
        lst += [0]*zeroCount
        return lst
