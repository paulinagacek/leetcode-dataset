class Solution:
    def carPooling(self, trips: List[List[int]], capacity: int) -> bool:
        dp = [0]*1000
        for trip in trips:
            dp[trip[1]] += trip[0];
            dp[trip[2]] -= trip[0];
            
        for stop in dp:
            capacity -= stop
            if capacity < 0:
                return False;
        return True
