class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        l = cur = best = 0
        for r in range(len(s)):
            cur += abs(ord(s[r]) - ord(t[r]))
            while cur > maxCost:
                cur -= abs(ord(s[l]) - ord(t[l]))
                l += 1
            best = max(best, r - l + 1)
        return best
