class Solution:
    def equalSubstring(self, s: str, t: str, maxCost: int) -> int:
        \'\'\'
        # Problem :
        - Max Length
        - Substring
        - Cost <= maxCost
        
        They are all sign of using a sliding window!
        
        # Idea:
        - Expand the window till curSum <= maxCost : curCost += |ord(s[i]) - ord(t[i])|
        - Shrink the window if curSum > maxCost 
        \'\'\'
        start,end =0,0
        out = float(\'-inf\')
        curCost =0
        while end < len(s):
            curCost += abs(ord(s[end]) - ord(t[end]))
            while curCost > maxCost:
                    curCost -= abs(ord(s[start]) - ord(t[start]))
                    start +=1
            out = max(out, end-start+1)
            end +=1
            
        return out
