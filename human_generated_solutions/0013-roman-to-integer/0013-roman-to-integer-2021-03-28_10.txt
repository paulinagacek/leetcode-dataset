def romanToInt(self, s: str) -> int:
      values = {\'I\': 1, \'V\': 5, \'X\': 10, \'L\': 50, \'C\': 100, \'D\': 500, \'M\': 1000}
      prev, ans = -1, 0
      for c in s:
          curr = values[c]
          if prev != -1 and curr > prev:
              ans -= prev
              ans += curr - prev
          else:
              ans += curr
          prev = curr
      return ans
