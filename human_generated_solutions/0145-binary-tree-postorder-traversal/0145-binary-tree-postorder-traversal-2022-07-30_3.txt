class Solution:
    def postorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        if not root:
            return []
        
        s = [root]
        res = []        
        
        while s:
            cur = s[-1]
            if not cur.left and not cur.right:
                s.pop()
                res.append(cur.val)

            if cur.right:
                s.append(cur.right)
                cur.right = None
            
            if cur.left:
                s.append(cur.left)
                cur.left = None
                
        return res
