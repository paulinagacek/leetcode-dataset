class Solution:
    def minOperations(self, n: int) -> int:
        # if there is only one element in list then its already equal so no need to perform any operation
        if n==1:return 0
        else:
            a=1 # First value
            b=2*(n-1)+1 # Last value
            avg=(b+1)//2  # Taking average
            res=0 # Counter for storing answer
            for i in range(n//2,n): # Loop for 2nd half portion of the array
                res+=abs((2*i+1)-avg)   # Taking the difference with average value and adding to the result
            return res
