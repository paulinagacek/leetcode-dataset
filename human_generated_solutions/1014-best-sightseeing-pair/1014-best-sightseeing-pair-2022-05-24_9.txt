class Solution:
    def maxScoreSightseeingPair(self, values: List[int]) -> int:
        max_score = 0
        
        decayed_max_so_far = values[0]
        
        for i in range(1, len(values)):
            decayed_max_so_far -= 1
            max_score = max(max_score, decayed_max_so_far + values[i])
            decayed_max_so_far = max(decayed_max_so_far, values[i])
            
        return max_score
