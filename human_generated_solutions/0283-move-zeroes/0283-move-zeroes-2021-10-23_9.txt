class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        n = len(nums)
        lastZeroPosition = n - 1
        for i in range(n-1, -1, -1):
            if nums[i] == 0:
                for j in range(i, lastZeroPosition):
                    nums[j], nums[j+1] = nums[j+1], nums[j]
                lastZeroPosition -= 1
