def moveZeroes(self, nums: List[int]) -> None:
        """
        Do not return anything, modify nums in-place instead.
        """
        count = 0
        length = len(nums)
        for i in range(length):
            if nums[i] != 0:
                nums[count] = nums[i]
                count += 1
        if count:
            for i in range(count, length):
                nums[i] = 0
