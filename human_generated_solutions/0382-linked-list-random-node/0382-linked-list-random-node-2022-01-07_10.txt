import random

class Solution:

	\'\'\'
	When the class gets intialized, we simply put all the nodes from the linked list 
	into an array (list), then generate a random index from 0 - length of array - 1 
	everytime the getRandom function is called, and return the value of the node at 
	that index.
	\'\'\'


    def __init__(self, head: Optional[ListNode]):
        
        self.arr_list = []
        self.limit = 0
        
        temp = head
        while temp:
            self.arr_list.append(temp)
            temp = temp.next
            self.limit += 1
        
        
    def getRandom(self) -> int:
        idx = random.randint(0, self.limit - 1)
        return self.arr_list[idx].val
