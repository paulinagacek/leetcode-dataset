def preorderTraversal(self, root: Optional[TreeNode]) -> List[int]:
        ans=[]
        if root==None:
            return None
        stack=[root]
        while len(stack)>0:
            temp=stack.pop(-1)
            ans.append(temp.val)
            if temp.right!=None:
                stack.append(temp.right)
            if temp.left!=None:
                stack.append(temp.left)
        return ans
