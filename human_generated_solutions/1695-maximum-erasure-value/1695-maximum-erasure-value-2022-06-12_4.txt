class Solution:
    def maximumUniqueSubarray(self, nums: List[int]) -> int:

        left, d = [0], defaultdict(int)
        pref, ans = 0, 1

        for n in nums:
            pref.append(pref[-1]+n)

        for idx, n in enumerate(nums):
            if d[n] > left: left = d[n]
            d[n] = idx+1

            ans = max(ans, pref[idx+1] - pref[left])

        return ans
