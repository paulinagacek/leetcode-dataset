class Solution:
    def reorderList(self, head: Optional[ListNode]) -> None:
        """
        Do not return anything, modify head in-place instead.
        """

        if not head: return
        
        # Find the middle node
        slow = fast = head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        
        # Reverse the second half
        prev, curr = None, slow
        while curr:
            temp = curr.next
            curr.next = prev
            prev = curr
            curr = temp
        
        # Merge the two sorts
        first, second = head, prev
        while second.next:
            temp1 = first.next
            first.next = second
            first = temp1
            
            temp2 = second.next
            second.next = first
            second = temp2
