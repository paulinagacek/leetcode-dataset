class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:

        mx = [nums[-1]]*len(nums)
        for i in range(len(nums)-2, 1, -1):
            mx[i] = max(mx[i+1], nums[i])
            
        mn = nums[0]
        for i in range(1, len(nums)-1):
            if mn < nums[i] < mx[i+1]: return True
            if nums[i] < mn: mn = nums[i]

        return False
