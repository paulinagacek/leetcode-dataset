class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        if len(nums) < 3:
            return False
        
        num1, num2 = nums[0], float(\'inf\')
        global_min = nums[0]
        
        for cur in nums[1:]:
            if num2 < cur:
                return True
            if cur < global_min:
                global_min = cur
            elif global_min < cur < num2:
                num1, num2 = global_min, cur
            elif num1 < cur < num2:
                num2 = cur
                
        return False
