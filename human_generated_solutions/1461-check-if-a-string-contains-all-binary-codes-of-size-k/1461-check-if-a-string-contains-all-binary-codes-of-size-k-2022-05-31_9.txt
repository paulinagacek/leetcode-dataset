def hasAllCodes(self, s: str, k: int) -> bool:
        d = {}
        i = 0
        n = len(s)
        
        while i+k<=n:
            if s[i:i+k] not in d:
                d[s[i:i+k]] = 1
            else:
                d[s[i:i+k]] += 1
            i += 1
        
        if len(d) == 2 ** k:
            return True
        else:
            return False
