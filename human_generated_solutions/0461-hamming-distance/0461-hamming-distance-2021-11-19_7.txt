class Solution:
    def hammingDistance(self, x: int, y: int) -> int:
        
        x_bin = format(x, \'0b\')
        y_bin = format(y, \'0b\')
        
        longest_bin = max(len(x_bin), len(y_bin))
        
        x_bin = x_bin.zfill(longest_bin)
        y_bin = y_bin.zfill(longest_bin)
        
        output = 0
        
        for a, b in zip(x_bin, y_bin):
            if a is not b:
                output += 1
        
        return output
