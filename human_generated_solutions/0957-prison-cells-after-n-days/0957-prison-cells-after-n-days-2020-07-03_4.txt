class Solution:
    def prisonAfterNDays(self, cells: List[int], N: int) -> List[int]:
        states = {}
        for i in range(N):
            tmp = [0] + [cells[i-1] ^ cells[i+1] ^ 1 for i in range(1, len(cells)-1)] + [0] 
            if (t := tuple(tmp)) in states:
                break
            states[t] = i
            cells[:] = tmp
        return sorted(states.keys(), key=states.get)[(N % len(states))-1]
