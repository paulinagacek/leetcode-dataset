class Solution:
    def frequencySort(self, s: str) -> str:
        from collections import Counter,deque
        dic = Counter(s)
        pq=[]
        result = deque([])
        # # Heap Sort (Min Heap)
        for char, freq in dic.items():
            heapq.heappush(pq, (freq,char))
        # Pop the char * freq to deque, and return reversed deque
        for i in range(len(pq)):
            char, freq = heapq.heappop(pq)
            result.extend(char*freq)
        
        result.reverse()
        return "".join(result)
