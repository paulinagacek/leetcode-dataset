class Solution:
    def removeDuplicates(self, s: str, k: int) -> str:
        stack = []
        for c in s:
            if not stack or c != stack[-1][0]:
                stack.append([c, 1])
            elif stack[-1][1] < k - 1:
                stack[-1][1] += 1
            else:
                stack.pop()
                
        result = []
        for c, count in stack:
            result += [c] * count

        return "".join(result)
