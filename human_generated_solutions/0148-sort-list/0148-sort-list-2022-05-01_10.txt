class Solution:
    def _findMid(self, head):  # return mid and its prev
        slow = fast = head
        prev = None
        while fast and fast.next:
            prev = slow
            slow = slow.next
            fast = fast.next.next
        return prev, slow
    
    def _merge2SortedList(self, l1, l2):
        cur = dummyHead = ListNode()
        while l1 and l2:
            if l1.val <= l2.val:
                cur.next = l1
                l1 = l1.next
            else:
                cur.next = l2
                l2 = l2.next
            cur = cur.next
        cur.next = l1 or l2
        return dummyHead.next
    
    def sortList(self, head: ListNode) -> ListNode:
        if head == None or head.next == None:
            return head
        
        prevMid, mid = self._findMid(head)
        prevMid.next = None
        
        left = self.sortList(head)
        right = self.sortList(mid)
        return self._merge2SortedList(left, right)
