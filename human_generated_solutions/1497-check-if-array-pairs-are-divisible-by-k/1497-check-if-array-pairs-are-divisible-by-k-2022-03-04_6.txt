class Solution:
    def canArrange(self, arr: List[int], k: int) -> bool:
        m = defaultdict(int)
        for n in arr:
            r = n % k
            t = k - r if r > 0 else 0
            if m[t] > 0:
                m[t] -= 1
            else:
                m[r] += 1
        return {v for v in m.values()} == {0}
