class Solution:
    def minCameraCover(self, root: TreeNode) -> int:
        ans = [0]
        #ans stores the count of cameras
        def dfs(root,ans):
            if root == None:
                return None
            l = dfs(root.left,ans)
            r = dfs(root.right,ans)
            if l == None and r == None:
				#the current node is a leaf
                return "leaf"
            if l=="leaf" or r == "leaf":
				#if any one of its child is a leaf than increment answer by 1 
                ans[0]+=1
                return "parent_also_covered"
            if l == "parent_also_covered" or r == "parent_also_covered":
				#if the node is covered by its child
                return None
        a = dfs(root,ans)
        if a == "leaf":
            ans[0]+=1
        return ans[0]
