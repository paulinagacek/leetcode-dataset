class Solution:
    def sumRootToLeaf(self, root: TreeNode) -> int:
        
        def calc(node, temp):
            if not node:
                return 0
            
            temp = temp*2 + node.val
            if (node.left is None) and (node.right is None):
                return temp
            
            return (calc(node.left, temp) + calc(node.right, temp))
        
        return calc(root, 0)
