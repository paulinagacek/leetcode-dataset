class Solution:
    def sumRootToLeaf(self, root: Optional[TreeNode]) -> int:
        self.ans = 0
        def dfs(node, soFar):
            soFar = soFar * 2 + node.val
            if not node.left and not node.right: self.ans += soFar
            if node.left: dfs(node.left, soFar)
            if node.right: dfs(node.right, soFar)
        
        dfs(root, 0)
        return self.ans
