class Solution:
    def numSub(self, s: str) -> int:
        x=[]
        n=len(s)
        i=0
        while i<n:
            c=1
            while i+1<n and s[i]==s[i+1]:
                c+=1
                i+=1
            if s[i]=="1":
                x.append("1"*c)
            i+=1
        c=0
        for i in x:
            n=len(i)
            c+=n*(n+1)//2
        return c%(10**9 + 7)
