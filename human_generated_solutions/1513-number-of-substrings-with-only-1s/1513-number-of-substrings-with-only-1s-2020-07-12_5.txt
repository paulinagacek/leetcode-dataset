class Solution:
    def numSub(self, s: str) -> int:
        res, curr = 0, 0
        for i in range(len(s)):
            if s[i] == \'1\':
                curr += 1
            else:
                res += (curr * (curr + 1))//2
                curr = 0
        if curr > 0:
            res += (curr * (curr + 1))//2
        return res % 1000000007
