class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        ans=[]
        for i in range(1,10):
            temp=i
            for j in range(i+1,10):
                temp=10*temp+j
                if temp in range(low,high+1):
                    ans.append(temp)
        ans.sort()
        return ans
