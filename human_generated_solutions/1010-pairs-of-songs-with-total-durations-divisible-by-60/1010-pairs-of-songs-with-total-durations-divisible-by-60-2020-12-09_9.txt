from collections import Counter
class Solution:
    def numPairsDivisibleBy60(self, time: List[int]) -> int:
        
		# note each song <= 500, total <= 1000
        totals = list(range(60, 1001, 60))
        
        counts = Counter()
        res = 0
        for t in time: 
            for total in totals:
                diff = total - t
                if diff > 0 and diff in counts: 
                    res += counts[diff]
                
            counts[t] += 1
            
        return res
