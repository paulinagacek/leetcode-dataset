# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def mergeNodes(self, head):
        """
        :type head: Optional[ListNode]
        :rtype: Optional[ListNode]
        """
        head1 = head
        head2 = head.next
        result = 0
        while head2:
            if head2.val == 0:
                head1 = head1.next
                head1.val = result
                result = 0
            else:
                result += head2.val
            head2 = head2.next
        head1.next=None
        return head.next
