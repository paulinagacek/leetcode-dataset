from collections import defaultdict

class Solution:
    def minSteps(self, s, t):
        s_freq, t_freq, steps = defaultdict(int), defaultdict(int), 0
        for c in s:
            s_freq[c] += 1
        for c in t:
            t_freq[c] += 1
        for l in s_freq:
            s_i = s_freq[l] - t_freq[l]
            if s_i > 0:
                steps += s_i
        return steps
