class Solution:
    def isUnivalTree(self, root: TreeNode) -> bool:
        if not root:
            return True
        
        currval = root.val
        if root.right and currval != root.right.val :
            return False
        elif root.left and currval != root.left.val :
            return False
        
        return self.isUnivalTree(root.right) and self.isUnivalTree(root.left)
