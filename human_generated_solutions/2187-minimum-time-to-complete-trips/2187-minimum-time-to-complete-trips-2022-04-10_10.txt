class Solution:
    def minimumTime(self, time: List[int], totalTrips: int) -> int:
        hi = totalTrips * min(time)
        lo=1

        def f(x):
            sm=0
            for z in time:
                sm+=(x//z)
            return sm>=totalTrips
    
        while lo < hi:
            mid = (lo + hi) // 2
            if not f(mid): 
                lo = mid + 1
            else: hi = mid
        return lo
