class NumArray:

    def __init__(self, nums: List[int]):
        accu_sum = 0
        self._pre_sum = [0,]
        for num in nums:
            accu_sum += num
            self._pre_sum.append(accu_sum)
            
    def sumRange(self, left: int, right: int) -> int:
        return self._pre_sum[right+1] - self._pre_sum[left]
