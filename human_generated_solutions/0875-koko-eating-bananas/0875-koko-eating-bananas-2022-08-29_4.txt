class Solution:
    def totalHoursTaken(self, piles, k):
        totalHours = 0
        for pile in piles:
            totalHours += math.ceil(pile / k)
        return totalHours
        
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        l, r = 1, max(piles)
        while l < r:
            mid = (l + r) // 2
            if self.totalHoursTaken(piles, mid) > h:
                l = mid + 1
            else:
                r = mid
        
        return l
