class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        l,r = 1, max(piles)
        while l<r:
            mid = (l+r)//2
            required = 0
            for i in piles:
                required+=ceil(i/mid)
            if h>=required:
                r = mid
            else:
                l = mid+1
        return l
