class Solution:
    def minEatingSpeed(self, piles: List[int], h: int) -> int:
        # T: O(n log (max(piles))), S: O(1)
        l, r = 1, max(piles)
        while l < r:
            k = (l + r) >> 1
            time_left = h
            
            for p in piles:
                time_left -= ceil(p / k)
                
            if time_left >= 0:
                r = k
            else:
                l = k + 1
        return l
