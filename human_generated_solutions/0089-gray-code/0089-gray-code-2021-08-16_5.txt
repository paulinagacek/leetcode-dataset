class Solution:
    def grayCode(self, n: int) -> List[int]:
        
        k = 2**n-1
        lt = [0]
        x=0
        while(x<n):
            p = len(lt)
            two = 2**x
            for i in range(p-1, -1, -1): #Reverse the output array of n-1
                lt.append(two | lt[i])
            x+=1
        return lt
