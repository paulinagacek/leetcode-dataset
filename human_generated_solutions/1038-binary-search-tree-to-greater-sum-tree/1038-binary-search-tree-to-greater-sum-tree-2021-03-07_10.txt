# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def __init__(self):
        self.values = []
        
    def bstToGst(self, root):
        self.getValues(root)
        self.updateValues(len(self.values) - 2)
        self.index = len(self.values) - 1
        self.updateTree( root)
        return root
            
    def getValues(self, root):
        if root:
            self.getValues(root.left)
            self.values.append(root.val)
            self.getValues(root.right)
            
    def updateValues(self,index):
        if index >= 0:
            self.values[index] += self.values[index + 1]
            self.updateValues(index - 1)
            
    def updateTree(self, root):
        if root:
            self.updateTree(root.right)
            root.val = self.values[ self.index ]
            self.index -= 1
            self.updateTree(root.left)
