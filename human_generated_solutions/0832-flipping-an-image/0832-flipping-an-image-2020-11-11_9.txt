def flipAndInvertImage(A):
    if len(A) == 0: return []
    flipped = []
    for row in A:
        flipped.append(row[::-1])
    invert = []
    for row in flipped:
        curr = []
        for element in row:
            if element == 1:
                curr.append(0)
            else:
                curr.append(1)
        invert.append(curr)
    return invert
