class Solution:
    def flipAndInvertImage(self, image: List[List[int]]) -> List[List[int]]:
        
        def reverse(listt, left, right):
            
            while left < right:                
                listt[left], listt[right] = listt[right], listt[left]
                
                left += 1
                right -= 1
        
        
        # 1. reverse every list in image
        for listt in image:
            reverse(listt, 0, len(listt) - 1)
        
        # 2. invert every elements in the matrix
        for i in range(len(image)):
            for j in range(len(image[0])):
                
                if image[i][j] == 0:
                    image[i][j] = 1
                    
                elif image[i][j] == 1:
                    image[i][j] = 0
        
        return image
