class Solution:
    def superEggDrop(self, egg: int, floor: int) -> int:
        dp = [[0]*(egg+1) for i in range(floor+1)]
        for i in range(1 , floor+1):
            for j in range(1 , egg + 1):
                dp[i][j] =  1 + dp[i-1][j] + dp[i-1][j-1]
                if dp[i][j] >= floor:
                    return i
