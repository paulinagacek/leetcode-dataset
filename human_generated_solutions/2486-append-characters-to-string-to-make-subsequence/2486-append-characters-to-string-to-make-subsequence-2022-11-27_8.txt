class Solution:
    def appendCharacters(self, s: str, t: str) -> int:
        count = 0
        i=j=0
        
        while i<len(s) and j<len(t):
            if s[i]==t[j]:
                count += 1
                i += 1
                j += 1
            else:
                i += 1
                
        if j==len(t):
            return 0
        elif i==len(s):
            return len(t)-count
