from collections import defaultdict


class Customer:
    
    def __init__(self, id, stationName, t):
        """
        A customer can only be checked into one place at a time
        That means we don\'t need to keep track of whether a customer
        checked in a similar station before.
        
        We are expecting one recent entry for each (customer, checkInStation, checkInTime) pair
        
        """
        self.id = id
        self.checkInStation = stationName
        self.checkInTime = t


class UndergroundSystem:

    def __init__(self):
        # keep a customer profile
        self.customer = defaultdict(list)
        
        # keep a station profile
        # where key is station pair names and value is [travel_time, number_of_travel]
        self.station_pair = defaultdict(list)

    def checkIn(self, id: int, stationName: str, t: int) -> None:
        if id not in self.customer:
            self.customer[id] = Customer(id, stationName, t)
        else:
            # update the checkin station and time
            customer = self.customer[id]
            customer.checkInStation = stationName
            customer.checkInTime = t

    def checkOut(self, id: int, stationName: str, t: int) -> None:
        customer = self.customer[id]
        checkInStation = customer.checkInStation
        checkInTime = customer.checkInTime
        
        checkOutStation, checkOutTime = stationName, t
        
        if (checkInStation, checkOutStation) not in self.station_pair:
            self.station_pair[(checkInStation, checkOutStation)] = [checkOutTime - checkInTime, 1]
        
        else:
            # already travelled by someone before
            # update the entry
            self.station_pair[(checkInStation, checkOutStation)][0] += checkOutTime - checkInTime
            # increase the number of travel
            self.station_pair[(checkInStation, checkOutStation)][1] += 1
    
    def getAverageTime(self, startStation: str, endStation: str) -> float:
        
        total_time, total_num = self.station_pair[(startStation, endStation)]
        return total_time / total_num
