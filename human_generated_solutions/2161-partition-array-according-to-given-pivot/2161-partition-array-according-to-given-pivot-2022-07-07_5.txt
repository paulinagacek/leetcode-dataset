class Solution:
    def pivotArray(self, nums: List[int], pivot: int) -> List[int]:
        l = []
        r = []
        for num in nums:
            if num < pivot:
                l.append(num)
            elif num > pivot:
                r.append(num)
		# add pivot
        l.extend([pivot] * (len(nums) - len(l) - len(r)))
        l.extend(r)
        return l
