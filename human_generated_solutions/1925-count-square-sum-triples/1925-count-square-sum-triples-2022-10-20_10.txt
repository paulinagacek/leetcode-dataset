def countTriples(self, n: int) -> int:
        count = 0
        for a in range(1, n-1):
            for b in range(a + 1, n):
                root = sqrt(a ** 2 + b ** 2)
                if root <= n and int(root) == root:
                    count += 2
                if root > n:
                    break
        return count
