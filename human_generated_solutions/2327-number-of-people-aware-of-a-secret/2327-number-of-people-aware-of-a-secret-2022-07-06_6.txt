class Solution:
    def peopleAwareOfSecret(self, n: int, delay: int, forget: int) -> int:
        modulus = 10 ** 9 + 7
        discovered = deque([0] * delay)
        know = discovered[0] = 1
        sharing = 0

        for day in range(delay, forget):
            sharing += discovered[-delay]
            discovered.append(sharing)
            know += sharing

        for day in range(forget, n):
            sharing += discovered[-delay] - discovered[0]
            discovered.append(sharing)
            know += sharing - discovered.popleft()
            
        return know % modulus
