MODULO = 10**9 + 7

def people_aware_of_secret(n: int, delay: int, forget: int) -> int:
    aware    = 1; share = 0
    d_change = collections.Counter({delay:  1})
    f_change = collections.Counter({forget: 1})
    for day in range(1, n):
        share += d_change[day] - f_change[day]
        aware += share - f_change[day]
        d_change[day + delay]  += share
        f_change[day + forget] += share
    return aware % MODULO
