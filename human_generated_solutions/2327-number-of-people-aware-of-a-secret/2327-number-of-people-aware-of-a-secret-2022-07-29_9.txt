def peopleAwareOfSecret(self, n: int, delay: int, forget: int) -> int:
        BIG_NUM = 10**9 + 7
        last = [1]+[0]*(forget-1)
        
        for _ in range(1,n):
            cur = [sum(last[delay-1:forget-1]) % BIG_MOD]+last[:-1]
            last = cur
            
        return sum(cur) % BIG_NUM
