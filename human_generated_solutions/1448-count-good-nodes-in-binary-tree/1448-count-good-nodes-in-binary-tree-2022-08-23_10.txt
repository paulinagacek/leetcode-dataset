class Solution:
    def goodNodes(self, root: TreeNode) -> int:
        self.res = 0
        def dfs(root, max_val):
            if root is None:
                return 0
            
            if root.val >= max_val:
                self.res += 1
                max_val = root.val
            
            dfs(root.left, max_val)
            dfs(root.right, max_val)
        
        dfs(root, float("-inf"))
        return self.res
