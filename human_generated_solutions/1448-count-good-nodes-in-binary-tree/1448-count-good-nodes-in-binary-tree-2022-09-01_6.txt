class Solution:
    def goodNodes(self, root: TreeNode) -> int:
        maxx = root.val
        count = 0
        def dfs(node,maxx):
            nonlocal count
            if node is None:
                return
            if node.val >= maxx:
                count +=1
                maxx = node.val
            dfs(node.left,maxx)
            dfs(node.right,maxx)
            return
        dfs(root,maxx)
        return count
