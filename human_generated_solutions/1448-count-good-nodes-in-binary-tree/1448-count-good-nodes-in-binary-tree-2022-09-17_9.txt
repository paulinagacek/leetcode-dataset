def goodNodes(self, root: TreeNode) -> int:
	stack = [(root, root.val)]
	
	# Initial count with root
	count = 1

	while stack:
		node, good = stack.pop()
		if node.right:
			if node.right.val >= good:
				count += 1
			stack.append((node.right, max(good, node.right.val)))
		if node.left:
			if node.left.val >= good:
				count += 1
			stack.append((node.left, max(good, node.left.val)))

	return count
