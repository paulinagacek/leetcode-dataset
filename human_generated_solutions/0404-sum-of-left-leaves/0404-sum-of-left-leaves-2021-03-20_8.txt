class Solution:
    def sumOfLeftLeaves(self, root: TreeNode) -> int:
        #
        ## bfs
        if not root or (not root.left and not root.right): return 0
        que = deque([(root,0)])
        res = 0
        while que:
            node,flag = que.popleft()
            if flag and not (node.left or node.right):
                res += node.val
            else:
                if node.left:
                    que.append((node.left,1))
                if node.right:
                    que.append((node.right,0))
        return res
