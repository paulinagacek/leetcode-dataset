class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        myDict = {}
        maxLength = 0
        windowStart = 0
        for windowEnd in range(len(s)):
            myDict[s[windowEnd]] = myDict.get(s[windowEnd],0)+1
            length = (windowEnd - windowStart) + 1
            if length - max(myDict.values()) > k:
                myDict[s[windowStart]] = myDict.get(s[windowStart],0)-1
                if myDict[s[windowStart]] == 0:
                    del myDict[s[windowStart]]
                windowStart += 1
            else:
                maxLength = max(maxLength,length)
        return maxLength
