class Solution:
    def maxLength(self, arr: List[str]) -> int:
        self.max_length = 0
        def traverse(i, path):
            if len(path) == len(set(path)):
                self.max_length = max(self.max_length, len(path))
            else:
                return
        
            for j in range(i, len(arr)):
                traverse(j, path + arr[j])
                
        traverse(0, "")
        return self.max_length
