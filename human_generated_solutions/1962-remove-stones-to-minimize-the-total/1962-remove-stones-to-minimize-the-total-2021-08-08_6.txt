class Solution:
    def minStoneSum(self, piles: List[int], k: int) -> int:
        import heapq
        q = [-i for i in piles]
        heapq.heapify(q)
        for i in range(k):
            cur = - heapq.heappop(q)
            cur -= cur//2
            heapq.heappush(q, -cur)
        return -sum(q)
