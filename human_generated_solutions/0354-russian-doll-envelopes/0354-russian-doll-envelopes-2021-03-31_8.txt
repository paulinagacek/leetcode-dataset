def maxEnvelopes(self, envelopes: List[List[int]]) -> int:
        envelopes.sort()   # Orders envelopes by width, then by height
        n = len(envelopes) # Number of envelopes
        c = [1] * n        # Count of how many can be nested for each envelope
        
        for i in range(n):    # Calculate c[i] for each envelope
            s = envelopes[i]  # s is the size of the current envelope
            for j in range(i):   # Try all envelopes smaller than s
                if c[j] >= c[i]: # Actually, only check if it would be more envelopes
                    t = envelopes[j]    # t is the size of the envelope to test
                    if s[0] > t[0] and s[1] > t[1]:  # Check if t is smaller than s
                        c[i] = c[j] + 1              # Update c
        
        return max(c)  # This will be the maximum number of envelopes that can be nested
