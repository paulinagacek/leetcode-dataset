class Solution:     
    def longestSubarray(self, nums: List[int]) -> int:

        ans, tally, mx = 0, 0, max(nums)            # <-- 1)

        for n in nums:
            if n == mx: tally += 1                  # <-- 2)
            else: ans, tally = max(ans, tally), 0   # <-- 3)

        return max(ans, tally)                      # <-- 4)
