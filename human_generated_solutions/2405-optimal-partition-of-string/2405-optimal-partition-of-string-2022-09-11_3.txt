# python greedy solution
    def partitionString(self, s):
        """
        :type s: str
        :rtype: int
        """        
        res = 0
        seen = set()
        for c in s:
            if c not in seen:
                seen.add(c)
            else:
                seen.clear()
                res += 1
                seen.add(c)
        res += 1
        return res
