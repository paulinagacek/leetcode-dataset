class Solution:
    def maxProduct(self, words: List[str]) -> int:
        max_len = 0
        for i in words:
            for j in words:
                if i !=j and self.check_words(i , j):
                    max_len = max( len(i) * len(j) , max_len)
        return max_len
    def check_words(self,word1,word2):
        if set(word1) & set(word2):
            return False
        return  True  
		
    def maxProduct(self, words: List[str]) -> int:
        cache = dict()
        for word in words:
            mask = 0
            for c in word:
                mask |= 1 << ord(c)
            cache[mask]= max(cache.get(mask,0), len(word)) 
        max_pr = 0
        for i in cache:
            for j in cache:
                if not i & j:
                    max_pr = max(max_pr , cache[i]*cache[j])
        return max_pr
