class Solution:
    def maxProduct(self, words: List[str]) -> int:
        
        def check(a,b):
            for char in a:
                if char in b:
                    return False
            return True
        
        max_ln = 0
        for indx,word in enumerate(words):
            for nxt_word in words[indx:]:
                prdct = len(word)*len(nxt_word)
                
                if check(word,nxt_word) and max_ln<prdct:
                    max_ln=prdct
                            
        return max_ln
