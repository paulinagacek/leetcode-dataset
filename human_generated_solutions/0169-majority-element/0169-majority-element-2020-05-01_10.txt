def using_hashtable(self, nums):
        counter = collections.Counter(nums)
        for i in set(nums):
            if counter.get(i) > len(nums)//2:
                return i
    
    def brute_force(self, nums):
        # TLE
        count = 0
        required = len(nums)//2
        for i,v in enumerate(nums):
            count = 0
            for j in range(i, len(nums)):
                if v == nums[j]: count += 1
            if count > required: return v
