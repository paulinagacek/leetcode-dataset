class Solution:
    def integerBreak(self, n: int) -> int:
        
        if n <= 3:
            # n = 2 = 1 + 1, product = 1
            # n = 3 = 2 + 1, product = 2
            return n-1
        
        remain, product = n, 1
        
        while remain > 4:
            # Keep decompose with 3
            product *= 3
            remain -=3
            
        return product * remain
