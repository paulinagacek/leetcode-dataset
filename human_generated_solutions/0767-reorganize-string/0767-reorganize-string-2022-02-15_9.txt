class Solution(object):
    def reorganizeString(self, s):
        maxheap =[]
        counter = Counter(s)
        for char, count in counter.items():
            heapq.heappush(maxheap, (-count, char))
        res = ""
        while maxheap:
            if len(maxheap) > 1:
                count1, char1 = heapq.heappop(maxheap)
                count2, char2 = heapq.heappop(maxheap)
                count1, count2 = -count1, -count2
                res += char1
                res += char2
                count1 -=1
                count2 -=1
                if count1 > 0:
                    heappush(maxheap, (-count1, char1))
                if count2 > 0:
                    heappush(maxheap, (-count2, char2))
            else: 
                # if maxheap only has 1 unique element (bc len(maxheap) measures num of unique chars)
                count, char = heapq.heappop(maxheap)
                if -count == 1:
                    res += char
                else:
                    # same element, multiple occurences
                    return "" 
        return res
