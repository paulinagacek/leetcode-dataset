class Solution(object):
    def minimumTotal(self, triangle):
        """
        :type triangle: List[List[int]]
        :rtype: int
        """
		# Edge cases for small input
        if not triangle: 
            return 0
        if len(triangle) < 2:
            return triangle[0][0]
        if len(triangle) < 3:
            return triangle[0][0] + min(triangle[1])
        
        n, m = len(triangle), len(triangle[-1])
        
		# Construct auxillary DP matrix and setting initial value with first in triangle input
        dp = [[0] * (m) for i in range(n)]
        dp[0][0] = triangle[0][0]
        
		# Calculate sums from start to bottom (vertically)
        for i in range(1, n):
            dp[i][0] = triangle[i][0] + dp[i-1][0]
            
		# Calculate sums from start to last value in last list (diagonally)
        for j in range(1, m):
            dp[j][j] = triangle[j][j] + dp[j-1][j-1]
            
		# Now that we\'ve calculated our initial border values we can now use previous info to build up our answer
        for i in range(2, n):
            for j in range(1, len(triangle[i])-1):
                dp[i][j] = min(dp[i-1][j], dp[i-1][j-1]) + triangle[i][j]
        
		# Bottom row will hold the path sums for all possible rows above, just find min of last row
        return min(dp[m-1])
