class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        def dfs(nums, curr) :
            if not nums :
                res.append(curr)
            for x in nums :
                rem = list(nums)
                rem.remove(x)
                dfs(rem, curr + [x])
        res = []
        dfs(nums, [])
        return res
