class Solution:
    def winnerOfGame(self, colors: str) -> bool:
        countA = 0
        countB = 0
        n = len(colors)
        for i in range(n):
            if i == 0 or i == n - 1:
                continue
            if colors[i] == \'A\' and colors[i + 1] == \'A\' and colors[i - 1] == \'A\':
                countA += 1
            elif colors[i] == \'B\' and colors[i + 1] == \'B\' and colors[i - 1] == \'B\':
                countB += 1
                
        return countA > countB
