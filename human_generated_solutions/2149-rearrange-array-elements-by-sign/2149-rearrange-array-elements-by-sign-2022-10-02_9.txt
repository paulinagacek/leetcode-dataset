class Solution:
    def rearrangeArray(self, nums: List[int]) -> List[int]:
        a=[]
        b=[]
        c=[]
        n=len(nums)//2
        for i in nums:
            if i>0:
                a.append(i)
            else:
                b.append(i)
        for i in range(n):
            c.append(a[i])
            c.append(b[i])
        return c
