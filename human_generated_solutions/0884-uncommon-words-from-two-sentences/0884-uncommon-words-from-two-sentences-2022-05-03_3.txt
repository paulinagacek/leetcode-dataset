class Solution:
    def uncommonFromSentences(self, s1: str, s2: str) -> List[str]:
        word_counts = {}
        for word in s1.split(\' \'):
            if word in word_counts:
                word_counts[word] += 1
            else:
                word_counts[word] = 1
        for word in s2.split(\' \'):
            if word in word_counts:
                word_counts[word] += 1
            else:
                word_counts[word] = 1
        return [word for word in word_counts if word_counts[word] == 1]
