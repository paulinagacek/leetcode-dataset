class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:
        if not root:
            return 0
        q=collections.deque()
        q.append((root,1))
        while len(q)>0:
            node,depth=q.popleft()            
            if not node.left and not node.right and not q:
                return depth
            if node.left:
                q.append((node.left,depth+1))
            if node.right:
                q.append((node.right, depth+1))
