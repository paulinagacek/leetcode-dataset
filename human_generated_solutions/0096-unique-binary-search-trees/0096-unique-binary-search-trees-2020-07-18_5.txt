class Solution:
    def numTrees(self, n: int) -> int:
        if n == 0 or n == 1:
            return 1
        unique_trees = 0
        for i in range(n):
            left_trees = self.numTrees(i)
            right_trees = self.numTrees(n-i-1)
            unique_trees += left_trees*right_trees
            
        return unique_trees
