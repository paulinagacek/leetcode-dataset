def numTrees(self, n: int) -> int:
		# index of dp i represents the number of nodes to form a tree -> range is [0, n], length is n+1
        # when 0 or 1 node, there\'s only one way to form a tree
        dp = [0]*(n+1)
        dp[0] = dp[1] = 1
        for i in range(2, len(dp)):
            # number of (current root + left nodes) is from 1 to j
            for j in range(1, i+1):
                # left subtree has j-1 nodes, 1 is current root, then the number of right subtree\'s nodes is i-j
                dp[i] += dp[j-1]*dp[i-j]
        return dp[n] # when there are n nodes
