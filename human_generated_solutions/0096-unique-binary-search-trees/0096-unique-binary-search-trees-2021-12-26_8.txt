import math
class Solution:
    def numTrees(self, n: int) -> int:
        # Total number of possible Binary Search Trees with n different nodes 
        
        # 1, 1, 2, 5, 14, 42, 132, 429, 1430, 4862
        
        # This series is CATALEN NUMBER series 
            
        # Formula = (2n)! / ((n + 1)! * n!) 
        
        
        return math.factorial((2*n)) // (math.factorial((n + 1)) * math.factorial(n))
