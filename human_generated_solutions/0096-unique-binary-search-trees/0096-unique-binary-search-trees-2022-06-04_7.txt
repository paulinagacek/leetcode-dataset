class Solution:
    def numTrees(self, n: int) -> int:
        
        a = [0]*n
        
        def solve(n):
            
            if n <= 1:
                return 1
            if a[ n - 1 ] != 0:
                return a[ n - 1 ]
            k = 0
            for i in range(n):
                k += ( solve(i) * solve( n-i-1 ) )
            a[n-1] = k
            return k
        
        return s( n )
