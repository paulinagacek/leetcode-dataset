class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        @cache
        def fn(i): 
            """Return max value obtained from stoneValue[i:]."""
            if i >= len(stoneValue): return 0 
            ans = -inf
            for ii in range(i, i+3): 
                ans = max(ans, sum(stoneValue[i:ii+1]) - fn(ii+1))
            return ans 
        
        ans = fn(0)
        if ans > 0: return "Alice"
        if ans == 0: return "Tie"
        return "Bob"
