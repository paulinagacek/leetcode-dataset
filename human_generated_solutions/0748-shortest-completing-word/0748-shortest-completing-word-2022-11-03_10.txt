class Solution:
    def shortestCompletingWord(self, licensePlate: str, words: List[str]) -> str:
        hashmap = {}
        for i in licensePlate:
            if i.isalpha():
                try:
                    hashmap[i.lower()] += 1
                except:
                    hashmap[i.lower()] = 1

        words.sort(key = len)

        for i in words:
            flag = 1
            for char, count in hashmap.items():
                if char in i.lower():
                    if count > i.lower().count(char):
                        flag = 0
                        break
                else:
                    flag = 0
                    break
            if flag == 1:
                return i
        return ""
