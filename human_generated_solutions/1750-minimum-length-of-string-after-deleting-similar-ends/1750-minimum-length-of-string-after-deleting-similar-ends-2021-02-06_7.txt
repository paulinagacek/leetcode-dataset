w: two pointers + greedy
h: 1) start from left end and right end
     while left end = right end, move left forward
     then while left end - 1 = right end, move right backward
     
   2) stop conditions:
        1. left > right (we can remove all the letter) OR
        2. left end != right end 
\'\'\'
class Solution:
    def minimumLength(self, s: str) -> int:
        left = 0
        right = len(s) - 1
        
        while left < right:
            while left < right and s[left] == s[right]:
                left += 1
            while left <= right and left > 0 and s[left-1] ==s[right]:
                right -= 1

            if s[left] != s[right]:
                break
                
        return 0 if left > right else right-left+1
