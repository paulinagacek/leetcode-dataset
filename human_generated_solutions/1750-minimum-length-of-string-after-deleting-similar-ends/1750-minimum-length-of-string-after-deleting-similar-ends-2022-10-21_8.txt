class Solution(object):
    def minimumLength(self, s):
        """
        :type s: str
        :rtype: int
        """
        l, r = 0, len(s)-1
        while l < r and s[l] == s[r]:
            while l < len(s)-1 and s[l] == s[l+1]: l += 1
            while r > 0 and s[r] == s[r-1]: r -= 1
            l += 1
            r -= 1
            if l > r: return 0
        return r - l + 1
