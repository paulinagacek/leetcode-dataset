class Solution:
    def minimumLength(self, s: str) -> int:
        l, r = 0, len(s) - 1    # left and right pointers
        while l < r and s[l] == s[r]:
            ele = s[l]
            while s[l] == ele and l < r:
                l += 1
            if l == r:  return 0    # the entire string will be wiped out
            while s[r] == ele:
                r -= 1
        return r - l + 1
