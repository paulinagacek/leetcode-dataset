class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        def combination(i, target, ds, res):
            if i == len(candidates):
                if target == 0 and ds not in res:
                    res.append(ds.copy())
                return
            if candidates[i] <= target:
                ds.append(candidates[i])
                combination(i, target - candidates[i], ds, res)
                ds.pop()
            combination(i + 1, target, ds, res)
            return res
        return combination(0, target, ds = [], res = [])
