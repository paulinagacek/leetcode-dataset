def removeDuplicates(self, nums: List[int]) -> int:
        cur_idx, count, n = 0, 1, len(nums)
        for i in range(n):
            count = count+1 if i+1 < n and nums[i] == nums[i+1] else 1
            nums[i], nums[cur_idx] = nums[cur_idx], nums[i]
            cur_idx += (count <= 2)
        return cur_idx
