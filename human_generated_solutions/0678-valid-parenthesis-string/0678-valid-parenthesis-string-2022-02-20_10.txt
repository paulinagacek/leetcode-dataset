def checkValidString(self, s: str) -> bool:
        open_stack = []
        star_stack = []
        for i in range(len(s)):
            if s[i] == \'(\':
                open_stack.append(i)
            elif s[i] == "*":
                star_stack.append(i)
            else:
                if len(open_stack) != 0:
                    open_stack.pop()
                elif len(star_stack) != 0:
                    star_stack.pop()
                else:
                    return False
        while len(open_stack) != 0:
            if len(star_stack) == 0:
                return False
            elif open_stack[-1] < star_stack[-1]:
                open_stack.pop()
                star_stack.pop()
            else:
                return False
        return True
