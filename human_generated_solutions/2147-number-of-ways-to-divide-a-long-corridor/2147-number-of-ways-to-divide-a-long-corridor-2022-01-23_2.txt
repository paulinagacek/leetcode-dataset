class Solution:
    def numberOfWays(self, corridor: str) -> int:
        n = len(corridor)
        scount = 0 # count the number of seats
        for i in range(n):
            if corridor[i] == "S":
                scount += 1
        
        # return 0 if number of seats are odd (or) zero
        if scount % 2 != 0 or scount == 0:
            return 0
        
        # count number of plants in between two dividors
        ans, seats, plants = 1, 0, 0
        for i in range(n): 
            if seats == 2 and corridor[i] == "P":
                plants += 1
            elif seats == 2 and corridor[i] == "S":
                ans = ans * (plants + 1) # answer is combinations of number of plants
                plants = 0
                seats = 1
            elif corridor[i] == "S":
                seats += 1
        return ans % ((10 ** 9) + 7)
