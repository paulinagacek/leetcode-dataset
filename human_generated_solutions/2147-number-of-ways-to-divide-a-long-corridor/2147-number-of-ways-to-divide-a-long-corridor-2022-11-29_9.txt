class Solution:
    def numberOfWays(self, corridor: str) -> int:
        
        l=len(corridor)
        count=0
        last=-1
        ans=1
        for i in range(l):
            if corridor[i]==\'S\':
                count+=1
                if count%2==0:
                    last=i
                elif count%2==1 and last!=-1:
                    ans*=(i-last)
        if count%2!=0 or count==0:
            return 0
        return ans%1000000007
