class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        length = len(lists)
        
		# The list is empty. 
        if not length:
            return None
        # There is only one Linked list - base condition. return the Linked list as is, because this is sorted
        if length == 1:
            return lists[0]
        
        mid = length//2
		# get the left sorted half and right sorted half
        left = self.mergeKLists(lists[0:mid])
        right = self.mergeKLists(lists[mid:])
        
        return self.mergeTwolists(left,right)
        
    # Merge two lists is another Leetcode question.  Visit - https://leetcode.com/problems/merge-two-sorted-lists/
    def mergeTwolists(self, l1, l2):
        tail = head = ListNode(0)        
        while l1 and l2:
            if l1.val < l2.val:
                tail.next = l1
                l1 = l1.next
            else:
                tail.next = l2
                l2 = l2.next
                
            tail = tail.next
            
        tail.next = l1 or l2
            
        return head.next
