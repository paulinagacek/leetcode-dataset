class Solution:
    def maxArea(self, h: int, w: int, horizontalCuts: List[int], verticalCuts: List[int]) -> int:
		# We need to append 0 and h to horizontalCuts, since they are a line too
        horizontalCuts.append(0)
        horizontalCuts.append(h)
        
		# We need to append 0 and w to verticalCuts, since they are a line too
        verticalCuts.append(0)
        verticalCuts.append(w)
        
		# Sort in order to compare adjacent lines
        horizontalCuts.sort()
        verticalCuts.sort()
        
        best_vertical_distance = 0
        best_horizontal_distance = 0
        
		# Find largest vertical distance
        for vertical_index in range(1, len(verticalCuts)):
            vertical_distance_between_adjacent = verticalCuts[vertical_index] - verticalCuts[vertical_index - 1]
            best_vertical_distance = max(best_vertical_distance, vertical_distance_between_adjacent)
    
		# find largest horizontal distance
        for horizontal_index in range(1, len(horizontalCuts)):
            horizontal_distance_between_adjacent = horizontalCuts[horizontal_index] - horizontalCuts[horizontal_index - 1]
            best_horizontal_distance = max(best_horizontal_distance, horizontal_distance_between_adjacent)
            
        
		# multiply the distances. Don\'t forget the mod
        return (best_vertical_distance * best_horizontal_distance) % (pow(10, 9) + 7)
