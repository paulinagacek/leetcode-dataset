class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if head==None or head.next==None:
            return head
			
        #fakeHead
        newHead=ListNode(\'a\',head)
        prev=head
        curr=head.next
        lastKnown=newHead
        flag=0
        while(curr):
            if curr.val==prev.val:
                flag=1
            else:
                if flag==1:
                    lastKnown.next=curr
                    flag=0
                else:
                    lastKnown=prev
            prev=curr
            curr=curr.next
        if flag==1:
            lastKnown.next=curr
        return newHead.next
