class Solution:
    def rangeSumBST(self, root: Optional[TreeNode], low: int, high: int) -> int:
        if root is None:
            return 0
        s=0
        if root.val>=low and root.val<=high:
            s=s+root.val
        s=s+self.rangeSumBST(root.left,low,high)
        s=s+self.rangeSumBST(root.right,low,high)
        return s
