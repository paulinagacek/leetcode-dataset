class Solution:
    def levelOrder(self, root: TreeNode) -> List[List[int]]:
        if not root: return []
        stack = [root]
        temp = root
        out = []
        push = []
        while len(stack):
            arr = []
            push = []
            while len(stack):
                temp = stack.pop(0)
                push.append(temp.val)
                if temp.left:
                    arr.append(temp.left)
                if temp.right: 
                    arr.append(temp.right)    
            out.append(push)
            stack = arr
        return out
