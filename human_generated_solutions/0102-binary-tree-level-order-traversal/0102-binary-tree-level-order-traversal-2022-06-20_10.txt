class Solution:
    def levelOrder(self, root):
        res = []
        if not root: return res
        queue = [root]
        
        while queue:
            cur_level = []
            for i in range(len(queue)):
                node = queue.pop(0)
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
                cur_level.append(node.val)
            res.append(cur_level)
        return res
