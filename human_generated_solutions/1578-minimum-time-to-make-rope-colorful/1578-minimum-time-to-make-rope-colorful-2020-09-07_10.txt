class Solution:
    def minCost(self, s: str, c: List[int]) -> int:
        res,st,end = 0,0,1
        while end < len(s):
            if s[st] == s[end]:
                if c[st] < c[end]:
                    res += c[st]
                    st = end
                else:
                    res += c[end]
            else:
                st = end
            end += 1
        return res
