class Solution:
    def validateStackSequences(self, pushed, popped):
        stack = []
        j = 0
        for i in pushed:
            stack.append(i)
            while stack[-1] == popped[j]:
                stack.pop()
                j += 1
                if len(stack) == 0:
                    break
                
        if len(stack) == 0:
            return True
        else:
            return False
