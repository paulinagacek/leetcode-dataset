class Solution:
    def arrayPairSum(self, nums: List[int]) -> int:
        n=len(nums)//2
        nums.sort()
        i=0
        j=n
        if n%2!=0:
            j+=1
        ans=0
        while(i<n):
            ans+=nums[i]
            if j<n+n:
                ans+=nums[j]
            i+=2
            j+=2
        return ans
