class Solution:
    def equalPairs(self, grid: List[List[int]]) -> int:
        
        row=len(grid)
        col=len(grid[0])
        
        rez = [[grid[j][i] for j in range(len(grid))] for i in range(len(grid[0]))]
        count =0
        for el in rez:
            
            if el in grid:
                count=count + grid.count(el)
        return count
