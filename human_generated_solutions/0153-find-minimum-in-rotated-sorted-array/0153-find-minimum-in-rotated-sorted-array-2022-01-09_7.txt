class Solution:
    def findMin(self, a: List[int]) -> int:
        left, right = 0,len(a)-1
        
        while(left<right):
            mid = left + (right-left)//2
            
            if(a[mid]>=a[right]):
                left = mid+1
                
            elif(a[mid]<=a[right]):
                right = mid
                
        return a[right]
