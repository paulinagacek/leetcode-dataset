class Solution(object):
    def removeKdigits(self, num, k):
        if len(num)<=k:
            return "0"
        elif k==0:
            return str(int(num))
        else:
            ansr = ""
            for i in num:
                while k and ansr and ansr[-1] > i:
                    k -= 1
                    ansr = ansr[:-1]
                if ansr or i!="0":
                    ansr += i
            for i in range(k):
                ansr = ansr[:-1]
            return ansr or "0"
