class Solution:
    def longestIdealString(self, s: str, k: int) -> int:
        freq = [0] * 26
        key = lambda x: abs(ord(x) - ord(\'a\'))
        
        for ch in s[::-1]:
            ch_key = key(ch)
            
            mx = max(freq[max(0, ch_key - k):min(26, ch_key + k + 1)])
            freq[ch_key] = mx + 1
            
        return max(freq)
