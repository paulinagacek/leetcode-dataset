class Solution:
    def kidsWithCandies(self, candies: List[int], extraCandies: int) -> List[bool]:
        highest = max(candies)
        values = []
        for i in range(0, len(candies)):
            reference = candies[i] + extraCandies
            if reference >= highest:
                values += [True]
            else:
                values += [False]
        return values
