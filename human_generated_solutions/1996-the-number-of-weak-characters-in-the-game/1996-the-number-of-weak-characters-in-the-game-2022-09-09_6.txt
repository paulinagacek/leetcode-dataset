class Solution:
    def numberOfWeakCharacters(self, A: List[List[int]]) -> int:
        A.sort(key=lambda x: (x[0], x[1]))
        n = len(A)
        heap = []
        weak = 0
        i = 0
        
        while i < n:
            attack, defense = A[i]
            level_defense = [defense]
			
			# find the largest defense for the same attack level
            while i+1 < n and A[i+1][0] == attack:
                level_defense.append(A[i+1][1])
                i += 1
            
			# the maximum defense for the attack level
            max_defense_on_level = level_defense[-1]
			
			# find how many elements have smaller defense
            while heap and heap[0] < max_defense_on_level:
                heapq.heappop(heap)
                weak += 1
            
			# add the current defense layer to our heap. We do it here to not interfere with the check above
            for ldefense in level_defense:
                heapq.heappush(heap, ldefense)
            
            i += 1
        
        return weak
