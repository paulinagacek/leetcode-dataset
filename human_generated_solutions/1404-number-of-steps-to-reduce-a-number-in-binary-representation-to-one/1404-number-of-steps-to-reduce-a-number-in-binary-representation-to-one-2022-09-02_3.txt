def numSteps(self, s):
        # Assignment
        num = int(s, 2)  # Turn binary str into its decimal counterpart
        moves = 0
    
        if num == 1:  # Threshold
            return moves

        # Reducing num into 1 while incrementing moves
        while num != 1:
            # If odd, make it even by addition
            if num % 2 == 1: 
                num += 1
                moves += 1
            # Inloop Threshold
            if num == 1:  
                return moves
            # Use division by 2 as much as possible
            num //= 2
            moves += 1

        return moves
