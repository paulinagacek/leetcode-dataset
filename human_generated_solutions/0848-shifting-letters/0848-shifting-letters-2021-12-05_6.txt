class Solution(object):    
    def shiftingLetters(self, s, shifts):
        fin = ""
        cumusum = shifts[len(shifts)-1]
        
        for sbk in range(len(shifts)-1, -1, -1):
			# we use cumulative shift % 26 since it shifts back to the original character every 26th anyways
			# then pass the original character and the shift number to getShiftedTo to get the shifted character
			# prepend to fin because we start iteration from the last letter
            fin = self.getShiftedTo(s[sbk], cumusum % 26) + fin 
			
			# the shift we will be making to nth character is the cumulative sum from the n to last shift
            cumusum += shifts[sbk-1]

        return fin
