class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        self.d={}
        self.trav(root, 0)
        return self.d.values()
        
    def trav(self, root, l):
        if root:
            if self.d.get(l, False): self.d[l].append(root.val)
            else: self.d[l] = [root.val]
            for node in root.children: self.trav(node, l+1)
