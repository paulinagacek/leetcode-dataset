class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        d={}        
        
        def trav(root, n):
            if(not root):
                return
            
            if(n not in d):
                d[n]=[root.val]
            else:
                d[n]+=[root.val]
            
            for x in root.children:
                trav(x, n+1)
        
        trav(root, 0)
        
        ans=list(d.values())
        
        return ans
