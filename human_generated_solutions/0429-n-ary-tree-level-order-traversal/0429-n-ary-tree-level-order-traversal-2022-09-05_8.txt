class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if not root: return []
        res = []
        levels = set()
        def dfs(node, level):
            if level not in levels:
                levels.add(level)
                res.append([])
            res[level].append(node.val)
            for child in node.children:
                dfs(child, level + 1)
                    
        dfs(root, 0)
        return res
