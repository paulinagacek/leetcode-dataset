def numSubmatrixSumTarget(self, mat: List[List[int]], target: int) -> int:
	m, n = len(mat), len(mat[0])
	for i in range(m):
		for j in range(1, n):
			mat[i][j] += mat[i][j-1]
		mat[i] = [0] + mat[i]
	ans = 0
	d = defaultdict(int)
	for col1 in range(n):
		for col2 in range(col1+1, n+1):
			temp = 0
			d[0] = 1
			for r in range(m):
				temp += mat[r][col2] - mat[r][col1]
				if(temp - target in d):
					ans += d[temp - target]
				d[temp] += 1
			d.clear()
	return ans
