class Solution:
    def missingRolls(self, rolls: List[int], mean: int, n: int) -> List[int]:
        m = len(rolls)
        remaining_sum = mean * (m + n) - sum(rolls)

        if not n <= remaining_sum <= 6 * n:
            return []

        # return self.generate_array_for_given_sum_method1(remaining_sum, n)
        return self.generate_array_for_given_sum_method2(remaining_sum, n)

    @staticmethod
    def generate_array_for_given_sum_method1(s: int, n: int) -> List[int]:
        dice_val, output = 6, []

        while s > 0:
            if s - dice_val >= n - 1:
                output.append(dice_val)

                s -= dice_val
                n -= 1
            else:
                dice_val -= 1

        return output

    @staticmethod
    def generate_array_for_given_sum_method2(s: int, n: int) -> List[int]:
        q, r = divmod(s, n)  # s = q * n + r

        output = [q] * n

        # since sum(output) = q * n so adding rest of "r" in output
        for i in range(r):
            output[i] += 1

        return output
