import string

class Solution:
    def validIPAddress(self, IP: str) -> str:
        def isIPv4(inputStr):
            # groups sep by \'.\'
            splitInput = inputStr.split(\'.\') 
			
            # no length > 4
            if len(splitInput) != 4:
                return False
				
            for group in splitInput:
                 # must have some value for a number
                if len(group) == 0:
                    return False
					
                # confirm no leading 0
                if group[0] == \'0\' and len(group) > 1: 
                    return False
					
                # decimal numbers (no alphabetical) in [0, 255]
                if not group.isdigit() or int(group) < 0 or int(group) > 255:
                    return False
					
            return True
        
        def isIPv6(inputStr):
            # groups sep by \':\'
            splitInput = inputStr.split(\':\') 
			
			 # 8 groups
            if len(splitInput) != 8:
                return False
            
            for group in splitInput:
                # each group 4 hex digits, but can ommit leading 0s
                if len(group) > 4 or len(group) < 1: 
                    return False
					
                # hex only
                for char in group:
                    if char not in string.hexdigits:
                        return False
            return True
                
        if isIPv4(IP):
            return \'IPv4\'
        if isIPv6(IP):
            return \'IPv6\'
        return \'Neither\'
