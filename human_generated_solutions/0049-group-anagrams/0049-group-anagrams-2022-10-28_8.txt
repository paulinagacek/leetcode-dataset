class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        dct=defaultdict(list)
        for i in range(len(strs)):
            st=[0 for _ in range(26)]
            for j in strs[i]:
                xi=ord(j)-97
                st[xi]+=1
            st=tuple(st)
            dct[st].append(strs[i])
        return dct.values()
