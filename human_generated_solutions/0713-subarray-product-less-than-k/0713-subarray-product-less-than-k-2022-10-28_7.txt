class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        i = j = res = 0
        mul = 1
        
        while j < len(nums):
            mul *= nums[j]
            
            while i < j and mul >= k: 
                mul //= nums[i]
                i += 1
            
            if mul < k: res += j - i + 1
            j += 1
            
        return res
