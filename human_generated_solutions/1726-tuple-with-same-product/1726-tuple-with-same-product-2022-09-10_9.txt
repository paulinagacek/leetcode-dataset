def tupleSameProduct(self, nums: List[int]) -> int:
        d = collections.defaultdict(lambda:0)
        for i in range(len(nums)):
            for j in range(i+1,len(nums)):
                d[nums[i]*nums[j]]+=1
        ans = 0
        for i in range(len(nums)):
            for j in range(i+1,len(nums)):
                ans += d[nums[i]*nums[j]]
        ans -= (len(nums)*(len(nums)-1))//2
        ans*=4
        return ans
