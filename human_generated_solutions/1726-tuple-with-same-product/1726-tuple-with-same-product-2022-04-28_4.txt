class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        
        n=len(nums)
        count=defaultdict(int)
        
        for i in range(n):
            for j in range(i+1,n):
                count[nums[i]*nums[j]]+=1
                
        res=0
        for i,j in count.items():
            res+=(j*(j-1))
        
        return res*4
