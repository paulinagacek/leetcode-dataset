class Solution:
    def tupleSameProduct(self, nums: List[int]) -> int:
        hashmap = {}
        for i in range(len(nums)):
            for j in range(i+1,len(nums)):
                product = nums[i]*nums[j]
                if product in hashmap.keys():
                    hashmap[product] += 1
                
                else:
                    hashmap[product] = 1
                    
        ans = 0
        for key,value in hashmap.items():
            if value==1:
                continue
            ans += ((value *(value-1))//2) * 8
            
        return ans
