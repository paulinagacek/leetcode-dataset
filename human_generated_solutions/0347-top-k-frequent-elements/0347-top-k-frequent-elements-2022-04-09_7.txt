class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        from collections import Counter
        ranked = [[] for _ in range(len(nums) + 1)]
        c = Counter(nums)
        for key, val in c.items():
            ranked[val].append(key)
        return list(chain(*ranked))[len(c) - k:]
