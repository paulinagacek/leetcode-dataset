def numRescueBoats(self, people: List[int], limit: int) -> int:
        n=len(people)
        people.sort()
        i=0
        j=n-1
        count=0
        while(i<=j):
            if (people[i]+people[j]<=limit):
                i+=1
                j-=1
                count+=1
            else:
                j-=1
                count+=1
        return count
