class Solution:
    def partitionArray(self, nums: List[int], k: int) -> int:
        nums.sort()
        l=[[nums[0]]]
        for i in range(1,len(nums)):
            if nums[i]-l[-1][0]<=k:
                l[-1].append(nums[i])
            
            else:
                l.append([nums[i]])
                
        return len(l)
