class Solution:
    def partitionArray(self, nums: List[int], k: int) -> int:
        nums.sort()
        res, i, j = 1, 0, 1
        while j < len(nums):
            if nums[j] - nums[i] <= k:
                j += 1
            else:
                i = j
                j += 1
                res += 1
        return res
