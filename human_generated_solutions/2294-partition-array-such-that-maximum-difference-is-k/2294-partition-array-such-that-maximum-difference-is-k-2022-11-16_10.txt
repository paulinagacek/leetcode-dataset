class Solution(object):
    def partitionArray(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        nums.sort()
        minimum = nums[0]
        count = 0
        for i in range(1,len(nums)):
            if (nums[i] - minimum) > k :
                count += 1
                minimum = nums[i]
        return count + 1
