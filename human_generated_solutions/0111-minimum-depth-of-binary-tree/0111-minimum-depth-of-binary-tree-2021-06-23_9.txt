from collections import deque

class Solution:
    def minDepth(self, root: TreeNode) -> int:
        if not root: return 0
        
        level = 0
        queue = deque()
        queue.append(root)
        while queue:
            n = len(queue)
            level += 1
            
            for i in range(n):
                cur = queue.popleft()
                if not cur.left and not cur.right:
                    return level
                if cur.left:
                    queue.append(cur.left)
                if cur.right:
                    queue.append(cur.right)
