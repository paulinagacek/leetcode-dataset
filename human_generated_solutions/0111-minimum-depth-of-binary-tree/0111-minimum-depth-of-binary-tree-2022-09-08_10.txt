def minDepth(self, root: Optional[TreeNode]) -> int:
        if(root==None):
            return 0
        else:
            lh,rh=self.minDepth(root.left),self.minDepth(root.right)
            if(lh==0 or rh==0):
                return 1+max(lh,rh)
            else:
                return 1+min(lh,rh)
